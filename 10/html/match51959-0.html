Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;

    public Rectangle() {};
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    @Override
    public double getArea() {
        return width * length;
    }
    @Override
    public double getPerimeter() {
        return (width + length) * 2.00;
    }
    public String toString() {
        return "Rectangle[width=" + width + ",length=" + length + ",color=" + color +
                ",filled=" + filled + "]";
    }

    public static void main(String[] args) {
        Rectangle rec1 = new Rectangle();
        System.out.println(rec1.toString());
        Rectangle rec2 = new Rectangle(1.0, 2.0, "blue", false);
        System.out.println(rec2.getWidth());
        System.out.println(rec2.getLength());
        System.out.println(rec2.getArea());
        System.out.println(rec2.getPerimeter());
        System.out.println(rec2.getColor());
        System.out.println(rec2.toString());
        Rectangle rec3 = rec2;
        rec3.setWidth(2.0);
        rec3.setLength(4.0);
        rec3.setColor("green");
        rec3.setFilled(true);
        System.out.println(rec3.getArea());
        System.out.println(rec3.getPerimeter());
        System.out.println(rec3.toString());
        Shape rec4 = new Rectangle(2.0, 4.0);
        System.out.println(rec4.toString());
        System.out.println(rec4.getArea());
    }

}
