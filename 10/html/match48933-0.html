//package com.company;
import java.lang.Math;
public class Circle extends Shape {
    protected double radius;

    public Circle(){}
    public Circle(double radius){
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled){
        super(color, filled);
        this.radius=radius;
    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public double getArea(){
        return Math.PI*radius*radius;
    }
    @Override
    public double getPerimeter(){
        return 2*Math.PI*radius;
    }
    public String toString(){
        return "Circle[radius="+radius+",color="+super.color+",filled="+super.filled+"]";
    }
}

//package com.company;

public class Square extends Rectangle {
    public Square(){}
    public Square(double side){
        super(side,side);
    }
    public Square(double side,String color,boolean filled){
        super(side,side,color,filled);
    }
    public double getSide(){return super.length;}
    public void setSide(double side){super.length=side;super.width=side;}
    @Override
    public void setWidth(double side) {
        super.width=side;
    }
    @Override
    public void setLength(double side) {
        super.length=side;
    }
    public String toString(){
        return "Square[side="+super.length+",color="+super.color+",filled="+super.filled+"]";
    }
}
