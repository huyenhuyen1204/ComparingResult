Circle.java

public class Circle extends Shape{
    protected double radius;
    protected final double PI = Math.PI;

    public Circle(){

    }

    public Circle( double radius){
        this.radius = radius;
    }

    public Circle ( double radius, String color, boolean filled) {
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return PI * this.radius * this.radius;
    }

    @Override
    public double getPerimeter() {
        return 2.0 * PI * this.radius;
    }

    @Override
    public String toString() {
        return "Circle[radius=" + this.radius + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }

    public static void main(String[] args) {
        Shape a = new Circle(11, "red", true);
        System.out.println(a);
        System.out.println(a.getArea());
        System.out.println(a.getPerimeter());
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;

    public Rectangle(){

    }

    public Rectangle(double width, double length){
        this.length = length;
        this.width = width;
    }

    public Rectangle ( double width, double length, String color, boolean filled) {
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return this.length * this.width;
    }

    @Override
    public double getPerimeter() {
        return 2 * ( this.width + this.length);
    }

    @Override
    public String toString() {
        return "Rectangle[width=" + this.width + ",length=" + this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

    public static void main(String[] args) {
        Shape b = new Rectangle(4, 5, "red", false);
        System.out.println(b.getArea());
        System.out.println(b.getPerimeter());
        //System.out.println(b.toString());
    }
}

public class Square extends  Rectangle{

    public Square() {

    }

    public Square ( double side) {
        setWidth(side);
        setLength(side);
    }

    public Square ( double side, String color, boolean filled) {
        super(side, side, color, filled);
    }

    public double getSide() {
        return this.getLength();
    }

    public void setSide(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public void setWidth(double side) {
        this.setSide(side);
    }

    @Override
    public void setLength(double side) {
        this.setSide(side);
    }

    @Override
    public String toString() {
        return "Square[side=" + getSide() + ",color=" + this.getColor() + ",filled=" +  filled + "]";
    }

    public static void main(String[] args) {
        Rectangle c = new Square(5, "red", true);
        System.out.println(c.getArea());
        System.out.println(c.getPerimeter());
        System.out.println(c.getLength());
        System.out.println(c.getWidth());
    }
}
