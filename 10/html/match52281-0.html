/**
 * @author Nguyen Quang Huy
 * @version 1.0
 * @since 12/08/2019
 */
public class Circle extends Shape {
    protected double radius;

    /**
     * Phuong thuc khoi tao khong tham so
     */
    public Circle() {
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param radius
     */
    public Circle(double radius) {
        this.radius = radius;
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param color
     * @param filled
     * @param radius
     */
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    /**
     * Getter cho radius
     * @return ban kinh hinh tron
     */
    public double getRadius() {
        return radius;
    }

    /**
     * Setter cho radius
     * @param radius
     */
    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return Math.PI * getRadius() * getRadius();
    }

    @Override
    public double getPerimeter() {
        return 2 * Math.PI * getRadius();
    }

    @Override
    public String toString() {
        return "Circle[" +
                "radius=" + radius +
                ",color=" + super.color +
                ",filled=" + super.filled +
                ']';
    }
}

/**
 * @author Nguyen Quang Huy
 * @version 1.0
 * @since 12/08/2019
 */
public class Square extends Rectangle {
    /**
     * Phuong thuc khoi tao khong tham so
     */
    public Square() {
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param side
     */
    public Square(double side) {
        super(side);
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param side
     * @param color
     * @param filled
     */
    public Square(double side, String color, boolean filled) {
        super(side, color, filled);
    }

    /**
     * Getter cho side
     * @return do dai canh hinh vuong
     */
    public double getSide() {
        return super.length;
    }

    /**
     * Setter cho side
     * @param side
     */
    public void setSide(double side) {
        super.width = side;
        super.length = side;
    }

    /**
     * Setter cho length
     * @param length
     */
    @Override
    public void setLength(double length) {
        super.length = length;
        super.width = length;
    }

    /**
     * Setter cho width
     * @param width
     */
    @Override
    public void setWidth(double width) {
        super.width = width;
        super.length = width;
    }

    @Override
    public String toString() {
        return "Square[" +
                "side=" + super.length +
                ",color=" + super.color +
                ",filled=" + super.filled +
                "]";
    }
}
