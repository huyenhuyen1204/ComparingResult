Circle.java


public class Circle extends Shape {
    protected double radius;
    public Circle() {
    }

    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle( double radius, String color, boolean filled) {
        super(color,filled);
        this.radius = radius;
    }
     public double getRadius() {
        return this.radius;
     }
     public void setRadius(double radius) {
        this.radius = radius;
     }
     public double getArea() {
        double r = (double)Math.round(radius * 100) / 100;
        return (double)Math.PI * r * r;
     }
     public double getPerimeter() {
        return (double)2 * Math.PI * radius;
     }
     public String toString() {
        return "Circle[radius=" + this.getRadius() + ",color=" + this.getColor() + ",filled=" + this.isFilled() +"]";
     }

}

Rectangle.java


public class Rectangle extends Shape {
    protected  double width;
    protected  double length;
    public Rectangle() {

    }
    public Rectangle(double width , double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width,double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public double getWidth() {
        return this.width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return this.length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public double getArea() {
        return (double)Math.round(this.getLength() * this.getWidth() * 100) / 100;
    }
    public double getPerimeter() {
        return (double)2 * (this.getWidth() + this.getLength());
    }
    public String toString() {
        return "Rectangle[width=" + this.getWidth() + ",length=" + this.getLength() + ",color=" + this.getColor() + ",filled=" + this.isFilled() +"]";
    }

}
