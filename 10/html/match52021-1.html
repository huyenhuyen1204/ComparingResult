public class Rectangle extends Shape {
    protected double width;
    protected double length;
    public 
Rectangle(){}
    public Rectangle(double w, double l){
        width = w;
        length = l;
    }
    public Rectangle(double w, double l, String _color, boolean _fill){
        width  = w;
        length = l;
        color  = _color;
        filled = _fill;
    }

    public double getWidth() {
        return width;
    }
    public double getLength(){
        return  length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }
    public double getArea(){
        return this.width*this.length;
    }
    public double getPerimeter(){
        return (this.width+this.length)*2.0;
    }
    public String toString(){
        return "Rectangle[width=" +this.width+",length=" + this.length + ",color=" +this.color +",filled="+this.filled +"]";
    }

}

        width  = _side;

    }
    public double getSide(){
        return  this.width;
    }

    public void setWidth(double width) {
        this.width = width;
        
this.length = width;
    }

    public void setLength(double length) {
        this.width = length;
        this.length = length;
    }
    public void setSide(double Side){
        this.length = Side;
        this.width = Side;
    }
    public String toString(){
        return "Square[side="+this.width +",color=" +this.color +",filled="+this.filled+"]";
