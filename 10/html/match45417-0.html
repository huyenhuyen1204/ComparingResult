Circle.java


public class Circle extends Shape{
	protected double radius;
	final private double PI = Math.PI; 
	
	public Circle() {this.radius = 0.0;}
	public Circle(double r) {this.radius = r;}
	public Circle(double r, String c, boolean f) {super(c, f); this.radius = r;}
	
	public void setRadius(double x) {this.radius = x;}
	public double getRadius() {return this.radius;}
	
	@Override 
	public double getArea(){
		return this.radius * this.radius * PI;
	}
	@Override
	public double getPerimeter(){
		return this.radius*2.0*PI;
	}
	
	public String toString(){
		return ("Circle[radius=" + this.radius +  ",color=" + this.color + ",filled=" + this.filled + "]");
	}
}

Rectangle.java

public class Rectangle extends Shape{
	protected double width;
	protected double length;
	
	public Rectangle() {
		this.width = 0.0;
		this.length = 0.0;
	}
	public Rectangle(double w, double l){
		this.width = w;
		this.length = l;
	}
	public Rectangle(double w, double l, String c, boolean f){
		super(c, f);
		this.width = w;
		this.length = l;
	}
	
	public double getWidth() {return this.width;}
	public void setWidth(double x) {this.width = x;}
	
	public double getLength() {return this.length;}
	public void setLength(double x) {this.length = x;}
	
	@Override
	public double getArea() {return (this.width * this.length);}
	@Override
	public double getPerimeter() {return (this.width + this.length) * 2.0;}
	
	public String toString(){
		return ("Rectangle[width=" + this.width + ",length=" + this.length 
				+ ",color=" + this.color + ",filled=" + this.filled + "]");
	}

}

Shape.java

public abstract class Shape {
	protected String color;
	protected boolean filled;
	
	public Shape() {this.filled = false;}
	public Shape(String c, boolean f) {this.color = c; this.filled = f;}
	public void setColor(String c) {this.color = c;}
	public String getColor() {return this.color;}
	public boolean isFilled() {return this.filled;}
	public void setFilled(boolean f) {this.filled = f;}
	abstract public double getArea();
	abstract public double getPerimeter();
	public String toString(){
		return ("Shape[color=" + this.color + "],filled=" + this.filled + "]"); 
	}
	
}

Square.java

public class Square extends Rectangle{

	public Square() {super(0.0, 0.0);}
	public Square(double x) {super(x, x);}
	public Square(double x, String c, boolean f) {super(x, x, c, f);}
	
	public void setSide(double x){
		super.setWidth(x);
		super.setLength(x);
	}
	public double getSide(){
		return super.getLength();
	}
	
	@Override
	public void setWidth(double x){
		super.setWidth(x);
		super.setLength(x);
	}
	@Override
	public double getWidth(){
		return super.getLength();
	}
	
	@Override
	public void setLength(double x){
		super.setWidth(x);
		super.setLength(x);
	}
	@Override
	public double getLength(){
		return super.getLength();
	}
	
	public String toString(){
		return ("Square[side=" + this.getLength() + ",color=" + this.color + ",filled=" + this.filled + "]");
	}
}
