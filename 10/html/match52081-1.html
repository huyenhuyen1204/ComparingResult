public class Circle extends Shape {
    protected double radius;
    public Circle(){
    }
    public Circle( double _radius){
        this.radius=_radius;
    }
    public Circle(double _radius, String _color, boolean _filled){
        super(_color, _filled);
        this.radius=_radius;

    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    public double getArea(){
        return 
this.getRadius()*this.getRadius()*Math.PI;
    }
    public double getPerimeter(){
        return 2*this.getRadius()*Math.PI;
    }
    public String toString(){
        return "Circle[radius="+this.getRadius()+",color="+super.getColor()+ ",filled="+super.isFilled()+"]";
    }
}
