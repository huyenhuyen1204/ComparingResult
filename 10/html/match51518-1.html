class Square extends Rectangle {

    public Square(){

    }

    public Square(double side){
        width = side;
        length = side;
    }

    public Square(double side, String color, boolean filled){
        this.color = color;
        this.filled = filled;
        width = side;
        length = side;
    }
    public double getSide()
    {
        return getWidth();
    }

    public void setSide(double side)
    {
        setWidth(side);
        setLength(side);
    
}
    public void setWidth(double width){
        this.width = width;
        this.length = width;
    }

    public void setLength(double length){
        this.width = length;
        this.length = length;
    }

    public String toString(){
        return "Square[side=" + getSide() + ",color=" + super.toString()
                + ",filled=" + super.toString();
    }
}
