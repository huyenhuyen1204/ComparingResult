import java.util.Scanner;

public class Fraction {
    int numerator, denominator;
    public Fraction() {
        this.denominator = 1;
    }
    public int getNumerator() {
        return numerator;
    }
    public void setNumerator(int _numerator) {
            this.numerator = _numerator;
    }
    public int getDenominator() {
        return denominator;
    }
    public void setDenominator(int _denominator) {
        if (_denominator == 0) {
            this.denominator = 1;
        }
        else {
            this.denominator = _denominator;
        }
    }

    public Fraction(int numerator1, int denominator1) {
        this.numerator = numerator1;
        if (denominator1 == 0) {
            this.denominator = 1;
        }
        else {
            this.denominator = denominator1;
        }
    }

    public int gcd(int a, int b) {
        int x = Math.min(a, b);
        int uc = 1;
        for (int i = x; i >= 1 ; i--) {
            if ((a % i == 0) && (b % i == 0)) {
                uc = i;
                break;
            }
        }
        return uc;
    }

    public Fraction reduce() {
        int uc = gcd (numerator, denominator);
        this.numerator = this.getNumerator() / uc;
        this.denominator = this.getDenominator()  /uc;
        return this;
    }

    public Fraction add(Fraction otherFraction) {
        this.numerator = this.numerator * otherFraction.denominator + otherFraction.numerator * this.denominator;
        this.denominator = this.denominator * otherFraction.denominator;
        reduce();
        return this;
    }

    public Fraction subtract(Fraction otherFraction) {
        this.numerator = this.numerator * otherFraction.denominator - otherFraction.numerator * this.denominator;
        this.denominator = this.denominator * otherFraction.denominator;
        reduce();
        return this;
    }

    public Fraction multiply (Fraction otherFraction) {
        this.numerator = this.numerator * otherFraction.numerator;
        this.denominator = this.denominator * otherFraction.denominator;
        reduce();
        return this;
    }

    public Fraction divide (Fraction otherFraction) {
        if (otherFraction.numerator == 0) {
            reduce();
            return this;
        }
        else {
            this.numerator = this.numerator * otherFraction.denominator;
            this.denominator = this.denominator * otherFraction.numerator;
            reduce();
            return this;
        }
    }

    public boolean equals(Object obj) {
        if (!(obj instanceof Fraction)) {
            return false;
        }
        Fraction other = (Fraction) obj;

        if ((this.numerator * other.denominator) == (other.numerator * this.denominator)) return true;
        return false;
    }
