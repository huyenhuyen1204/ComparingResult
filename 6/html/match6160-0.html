public class Fraction<Frantion> {
    private int  numerator;
    private int denominator = 1 ;
    Fraction(int numerator_ , int denominator_ ){
        if (denominator_ != 0){
            this.numerator = numerator_;
            this.denominator  = denominator_;
        }
    }

    public int getNumerator(){
        return numerator;
    }

    public int getDenominator(){
        return denominator ;
    }
    public void setNumerator(int numerator_){
        this.numerator = numerator_;
    }

    public void setDenominator(int denominator_){
        if (denominator_ != 0){
            this.denominator = denominator_;
        }

    }

    public static int ucln(int a, int b){
        if (b == 0) return a;
        return ucln(b, a % b);
    }

    public Fraction reduce(){
        int t = ucln(numerator,denominator);
        this.setNumerator(this.getNumerator() / t);
        this.setDenominator(this.getDenominator() / t);
        return this;
    }

    public Fraction add(Fraction other){
        numerator = numerator * other.denominator + denominator * other.numerator;
        denominator = denominator * other.denominator;
        this.reduce();
        return this;
    }

    public Fraction subtract(Fraction other){

        numerator = numerator * other.denominator - denominator * other.numerator;
        denominator = denominator * other.denominator;
        this.reduce();
        return this;
    }

    public Fraction multiply(Fraction other){

        numerator = numerator * other.numerator ;
        denominator = denominator * other.denominator;
        this.reduce();
        return this;
    }

    public Fraction divide(Fraction other){
        if (other.numerator != 0 && other.denominator != 0){

            numerator = numerator * other.denominator ;
            denominator = denominator * other.numerator;
            this.reduce();
            return this;
        }
        return this.reduce();
    }

    public boolean equals(Object obj){
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;

            // compare this vs other here
            int a = this.numerator*other.denominator;
            int b = this.denominator*other.numerator;
            return a == b;

        }
        return false;
    }
}
