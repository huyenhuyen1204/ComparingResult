class Fraction {
    private int numerator;
    private int denominator;
    public int getNumerator() {
        return this.numerator;
    }
    public int getDenominator() {
        return this.denominator;
    }
    public void setNumerator(int numerator1) {
        this.numerator = numerator1;
    }
    public void setDenominator(int denominator1) {
        if(denominator1 != 0) this.denominator = denominator1;
    }
    public Fraction ()
    {
        this.numerator = 0;
        this.denominator = 1;
    }
    public Fraction (int a,int b)
    {
        this.numerator = a;
        if(b == 0) this.denominator = 1;
        else this.denominator = b;
    }
    public Fraction (int a)
    {
        this.numerator = a;
        this.denominator = 1;
    }
    public static int gcd(int a,int b)
    {
        if(a == 0) return 1;
        if(a>b) 
        {
            a = a-b;
            return gcd(a,b);
        }
        if(a<b) 
        {
            b = b-a;
            return gcd(a,b);
        }
        return a;
    }
    public Fraction reduce()
    {
        int a = gcd(this.numerator,this.denominator);
        this.numerator = this.numerator/a;
        this.denominator = this.denominator/a;
        return this;
    }
    public Fraction add(Fraction a)
    {
        
this.numerator = this.numerator*a.getDenominator() + this.denominator*a.getNumerator();
        this.denominator = this.denominator*a.getDenominator();
        boolean x = true;
        if(this.numerator < 0)
        {
            this.numerator = -this.numerator;
            x = false;
        }
        this.reduce();
        if(x == false)
        {
            this.numerator = -this.numerator;
        }
        return this;
    }
    public Fraction divide(Fraction a)
    {
        if(a.getDenominator() != 0 && a.getNumerator() != 0)
        {
            this.numerator = this.numerator*a.getDenominator();
            this.denominator = this.denominator*a.getNumerator();
            this.reduce(); 
        }
        return this;
    }
    public Fraction subtract(Fraction a)
    {
        this.numerator = this.numerator*a.getDenominator() - this.denominator*a.getNumerator();
        this.denominator = this.denominator*a.getDenominator();
        boolean x = true;
        if(this.numerator < 0)
        {
            this.numerator = -this.numerator;
            x = false;
        }
        this.reduce();
        if(x == false)
        {
            this.numerator = -this.numerator;
        }
        return this;
    }
    public Fraction multiply(Fraction a)
    {
        this.numerator = this.numerator*a.getNumerator();
        this.denominator = this.denominator*a.getDenominator();
        this.reduce();
        return this;
    }
    public boolean equals(Object obj)
    {
        if (obj instanceof Fraction)
        {
            Fraction a = (Fraction) obj;
            if(this.numerator*(a).denominator == this.denominator*(a).numerator)
            {
                return true;
            }
        }
        return false;
    }
}
