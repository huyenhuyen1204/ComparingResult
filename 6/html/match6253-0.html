import java.util.Scanner;

public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator(){ return numerator;}
    public void setNumerator(int num){ this.numerator = num;}

    public  int getDenominator(){ return denominator;}
    public  void setDenominator(int den){
        if (den ==0 ) this.denominator = 1;
        else this.denominator = den;
    }

    public Fraction(int num, int den){
        this.numerator = num;
        if (den == 0 ) this.denominator = 1;
        else this.denominator = den;
    }

    public static int gcd(int a, int b){
        while (b>0){
            int r=a%b;
            a=b;
            b=r;
        }
        return a;
    }

    public Fraction reduce(){
        int a = this.getNumerator();
        int b = this.getDenominator();
        int c = gcd(a,b);

        Fraction m = new Fraction(a/c, b/c);

        this.setNumerator(a/c);
        this.setDenominator(b/c);
        return m;
    }

    public Fraction add(Fraction other)
    {
        this.numerator = this.getNumerator() * other.getDenominator() + this.getDenominator() * other.getNumerator();
        this.denominator = this.getDenominator() * other.getDenominator();
        return this.reduce();
    }

    public Fraction subtract(Fraction other)
    {
        this.numerator = this.getNumerator() * other.getDenominator() - this.getDenominator() * other.getNumerator();
        this.denominator = this.getDenominator() * other.getDenominator();
        return this.reduce();
    }

    public Fraction multiply(Fraction other)
    {
        this.numerator = this.getNumerator() * other.getNumerator();
        this.denominator = this.getDenominator() * other.getDenominator();
        return this.reduce();
    }

    public Fraction divide(Fraction other)
    {
        if (other.getNumerator() == 0) return this.reduce();
        this.numerator = this.getNumerator() * other.getDenominator();
        this.denominator = this.getDenominator() * other.getNumerator();
        return this.reduce() ;
    }

    public boolean equals(Object obj){
        if (!(obj instanceof Fraction))  return false;
        Fraction other = (Fraction) obj;
        if (this.numerator * other.getDenominator() == this.denominator * other.getNumerator()) return true;
        return false;
    }

}
