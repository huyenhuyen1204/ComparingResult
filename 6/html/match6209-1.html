import java.util.Scanner;

public class Fraction {
    private int numerator, denominator;

    public int getNumerator() { return numerator; };
    public int getDenominator() {
        return denominator;
    }
    public void setDenominator(int denominator) {
        if (denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public Fraction(int numerator, int denominator) {
        if (denominator == 0) denominator = 1;
        this.numerator = numerator;
        this.denominator = denominator;
    }

    private int gcd(int a, int b) {
        if (a == 0 || b == 0) return 1;
        int t = b % a;
        
while (t != 0) {
            t = a % b;
            a = b;
            b = t;
        }
        return a;
    }

    public Fraction reduce() {
        int tuSo = this.getNumerator();
        int mauSo = this.getDenominator();
        int ucln = this.gcd(tuSo, mauSo);
        Fraction finalFraction = new Fraction(tuSo / ucln, mauSo / ucln);
        
this.setNumerator(finalFraction.getNumerator());
        this.setDenominator(finalFraction.getDenominator());
        return this;
    }

    public Fraction add(Fraction other) {
        int tuSo = this.getNumerator() * other.getDenominator() + other.getNumerator() * this.getDenominator();
        int mauSo = this.getDenominator() * other.getDenominator();
        Fraction finalFraction = new Fraction(tuSo, mauSo);
        
finalFraction.reduce();
        return finalFraction;
    }

    public Fraction subtract(Fraction other) {
        int tuSo = this.getNumerator() * other.getDenominator() - other.getNumerator() * this.getDenominator();
        int mauSo = this.getDenominator() * other.getDenominator();
        Fraction finalFraction = new Fraction(tuSo, mauSo);
        
finalFraction.reduce();
        return finalFraction;
    }

    public Fraction multiply(Fraction other) {
        int tuSo = this.getNumerator() *  other.getNumerator();
        int mauSo = this.getDenominator() * other.getDenominator();
        Fraction finalFraction = new Fraction(tuSo, mauSo);
        
finalFraction.reduce();
        return finalFraction;
    }

    public Fraction divide(Fraction other) {
        if (other.getNumerator() == 0) return this;
        int tuSo = this.getNumerator() *  other.getDenominator();
        int mauSo = this.getDenominator() * other.getNumerator();
        Fraction finalFraction = new Fraction(tuSo, mauSo);
        finalFraction.reduce();
        return finalFraction;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            Fraction cc = this.subtract(other);
            if (cc.getNumerator() == 0) return true;
        }
        return false;
    }
}
