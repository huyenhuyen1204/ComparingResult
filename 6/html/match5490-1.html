class Fraction {
    private int numerator;
    private int denominator;
    public Fraction(){
        this.numerator=0;
        this.denominator=1;
    }

    public Fraction(int numerator, int denominator) {
        this.denominator = denominator;
        this.numerator = numerator;
        if(denominator==0) this.denominator=1;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        this.denominator = denominator;
        if(denominator==0) this.denominator=1;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public Fraction reduce() {
        int r = 1;
        int a = this.numerator;
        int b = this.denominator;
        while (r != 0) {
            r = a % b;
            a = b;
            b = r;
        }
        this.numerator = this.numerator / a;
        this.denominator = this.denominator / a;
        return this;

    }

    public Fraction add(Fraction a) {
        int x, y;
        y = this.denominator * a.denominator;
        x = this.numerator * a.denominator + a.numerator * this.denominator;
        this.numerator = x;
        this.denominator = y;
        return this;
    }

    public Fraction subtract(Fraction a) {
        int x, y;
        y = this.denominator * a.denominator;
        x = this.numerator * a.denominator - a.numerator * this.denominator;
        this.numerator = x;
        this.denominator = y;
        return this;
    }

    public Fraction multiply(Fraction a) {
        this.numerator = this.numerator * a.numerator;
        this.denominator = this.denominator * a.denominator;
        return this;
    }

    public Fraction divide(Fraction a) {
        this.numerator = this.numerator * a.denominator;
        this.denominator = this.denominator * a.numerator;
        return this;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;
            int x, y;
            x = this.numerator * other.denominator;
            y = other.numerator * this.denominator;
            if (x == y) return true;
        }
        return false;


    }

    }
}
