public class Fraction {
        private int numerator;
        private int denominator;

        public int getNumerator() {
            return numerator;
        }
        public void setNumerator(int n) {
            this.numerator = n;
        }
        public void setDenominator(int n) {
            if ( n!=0 ) this.denominator = n;
            else this.denominator = 1;
        }
        public int getDenominator() {
            return denominator;
        }
        public Fraction(int x, int y){
            this.numerator = x;
            if (y == 0) this.denominator = 1;
            else this.denominator = y;
        }
    public int gcd(int a, int b){
        if (b == 0) return a;
        else return gcd(b,a%b);
    }
    public Fraction reduce(){
        int k = gcd(this.getNumerator(),this.getDenominator());
        this.setNumerator(this.getNumerator() / k);
        this.setDenominator(this.getDenominator() / k);
        return this;
    }
    public Fraction add(Fraction ps){
        int x = this.getNumerator() * ps.getDenominator() + this.getDenominator() * ps.getNumerator();
        int y = this.getDenominator() * ps.getDenominator();
        this.setNumerator(x);
        this.setDenominator(y);
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction ps){
        int x = this.getNumerator() * ps.getDenominator() - this.getDenominator() * ps.getNumerator();
        int y = this.getDenominator() * ps.getDenominator();
        this.setNumerator(x);
        this.setDenominator(y);
        this.reduce();
        return this;
    }
    public Fraction multiply(Fraction ps){
        int x = this.getNumerator() * ps.getNumerator();
        int y = this.getDenominator() * ps.getDenominator();
        this.setNumerator(x);
        this.setDenominator(y);
        this.reduce();
        return this;
    }
    public Fraction divide(Fraction ps){
        int x = this.getNumerator() * ps.getDenominator();
        int y = this.getDenominator() * ps.getNumerator();
        this.setNumerator(x);
        this.setDenominator(y);
        this.reduce();
        return this;
    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            this.reduce();
            other.reduce();
            if (this.numerator == other.numerator && this.denominator == other.denominator) return true;
        }
        return false;
    }
}
