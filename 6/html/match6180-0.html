public class Fraction {
    private int numerator, denominator;

    int getNumerator() {
        return numerator;
    }

    void setNumerator(int _numerator) {
        numerator = _numerator;
    }

    int getDenominator() {
        return denominator;
    }

    void setDenominator(int _denominator) {
        if (_denominator != 0)
            denominator = _denominator;
    }

    int gcd(int a, int b) {
        if (a < b)
            return gcd(b, a);
        return (b == 0 ? a : gcd(b, a % b));
    }

    Fraction() {
        denominator = 1;
    }

    Fraction(int _numerator, int _denominator) {
        if (_denominator != 0) {
            numerator = _numerator;
            denominator = _denominator;
        } else {
            if (denominator == 0)
                denominator = 1;
        }
    }

    Fraction reduce() {
        Fraction ans = new Fraction(numerator, denominator);
        int _gcd = gcd(ans.numerator, ans.denominator);
        ans.numerator /= _gcd;
        ans.denominator /= _gcd;
        numerator = ans.numerator;
        denominator = ans.denominator;
        return this;
    }

    Fraction add(Fraction b) {
        Fraction ans = new Fraction(this.numerator * b.denominator + this.denominator * b.numerator,
                this.denominator * b.denominator);
        numerator = ans.numerator;
        denominator = ans.denominator;
        return this;
    }

    Fraction subtract(Fraction b) {
        Fraction newb = new Fraction(-1 * b.numerator, b.denominator);
        return add(newb);
    }

    Fraction multiply(Fraction b) {
        Fraction ans = new Fraction(this.numerator * b.numerator, this.denominator * b.denominator);
        numerator = ans.numerator;
        denominator = ans.denominator;
        return this;
    }

    Fraction divide(Fraction b) {
        if (b.getNumerator() == 0)
            return this;
        Fraction newb = new Fraction(b.denominator, b.numerator);
        return multiply(newb);
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction cur = (Fraction) obj;
            return (this.reduce().getNumerator() == cur.reduce().getNumerator()
                    && this.reduce().getDenominator() == cur.reduce().getDenominator());
        }
        return false;
    }
