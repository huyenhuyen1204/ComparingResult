import java.util.Scanner;

public class Fraction {
    private int numerator , denominator;

    public Fraction ( int numerator , int denominator ){
        if ( denominator != 0 )
        {
            this.numerator = numerator;
            this.denominator = denominator;
        }
        else this.denominator = 1;

    }
    public int getNumerator (){
        return numerator;
    }
    public void setNumerator ( int numerator ){
        this.numerator = numerator;
    }
    public int getDenominator (){
        return denominator;
    }
    public void setDenominator ( int denominator ){
        if ( denominator != 0) this.denominator = denominator;
        else 
this.denominator = 1;
    }

    private int GCD(int a, int b) {
        if (a == 0 || b == 0) return 1;
        int t = b % a;
        while (t != 0) {
            t = a % b;
            a = b;
            b = t;
        }
        return a;
    }

    public Fraction reduce (){
        if ( this.getNumerator() > 0 && this.getDenominator() > 0 ) {
            int k = GCD(this.getNumerator() , this.getDenominator() );
            this.setNumerator(this.getNumerator()/k);
            this.setDenominator(this.getDenominator()/k);
        }
        else{
            int tu = -this.getNumerator();
            int mau = -this.getDenominator();
            int x = GCD(tu , mau);
            this.setNumerator(tu/x);
            this.setDenominator(mau/x);
        }
        Fraction PS = new Fraction(this.getNumerator() , this.getDenominator());
        return PS;
    }
    // add
    public Fraction add(Fraction other) {
        int tu = this.getNumerator()*other.getDenominator() + this.getDenominator()*other.getNumerator();
        int mau = this.getDenominator()*other.getDenominator();
        Fraction newPS = new Fraction( tu , mau );
        newPS.reduce();
        return newPS;
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int tu = this.getNumerator()*other.getDenominator() - this.getDenominator()*other.getNumerator();
        int mau = this.getDenominator()*other.getDenominator();
        Fraction newPS = new Fraction( tu , mau );
        newPS.reduce();
        return newPS;
    }

    // multiple
    public Fraction multiply (Fraction other) {
        int tu = this.getNumerator()*other.getNumerator();
        int mau = this.getDenominator()*other.getDenominator();
        Fraction newPS = new Fraction( tu , mau );
        newPS.reduce();
        return newPS;
    }

    // divide
    public Fraction divide(Fraction other) {
        if ( numerator == 0 ) return this;
        int tu = this.getNumerator()*other.getDenominator();
        int mau = this.getDenominator()*other.getNumerator();
        Fraction newPS = new Fraction( tu , mau );
        newPS.reduce();
        return newPS;
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals (Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            Fraction PS = new Fraction( this.getNumerator() , this.getDenominator());
            Fraction newPS = PS.divide(other);
            if ( newPS.getNumerator() == newPS.getDenominator() )
                return true;
        }
        return false;
    }

    }
}
