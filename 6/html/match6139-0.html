public class Fraction {
    private int numerator, denominator;


    public Fraction(int nume, int deno) {
        this.numerator = nume;
        if (deno != 0) {
            this.denominator = deno;
        }else {
            this.denominator = 1;
        }
    }


    public void setNumerator(int nume) {
        this.numerator = nume;
    }

    public void setDenominator(int deno) {
        if (deno != 0) {
            this.denominator = deno;
        }else{
            this.denominator = 1;
        }
    }

    public int getNumerator() {
        return numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public static int gcd(int a, int b) {
        if (a < 0) a = -a;
        if (b < 0) b = -b;
        while (a * b != 0) {
            if (a > b) {
                a %= b;
            } else {
                b %= a;
            }
        }
        return a + b;
    }

    public Fraction reduce() {
        int tmp = gcd(this.numerator, this.denominator);
        this.numerator /= tmp;
        this.denominator /= tmp;
        Fraction fraction = new Fraction(this.numerator, this.denominator);
        return fraction;
    }
