import java.util.Scanner;

public class Fraction {
    private int numerator;
    private int denominator;

    Fraction (){
        this.numerator = 1;
        this.denominator = 1;
    }

    Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if (denominator == 0)
            denominator = 1;
        this.denominator = denominator;
    }

    public void setNumerator(int numerator){
        this.numerator = numerator;
    }
    public int getNumerator(){
        return this.numerator;
    }

    public void setDenominator(int denominator){
        this.denominator = denominator;
    }
    public int getDenominator(){
        return this.denominator;
    }

    public int gcd(int a, int b)
    {
        int r;
        
while (b != 0)
        {
            r = a%b;
            a = b;
            b = r;
        }
        return a;
    }

    public Fraction reduce(){
        int ucln = gcd(this.getNumerator(), 
this.getDenominator());
        Fraction f = new Fraction();
        f.setNumerator(this.getNumerator() / ucln);
        f.setDenominator(this.getDenominator() / ucln);
        return f;
    }

    public Fraction add(Fraction other){
        int Fn = this.getNumerator()*other.getDenominator() + this.getDenominator()*other.getNumerator();
        int Fd = this.getDenominator()*other.getDenominator();
        Fraction f = new Fraction(Fn, Fd);
        return f.reduce();
    
}

    public Fraction subtract(Fraction other){
        int Fn = this.getNumerator()*other.getDenominator() - this.getDenominator()*other.getNumerator();
        int Fd = this.getDenominator()*other.getDenominator();
        Fraction f = new Fraction(Fn, Fd);
        return f.reduce();
    
}

    public  Fraction multiply(Fraction other){
        int Fn = this.getNumerator()*other.getNumerator();
        int Fd = this.getDenominator()*other.getDenominator();
        Fraction f = new Fraction(Fn, Fd);
        return f.reduce();
    }

    public Fraction divide(Fraction other){
        int 
Fn = this.getNumerator()*other.getDenominator();
        int Fd = this.getDenominator()*other.getNumerator();
        Fraction f = new Fraction(Fn, Fd);
        return f.reduce();
    }

    public  boolean equals(Object obj){
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;
            if (this.reduce().getNumerator() == other.reduce().getNumerator() && this.reduce().getDenominator() == other.reduce().getDenominator())
                return true;
            else return false;
        }
        return super.equals(obj);
    }

    public  static void  main(String[] args){
        Fraction a = new Fraction(2,4);
        Fraction b = new Fraction(3, 6);
        System.out.println(a.add(b).numerator + " "+  a.add(b).denominator);
        System.out.println(a.equals(b));
    }
}
