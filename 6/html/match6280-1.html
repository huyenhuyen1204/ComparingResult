class Fraction {
    private
    int  numerator, denominator;

    public int getNumerator()
    {
        return numerator;
    }
    public int getDenominator()
    {
        return denominator;
    }
    public void setNumerator(int numerator)
    {
        this.numerator=numerator;
    }
    public void setDenominator(int denominator)
    {   if(denominator==0) this.denominator=1;
        else this.denominator=denominator;
    }
    public Fraction(int numerator, int denominator)
    {
        this.numerator=numerator;
        if(denominator==0) this.denominator=1;
        else this.denominator=denominator;
    }

    public int UCLN(int a, int b)
    {   int i;
        a=Math.abs(a);
        b=Math.abs(b);
        for( i=b;i>=1;--i){
            if(a%i==0&&b%i==0) break;
        }
        return i;
    }
    public Fraction reduce()
    {
        int i=UCLN(this.numerator,this.denominator);
        this.numerator= this. numerator/i;
        this.denominator=this . denominator/i;
        if(this.denominator<0) {
            this.numerator=this.numerator*(-1);
            this.denominator=this.denominator*(-1);
        }
        return this;
    }

    public Fraction add(Fraction ps)
    {
        
this.numerator=(this.numerator*ps.denominator) + (ps.numerator*this.denominator);
        this.denominator=ps.denominator*this.denominator;
        this.reduce();
        return this;
    }
    public  Fraction subtract(Fraction ps)
    {
        this.numerator=(this.numerator*ps.denominator) - (ps.numerator*this.denominator);
        this.denominator=ps.denominator*this.denominator;
        this.reduce();
        return this;
    }
    Fraction multiply(Fraction ps)
    {
        
this.numerator=ps.numerator*this.numerator;
        this.denominator=ps.denominator*this.denominator;
        this.reduce();
        return this;
    }
    Fraction divide(Fraction ps)
    {   this.numerator=ps.denominator*this.numerator;
        this.denominator=ps.numerator*this.denominator;
        this.reduce();
        return this;
    }
    public boolean equals(Object obj){
        if(obj instanceof Fraction==false) return false;
        else{

            Fraction a= (Fraction) obj;
            if(a.denominator==0||this.denominator==0) return false;
            if(this.numerator*a.denominator-this.denominator*a.numerator==0) return true;
            else return false;
        }
    }
