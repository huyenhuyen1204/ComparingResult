class Fraction{
    private int numerator;
    private int denominator;
    public  Fraction()
    {
        this.numerator = 0;
        this.denominator = 1;
    }
    public  Fraction(int numerator, int denominator)
    {
        this.setDenominator(denominator);
        this.setNumerator(numerator);
    }
    public int gcd(int a, int b)
    {
        while(true)
        {
            if(a == 0) return  b;
            if(b == 0) return  a;
            if(a > b)
            {
                a = a - b;
            }
            else b = b - a;
        }
    
}
    public int getNumerator(){return  this.numerator;}
    public int getDenominator(){return  this.denominator;}
    public void  setNumerator(int numerator){ this.numerator = numerator;}
    public  void  setDenominator(int denominator){
        if(denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }
    public  Fraction reduce()
    {
        int k = gcd(Math.abs(this.numerator), Math.abs(this.denominator));
        this.numerator /= k;
        this.denominator /= k;
        return  this;
    }
    public boolean equals(Object obj)
    {
        if(obj instanceof Fraction == false) return  false;
        Fraction 
other = (Fraction) obj;
        other.reduce();
        return other.getNumerator() * this.denominator == this.getNumerator()* other.getDenominator();
    }
    public  Fraction add(Fraction other)
    {
        Fraction res = new Fraction();
        res.setDenominator(this.getDenominator() * other.getDenominator());
        res.setNumerator(this.getNumerator() * other.getDenominator() + this.getDenominator()* other.getNumerator());
        return res;

    }
    public  Fraction subtract(Fraction other)
    {
        Fraction res = new Fraction();
        res.setDenominator(this.getDenominator() * other.getDenominator());
        res.setNumerator(this.getNumerator() * other.getDenominator() -  this.getDenominator()* other.getNumerator());
        return res;
    }
    public  Fraction multiply(Fraction other)
    {
        Fraction res = new Fraction();
        res.setDenominator(this.getDenominator() * other.getDenominator());
        res.setNumerator(this.getNumerator() * other.getNumerator());
        return res;
    }
    public  Fraction divide(Fraction other)
    {
        Fraction res = new Fraction();
        res.setDenominator(this.getDenominator() * other.getNumerator());
        res.setNumerator(this.getNumerator()*other.getDenominator() );
        res.reduce();
        return res;
    }
    public  void Print()
    {
        this.reduce();
        System.out.println(this.numerator + "/" + this.denominator);
    }
