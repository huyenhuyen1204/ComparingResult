public class Fraction {
    private int numerator, denominator;
    //private boolean t1;
    // private boolean getnumerator;
    //  private String t;

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator()
    {


        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator==0)
            denominator=1;
        this.denominator = denominator;
    }

    public Fraction(int numerator, int denominator) {
        if (denominator==0)
            denominator=1;
        this.numerator = numerator;
        this.denominator = denominator;
    }
  //  Fraction(){}

    static int gcd(int a, int b) {
        if (b == 0) {
            return a;
        } else
            return gcd(b, a % b);
    }

    public Fraction reduce() {
        int i = gcd(this.numerator, this.denominator);
        this.setNumerator(
this.getNumerator() / i);
        this.setDenominator(this.denominator / i);
return  this;

    }

    public Fraction add(Fraction other) {
        int t = this.getNumerator() * other.getDenominator() + other.getNumerator() * this.getDenominator();
        int m = this.getDenominator() * other.getDenominator();
        Fraction tt = new Fraction(t, m);
        tt.reduce();
       // System.out.println(tt.numerator + "/" + tt.denominator);
        return  tt;

    }

    public Fraction subtract(Fraction other) {
        int t = this.getNumerator() * other.getDenominator() - other.getNumerator() * this.getDenominator();
        int m = this.getDenominator() * other.getDenominator();
        Fraction tt = new Fraction(t, m);
        tt.reduce();
       // System.out.println(tt.numerator + "/" + tt.denominator);
        return  tt;
    }

    public Fraction multiply(Fraction other) {
        int t = this.getNumerator() *other.getNumerator();
        int m = this.getDenominator() * other.getDenominator();
        Fraction tt = new Fraction(t, m);
        tt.reduce();
       // System.out.println(tt.numerator + "/" + tt.denominator);
        return  tt;
    }

    public Fraction divide(Fraction other) {

        int 
t = this.getNumerator() * other.getDenominator();
        int m = this.getDenominator() * other.getNumerator();
        Fraction tt = new Fraction(t, m);
        tt.reduce();
       // System.out.println(tt.numerator + "/" + tt.denominator);
        return tt;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;
          // int tu=this.getNumerator()-other.getNumerator();
         //   int mau=this.getDenominator()-other.getDenominator();
            int t1=this.getNumerator();
            int 
m1=this.getDenominator();
            int t2=other.getNumerator();
            int m2=other.getDenominator();
            Fraction ps1 = new Fraction(t1,m1);
            Fraction ps2=new Fraction(t2,m2);
            ps1.reduce();
            ps2.reduce();
            if (ps1.numerator==ps2.numerator&&ps1.denominator==ps2.denominator)
                return true;

        }
        return false;
    }
