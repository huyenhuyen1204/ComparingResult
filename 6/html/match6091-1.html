public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator(){
        return numerator;
    }
    public void setNumerator(int numerator){
        this.numerator=numerator;
    }

    public int getDenominator(){
        return denominator;
    }
    public void setDenominator(int denominator){
        if(denominator==0) this.denominator =1;
        else
            
this.denominator=denominator;
    }

    Fraction(){
        this.numerator=0;
        this.denominator=1;
    }
    Fraction(int numerator, int denominator){
        if(denominator==0) this.denominator=1;
        else {
            this.numerator = numerator;
            this.denominator = denominator;
        }
    }


    public static int getUCLN(int a, int b){
        if(b==0) return a;
        else return getUCLN(b, a%b);
    }

    public Fraction reduce(){
        int n = getUCLN(this.numerator, this.denominator);
        this.numerator=this.numerator/n;
        this.denominator=this.denominator/n;
        return this;
    }

    public Fraction add(Fraction fraction){
        int 
n=(this.denominator * fraction.getDenominator())/getUCLN(this.denominator, fraction.getDenominator());
        int x2 = n/fraction.getDenominator();
        int x1 = n/this.denominator;
        Fraction other= new Fraction();
        other.numerator=this.numerator*x1 + fraction.getNumerator()*x2;
        other.denominator=this.denominator*x1;
        return other;
    }
    public Fraction subtract(Fraction fraction){
        int n=(this.denominator * fraction.getDenominator())/getUCLN(this.denominator, fraction.getDenominator());
        int x2 = n/fraction.getDenominator();
        int x1 = n/this.denominator;
        Fraction other= new Fraction();
        other.numerator=this.numerator*x1 - fraction.getNumerator()*x2;
        other.denominator=this.denominator*x1;
        return other;
    }
    public Fraction multiply(Fraction fraction){
        Fraction other = new Fraction();
        other.numerator=this.numerator * fraction.getNumerator();
        other.denominator=this.denominator * fraction.getDenominator();
        return other;
    }
    public Fraction divide(Fraction fraction){
        Fraction other = new Fraction();
        if(fraction.denominator==0 || fraction.numerator==0) return this;
        else {
            other.numerator = this.numerator * fraction.getDenominator();
            other.denominator = this.denominator * fraction.getNumerator();
            return other;
        }
    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if(this.numerator*other.getDenominator()-this.denominator*other.getNumerator()==0) return true;
            else return false;
        }
        else return false;
    }
}
