public class Fraction {
    private int numerator;
    private int denominator;


    public int getNumerator() {
        return numerator;
    }
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public int getDenominator() {
        return denominator;
    }
    public void setDenominator(int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
        }
        else {
            this.denominator = 1;
        }
    }

    public Fraction() {
        this.denominator = 1;
    }

    public 
Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        if (denominator == 0) {
            this.denominator = 1;
        }
        else {
            this.denominator = denominator;
        }
    }

    static int gcd(int numerator, int denominator) {
        int ucln = 0;
        if (numerator < denominator) {
            for (int i = 1; i <= numerator; i++) {
                if (numerator % i == 0 && denominator % i == 0) {
                   ucln = i;
                }
            }
        }
        if (numerator >= denominator) {
            for (int i = 1; i <= denominator; i++) {
                if (numerator % i == 0 && denominator % i == 0) {
                    ucln = i;
                }
            }
        }
        return ucln;
    }

    public Fraction reduce(){
        int k = gcd(Math.abs(this.getNumerator()), Math.abs(this.getDenominator()));
        this.numerator = this.getNumerator() / k;
        this.denominator = 
this.getDenominator() / k;
        return this;
    }

    public Fraction add(Fraction other) {
        this.numerator = this.getNumerator()*other.getDenominator() + this.getDenominator()*other.getNumerator();
        this.denominator = this.getDenominator()*other.getDenominator();
        return this.reduce();
    }

    public Fraction subtract(Fraction other) {
        this.numerator = this.getNumerator()*other.getDenominator() - this.getDenominator()*other.getNumerator();
        this.denominator = this.getDenominator()*other.getDenominator();
        return this.reduce();
    }

    public Fraction multiple(Fraction other) {
        this.numerator = this.getNumerator()*other.getNumerator();
        this.denominator = this.getDenominator()*other.getDenominator();
        return this.reduce();
    }

    public Fraction divide(Fraction other) {
        
this.numerator = this.getNumerator()*other.getDenominator();
        this.denominator = this.getDenominator()*other.getNumerator();
        return this.reduce();
    }

    public boolean equals(Object obj){
        if (!(obj instanceof Fraction))  return false;
        Fraction other = (Fraction) obj;
        if (this.numerator * other.getDenominator() == this.denominator * other.getNumerator()) return true;
        return false;
    }



}
