public 
class Fraction {
    /**
     * declare attributes here
     */
    private int numerator;
    private int denominator;


    /**
     * support function
     */
    int gcd(int a, int b) {
        // return;
        if (a<0) a = -a;
        if (b<0) b = -b;
        if (b % a == 0) return a;
        else return gcd( b % a, a);
    }
    /**
     * declare constructor here
     */
    public Fraction() {
        this.numerator = 1;
        this.denominator = 1;
    }

    public Fraction(int numerator, int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
            this.numerator = numerator;
        }
        else {
            this.denominator = 1;
            this.numerator = numerator;
        }
    }

    public Fraction(Fraction other){
        this.denominator = other.denominator;
        this.numerator = other.numerator;

    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator!=0) this.denominator = denominator;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    /**
     * methods
     */
    //reduce
    public Fraction reduce(){
        if (this.denominator!=0 && this.numerator!=0) {
            int gcd = gcd(this.denominator, this.numerator);
            if (this.denominator < 0) gcd = -gcd;
            this.denominator /= gcd;
            this.numerator /= gcd;
        }
        return this;
    }
    // add
    public Fraction add(Fraction other) {
        if (this.denominator!= 0 && other.denominator != 0) {
            int commonDenominator = this.denominator * other.denominator / gcd(this.denominator, other.denominator);
            this.numerator = this.numerator * (commonDenominator / this.denominator) + other.numerator * (commonDenominator / other.denominator);
            this.denominator = commonDenominator;
        }
        return this.reduce();
    }

    // subtract
    public Fraction subtract(Fraction other) {
        if (this.denominator!= 0 && other.denominator != 0) {
            int commonDenominator = this.denominator * other.denominator / gcd(this.denominator, other.denominator);
            this.numerator = this.numerator * (commonDenominator / this.denominator) - other.numerator * (commonDenominator / other.denominator);
            this.denominator = commonDenominator;
        }
        return this.reduce();
    }

    // multiple
    public Fraction multiply(Fraction other) {
        if (this.denominator!= 0 && other.denominator != 0) {
            this.denominator *= other.denominator;
            this.numerator *= other.numerator;
        }
        return this;
    }

    // divide
    public Fraction divide(Fraction other) {
        if (other.numerator!= 0 && this.denominator!= 0 && other.denominator != 0){
            this.numerator *= other.denominator;
            this.denominator *= other.numerator;
        }
        return this.reduce();
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        boolean result = false;
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;

            // compare this vs other here
            if (this.numerator*1.0/ this.denominator  == other.numerator*1.0/other.denominator) result = true;

        }
        return result;
    }

}
