public class Fraction {
    private int numerator, denominator;

    /**
     * Getter cho numerator
     * @return tu so
     */
    public int getNumerator() {
        return numerator;
    }

    /**
     * Setter cho numerator
     * @param numerator
     */
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    /**
     * Getter cho denominator
     * @return mau so
     */
    public int getDenominator() {
        return denominator;
    }

    /**
     * Setter cho denominator
     * @param denominator
     */
    public void setDenominator(int denominator) {
        if(denominator != 0)
            this.denominator = denominator;
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param numerator
     * @param denominator
     */
    public Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        this.denominator = denominator;
        if(denominator == 0)
            this.denominator = 1;
    }

    /**
     * Tim uoc chung lon nhat
     * @param a
     * @param b
     */
    public int GreatestCommonDivisor(int a, int b){
        return b==0 ? a : GreatestCommonDivisor(b, a%b);
    }

    /**
     * Ham lay tri tuyet doi
     * @param n
     * @return tri tuyet doi
     */
    public int Absolute(int n){
        return n<0 ? (-n) : n;
    }

    /**
     * Ham rut gon phan so
     * @return phan so da rut gon
     */
    public Fraction reduce(){
        int gcd = Absolute(GreatestCommonDivisor(this.getNumerator(), this.getDenominator()));
        Fraction reduceF = new Fraction(this.getNumerator() / gcd, this.getDenominator() / gcd);
        if( reduceF.getNumerator() > 0 && reduceF.getDenominator() < 0 ||
            reduceF.getNumerator() < 0 && reduceF.getDenominator() < 0) {
                reduceF.setNumerator(-reduceF.getNumerator());
                reduceF.setDenominator(-reduceF.getDenominator());
        }

        return reduceF;
    }

    /**
     * Ham cong hai phan so
     * @param f
     * @return tong hai phan so
     */
    public Fraction add(Fraction f){
        int num = this.getNumerator() * f.getDenominator() + f.getNumerator() * this.getDenominator();
        int den = this.getDenominator() * f.getDenominator();
        Fraction addF = new Fraction(num, den);
        return addF.reduce();

    }

    /**
     * Ham tru hai phan so
     * @param f
     * @return hieu hai phan so
     */
    public Fraction subtract(Fraction f){
        int num = this.getNumerator() * f.getDenominator() - f.getNumerator() * this.getDenominator();
        int den = this.getDenominator() * f.getDenominator();
        Fraction subtractF = new Fraction(num, den);
        return subtractF.reduce();
    }

    /**
     * Ham nhan hai phan so
     * @param f
     * @return tich hai phan so
     */
    public Fraction multiply(Fraction f){
        int num = this.getNumerator() * f.getNumerator();
        int den = this.getDenominator() * f.getDenominator();
        Fraction multiplyF = new Fraction(num, den);
        return multiplyF.reduce();
    }

    /**
     * Ham chia hai phan so
     * @param f
     * @return thuong hai phan so
     */
    public Fraction divide(Fraction f){
        int num = this.getNumerator() * f.getDenominator();
        int den = this.getDenominator() * f.getNumerator();
        Fraction divideF = new Fraction(num, den);
        return divideF.reduce();
    }

    /**
     * Ham so sanh hai phan so
     * @return true hoac false
     */
    public boolean equals(Object obj){
        if(obj instanceof Fraction) {
            Fraction f = (Fraction) obj;
            if( this.reduce().getNumerator() == f.reduce().getNumerator() &&
                this.reduce().getDenominator() == f.reduce().getDenominator() ||
                this.getNumerator() == 0 && f.getNumerator() == 0
            )
                return true;
        }
        return false;
    }
}
