public class Fraction {
    private int numerator;
    private int denominator;

    public int getDenominator() {

        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator == 0) this.denominator = 1;
        else this.denominator = denominator;
    }

    public int getNumerator() {

        return numerator;
    }

    public void setNumerator(int numerator) {

        this.numerator = numerator;
    }

    public Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        this.denominator = denominator;
        if (this.denominator == 0)
            this.denominator = 1;
    }

    private int gcd(int a, int b) {
        if (b == 0)
            return a;
        return gcd(b, a % b);
    }

    public Fraction reduce() {
        int mis = gcd(this.numerator, this.denominator);
        this.numerator = this.numerator / mis;
        this.denominator = this.denominator / mis;
        return this;
    }

    public Fraction add(Fraction f2) {
        this.numerator = (this.numerator * f2.denominator) + (f2.numerator * this.denominator);
        this.denominator = this.denominator * f2.denominator;

        return this;
    }

    public Fraction subtract(Fraction f2) {
        this.numerator = (this.numerator * f2.denominator) - (f2.numerator * this.denominator);
        this.denominator = this.denominator * f2.denominator;

        return this;
    }

    public Fraction multiply(Fraction f2) {
            this.numerator = this.numerator * f2.numerator;
            this.denominator = this.denominator * f2.denominator;
            this.reduce();
            return this;
    }

    public Fraction divide(Fraction f2) {
        Fraction f1 = this;
	    if (f2.numerator != 0) {
            this.numerator = f1.numerator * f2.denominator;
            this.denominator = f1.denominator * f2.numerator;
            this.reduce();
            return this;
	} else
	    return this;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if (this.numerator * other.denominator == other.numerator * this.denominator)
                return true;
            return false;
        }
        return false;
    }
}
