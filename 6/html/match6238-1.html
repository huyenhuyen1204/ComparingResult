import java.util.Scanner ;
public 
class Fraction
{
    private int numerator ;
    private int denominator = 1 ;

    public int getNumerator()
    {
        return numerator ;
    }
    public int getDenominator()
    {
        return denominator ;
    }
    public void setNumerator(int n)
    {
        this.numerator = n ;
    }

    public void setDenominator(int n)
    {
        if (n != 0) this.denominator = n ;
    }

    public Fraction (int a , int b)
    {
        this.numerator = a ;
        if (b != 0)
        {
            this.denominator = b ;
        }
    }

    public int gcd(int a, int b)
    {
        if (b == 0) return a ;
        return gcd(b,a % b) ;
    }

    public Fraction reduce()
    {
        int tmp = gcd(numerator , denominator) ;
        this.numerator = numerator / tmp ;
        this.denominator = denominator /tmp ;
        return this ;
    }

    public Fraction add(Fraction b)
    {
        int uc = gcd(this.denominator , b.denominator) ;
        int bc = (this.denominator * b.denominator) / uc ;
        this.numerator = this.numerator * (bc / this.denominator) + b.numerator * (bc / b.denominator) ;
        this.denominator = bc ;
        return this.reduce() ;
    }

    public Fraction subtract (Fraction b)
    {
        int uc = gcd(this.denominator , b.denominator) ;
        int bc = (this.denominator * b.denominator) / uc ;
        
this.numerator = this.numerator * (bc / this.denominator) - b.numerator * (bc / b.denominator) ;
        this.denominator = bc ;
        return this.reduce() ;
    }

    public Fraction multiply (Fraction b)
    {
        this.numerator = this.numerator * b.numerator ;
        this.denominator = this.denominator * b.denominator ;
        return this.reduce() ;
    }

    public Fraction divide (Fraction b)
    {
        if ((b.numerator != 0) && (b.denominator != 0))
        {
            this.numerator = numerator * b.denominator ;
            this.denominator = denominator * b.numerator ;
            return this.reduce() ;
        }
        else return this.reduce() ;
    }

    public boolean equals(Object obj)
    {
        if (obj instanceof Fraction)
        {
            Fraction other = (Fraction) obj;
            this.reduce ();
            other.reduce();
            if ((this.numerator == other.numerator) && (this.denominator == other.denominator)) return true ;
            else return false ;
        }
        else return false ;
    }

    public static void main (String[] args)
    {
        Fraction s1 = new Fraction(5,7) ;
        Fraction s2 = new Fraction(10,14) ;
        Fraction s3 = s1.divide(s2) ;
        System.out.println(s3.numerator) ;
        System.out.println(s3.denominator) ;

    }
}
