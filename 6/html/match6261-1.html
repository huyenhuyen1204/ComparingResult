public class Fraction {
    private int numerator,denominator;
    public int getNumerator(){
        return numerator;
    }
    public void setNumerator(int tu){
        this.numerator=tu;
    }
    public int getDenominator(){
        return denominator;
    }
    public void setDenominator(int mau){
        if (mau==0) this.denominator=1;
        else 
this.denominator=mau;
    }
    public Fraction(int tu, int mau){
        this.numerator=tu;
        if (mau==0) this.denominator=1;
        else this.denominator=mau;
    }
    public Fraction(){
        this.numerator=0;
        this.denominator=1;
    }
    public static int gcd(int a, int b) {
        if (b==0) return a;
        return gcd(b,a%b);
    }
    public Fraction reduce()
    {
        //Fraction result = new Fraction();
        int common = 0;
        int num = Math.abs(numerator);
        int den = Math.abs(denominator);
        common=gcd(num,den);
        //common = num;

        this.numerator = numerator / common;
        this.denominator = denominator / common;
        
return this;
    }
    public Fraction add(Fraction b)
    {
        //Fraction result = new Fraction();
        this.numerator=this.numerator*b.denominator+this.denominator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction b)
    {
        //Fraction result = new Fraction();
        this.numerator=this.numerator*b.denominator-this.denominator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        this.reduce();
        return this;
    }
    public Fraction multiply(Fraction b)
    {
        //Fraction result = new Fraction();
        this.numerator=this.numerator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        this.reduce();
        return this;
    }
    public Fraction divide(Fraction b)
    {
        //Fraction result = new Fraction();
        if (b.numerator==0) 
return this;
        else{
        this.numerator=this.numerator*b.denominator;
        this.denominator=this.denominator*b.numerator;
        this.reduce();
        return this;}
    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            this.reduce();
            other.reduce();
            if (this.numerator==other.numerator&&this.denominator==other.denominator)
                return true;
            else return false;
        }
        else return false;
    }
