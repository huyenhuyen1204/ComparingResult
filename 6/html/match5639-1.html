class Fraction {
    private int numerator, denominator = 1;

    public int getNumerator(){
        return numerator;
    }

    public int getDenominator(){
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) this.denominator = denominator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public Fraction(int numerator, int denominator) {
        if (denominator != 0) {
            this.numerator = numerator;
            this.denominator = denominator;
        }
    }

    public Fraction reduce() {
        int temp = GCD_LCM.gcd(numerator,denominator);
        numerator = numerator / temp;
        denominator = denominator / temp;
        return this;
    }

    public Fraction add (Fraction other) {
        if (this.denominator == other.denominator) {
            
this.numerator += other.numerator;
        }
        else {
            int temp = GCD_LCM.lcm(this.denominator, other.denominator);
            this.numerator = temp / this.denominator * this.numerator +
                    temp / other.denominator * other.numerator;
            this.denominator = temp;
        }
        return this;
    }

    public Fraction subtract (Fraction other) {
        if (this.denominator == other.denominator) {
            this.numerator -= other.numerator;
        }
        else {
            int temp = GCD_LCM.lcm(this.denominator, other.denominator);
            this.numerator = temp / this.denominator * this.numerator -
                    temp / other.denominator * other.numerator;
            this.denominator = temp;
        }
        return this;
    }

    public Fraction multiply (Fraction other) {
        this.numerator = this.numerator * other.numerator;
        this.denominator = this. denominator * other.denominator;
        return this;
    }

    public Fraction divide(Fraction other) {
        if (other.numerator != 0) {
            this.numerator = this.numerator * other.denominator;
            this.denominator = this. denominator * other.numerator;
        }
        return this;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            int temp = GCD_LCM.lcm(this.denominator,other.denominator);
            return ((temp / this.denominator * this.numerator) == (temp / other.denominator * other.numerator));
        }
        return false;
    }
}

public 
class GCD_LCM {
    public static int gcd(int a, int b) {
        if (b == 0) return a;
        if (a > b) return gcd (b, a % b);
        else return gcd(a, b % a);
    }

    public static int lcm(int a, int b) {
        if ((a % b) == 0) return a;
        else
        if ((b % a) == 0) return b;
        else return a * b / gcd(a,b);
    }
}
