class Fraction {
    private int numerator;
    private int denominator;

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public int getNumerator() {
        return numerator;
    }
    public void setDenominator(int denominator) {
        if(denominator==0) this.denominator=1;
        else this.denominator= denominator;
    }
    public int getDenominator()
    {
        return denominator;
    }
    public Fraction() {
        this.denominator = 1;
    }

    public Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        if (denominator == 0) {
            this.denominator = 1;
        } else {
            this.denominator = denominator;
        }
    }
    public int UCLN(int num1, int num2) {
        while ( num1 != 0){
            int tmp;
            tmp = num2 % num1;
            num2 = num1;
            num1 = tmp;
        }
        return num2;
    }

    public Fraction reduce() {
        int ucln = UCLN(numerator, denominator);
        this.numerator/=ucln;
        this.denominator/=ucln;
        return this;
    }

    // add
    public Fraction add(Fraction other) {
        int new_numerator = this.getNumerator() * other.getDenominator() + other.getNumerator() * this.getDenominator();
        int new_denominator = this.getDenominator() * other.getDenominator();
        Fraction sumOfFraction = new Fraction(new_numerator, new_denominator);
        sumOfFraction.reduce();
        return sumOfFraction;
        //System.out.println("Tổng hai phân số = " + sumOfFraction.numerator + "/" + sumOfFraction.denominator);
    }

    public Fraction subtract(Fraction other) {
        int new_numerator = this.getNumerator() * other.getDenominator() - other.getNumerator() * this.getDenominator();
        int new_denominator = this.getDenominator() * other.getDenominator();
        Fraction differenceOfFraction = new Fraction(new_numerator, new_denominator);
        differenceOfFraction.reduce();
        return differenceOfFraction;
        //System.out.println("Hiệu hai phân số = " + differenceOfFraction.numerator + "/" + differenceOfFraction.denominator);
    }

    // multiple
    public Fraction multiply(Fraction other) {
        int new_numerator = this.getNumerator() * other.getNumerator();
        int new_denominator = this.getDenominator() * other.getDenominator();
        Fraction productOfFraction = new Fraction(new_numerator, new_denominator);
        productOfFraction.reduce();
        return productOfFraction;
        //System.out.println("Tích hai phân số = " + productOfFraction.numerator + "/" + productOfFraction.denominator);
    }

    // divide
    public Fraction divide(Fraction other) {
        if (other.getNumerator()==0)
        {
            return this;
        }
        else {
            int new_numerator = this.getNumerator() * other.getDenominator();
            int new_denominator = this.getDenominator() * other.getNumerator();
            Fraction quotientOfFraction = new Fraction(new_numerator, new_denominator);
            quotientOfFraction.reduce();
            return quotientOfFraction;
            //System.out.println("Thương hai phân số = " + quotientOfFraction.numerator + "/" + quotientOfFraction.denominator);
        }
    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if(this.numerator*other.denominator == this.denominator*other.numerator) return true;
        }
        return false;
    }
}
