public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }

    Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if(denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }
    Fraction(){
        this.denominator = 1;
    }

    public Fraction reduce(){
        int gcd = 0;
        int a = Math.abs(this.numerator);
        int b = Math.abs(this.denominator);

        while(a * b != 0){
            if(a < b) {
                int tmp = a;
                a = b; b = tmp;
            }
            int tmp = a;
            a = b;
            b = tmp % b;
        
}
        if(a * b == 0) gcd = a + b;
        if(gcd != 0) {
            this.numerator /= gcd;
            this.denominator /= gcd;
        }
        return this;
    }

    public Fraction add(Fraction c){
        int ts = this.numerator * c.getDenominator() + c.getNumerator() * this.denominator;
        int ms = this.denominator * c.getDenominator();
        Fraction res = new Fraction(ts, ms);
        res = res.reduce();
        return res;
    }

    public Fraction subtract(Fraction c){
        int ts = this.numerator * c.getDenominator() - c.getNumerator() * this.denominator;
        int ms = this.denominator * c.getDenominator();
        Fraction res = new Fraction(ts, ms);
        res = res.reduce();
        return res;
    }

    public Fraction multiply(Fraction c){
        int ts = this.numerator * c.getNumerator() ;
        int ms = this.denominator * c.getDenominator();
        Fraction res = new Fraction(ts, ms);
        res = res.reduce();
        return res;
    }

    public Fraction divide(Fraction c){
        int ts = this.numerator * c.getDenominator() ;
        int ms = this.denominator * c.getNumerator();
        Fraction res = new Fraction(ts, ms);
        res = res.reduce();;
        return res;
    }

    public boolean equals(Object obj){
        if(obj instanceof Fraction){
            if(this.getNumerator() * ((Fraction) obj).getDenominator() == this.getDenominator() * ((Fraction) obj).getNumerator()) return true;
        }
        return false;
    }

    public static void main(String[] agrs){
        Fraction fc = new Fraction();
        fc.setDenominator(0);
        fc.setNumerator(0);
        fc.reduce();
        Fraction ffc = new Fraction(0, 5);
        fc.divide(ffc);
        System.out.print(fc.getNumerator() + "/" + fc.getDenominator());
    }
}
