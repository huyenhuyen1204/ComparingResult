import java.util.Scanner;

public class Fraction {
    private int numerator;
    private int denominator;
    public int getNumerator() {return numerator;}
    public void setNumerator(int numerator1) {
        this.numerator = numerator1;
    }
    public int getDenominator() {return denominator;}
    public void setDenominator(int denominator1) {
        if(denominator1 != 0) this.denominator = denominator1;
        else this.denominator = 1;
    }
    static int gcd(int a, int b) {
        if(a == 0) return b;
        return gcd(b % a, a);
    }
    Fraction(int numerator1, int denominator1) {
        this.numerator = numerator1;
        if(denominator1 != 0) this.denominator = denominator1;
        else this.denominator = 1;
    }
    public Fraction reduce() {
        int foo = gcd(Math.abs(this.numerator), Math.abs(this.denominator));
        this.numerator /= foo;
        this.denominator /= foo;
        return this;
    }
    public Fraction add(Fraction f) {
        this.numerator = this.numerator * f.getDenominator() + f.getNumerator() * this.denominator;
        this.denominator *= f.getDenominator();
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction f) {
        this.add(new Fraction(f.getNumerator() * (-1), f.getDenominator()));
        return this;
    }
    public Fraction multiply(Fraction f) {
        this.numerator *= f.getNumerator();
        this.denominator *= f.getDenominator();
        this.reduce();
        return this;
    }
    public Fraction divide(Fraction f) {
        this.multiply(new Fraction(f.getDenominator(), f.getNumerator()));
        return this;
    }
    public boolean equals(Object obj) {
        if(obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(this.numerator == other.getNumerator() && this.denominator == other.getDenominator()) return true;
            return false;
        }
        else return false;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        Fraction fr = new Fraction(1,2);
        Fraction k = new Fraction(3, 4);
        int n = 10;
        System.out.println(fr.equals(k));
    }
}
