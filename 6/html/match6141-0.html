public class Fraction{
    private int numerator, denominator;
    
    public int getNumerator(){
        return this.numerator;
    }
    public void setNumerator(int numerator){
        this.numerator = numerator;
    }
    public int getDenominator(){
        return this.denominator;
    }
    public void setDenominator(int denominator){
        if(denominator != 0){
            this.denominator = denominator;
        }
    }

    Fraction(int numerator, int denominator){
        if(denominator != 0){
            this.numerator = numerator;
            this.denominator = denominator;
        }
        else{
            this.numerator = numerator;
            this.denominator = 1;
        }
    }
    Fraction(){
        this.denominator = 1;
    }

    public static int gcd(int a, int b) {
        return a == 0 ? b : gcd(b % a, a);
    }
    public static long gcd(long a, long b) {
        return a == 0 ? b : gcd(b % a, a);
    }

    public Fraction reduce(){
        int gcd = gcd(this.numerator, this.denominator);
        int _numerator = this.numerator / gcd;
        int _denominator = this.denominator / gcd;
        return new Fraction(_numerator, _denominator);
    }
    
    public Fraction add(Fraction f){
        Fraction a = this.reduce();
        Fraction b = f.reduce();
        int gcd1 = gcd(a.getDenominator(),b.getDenominator());
        this.denominator = (a.getDenominator()/gcd1)*b.getDenominator();
        this.numerator = ( a.getNumerator() * (this.denominator/a.getDenominator()) ) + ( b.getNumerator() * (this.denominator/b.getDenominator()) );
        return this;
    }

    public Fraction subtract(Fraction f){
        Fraction ans = this.add(new Fraction(-f.getNumerator(), f.getDenominator()));
        return ans;
    }

    public Fraction multiply(Fraction f){
        Fraction a = this.reduce();
        Fraction b = f.reduce();
        this.numerator = a.getNumerator()*b.getNumerator();
        this.denominator = a.getDenominator()*b.getDenominator();
        return this;
    }
    public Fraction divide(Fraction f){
        if(f.getNumerator() == 0) return this;
        Fraction ans = this.multiply(new Fraction(f.getDenominator(), f.getNumerator()));
        return ans;
    }
    public boolean equals(Object obj){
        if(obj instanceof Fraction){
            Fraction a = ((Fraction) obj).reduce();
            Fraction b = this.reduce();
            
            if((a.getNumerator()/a.getDenominator() == b.numerator/b.denominator)){
                return true;
            }
            return false;
        }
        return false;
    }
}
