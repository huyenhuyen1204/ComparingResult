public class Fraction {
    private int numerator = 0;
    private int denominator = 1;

    public static int gcd(int a, int b) {
        while (b != 0) {
            int t = a;
            a = b;
            b = t % b;
        }
        return a;
    }
    public Fraction(int numerator, int denominator) {
        if (denominator != 0) {
            this.numerator = numerator;
            this.denominator = denominator;
        }
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
        }
    }

    public int getNumerator() {
        return this.numerator;
    }

    public int getDenominator() {
        return this.denominator;
    }

    public Fraction reduce() {
        int t = gcd(this.numerator, this.denominator);
        this.numerator = this.numerator / t;
        this.denominator = this.denominator /t;
        return new Fraction(this.numerator, this.denominator);
    }

    public Fraction add(Fraction other) {
        int tDenominator = this.denominator * other.denominator;
        int tNumerator = this.numerator * other.denominator + this.denominator * other.numerator;
        int t = tNumerator;
        tNumerator = tNumerator/gcd(tNumerator, tDenominator);
        tDenominator = tDenominator/gcd(t, tDenominator);
        return new Fraction(tNumerator, tDenominator);
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int tDenominator = this.denominator * other.denominator;
        int tNumerator = this.numerator * other.denominator - this.denominator * other.numerator;
        int t = tNumerator;
        tNumerator = tNumerator/gcd(tNumerator, tDenominator);
        tDenominator = tDenominator/gcd(t, tDenominator);
        return new Fraction(tNumerator, tDenominator);
    }

    // multiple
    public Fraction multiply(Fraction other) {
        int tDenominator = this.denominator * other.denominator;
        int tNumerator = this.numerator * other.numerator;
        return new Fraction(tNumerator / gcd(tDenominator, tNumerator), tDenominator / gcd(tDenominator, tNumerator));
    }

    // divide
    public Fraction divide(Fraction other) {
        if(other.denominator == 0)
        {
            return this;
        }
        int tDenominator = this.denominator * other.numerator;
        int tNumerator = this.numerator * other.denominator;
        return new Fraction(tNumerator / gcd(tDenominator, tNumerator), tDenominator / gcd(tDenominator, tNumerator));
    }


    public boolean equals(Object obj) {

        if (obj == null) {
            return false;
        }
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            int numeratora = this.numerator / gcd(this.numerator, this.denominator);
            int denominatora = this.denominator / gcd(this.numerator, this.denominator);
            int numeratorb = other.numerator / gcd(other.numerator, other.denominator);
            int denominatorb = other.denominator / gcd(other.numerator, other.denominator);
            if(numeratora == numeratorb && denominatora == denominatorb)
            {
                return true;
            }
        }
        return false;
    }
}
