public class Fraction {
    private int numerator, denominator;

    int getNumerator() {
        return numerator;
    }

    void setNumerator(int x) {
        numerator = x;
    }

    int getDenominator() {
        return denominator;
    }

    void setDenominator(int x) {
        if (x != 0)
            denominator = x
;
    }

    int gcd(int a, int b) {
        if (a < b)
            return gcd(b, a);
        return (b == 0 ? a : gcd(b, a % b));
    }

    Fraction() {
        denominator = 1;
    }

    Fraction(int x, int y) {
        if (y != 0) {
            numerator = x;
            denominator = y;
        } else {
            if (denominator == 0)
                denominator = 1;
        }
    }

    Fraction reduce() {
        Fraction tmp = new Fraction(numerator, denominator);
        int x = gcd(tmp.numerator, tmp.denominator);
        tmp.numerator /= x;
        tmp.denominator /= x;
        numerator = tmp.numerator;
        denominator = tmp.denominator;
        return this;
    }

    Fraction add(Fraction a) {
        Fraction tmp = new Fraction(this.numerator * a.denominator + this.denominator * a.numerator, this.denominator * a.denominator);
        numerator = tmp.numerator;
        denominator = tmp.denominator;
        return this.reduce();
    }

    Fraction subtract(Fraction a) {
        Fraction tmp = new Fraction(-1 * a.numerator, a.denominator);
        return add(tmp);
    }

    Fraction multiply(Fraction a) {
        Fraction tmp = new Fraction(this.numerator * a.numerator, this.denominator * a.denominator);
        numerator = tmp.numerator;
        denominator = tmp.denominator;
        return this;
    }

    Fraction divide(Fraction a) {
        
if (a.getNumerator() == 0)
            return this;
        Fraction tmp = new Fraction(a.denominator, a.numerator);
        return multiply(tmp);
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            return (this.reduce().getNumerator() == other.reduce().getNumerator() && this.reduce().getDenominator() == other.reduce().getDenominator());
        }
        return false;
    }
}
