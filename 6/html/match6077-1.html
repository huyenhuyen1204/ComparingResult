public class Fraction{
    private int numerator;
    private int denominator = 1;
    public int getNumerator(){
        return numerator;
    }
    public void setNumerator(int numerator){
        this.numerator=numerator;
    }
    public int getDenominator(){
        return denominator;
    }
    public void setDenominator(int denominator){
        if(denominator != 0)this.denominator = denominator;
    }
    public Fraction(){
        this.denominator = 1;
    }
    public Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if (denominator != 0)
            this.denominator = denominator;
    }
    //hàm tìm ước chung lớn nhất
    public int UCLN(int a, int b){
        if (b != 0)
            return UCLN(b,a % b);
        else return a;
    }
    // hàm rút gọn phân số
    public Fraction reduce(){
        int u = UCLN(this.getDenominator(),
this.getNumerator());
        this.setNumerator(this.getNumerator() / u);
        this.setDenominator(this.getDenominator() / u);
        if (this.getDenominator() <0 && this.getNumerator() > 0)
        {
            this.setDenominator(this.getDenominator() * -1);
            this.setNumerator(this.getNumerator() * -1);
        }
        return this;
    }
    // hàm cộng phận sô
    public Fraction add(Fraction fr)
    {
        if(fr.denominator==0 || this.denominator==0)
        {
            return this;
        }
        else
        {
            this.setNumerator(this.getNumerator()*fr.denominator+fr.numerator*this.getDenominator());
            this.setDenominator(this.getDenominator()*fr.denominator);
            return this;
        }
 
    }
    //hàm trừ phân số
    public Fraction subtract(Fraction fr){
        if (fr.denominator == 0 || this.denominator == 0)
            return this;
        else {
            this.setNumerator(this.getNumerator()*fr.denominator-fr.numerator*this.getDenominator());
            this.setDenominator(this.getDenominator()*fr.denominator);
            return this;
        }
    }
    //hàm nhân phân số
    public Fraction multiply(Fraction fr){
        if (fr.denominator == 0 || this.denominator == 0)
            return this;
        else {
            this.setNumerator(this.getNumerator()*fr.numerator);
            this.setDenominator(this.getDenominator()*fr.denominator);
            return this;
        }
    }
    //hàm chia phân sô
    public Fraction divide(Fraction fr){
        if (fr.denominator == 0 || this.denominator == 0)
            return this;
        else{
            this.setNumerator(this.getNumerator() * fr.denominator);
            this.setDenominator(this.getDenominator() * fr.numerator);
            return this;
        }
    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction){
            Fraction other = (Fraction) obj;
            if (this.getDenominator() * other.getNumerator() - this.getNumerator() * other.getDenominator() == 0)
                return true;
        }
        return false;
    }

}
