class Fraction {
    private int numerator;
    private int denominator ;
    public Fraction()
    {
        this.numerator = 0;
        this.denominator =1;
    }
    public Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        if( denominator != 0)

        {
            this.denominator = denominator;

        }
        else
        {
            this.denominator = 1;
            this.numerator = 0;
        }
    }

    public int getNumerator() {
        return numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setNumerator(int numerator) {

        this.numerator = numerator;
    }

    public void setDenominator(int denominator) {
        if(denominator != 0)
            this.denominator = denominator;
    }
    public static int gcd(int a, int b) {
        while (a != 0 && b != 0)
        {
            if (a > b) a = a - b;
            else if ( b>a ) b = b-a;
            else return a;
        }
        return a+b;
    }
    public Fraction reduce ( )
    {
        int k = gcd(this.getDenominator(),this.getNumerator());
        Fraction l = new Fraction(this.getNumerator(),this.getDenominator());

        l.setDenominator(this.getDenominator()/k);
        l.setNumerator(this.getNumerator()/k);
        this.setDenominator(l.getDenominator());
        this.setNumerator(l.getNumerator());
        return  l;
    }
    public Fraction add (Fraction s)
    {

        Fraction k = new Fraction(1,1);
        k.setNumerator( this.numerator*s.getDenominator() +s.getNumerator()*this.denominator);
        k.setDenominator(s.getDenominator()*this.denominator);
        this.setDenominator(k.getDenominator());
        this.setNumerator(k.getNumerator());
        return  k;

    }
    public Fraction subtract (Fraction s)
    {
        Fraction k = new Fraction(1,1);
        k.setNumerator( this.numerator*s.getDenominator() -s.getNumerator()*this.denominator);
        k.setDenominator(s.getDenominator()*this.denominator);
        this.setDenominator(k.getDenominator());
        this.setNumerator(k.getNumerator());
        return  k;

    }
    public Fraction multiply (Fraction s)
    {
        Fraction k = new Fraction(1,1);

        k.setNumerator(this.getNumerator()*s.getNumerator());
        k.setDenominator(this.getDenominator()*s.getDenominator());
        this.setDenominator(k.getDenominator());
        this.setNumerator(k.getNumerator());
        return  k;

    }

    public Fraction divide (Fraction s)

    {
        Fraction k = new Fraction(this.getNumerator(),this.getDenominator());

        if ( s.getNumerator() != 0)
        {
            k.setNumerator(this.getNumerator()*s.getDenominator());
            k.setDenominator(this.getDenominator()*s.getNumerator());
            this.setDenominator(k.getDenominator());
            this.setNumerator(k.getNumerator());

            return  k;
        }

        return  k;

    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            Fraction k = new Fraction(this.getNumerator(),this.getDenominator());

            if ( (k.subtract(other)).numerator ==0 ) return true;
        }
        return false;
    }

    public static void main(String[] args) {
        Fraction a = new Fraction(4,6);
        Fraction b = new Fraction(6,0);
        a.multiply(b);
        System.out.print(a.getNumerator()+ " "+ a.getDenominator());
    }




}
