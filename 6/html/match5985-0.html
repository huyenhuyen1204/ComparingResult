public class Fraction {
    private int  numerator ;
    private int denominator;

    public int getNumerator()
    {
        return numerator;
    }
    public int getDenominator()
    {
        return denominator;
    }

    public void setNumerator (int numerator)
    {
        this.numerator= numerator;
    }
    public void setDenominator (int denominator)
    {
        if(denominator==0) {
            denominator =1;
        }
        this.denominator=denominator;
    }
   public Fraction(int numerator,int denominator)
    {
        if (denominator == 0) denominator = 1;
        this.numerator=numerator;
        this.denominator=denominator;
    }

   public int gcd(int numerator, int denominator)
    {
        int a = Math.abs(numerator);
        int b = Math.abs(denominator);
        if(a==0||b==0) return (a+b);
        else {
            while (a != b) {
                if (a > b) a = a - b;
                else b = b - a;
            }
            return a;
        }
    }
   public Fraction reduce()
    {
        int UCLN = gcd(numerator,denominator);
        this.denominator=denominator/UCLN;
        this.numerator=numerator/UCLN;
        return this;
    }
   public Fraction add(Fraction x)
    {
        numerator = numerator * x.denominator + denominator * x.numerator;
        denominator = denominator * x.denominator;
        this.reduce();
        return this;
    }
   public Fraction subtract(Fraction x)
    {
        numerator = numerator * x.denominator - denominator * x.numerator;
        denominator = denominator * x.denominator;
        this.reduce();
        return this;
    }
    public Fraction divide(Fraction x) {
        numerator = x.denominator * numerator;
        denominator = denominator * x.numerator;
        this.reduce();
        return this;
    }

    public Fraction multiply(Fraction x)
    {
        numerator = numerator * x.numerator;
        denominator = denominator * x.denominator;
        reduce();
        return this;
    }
    public boolean equals(Object obj)
    {
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;
// compare this vs other here
            if(this.numerator*other.denominator==this.denominator*other.numerator)
              return true;
            else
                return false;
        }
        else
            return false;
    }

}
