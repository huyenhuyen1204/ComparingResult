public class Fraction {
        private int numerator;
        private int denominator;

        public int getNumerator() {return this.numerator;}
        public int getDenominator() {return this.denominator;}

        public void setNumerator(int numerator) {this.numerator = numerator;}
        public void setDenominator(int denominator) {
            if (denominator == 0)
                this.denominator = 1 ;
            else this.denominator = denominator ;
        }


        public Fraction()
        {
            this.denominator = 1;
        }

        public Fraction(int ts)
        {
            this.numerator = ts ; this.denominator = 1 ;
        }

        public Fraction(int ts, int ms)
        {
            if(ms == 0)
            {this.numerator = ts; this.denominator = 1 ;}
            else this.numerator = ts; this.denominator = ms ;
        }

        public Fraction(Fraction f)
        {
            this.numerator = f.getNumerator();
            this.denominator = f.getDenominator();
        }

        public Fraction reduce()
        {
            int a = Math.abs(this.numerator);
            int b = Math.abs(this.denominator);
            while (a*b != 0){
                if (a > b){
                    a %= b;
                }else{
                    b %= a;
                }
            }
            this.numerator = this.numerator/(a+b);
            this.denominator = this.denominator/(a+b);
            return this ;
        }

        public void display()
        {
            if (this.numerator == 0) {System.out.println(0);return;}
            else System.out.print(this.numerator);

            if (this.denominator==1) return;
            else System.out.println("/"+this.denominator);

        }

        public Fraction add(Fraction f2) {
            this.numerator = this.numerator * f2.denominator + this.denominator* f2.numerator ;
            this.denominator = this.denominator * f2.denominator ;
            this.reduce();
            return this ;
        }
        public Fraction subtract(Fraction f2) {
            this.numerator = this.numerator * f2.denominator - this.denominator* f2.numerator ;
            this.denominator = this.denominator * f2.denominator ;
            this.reduce();
            return this ;
        }
        public Fraction multiply(Fraction f2){
            this.denominator = this.denominator * f2.denominator ;
            this.numerator = this.numerator *  f2.numerator ;
            this.reduce();
            return this ;
        }
        public Fraction divide (Fraction f2){
            if(f2.numerator == 0 ) return this;
            this.numerator = this.numerator *  f2.denominator ;
            this.denominator = this.denominator * f2.numerator ;
            this.reduce();
            return this ;
        }

        public boolean equals(Object obj)
        {
            if (obj instanceof Fraction) {

                Fraction other = (Fraction) obj;
                other.reduce();
                this.reduce();
                if(this.numerator == other.numerator && this.denominator == other.denominator) return true;
            }
            return false ;
        }
}
