public class Fraction {

    private int numerator;
    private int denominator;

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public int getNumerator() {
        return numerator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
        
}
    }
    public int getDenominator() {
        return denominator;
    }


    public Fraction() {
        this.numerator = 0;
        this.denominator = 1;
    }
    public Fraction(int numerator) {
        this.numerator = numerator;
        this.denominator = 1;
    };
    public Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        this.denominator = denominator;
        if (denominator == 0) {
            this.denominator = 1;
        }
    }

    public static int absoluteValue(int n) {
        if (n >= 0) {
            return n;
        }
        else {
            return -n;
        }
    }

    public static int gcd(int a, int b) {
        if (a != 0 && b != 0) {
            if (a == b) {
                return a;
            }
            else if (a > b) {
                return gcd(a - b, b);
            }
            else {
                return gcd(a, b - a);
            }
        }
        else if (a == 0 && b != 0) {
            return b;
        }
        else {
            return a;
        }
    }
    public 
void reduce() {
        if (denominator < 0) {
            denominator = absoluteValue(denominator);
            numerator = -numerator;
        }
        numerator = numerator / gcd(absoluteValue(numerator), denominator);
        denominator = denominator / gcd(absoluteValue(numerator), denominator);
    
}

    public static int lcm(int a, int b) {
        return a * b / gcd(a, b);
    }

    public Fraction add(Fraction other) {
        Fraction sum = new Fraction();
        if (other.denominator != 0) {
            this.reduce();
            other.reduce();
            int n = lcm(this.denominator, other.denominator);
            sum.denominator = n;
            sum.numerator = this.numerator * (n / this.denominator) + other.numerator * (n / other.denominator);
            sum.reduce();
        }
        return sum;
    }
    public Fraction subtract(Fraction other) {
        Fraction difference = new Fraction();
        if (other.denominator != 0) {
            this.reduce();
            other.reduce();
            int n = lcm(this.denominator, other.denominator);
            difference.denominator = n;
            difference.numerator = this.numerator * (n / this.denominator) - other.numerator * (n / other.denominator);
            difference.reduce();
        }
        return difference;
    }

    public Fraction multiply(Fraction other) {
        Fraction product = new Fraction();
        if (other.denominator != 0) {
            this.reduce();
            other.reduce();
            product.numerator = this.numerator * other.numerator;
            product.denominator = this.denominator * other.denominator;
            product.reduce();
        }
        return product;
    }
    public Fraction divide(Fraction other) {
        Fraction quotient = new Fraction();
        if (other.numerator != 0 && other.denominator != 0) {
            this.reduce();
            other.reduce();
            quotient.numerator = this.numerator * other.denominator;
            quotient.denominator = this.denominator * other.numerator;
            quotient.reduce();
        }
        
return quotient;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction another = (Fraction) obj;
            this.reduce();
            another.reduce();
            if ((this.numerator == another.numerator && this.denominator == another.denominator) || (this.numerator == 0 && another.numerator == 0)) {
                return true;
            }
        }
        return false;
    }
}
