public class Fraction {
    private int numerator, denominator=1;

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public int getNumerator() {
        return numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) this.denominator = denominator;
    }

    Fraction(int Numerator, int Denominator)
    {

        if(Denominator!=0)
        {
            this.numerator = Numerator;
            this.denominator = Denominator;
        }
        else    this.denominator =1;
    }
    Fraction()
    {
        this.denominator =1;
    }

    public Fraction reduce() {
        int a = Math.abs(this.numerator);
        int b = Math.abs(this.denominator);
        while (a != b) {
            if (a > b) a -= b;
            else b -= a;
        }
        Fraction res = new Fraction(this.numerator/a,this.denominator/b);
        return res;
    }

    public Fraction add(Fraction Fraction_1) {
        int a = this.numerator * Fraction_1.getDenominator() + Fraction_1.getNumerator()* this.denominator;
        int b = this.denominator * Fraction_1.getDenominator();
        Fraction res = new Fraction(a,b);
        return res.reduce();
    }

    public Fraction subtract(Fraction Fraction_1) {
        int a = this.numerator * Fraction_1.getDenominator() - Fraction_1.getNumerator() * this.denominator;
        int b = this.denominator * Fraction_1.getDenominator();
        Fraction res = new Fraction(a,b);
        return res.reduce();
    }
    public Fraction multiply(Fraction Fraction_1)
    {
        int a = this.numerator* Fraction_1.getNumerator();
        int b = this.denominator*Fraction_1.getDenominator();
        Fraction res = new Fraction(a,b);
        return res.reduce();
    }
    public Fraction divide(Fraction Fraction_1)
    {
        if(Fraction_1.getDenominator()!=0)
        {
            int a = this.numerator * Fraction_1.getDenominator();
            int b = this.denominator * Fraction_1.getNumerator();
            Fraction res = new Fraction(a,b);
            return res.reduce();
        }
        else return  null;

    }
    public boolean equals(Object obj)
    {
        boolean test = obj instanceof Fraction;
        if(test)
        {
            int a = ((Fraction) obj).getNumerator();
            int b = ((Fraction) obj).getDenominator();
            if((this.numerator * b - this.denominator * a)==0)  return true;
            else    return false;

        }
        else    return false;
    }

    }
}
