public class Fraction
{       private int numerator;
        private int denominator;

        public int getDenominator(){return denominator;}
        public void setDenominator(int denominator)

            if(denominator==0) denominator=1
;

        public Fraction(int numerator,int denominator)
        {
            this.numerator=numerator;
            if(denominator==0) denominator=1;
            this.denominator=denominator;
        }
        public int gcd(int a,int b)
        {
            if(a==b)return a;
            else if(b%a==0) return a;
            return gcd(b,a%b);

        }
        public Fraction reduce()
        {
            if(this.denominator==0)this.denominator=1;
            int a=gcd(this.numerator,this.denominator);
            this.numerator=this.numerator/a;
            this.denominator=this.denominator/a;
            return this;
        }
        public Fraction add (Fraction s)
        {
            if(s.denominator==0) 
s.denominator=1;
            this.numerator=this.numerator*s.denominator+this.denominator*s.numerator;
            this.denominator=this.denominator*s.denominator;
            return this.reduce();
        }
        public Fraction subtract (Fraction s)
        {
            if(s.denominator==0) s.denominator=1;
            this.numerator=this.numerator*s.denominator-this.denominator*s.numerator;
            this.denominator=this.denominator*s.denominator;
            return this.reduce();
        }
        public Fraction multiply (Fraction s)
        {
            if(s.denominator==0) s.denominator=1;
            this.numerator=this.numerator*s.numerator;
            this.denominator=this.denominator*s.denominator;
            return this.reduce();
        }
        public Fraction divide (Fraction s)
        {
            if(s.numerator==0)return this;
            if(s.denominator==0) s.denominator=1;
            this.numerator=this.numerator*s.denominator;
            this.denominator=this.denominator*s.numerator;
            return this.reduce();
        }
        public boolean equals(Object obj)
        {
            if(obj instanceof Fraction)
            {
                Fraction s= (Fraction) obj;
                s.reduce();
                this.reduce();
                if(this.numerator==s.numerator&&this.denominator==s.denominator) return true;
                return false;
            }
            else return false;
        }
