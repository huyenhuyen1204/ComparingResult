public class Fraction {
    private  int numerator ;
    private  int denominator;

    void setNumerator(int numerator){
        this.numerator = numerator;
    }

    void setDenominator(int denominator){
        if (denominator != 0){
            this.denominator = denominator;
        }
    }

    int getNumerator(){
        return this.numerator;
    }
    int getDenominator(){
        return this.denominator;
    }

    void getInfo(){
        System.out.println(this.numerator + " " + this.denominator);
    }

    Fraction(){
        this.denominator = 1;
    }

    Fraction(int numerator, int denominator){
        if (denominator != 0){
            this.numerator = numerator;
            this.denominator = denominator;
        }
        else if (denominator == 0){
            this.denominator = 1;
        }
    }

    int gcd(int a, int b){
        int temp = a;
        while (a != 0) {
            temp = b;
            b = a;
            a = temp % a;
        }
        return b;
    }

    Fraction add (Fraction a){
        Fraction res = new Fraction();

        int commomDenominator = (this.denominator * a.getDenominator()) / gcd(this.denominator, a.getDenominator());
        res.setNumerator(this.numerator*(commomDenominator / this.denominator) + a.getNumerator()*(commomDenominator/a.getDenominator()));
        res.setDenominator(commomDenominator);

        return res;
    }

   Fraction subtract (Fraction a){
        Fraction temp = new Fraction();
        temp.setNumerator(-1 * a.getNumerator());
        temp.setDenominator(a.getDenominator());

        return add(temp);
    }

    Fraction multiply (Fraction a){
        Fraction res = new Fraction();

        res.setNumerator(a.getNumerator() * this.numerator);
        res.setDenominator(a.getDenominator() * this.denominator);

        return res;
    }

    Fraction divide (Fraction b){
        if (b.getNumerator() == 0){
            return this;
        }

        Fraction res = new Fraction();

        res.setNumerator(this.numerator * b.getDenominator());
        res.setDenominator(this.denominator * b.getNumerator());

        return res;
    }



    Fraction reduce(){

        int a = gcd (this.numerator, this.denominator);

        this.numerator /= a;
        this.denominator /= a;

        return this;
    }

    public boolean equals (Object obj){

        Fraction other;

        if (obj instanceof Fraction){
            other = (Fraction) obj;
        }
        else return false;
        this.reduce();
        other = other.reduce();
        if ((this.getNumerator() == other.getNumerator()) && (this.getDenominator() == other.getDenominator())){
            return true;
        }
        else return false;
    }

    public static void main(String[] args) {
    }
}
