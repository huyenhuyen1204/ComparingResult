public class Fraction {
    /**
     * declare attributes here
     */
    private int numerator;
    private int denominator;
    /**
     * declare constructor here
     */
    Fraction() {
        this.numerator = 1;
        this.denominator = 1;
    }
    Fraction(int numerator, int denominator) {
        this.numerator = numerator;
        if (denominator == 0) denominator = 1;
        this.denominator = denominator;
    }

    Fraction(Fraction s) {
        this.numerator = s.numerator;
        this.denominator = s.denominator;
    }
    /**
     * methods
     */
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getNumerator() {
        return this.numerator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }

    public int getDenominator() {
        return denominator;
    }

    public int gcd(int a, int b) {
        int tmp;
        while ( b != 0) {
            tmp = a%b;
            a = b;
            b = tmp;
        }
        return a;
    }

    public Fraction reduce() {
        int tmp = gcd(this.getNumerator(), this.getDenominator());
        Fraction F = new Fraction(this.getNumerator()/tmp, this.getDenominator()/tmp);
        
return F;
    }

    // add
    public Fraction add(Fraction other) {
        int FN = this.getNumerator() * other.getDenominator() + this.getDenominator() * other.getNumerator();
        int FD = this.getDenominator() * other.getDenominator();
        Fraction result = new Fraction(FN, FD);
        return result.reduce();
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int FN = this.getNumerator()*other.getDenominator() - this.getDenominator()*other.getNumerator();
        int FD = this.getDenominator()*other.getDenominator();
        Fraction result = new Fraction(FN, FD);
        return result.reduce();
    }

    // multiply
    public Fraction multiply(Fraction other) {
        int FN = this.getNumerator()*other.getNumerator();
        int FD = this.getDenominator()*other.getDenominator();
        Fraction result = new Fraction(FN, FD);
        return result.reduce();
    }

    // divide
    public Fraction divide(Fraction other) {
        int FN = this.getNumerator()*other.getDenominator();
        int FD = this.getDenominator()*other.getNumerator();
        Fraction result = new Fraction(FN, FD);
        return result.reduce();
    }

        Fraction p1 = new Fraction(4,7);
        Fraction p2 = new Fraction(8, 14);
        System.out.println(p1.add(p2).getNumerator() + " " + p1.add(p2).getDenominator());

}
