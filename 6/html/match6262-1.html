import java.lang.Object;

import static java.lang.Math.abs;

public class Fraction {
    private int numerator;
    private int denominator=1;

    public int getNumerator(){return this.numerator;}
    public int getDenominator(){return this.denominator;}
    public void setNumerator(int numerator){
        if (numerator==0) System.out.println("0");
        this.numerator=numerator;}
    public void setDenominator(int denominator){
        if(denominator == 0)
            System.out.println(" ");
        
else
            this.denominator=denominator;
    }

    public Fraction(){}

    public Fraction(int numerator,int denominator) {
        this.setNumerator(numerator);
        this.setDenominator(denominator);
    }

    public Fraction reduce (){
        Fraction fraction =new Fraction();
        int a,b;
        a= abs(this.numerator);
        b= abs(this.denominator);
        while (a!=b) {
            if (a>b) a=a-b;
            else b=b-a;
        }
        if (this.numerator*this.denominator>0) {
            fraction.setNumerator(abs(this.numerator)/a);
            fraction.setDenominator(abs(this.denominator)/a);
        };
        if (this.numerator*this.denominator<0) {
            fraction.setNumerator(abs(this.numerator)/(-a));
            fraction.setDenominator(abs(this.denominator)/a);
        };
        return fraction;
    }

    public Fraction add(Fraction a) {
        Fraction fraction = new Fraction();
        fraction.setNumerator(a.getNumerator()*this.denominator+this.numerator*a.getDenominator());
        fraction.setDenominator(a.getDenominator()*this.denominator);
        fraction=fraction.reduce();
        return fraction;
    }

    public Fraction subtract(Fraction a) {
        Fraction fraction = new Fraction();
        fraction.setNumerator(this.numerator*a.getDenominator()-a.getNumerator()*this.denominator);
        fraction.setDenominator(a.getDenominator()*this.denominator);
        fraction=fraction.reduce();
        return fraction;
    }

    public Fraction multiply(Fraction a) {
        Fraction fraction = new Fraction();
        fraction.setNumerator(a.getNumerator()*this.numerator);
        fraction.setDenominator(a.getDenominator()*this.denominator);
        fraction=fraction.reduce();
        return fraction;
    }

    public Fraction divide(Fraction a) {
        Fraction fraction = new Fraction();
        fraction.setNumerator(this.numerator*a.getDenominator());
        fraction.setDenominator(this.denominator*a.getNumerator());
        fraction=fraction.reduce();
        return fraction;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            if (this.numerator*((Fraction) obj).getDenominator()==this.denominator*((Fraction) obj).getNumerator()) return true;
        }
        return false;
    }
