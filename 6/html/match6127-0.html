public class Fraction{
    private int numerator;
    private int denominator;

    public int getNumerator(){
        return numerator;
    }

    public void setNumerator(int num){
        this.numerator = num;
    }

    public int getDenominator(){
        return denominator;
    }

    public void setDenominator(int num){
        if(num != 0) this.denominator = num;
    }

    public Fraction(int a, int b){
        if(b == 0){
            this.denominator = 1;
            this.numerator = a;
        }
        else{
            this.numerator = a;
            this.denominator = b;
        }
    }

    public static int gcd(int a, int b){
        int ucln = 1;
        for(int i = 1; i <= a; i++){
            if(a%i == 0 && b % i == 0 && ucln < i)
                ucln = i;
        }
        return ucln;
    }

    public Fraction reduce(){
        int t = gcd(this.numerator, this.denominator);
        this.numerator = this.numerator / t;
        this.denominator = this.denominator / t;
        return this;
    }

    public Fraction add(Fraction a){
        this.numerator = this.numerator * a.denominator + this.denominator * a.numerator;
        this.denominator = this.denominator * a.denominator;
        return this;
    }

    public Fraction subtract(Fraction a){
        this.numerator = this.numerator * a.denominator - this.denominator * a.numerator;
        this.denominator = this.denominator * a.denominator;
        return this;
    }

    public Fraction multiply(Fraction a){
        this.numerator = this.numerator * a.numerator;
        this.denominator = this.denominator * a.denominator;
        return this;
    }

    public Fraction divide(Fraction a){
        if(a.numerator == 0) return this;
        else{
            this.numerator = this.numerator * a.denominator;
            this.denominator = this.denominator * a.numerator;
            return this;
        }
    }

    public boolean equals(Object obj){
        if(obj instanceof Fraction){
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(other.numerator == this.numerator && other.denominator == this.denominator)
                return true;
        }
        return false;
    }
}
