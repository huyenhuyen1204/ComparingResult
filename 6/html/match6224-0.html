public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator() {return numerator;}
    public void setNumerator(int a) {this.numerator = a;}

    public int getDenominator() {return denominator;}
    public void setDenominator(int a) {
        if (a == 0) this.denominator=1;
        else {
            this.denominator = a;
        }
    }

    public Fraction(int num1, int num2)
    {
        this.numerator = num1;
        if(num2 == 0)
        {
            this.denominator = 1;
        }
        else
        {
            this.denominator = num2;
        }
    }

    public static int ucln(int firstNumber, int secondNumber)
    {
        if(secondNumber == 0)
        {
            return firstNumber;
        }
        else
        {
            return ucln(secondNumber, firstNumber%secondNumber);

        }
    }

    public Fraction reduce()
    {
        int temp = ucln(this.getNumerator(), this.getDenominator());
        this.setNumerator(this.getNumerator()/temp);
        this.setDenominator(this.getDenominator()/temp);

        return this;
    }
    public Fraction add(Fraction fraction)
    {
        int x = this.denominator*fraction.getDenominator();
        int y = this.numerator*fraction.getDenominator() + this.denominator*fraction.getNumerator();
        this.setDenominator(x);
        this.setNumerator(y);
        this.reduce();

        return this;
    }
    public Fraction subtract(Fraction fraction)
    {
        int x = this.denominator*fraction.getDenominator();
        int y = this.numerator*fraction.getDenominator() - this.denominator*fraction.getNumerator();
        this.setDenominator(x);
        this.setNumerator(y);
        this.reduce();

        return this;
    }
    public Fraction multiply(Fraction fraction)
    {
        this.denominator = this.denominator*fraction.getDenominator();
        this.numerator = this.numerator*fraction.getNumerator();
        this.reduce();

        return this;
    }
    public Fraction divide(Fraction fraction)
    {
        this.denominator = this.denominator*fraction.getNumerator();
        this.numerator = this.numerator*fraction.getDenominator();
        this.reduce();

        return this;
    }
    public boolean equals(Object obj)
    {
        if(obj instanceof Fraction)
        {
            Fraction fraction1 = (Fraction) obj;
            fraction1.reduce();
            this.reduce();
            if(this.getNumerator() == fraction1.getNumerator() && this.getDenominator() == fraction1.getDenominator())
            {
                return true;
            }
            else
            {
                return false;
            }
        }
        else
        {
            return false;
        }
    }
}
