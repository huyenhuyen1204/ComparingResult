public class Fraction {
    private int numerator , denominator ;
    public int getNumerator ()
    {
        return numerator ;
    }
    public int getDenominator ()
    {
        return denominator ;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public void setDenominator(int denominator) {
        if ( denominator != 0 ) this.denominator = denominator ;
        else this.denominator = 1 ;
    }
    public  Fraction()
    {
        this.denominator = 1 ;
    }
    public Fraction ( int numerator , int denominator )
    {
        this.numerator = numerator ;
        if ( denominator != 0 )
            this.denominator = denominator ;
        else
            this.denominator = 1 ;
    }
    public int gcd ( int a , int b ) {
        if ( a == 0 || b == 0 )
            return  a + b ;
        else if (a > b)
            return gcd(a % b, b);
        else if (a < b)
            return gcd(a, b % a);
        if ( a == b ) return a ;
        return 1;
    }
    public Fraction reduce ()
    {
        int k = gcd( getNumerator() , getDenominator() ) ;
        int c = getDenominator()/k;
        int x = getNumerator()/k ;
        Fraction xxx = new Fraction(x,c) ;
        return xxx ;
    }
    public Fraction add(Fraction other) {
        int h = getNumerator()*other.getDenominator() + getDenominator()*other.getNumerator() ;
        int k = getDenominator()*other.getDenominator() ;
        Fraction xxx = new Fraction( h , k ) ;
        return xxx.reduce() ;
    }
    public Fraction subtract(Fraction other) {
        int h = getNumerator()*other.getDenominator() - getDenominator()*other.getNumerator() ;
        int k = getDenominator()*other.getDenominator() ;
        Fraction xxx = new Fraction( h , k ) ;
        return xxx.reduce() ;
    }
    public Fraction multiply(Fraction other) {
        int h = getNumerator()*other.getNumerator() ;
        int k = getDenominator()*other.getDenominator() ;
        Fraction xxx = new Fraction( h , k ) ;
        return xxx.reduce() ;
    }
    public Fraction divide(Fraction other) {
        if ( other.getNumerator() != 0 )
        {
            int h = getNumerator()*other.getDenominator() ;
            int k = getDenominator()*other.getNumerator() ;
            Fraction xxx = new Fraction( h , k ) ;
            return xxx.reduce() ;
        }
        else{
            int h = getNumerator() ;
            int k = getDenominator() ;
            Fraction xxx = new Fraction( h , k ) ;
            return xxx.reduce() ;
        }
    }
    public boolean equals(Object obj)
    {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            int h = getNumerator() ;
            int k = getDenominator() ;
            Fraction xxx = new Fraction( h , k ) ;
            if ( xxx.reduce().getNumerator() == other.reduce().getNumerator()
            && xxx.reduce().getDenominator() == other.reduce().getDenominator() )
                return true ;
            else return false ;
        }
        return false ;
    }
}
