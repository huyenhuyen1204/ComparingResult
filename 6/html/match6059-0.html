public class Fraction {
    /**
     * declare attributes here
     */
    private int numerator;
    private int denominator;
    public int getNumerator() {
        return numerator;
    }

    public int getDenominator() {
        if (denominator==0) return 1;
        else
            return denominator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public void setDenominator(int denominator) {
        if (denominator!=0)
            this.denominator = denominator;
        else this.denominator=1;
    }
    /**
     * declare constructor here
     */
    public Fraction() {}
    public Fraction(int numerator, int denominator) {
        this.numerator=numerator;
        if (denominator!=0) this.denominator=denominator;
                else this.denominator=1;
    }
    public static int gcd(int a, int b) {
        if (b==0) return a;
        else return gcd(b,a%b);
    }
    public Fraction reduce() {
        int ucln= gcd(Math.abs(numerator),Math.abs(denominator));
        //Fraction f= new Fraction();
        if (numerator<0 && denominator<0) ucln=ucln*(-1);
        numerator=numerator/ucln;
        denominator= denominator/ucln;
        return this;
    }
    public Fraction add(Fraction other) {
        int num=this.numerator*other.denominator+ this.denominator*other.numerator;
        int den=this.denominator*other.denominator;
        numerator=num;
        denominator=den;
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction other) {
        int num=this.numerator*other.denominator- this.denominator*other.numerator;
        int den=this.denominator*other.denominator;
        numerator=num;
        denominator=den;
        this.reduce();
        return this;
    }

    // multiple
    public Fraction multiply(Fraction other) {
        int num=this.numerator*other.numerator;
        int den=this.denominator*other.denominator;
        numerator=num;
        denominator=den;
        this.reduce();
        return this;
    }
    // divide
    public Fraction divide(Fraction other) {
        int num=this.numerator*other.denominator;
        int den=this.denominator*other.numerator;
        numerator=num;
        denominator=den;
        this.reduce();
        return this;

    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if (numerator*other.denominator- denominator*other.numerator==0)
                return true;
            else return false;

        } else return false;
    }

}
