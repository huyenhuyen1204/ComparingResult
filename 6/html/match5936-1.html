public class Fraction {
    int ucln(int a, int b) {
        int tmp;
        while(b!= 0) {
            tmp = a % b;
            a = b;
            b = tmp;
        }
        return a;
    }
    int bcnn(int a,int b){
        return (a*b)/ucln(a,b);
    }
    
    private int numerator;
    private int denominator;
    public Fraction(){
        numerator=1;
        denominator=1;
    }
    public Fraction(int a,int b){
        if (b!=0){
            this.denominator= b;
            this.numerator = a;
        }else {
            this.denominator= 1;
            this.numerator = a;
        }
    }
 

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator == 0) this.denominator = 1;
        else 
this.denominator = denominator;
    }
    
    public Fraction reduce(){
        this.denominator=this.denominator/ucln(this.denominator,this.numerator);
        this.numerator=this.numerator/ucln(this.denominator,this.numerator);
        return this;
    } 
    public Fraction add(Fraction c){
        int k=bcnn(this.denominator,c.denominator);
        this.numerator=this.numerator*(k/this.denominator)+c.numerator*(k/c.denominator);
        this.denominator=k;
        return this;
    }
    public Fraction subtract(Fraction d){
        int h=bcnn(this.denominator,d.denominator);
        this.numerator=this.numerator*(h/this.denominator)-d.numerator*(h/d.denominator);
        this.denominator=h;
        return this;
    }
    public Fraction multiply(Fraction e){
      
        int 
a=this.numerator*e.numerator;
        int b=this.denominator*e.denominator;
        int k=ucln(a,b);
        this.numerator=a/k;
        this.denominator=b/k;
        return this;
    }
    public Fraction divide(Fraction e){
        if (e.numerator == 0) return this;
        int a=this.numerator*e.denominator;
        int b=this.denominator*e.numerator;
        int k=ucln(a,b);
        this.numerator=a/k;
        this.denominator=b/k;
        return this;
    }
    public boolean equals(Object obj){
        boolean result = false;
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;

            // compare this vs other here
            if (this.numerator * other.denominator == this.denominator* other.numerator) result = true;
        
        }
        return result;
    }
    public static void main(String[] args){
        
        
        Fraction d=new Fraction(2,4);
        Fraction e=new Fraction(3,5);
        
        System.out.print(d.add(e).numerator);
        
        
    }
    
}
