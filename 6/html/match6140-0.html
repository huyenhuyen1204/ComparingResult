public class Fraction{
    private int numerator, denominator;
    
    public int getNumerator(){
        return this.numerator;
    }
    public void setNumerator(int numerator){
        this.numerator = numerator;
    }
    public int getDenominator(){
        return this.denominator;
    }
    public void setDenominator(int denominator){
        if(denominator == 0){
            this.denominator = denominator;
        }
    }

    Fraction(int numerator, int denominator){
        if(denominator != 0){
            this.numerator = numerator;
            this.denominator = denominator;
        }
        else{
            this.numerator = numerator;
            this.denominator = 1;
        }
    }
    Fraction(){
        this.denominator = 1;
    }

    public static int gcd(int a, int b) {
        return a == 0 ? b : gcd(b % a, a);
    }
    public static long gcd(long a, long b) {
        return a == 0 ? b : gcd(b % a, a);
    }

    public Fraction reduce(){
        int gcd = gcd(this.numerator, this.denominator);
        int _numerator = this.numerator / gcd;
        int _denominator = this.denominator / gcd;
        return new Fraction(_numerator, _denominator);
    }
    
    public Fraction add(Fraction f){
        Fraction a = this.reduce();
        Fraction b = f.reduce();
        int gcd1 = gcd(a.getDenominator(),b.getDenominator());
        long resDenominator = (long)((long)a.getDenominator()/(long)gcd1)*b.getDenominator();
        long resNumerator = (long)a.getNumerator()*(resDenominator / (long)a.getDenominator()) + (long)b.getNumerator()*(resDenominator / (long)b.getDenominator());
        long gcd2 = gcd(resNumerator, resDenominator);
        this.numerator = (int) (resNumerator / gcd2);
        this.denominator = (int)(resDenominator / gcd2);
        return this;
    //    this.monkaGun();
    }

    public Fraction subtract(Fraction f){
        Fraction b = new Fraction(-f.getNumerator(), f.getDenominator());
        this.add(b);
        return this;
    }

    public Fraction multiply(Fraction f){
        Fraction a = this.reduce();
        Fraction b = f.reduce();
        int gcd1 = gcd(a.getNumerator(), b.getDenominator());
        int gcd2 = gcd(a.getDenominator(), b.getNumerator());
        long resNumerator = (long)(a.getNumerator()/gcd1) * (long)(b.getNumerator()/gcd2);
        long resDenominator = (long)(a.getDenominator()/gcd2)*(long)(b.getDenominator()/gcd1);
        long gcd3 = gcd(resDenominator, resNumerator);
        this.numerator = (int) (resNumerator / gcd3);
        this.denominator = (int) (resDenominator / gcd3);
        return this;
     //   this.monkaGun();
    }
    public Fraction divide(Fraction f){
        if(f.getNumerator() == 0) return this;
        this.multiply(new Fraction(f.getDenominator(), f.getNumerator()));
        return this;
    }
    public boolean equals(Object obj){
        if(obj instanceof Fraction){
            Fraction a = ((Fraction) obj).reduce();
            Fraction b = this.reduce();
            
            if((a.getNumerator()/a.getDenominator() == b.numerator/b.denominator)){
                return true;
            }
            return false;
        }
        return false;
    }
}
