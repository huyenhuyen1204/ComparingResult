public class Fraction {
    private int numerator;
    private int denominator;

    public Fraction() {
        this.denominator = 1;
    }

    public Fraction(int numerator, int denominator) {
        if (denominator == 0) denominator = 1;
        this.numerator = numerator;
        this.denominator = denominator;
    }

    public Fraction reduce() {
        int a = Math.abs(this.numerator);
        int b = Math.abs(this.denominator);
        int temp;
        while (b != 0) {
            temp = a % b;
            a = b;
            b = temp;
        }
        temp = a;
        Fraction result = new Fraction(this.numerator / temp, this.denominator / temp);
        return result;
    }

    public Fraction add(Fraction temp){
        int a = this.numerator*temp.denominator + this.denominator*temp.numerator;
        int b = this.denominator*temp.denominator;
        Fraction result = new Fraction(a, b);
        result = result.reduce();
        return result;
    }

    public Fraction subtract(Fraction temp){
        int a = this.numerator*temp.denominator - this.denominator*temp.numerator;
        int b = this.denominator*temp.denominator;
        Fraction result = new Fraction(a, b);
        result = result.reduce();
        return result;
    }

    public Fraction multiply(Fraction temp){
        int a = this.numerator*temp.numerator;
        int b = this.denominator*temp.denominator;
        Fraction result = new Fraction(a, b);
        result = result.reduce();
        return result;
    }

    public Fraction divide(Fraction temp){
        int a = this.numerator*temp.denominator;
        int b = this.denominator*temp.numerator;
        Fraction result = new Fraction(a, b);
        result = result.reduce();
        return result;
    }
    public boolean equals(Object obj){
        Fraction temp = new Fraction(this.numerator, this.denominator);
        temp = temp.reduce();
        if(obj instanceof Fraction){
            Fraction other = (Fraction) obj;
            if(other.numerator*temp.denominator==other.denominator*temp.numerator){
                return true;
            }
        }
            return false;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        this.denominator = denominator;
    }
}
