public class Fraction {
    private int numerator;
    private int denominator = 1;

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator != 0 ) {
            this.denominator = denominator;
        }
    }
    public Fraction(int c, int b){
        if ( b != 0 ){
            this.numerator = c;
            this.denominator = b;
        }
    }
    public Fraction reduce(){
        int a1,b1,c1;
        a1 = this.numerator;
        b1 = this.denominator;
        while(b1 != 0 ){
            c1 = a1;
            a1 = b1;
            b1 = c1%b1;
        }
        this.numerator = this.numerator/a1;
        this.denominator = this.denominator/a1;
        return this;
    }
    public Fraction add(Fraction a){
        int tu,mau;
        tu = a.getNumerator() * this.getDenominator() + this.getNumerator() * a.getDenominator();
        mau = a.getDenominator() * this.getDenominator();
        this.setNumerator(tu);
        this.setDenominator(mau);
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction a){
        int tu,mau;
        tu = this.getNumerator() * a.getDenominator() - a.getNumerator() * this.getDenominator();
        mau = a.getDenominator() * this.getDenominator();
        this.setNumerator(tu);
        this.setDenominator(mau);
        this.reduce();
        return this;
    }
    public Fraction multiply(Fraction a){
        int tu,mau;
        tu = a.getNumerator() * this.getNumerator();
        mau = a.getDenominator() * this.getDenominator();
        this.setNumerator(tu);
        this.setDenominator(mau);
        return this;
    }
    public Fraction divide(Fraction a){
        if (a.getNumerator() == 0){
            return this;
        }
        int tu,mau;
        tu = this.getNumerator() * a.getDenominator();
        mau = this.getDenominator() * a.getNumerator();
        this.setNumerator(tu);
        this.setDenominator(mau);
        return this;
    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(other.getNumerator() == this.getNumerator() && other.getDenominator() == this.getDenominator()){
                return true;
            }
        }
        return false;
    }
}
