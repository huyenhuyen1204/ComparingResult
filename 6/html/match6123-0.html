public class Fraction {

    private int numerator;
    private int denominator;

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator!=0) this.denominator = denominator;
    }
    
    Fraction(int numerator,int denominator ){
        this.numerator=numerator;
        if(denominator==0){
            this.denominator=1;
        }
        else {
            this.denominator=denominator;
        }
    }
    Fraction(){
        this.denominator=1;
    }
    public int ucln(int a,int b){
        if(b==0){return a;}
        else return ucln(b,a%b);
    
    }
    public Fraction reduce(){
        int a;
        a=ucln(this.numerator,this.denominator);
        this.numerator=this.numerator/a;
        this.denominator=this.denominator/a;
        return this;
    }
    public Fraction add(Fraction b){
        this.numerator=this.numerator*b.denominator+this.denominator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        return this.reduce();
    }
    public Fraction subtract(Fraction b){
        this.numerator=this.numerator*b.denominator-this.denominator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        return this.reduce();
    }
    public Fraction multiply(Fraction b){
        this.numerator=this.numerator*b.numerator;
        this.denominator=this.denominator*b.denominator;
        return this.reduce();
    }
    public Fraction divide(Fraction b){
        if(b.denominator!=0&&b.numerator!=0){
            this.numerator=this.numerator*b.denominator;
            this.denominator=this.denominator*b.numerator;
            
        }
        return this.reduce();
        
    }
    
    public boolean equals(Object obj){
        boolean c=false;
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(this.numerator==this.numerator&&this.denominator==other.denominator){c=true;}
        }
        return c;
    }

}
