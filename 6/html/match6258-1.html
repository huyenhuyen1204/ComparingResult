public class Fraction {
    int numerator, denominator;
    public int getNumerator(){
        return numerator;
    }
    public void setNumerator(int numerator){

        this.numerator = numerator;
    }

    public int getDenominator(){
        return denominator;
    }
    public void setDenominator(int denominator){
        if (denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }

    
Fraction(int numerator, int denominator){
        this.setNumerator(numerator);
        if(denominator !=0)
            this.setDenominator(denominator);
        else this.denominator =1;
    }

    static int UCLN(int a, int b){
        while(true){
            if (a==0){
                return b;
            }
            else{
                int temp = a;
                a = b % a;
                b =  temp;
            }
        }
    }

    public Fraction reduce(){
        int a = this.numerator;
        int b = this.denominator;
        int 
c = UCLN(this.numerator,this.denominator);
        Fraction f = new Fraction(a/c, b/c);
        return f;
    }

    Fraction add(Fraction f){
        int a = f.numerator*this.denominator + f.denominator*this.numerator;
        int b = f.denominator*this.denominator;
        f = new Fraction(a,b);
        f.reduce();
        return f;
    }

    Fraction subtract(Fraction f){
        int a = this.numerator*f.denominator - this.denominator*f.numerator;
        int b = f.denominator*this.denominator;
        f = new Fraction(a,b);
        f.reduce();
        return f;
    }

    Fraction multiply(Fraction f){
        int a = f.numerator*this.numerator;
        int b = f.denominator*this.denominator;
        f = new Fraction(a,b);
        f.reduce();
        return f;
    }

    Fraction divide(Fraction f){
        int a = this.numerator*f.denominator;
        int b = this.denominator*f.numerator;
        f = new Fraction(a,b);
        f.reduce();
        return f;
    
}

    public boolean equals(Object obj){
        if(obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if(other.reduce().numerator==this.reduce().numerator && other.reduce().denominator==this.reduce().denominator){
                return true;
            }
            else {
                return false;
            }
        }
        return false;
    }

    public static void main(String[] args){
        Fraction f = new Fraction(10,8);
        Fraction n = new Fraction(3,2);

        System.out.println(f.reduce().getNumerator());

    }
}
