public class Fraction {

    private int numerator;
    private int denominator;

    static int gcd(int a, int b){
        if (b == 0) return a;
        return gcd(b, a % b);
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) this.denominator = denominator;
        else this.denominator = 1;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if (denominator != 0 ) {
            this.denominator = denominator;
        } else {
            this.denominator = 1;
        }
    }
    Fraction(){
        this.denominator = 1;
    }

    Fraction reduce(){
        int UCLN = gcd(numerator, denominator);
        numerator = numerator/UCLN;
        denominator= denominator/UCLN;
        return this;
    }

    Fraction add(Fraction other){
        
this.numerator = this.numerator * other.denominator + other.numerator * this.denominator;
        this.denominator = this.denominator * other.denominator;
        this.reduce();
        return this;
    }

    Fraction subtract(Fraction other){
        this.numerator = this.numerator * other.denominator - other.numerator * this.denominator;
        this.denominator = this.denominator * other.denominator;
        this.reduce();
        return this;
    }

    Fraction multiply(Fraction other){
        this.numerator = this.numerator * other.numerator;
        this.denominator = this.denominator * other.denominator;
        this.reduce();
        return this;
    }

    Fraction divide(Fraction other){
        if (other.numerator != 0){
            this.numerator = this.numerator * other.denominator;
            this.denominator = this.denominator * other.numerator;
            this.reduce();
            return this;
        } else return this;
    
}
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if (this.numerator * other.denominator - other.numerator * this.denominator == 0) return true;
            else return false;
        } else {
            return false;
        }
    }
}
