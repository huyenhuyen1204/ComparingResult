import static java.lang.Math.abs;

public class Fraction {

        private int numerator;
        private int denominator;

        public int getNumerator() {
            return numerator;
        }

        public int getDenominator() {
            return denominator;
        }

        public void setNumerator(int numerator) {
            this.numerator = numerator;
        }

        public void setDenominator(int denominator){
            if(denominator!=0)
            this.denominator = denominator;
        }
       // public Fraction(){};

        public Fraction(int a,int b) {
            this.denominator = b;
            this.numerator = a;

        }
    public int ucln(int a, int b){
        if(a==b) return a;
       int c,d;
       c=abs(a);
       d=abs(b);
        while(c!=d){
            if (c>d) c=c-d;
            else d=d-c;
        }
        return c;
    }

    public Fraction reduce(){
            if (this.denominator==0) return null;
            if (this.getNumerator()==0){
            Fraction a=new Fraction(0,1);
            return a;
        }
        int ucln=this.ucln(this.getDenominator(),this.getNumerator());
        Fraction x=new Fraction(this.getNumerator()/ucln,this.getDenominator()/ucln);
        return x;

    }
    public Fraction add(Fraction a){
            Fraction x= new Fraction(this.getNumerator()*a.getDenominator()+this.getDenominator()*a.getNumerator(),this.getDenominator()*a.getDenominator());
            x=x.reduce();
            return x;

    }
    public Fraction subtract(Fraction a){
        Fraction x= new Fraction(this.getNumerator()*a.getDenominator()-this.getDenominator()*a.getNumerator(),this.getDenominator()*a.getDenominator());
        x=x.reduce();
        return x;
        }
    public Fraction multiply(Fraction a){
        Fraction x= new Fraction(this.getNumerator()*a.getNumerator(),this.getDenominator()*a.getDenominator());
        x=x.reduce();
        return x;
    }
    public Fraction divide(Fraction a){
        Fraction x= new Fraction(this.getNumerator()*a.getDenominator(),this.getDenominator()*a.getNumerator());
        return x;
    }
    public boolean equals(Object obj){
            if(obj instanceof  Fraction){
            Fraction fr = (Fraction) obj;
            fr=fr.reduce();
            Fraction a=new Fraction(this.getNumerator(),this.getDenominator());
            a=a.reduce();
            if (a.getNumerator()==fr.getNumerator() && a.getDenominator()==fr.getDenominator())
                return true;
            else return false;
            }
            return false;

    }
