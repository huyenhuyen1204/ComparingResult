class Fraction {
    private int numerator, denominator;

    public Fraction(int a, int b) {
        if(b == 0){
            b = 1;
        }
        this.numerator = a;
        this.denominator = b;
    }

    public Fraction() {
        this.numerator = 0;
        this.denominator = 1;
    }

    public int getNumerator() {
        return numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public void setDenominator(int denominator) {
        if(denominator == 0){
            return;
        }
        this.denominator = denominator;
    }

    public Fraction reduce() {
        int a = this.numerator;
        int b = this.denominator;
        int UCLN = 0;
        if (a == 0) {
            UCLN = a;
        } else {
            while (a != b) {
                if (a > b) {
                    a -= b;
                } else if (a < b) {
                    b -= a;
                }
            }
            UCLN = a;
        }
        
if(UCLN == 0)UCLN = 1;
        this.numerator = this.numerator / UCLN;
        this.denominator = this.denominator / UCLN;
        return this;
    }

    public Fraction add(Fraction x) {
        this.numerator = this.numerator * x.denominator + x.numerator * this.denominator;
        this.denominator = this.denominator * x.denominator;
        //reduce();
        return this;
    }

    public Fraction subtract(Fraction x) {
        this.numerator = this.numerator * x.denominator - x.numerator * this.denominator;
        this.denominator = this.denominator * x.denominator;
        //reduce();
        return this;
    }

    public Fraction multiply(Fraction x) {
        this.numerator = this.numerator * x.numerator;
        this.denominator = this.denominator * x.denominator;
        //reduce();
        return this;
    }

    public Fraction divide(Fraction x) {
        if(x.equals(new Fraction(0,1))){
            return this;
        }
        this.numerator = this.numerator * x.denominator;
        this.denominator = this.denominator * x.numerator;
        //reduce();
        
return this;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            reduce();
            other.reduce();
            if (this.numerator == other.numerator && this.denominator == other.denominator) return true;
            else return false;
        }
        return false;
    }


    public static void main(String[] args) {
        Fraction a = new Fraction(0, 2);
        Fraction b = new Fraction(0, 13);
        a.divide(b);
        System.out.print(a.getNumerator() + "/" + a.getDenominator());
    }
}
