public class Fraction {
    private int numerator;
    private int denominator = 1;

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator != 0 ) {
            this.denominator = denominator;
        
}
    }
    public Fraction(int a, int b){
        if ( b != 0 ){
            this.numerator = a;
            this.denominator = b;
        }
    }
    public Fraction reduce(){
        int a,b,c;
        a = this.numerator;
        b = this.denominator;
        while(b != 0 ){
            c = a;
            a = b;
            b = c%b;
        }
        this.numerator = this.numerator/a;
        this.denominator = this.denominator/a;
        return this;
    }
    public Fraction add(Fraction a){
        int c,d;
        c = a.getNumerator() * this.getDenominator() + this.getNumerator() * a.getDenominator();
        d = a.getDenominator() * this.getDenominator();
        this.setNumerator(c);
        this.setDenominator(d);
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction a){
        int c,d;
        c = this.getNumerator() * a.getDenominator() - a.getNumerator() * this.getDenominator();
        d = a.getDenominator() * this.getDenominator();
        this.setNumerator(c);
        this.setDenominator(d);
        this.reduce();
        return this;
    }
    public Fraction multiply(Fraction a){
        int c,d;
        c = a.getNumerator() * this.getNumerator();
        d = a.getDenominator() * this.getDenominator();
        this.setNumerator(c);
        this.setDenominator(d);
        return this;
    }
    public Fraction divide(Fraction a){
        if (a.getNumerator() == 0){
            return this;
        }
        int c,d;
        c = this.getNumerator() * a.getDenominator();
        d = this.getDenominator() * a.getNumerator();
        this.setNumerator(c);
        this.setDenominator(d);
        return this;
    }
    public boolean equals(Object obj){
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(other.getNumerator() == this.getNumerator() && other.getDenominator() == this.getDenominator()){
                return true;
            }
        }
        return false;
    }
}
