public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator() {
        return this.numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return this.denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator != 0)
        {
            this.denominator = denominator;
        }
        else this.denominator = 1;
    }
    Fraction(){
        this.denominator = 1;
    }

    
Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if(denominator != 0)
        {
            this.denominator = denominator;
        }
        else this.denominator = 1;
    }


    public Fraction reduce(){
        int gcd = 0;
        int a = this.numerator;
        int b = this.denominator;

        while(a * b != 0){
            if (a > b){
                a %= b;
            }else{
                b %= a;
            }
        }
        if(a * b == 0) gcd = a + b;
        if(gcd != 0) {
            this.numerator /= gcd;
            this.denominator /= gcd;
        }
        return this;
    }

    public Fraction add(Fraction x){
        int 
tu = this.numerator * x.getDenominator() + x.getNumerator() * this.denominator;
        int mau = this.denominator * x.getDenominator();
        Fraction ans = new Fraction(tu, mau);
        ans = 
ans.reduce();
        return ans;
    }

    public Fraction subtract(Fraction x){
        int tu = this.numerator * x.getDenominator() - x.getNumerator() * this.denominator;
        int mau = this.denominator * x.getDenominator();
        Fraction ans = new Fraction(tu, mau);
        ans = 
ans.reduce();
        return ans;
    }

    public Fraction multiply(Fraction x){
        int tu = this.numerator * x.getNumerator() ;
        int mau = this.denominator * x.getDenominator();
        Fraction ans = new Fraction(tu, mau);
        ans = 
ans.reduce();
        return ans;
    }

    public Fraction divide(Fraction x){
        int tu = this.numerator * x.getDenominator() ;
        int mau = this.denominator * x.getNumerator();
        Fraction ans = new Fraction(tu, mau);
        ans = ans.reduce();;
        return ans;
    }

    public boolean equals(Object obj){
        if(obj instanceof Fraction){
            if(this.getNumerator() * ((Fraction) obj).getDenominator() == this.getDenominator() * ((Fraction) obj).getNumerator()) return true;
        }
        return false;
    }


}
