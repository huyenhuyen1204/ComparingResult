public class Fraction {
    private int numerator;
    private int denominator;

    public int getNumerator(){
        return numerator;
    }
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator(){
        return denominator;
    }
    public void setDenominator(int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
        }
    }

    public Fraction(){
        this.numerator = 1;
        this.denominator = 1;
    }

    public Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if (denominator != 0) {
            this.denominator = denominator;
        }
        else{
            this.denominator = 1;
            this.numerator = 1;
        }
    }

    static int UCLN(int a, int b) {
        int ucln = 1;
        if (a > b) {
            for (int i = 1; i <= b; i++) {
                if (b % i == 0 && a % i == 0) {
                    ucln = i;
                }
            }
        } else {
            for (int i = 1; i <= a; i++) {
                if (a % i == 0 && b % i == 0) {
                    ucln = i;
                }
            }
        }
        return ucln;
    }

    public Fraction reduce(){
        int ucln = UCLN(this.getNumerator(), this.getDenominator());
        Fraction frac = new Fraction(this.getNumerator(), this.getDenominator());
        frac.setNumerator(this.getNumerator()/ ucln);
        frac.setDenominator(this.getDenominator()/ ucln);
        this.setNumerator(frac.getNumerator());
        this.setDenominator(frac.getDenominator());
        return frac;
    }

    public Fraction add(Fraction frac2){
        Fraction frac = new Fraction(1,1);
        frac.setNumerator(this.numerator * frac2.getDenominator() + this.denominator * frac2.getNumerator());
        frac.setDenominator(this.denominator * frac2.getDenominator());
        this.setNumerator(frac.getNumerator());
        this.setDenominator(frac.getDenominator());
        return frac;
    }

    public Fraction subtract(Fraction frac2){
        Fraction frac = new Fraction(1, 1);
        frac.setNumerator(this.numerator * frac2.getDenominator() - this.denominator * frac2.getNumerator());
        frac.setDenominator(this.denominator * frac2.getDenominator());
        this.setNumerator(frac.getNumerator());
        this.setDenominator(frac.getDenominator());
        return frac;
    }

    public Fraction multiply(Fraction frac2){
        Fraction frac = new Fraction(1, 1);
        frac.setNumerator(this.numerator * frac2.getNumerator());
        frac.setDenominator(this.denominator * frac2.getDenominator());
        this.setNumerator(frac.getNumerator());
        this.setDenominator(frac.getDenominator());
        return frac;
    }

    public Fraction divide(Fraction frac2){
        Fraction frac = new Fraction(1, 1);
        if (frac2.getNumerator() != 0) {
            frac.setNumerator(this.numerator * frac2.getDenominator());
            frac.setDenominator(this.denominator * frac2.getNumerator());
            this.setNumerator(frac.getNumerator());
            this.setDenominator(frac.getDenominator());
            return frac;
        }
        return frac;
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            Fraction frac = new Fraction(this.getNumerator(), this.getDenominator());
            if (frac.subtract(other).numerator == 0){
                return true;
            }
            else{
                return false;
            }
        }
        else{
            return false;
        }
    }

    public static void main(String[] args) {
        Fraction frac1 = new Fraction(2,6);
        Fraction frac2 = new Fraction(1,5);
        Fraction frac3 = new Fraction(1, 2);
        frac1.reduce();
        System.out.println(frac1.getNumerator()+ " " + frac1.getDenominator());
        frac1.add(frac2);
        System.out.println(frac1.getNumerator()+ " " + frac1.getDenominator());
        frac1.subtract(frac3);
        System.out.println(frac1.getNumerator()+ " " + frac1.getDenominator());
        frac1.multiply(frac2);
        System.out.println(frac1.getNumerator()+ " " + frac1.getDenominator());
        frac1.divide(frac3);
        System.out.println(frac1.getNumerator()+ " " + frac1.getDenominator());

    }
}
