public class Fraction {
    private int numerator;
    private int denominator;
    public int getNumerator() {return numerator;}
    public void setNumerator(int numerator) {this.numerator=numerator;}
    public int getDenominator() {return denominator;}
    public void setDenominator(int denominator)
    {
        if(denominator != 0)this.denominator=denominator;
        else 
this.denominator = 1;
    }

    public Fraction ( int numerator , int denominator )
    {
        this.numerator = numerator;
        if ( denominator != 0 ) this.denominator = denominator;
        else this.denominator = 1;
    }

    public int gcd(int a,int b) {
        if(a < 0 && b < 0) return gcd(-a, -b);
        if(a < 0 && b > 0) return gcd(-a, b);
        if(a > 0 && b < 0) return gcd(a, -b);
        if (a==0|| b==0) return a+b;
        while (a!=b) {
            if (a>b) {
                a-=b;
            }else if (b>a) {
                b-=a;
            }
        }
        return a;
    }

    public Fraction reduce() {
        int k=gcd(getNumerator(),getDenominator());

        Fraction res = (getNumerator() < 0 && getDenominator() < 0) ?
                        new Fraction(-getNumerator()/k, -getDenominator()/k) :
                        new Fraction(getNumerator()/k, getDenominator()/k);
        return res;
    }

    public Fraction add(Fraction ps) {
        int a=(this.numerator*ps.denominator)+(this.denominator*ps.numerator);
        int b=(this.denominator*ps.denominator);
        Fraction newPs =new Fraction(a,b);
        return newPs.reduce();
    }

    public Fraction subtract(Fraction ps) {
        int a=(this.numerator*ps.denominator)-(this.denominator*ps.numerator);
        int b=(this.denominator*ps.denominator);
        Fraction newPs =new Fraction(a,b);
        return newPs.reduce();
    }

    public Fraction multiply(Fraction ps) {
        int a=(this.numerator*ps.numerator);
        int b=(this.denominator*ps.denominator);
        Fraction newPs =new Fraction(a,b);
        return newPs.reduce();
    }

    public Fraction divide(Fraction ps) {
        int a=(this.numerator*ps.denominator);
        int b=(this.denominator*ps.numerator);
        Fraction newPS =new Fraction(a,b);
        return newPS.reduce();
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            Fraction div = (new Fraction(getNumerator(), getDenominator()).divide(other));
            if(div.getNumerator() == div.getDenominator()) return true;
            return false;
        }
        return false;
    }

}
