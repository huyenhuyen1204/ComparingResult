public class Fraction {
    private  int numerator, denominator ;
    public int getNumerator() {
        return numerator;
    }
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }
    public void setDenominator(int denominator) {
        if (denominator == 0) {this.denominator = 1 ;}
        else {
            this.denominator = denominator ;
        }
    }
    /* public String getFrac() {
        return this.getNumerator() + "/" + this.getDenominator() ;
    }*/
    public Fraction() {
        this.denominator = 1 ;
    }

    public Fraction(int numerator, int denominator) {
        this.numerator = numerator ;
        if (denominator == 0 ) {
            this.denominator = 1 ;
        }
        else {
            this.denominator = denominator ;
        }
    }

    public int gcd(int a, int b){
        while (b > 0){
            int r = a % b;
            a = b;
            b = r;
        }
        return a;
    }

    public Fraction reduce(){
        int k = gcd(Math.abs(this.getNumerator()), Math.abs(this.getDenominator()));
        this.numerator = this.getNumerator() / k;
        this.denominator = this.getDenominator() / k;
        return this;
    }

    public Fraction add(Fraction f1)
    {
        this.numerator = this.getNumerator() * f1.getDenominator() + this.getDenominator() * f1.getNumerator();
        this.denominator = this.getDenominator() * f1.getDenominator();
        return this.reduce();
    }

    public Fraction subtract(Fraction f1)
    {
        this.numerator = this.getNumerator() * f1.getDenominator() - this.getDenominator() * f1.getNumerator();
        this.denominator = this.getDenominator() * f1.getDenominator();
        return this.reduce();
    }

    public Fraction multiply(Fraction f1)
    {
        this.numerator = this.getNumerator() * f1.getNumerator();
        this.denominator = this.getDenominator() * f1.getDenominator();
        return this.reduce();
    }

    public Fraction divide(Fraction f1)
    {
        if (f1.getNumerator() == 0) return this.reduce();
        this.numerator = this.getNumerator() * f1.getDenominator();
        this.denominator = this.getDenominator() * f1.getNumerator();
        return this.reduce() ;
    }

    public boolean equals(Object obj){
        if (!(obj instanceof Fraction))  return false;
        Fraction other = (Fraction) obj;
        if (this.numerator * other.getDenominator() == this.denominator * other.getNumerator()) return true;
        return false;
    }

    public static void main(String[] args){
        Fraction f1 = new Fraction(1, 2);
        Fraction f2 = new Fraction(4, 12);
        System.out.println(f1.getNumerator() + " " + f1.getDenominator());
        System.out.println(f1.add(f2).getNumerator() + " " + f1.add(f2).getDenominator());
        //System.out.println(f2.getNumerator() + " " + f2.getDenominator());
        //System.out.println(f1.getNumerator() + " " + f1.getDenominator());
        System.out.println(f1.getNumerator() + " " + f1.getDenominator());
    }
}
