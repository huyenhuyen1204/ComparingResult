        public static int gcd(int a, int b){
            if(b==0) return a;

        public  Fraction add(Fraction a){
            Fraction convert=new Fraction(this.reduce());
            if(convert.denominator==a.denominator) convert.numerator+=a.numerator;
            else {
                convert.numerator=convert.numerator*a.denominator+convert.denominator*a.numerator;
                convert.denominator=convert.denominator*a.denominator;
            }
            convert.reduce();
            return convert;
        }
        public  Fraction subtract(Fraction a){
            Fraction convert=new Fraction(this.reduce());

                convert.denominator=convert.denominator*a.denominator;

            }
            convert.reduce();
            return convert;
        }
