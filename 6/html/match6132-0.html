public class Fraction {
    /**
     * declare attributes here
     */

    private int numerator;
    private int denominator;

    /**
     * declare constructor here
     */
    public Fraction() {
        //this.numerator = 1;
        denominator = 1;
    }

    public Fraction(Fraction frac) {
        this.numerator = frac.numerator;
        this.denominator = frac.denominator;
    }

    public Fraction(int num, int den) {
        if (den == 0) {
            denominator = 1;
            return;
        }
        numerator = num;
        denominator = den;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setDenominator(int denominator) {
        if (denominator == 0) return;
        this.denominator = denominator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void reduce() {
        int num = Math.abs(this.numerator);
        int den = Math.abs(this.denominator);

        while (num != 0 && den != 0) {
            if (num > den) num %= den;
            else den %= num;
        }

        this.numerator /= (num + den);
        this.denominator /= (num + den);
    }

    // add
    public Fraction add(Fraction other) {
        this.setNumerator(this.getNumerator() * other.getDenominator() + other.getNumerator() * this.getDenominator());
        this.setDenominator(other.getDenominator() * this.getDenominator());

        this.reduce();
        return this;
    }

    // subtract
    public Fraction subtract(Fraction other) {
        this.setNumerator(this.getNumerator() * other.getDenominator() - other.getNumerator() * this.getDenominator());
        this.setDenominator(other.getDenominator() * this.getDenominator());

        this.reduce();
        return this;
    }

    // multiple
    public Fraction multiply(Fraction other) {
        if (this.getDenominator() * other.getDenominator() == 0) return this;
        this.setNumerator(this.getNumerator() * other.getNumerator());
        this.setDenominator(this.getDenominator() * other.getDenominator());
        this.reduce();
        return this;
    }

    // devide
    public Fraction divide(Fraction other) {
        if (this.getDenominator() * other.getNumerator() == 0) return this;
        this.setNumerator(this.getNumerator() * other.getDenominator());
        this.setDenominator(this.getDenominator() * other.getNumerator());
        this.reduce();
        return this;
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            return (((double) this.numerator / (double) this.denominator) == ((double) other.numerator / (double) other.denominator));
        }
        return false;
    }
}
