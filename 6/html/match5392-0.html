public class Fraction {
    private int numerator, denominator = 1;

    private int gcd(int a, int b) {
        if (b == 0) return a;
        return gcd(b, a % b);
    }

    public Fraction(int numerator, int denominator) {
        if (denominator != 0) {
            this.numerator = numerator;
            this.denominator = denominator;
        }
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (denominator != 0) {
            this.denominator = denominator;
        }
    }

    public Fraction reduce() {
        int x = gcd(numerator, denominator);
        this.setNumerator(this.getNumerator() / x);
        this.setDenominator(this.getDenominator() / x);
        return this;
    }

    public Fraction add(Fraction fr) {
        this.setNumerator(this.getNumerator() * fr.getDenominator() + this.getDenominator() * fr.getNumerator());
        this.setDenominator(this.getDenominator() * fr.getDenominator());
        return this.reduce();
    }

    public Fraction subtract(Fraction fr) {
        this.setNumerator(this.getNumerator() * fr.getDenominator() - this.getDenominator() * fr.getNumerator());
        this.setDenominator(this.getDenominator() * fr.getDenominator());
        return this.reduce();
    }

    public Fraction multiply(Fraction fr) {
        this.setNumerator(this.getNumerator() * fr.getNumerator());
        this.setDenominator(this.getDenominator() * fr.getDenominator());
        return this.reduce();
    }

    public Fraction divide(Fraction fr) {
        if (fr.getNumerator() != 0) {
            this.setNumerator(this.getNumerator() * fr.getDenominator());
            this.setDenominator(this.getDenominator() * fr.getNumerator());
            return this.reduce();
        }
        return this.reduce();
    }

    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction fr = (Fraction) obj;
            return this.reduce().getNumerator()==fr.reduce().getNumerator() && this.reduce().getDenominator()==fr.reduce().getDenominator();
        }
        return false;
    }
}
