public class Fraction {
    private int numerator;
    private int denominator = 1;
    public int getNumerator() { return numerator;}
    public int getDenominator(){return denominator;}
    public void setDenominator(int denominator) {
        if (denominator != 0 )
        this.denominator = denominator;
    }
    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public Fraction (int numerator,int denominator) {
        if (denominator != 0)
        this.denominator = denominator;
        this.numerator = numerator;
    }
    public int ucln(int a,int b) {
        int t;
        while ( b != 0 ) {
            t = a%b;
            a=b;
            b=t;
        }
        return a;
    }
    public Fraction reduce () {
        int b = ucln(this.getNumerator(),this.getDenominator());
        this.setDenominator(this.getDenominator()/b);
        this.setNumerator(this.getNumerator()/b);
        return this;
    }
    public Fraction add(Fraction a) {
        this.setNumerator(a.getNumerator()*this.getDenominator() + a.getDenominator()*this.getNumerator());
        this.setDenominator(a.getDenominator()*this.getDenominator());
        this.reduce();
        return this;
    }
    public Fraction subtract(Fraction a) {
        this.setNumerator(this.getNumerator()*a.getDenominator() - this.getDenominator()*a.getNumerator());
        this.setDenominator(this.getDenominator()*a.getDenominator());
        this.reduce();
        return this;
    }
    public Fraction multiply(Fraction a) {
        this.setNumerator(a.getNumerator()*this.getNumerator());
        this.setDenominator(a.getDenominator()*this.getDenominator());
        this.reduce();
        return this;
    }
    public Fraction divide(Fraction a) {
        this.setNumerator(this.getNumerator()*a.getDenominator());
        this.setDenominator(this.getDenominator()*a.getNumerator());
        this.reduce();
        return this;
    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
           if ((float)this.getNumerator()/(float)this.getDenominator() - (float)other.getNumerator()/(float)other.getDenominator() == 0)
                return true;
           else return false;
        }
        else return false;
    }
}
