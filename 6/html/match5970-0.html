public class Fraction {
    private int numerator;
    private int denominator;

    public Fraction(int numerator, int denominator) {
        setNumerator(numerator);
        setDenominator(1);
        setDenominator(denominator);
    }

    public Fraction(int numerator) {
        this(numerator, 1);
    }

    public Fraction() {
        this(0, 1);
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
        if (numerator == 0) setDenominator(1);
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if (getNumerator() != 0 && denominator != 0) this.denominator = denominator;
    }

    private int gcd(int a, int b) {
        while (b != 0) {
            int r = a % b;
            a = b;
            b = r;
        }
        return Math.abs(a);
    }

    public Fraction reduce() {
        int d = gcd(getNumerator(), getDenominator());
        if (d != 0) {
            setNumerator(getNumerator() / d);
            setDenominator(getDenominator() / d);
        }
        if (getDenominator() < 0) {
            setNumerator(-getNumerator());
            setDenominator(-getDenominator());
        }
        return this;
    }

    public Fraction add(Fraction other) {
        if (other == null) return this;
        int newNumerator = this.getNumerator() * other.getDenominator() + other.getNumerator() * this.getDenominator();
        int newDenominator = this.getDenominator() * other.getDenominator();
        setDenominator(newDenominator);
        setNumerator(newNumerator);
        return reduce();
    }

    public Fraction subtract(Fraction other) {
        if (other == null) return this;
        int newNumerator = this.getNumerator() * other.getDenominator() - other.getNumerator() * this.getDenominator();
        int newDenominator = this.getDenominator() * other.getDenominator();
        setDenominator(newDenominator);
        setNumerator(newNumerator);
        return reduce();
    }

    public Fraction multiply(Fraction other) {
        if (other == null) return this;
        int newNumerator = this.getNumerator() * other.getNumerator();
        int newDenominator = this.getDenominator() * other.getDenominator();
        setDenominator(newDenominator);
        setNumerator(newNumerator);
        return reduce();
    }

    public Fraction divide(Fraction other) {
        if (other == null) return this;
        if (other.getNumerator() == 0) return this;
        int newNumerator = this.getNumerator() * other.getDenominator();
        int newDenominator = this.getDenominator() * other.getNumerator();
        setNumerator(newNumerator);
        setDenominator(newDenominator);
        return reduce();
    }

    public boolean equals(Object obj) {
        if (obj == null) return false;
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            long a = (long) this.numerator * other.denominator;
            long b = (long) this.denominator * other.numerator;
            return a == b;
        }
        else return false;
    }

    public String toString() {
        return String.format("%d/%d", getNumerator(), getDenominator());
    }

    public static void main(String[] args) {
        Fraction a = new Fraction(-2, -4);
        Fraction b = new Fraction(+2, +3);
        System.out.print(a + " + " + b + " = ");
        System.out.println(a.add(b));
        System.out.print(a + " - " + b + " = ");
        System.out.println(a.subtract(b));
        System.out.print(a + " * " + b + " = ");
        System.out.println(a.multiply(b));
        System.out.print(a + " / " + b + " = ");
        System.out.println(a.divide(b));

        Fraction reducedA = new Fraction(-2, 4);
        System.out.println(a + " is " + (a.equals(reducedA) ? "" : "not ") + "equal to " + reducedA);
        System.out.println(a + " is " + (a.equals(b) ? "" : "not ") + "equal to " + b);
    }
}
