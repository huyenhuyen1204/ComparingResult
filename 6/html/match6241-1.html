public class Fraction {
    private int numerator;
    private int denominator;

    public Fraction(int numerator,int denominator){
        super();
        if(denominator!=0){
            this.numerator=numerator;
            this.denominator=denominator;
        }
        else if(denominator==0){
            this.denominator = 1;
        }
    }

    public int getNumerator(){
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator(){
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator!=0) this.denominator = denominator;
    }

    public Fraction(){
        this.denominator =1;
    }
    public int UCLN(int a, int b) {
        while (a*b != 0) {
            if (a > b) {
                a = a%b;
            } else {
                b = b%a;
            }
        }
        return a+b;
    }

    public Fraction reduce(){
        int n = UCLN(this.getNumerator(),this.getDenominator());
        this.numerator /= n;
        this.denominator /= n;
        return this;
    }

    public  Fraction add(Fraction ps){
        int 
ts = this.getNumerator()*ps.getDenominator()+this.getDenominator()*ps.getNumerator();
        int ms = this.getDenominator()*ps.getDenominator();
        Fraction newadd = new Fraction(ts,ms);
        newadd.reduce();
        return  newadd;
    }

    public  Fraction subtract(Fraction ps){
        int ts = this.getNumerator()*ps.getDenominator() - this.getDenominator()*ps.getNumerator();
        int ms = this.getDenominator()*ps.getDenominator();
        Fraction newsubtract = new Fraction(ts,ms);
        newsubtract.reduce();
        return newsubtract;
    }

    public  Fraction multiply(Fraction ps){
        int ts = this.getNumerator()*ps.getNumerator();
        int ms = this.getDenominator()*ps.getDenominator();
        Fraction newmultiply = new Fraction(ts,ms);
        newmultiply.reduce();
        return newmultiply;
    }

    public  Fraction divide(Fraction ps){
        if(ps.getNumerator()!=0) {
            int 
ts = this.getNumerator() * ps.getDenominator();
            int ms = this.getDenominator() * ps.getNumerator();
            Fraction newdivide = new Fraction(ts, ms);
            newdivide.reduce();
            return newdivide;
        }
        else return this;
    }

    public boolean equals(Object obj){
        Fraction other;
        if (obj instanceof Fraction) {

            other = (Fraction) obj;
        }
        else return false;
        this.reduce();
        other = other.reduce();
        if (this.getNumerator()==other.getNumerator()&&this.getDenominator()==other.getDenominator())
        {return true;}
        else return false;
    }

    }
}
