public class Fraction {
    /**
     * declare attributes here
     */
    private int  numerator;
    private int  denominator ;

    public int getNumerator()
    {
        return this.numerator;

    }
    public  void setNumerator(int newNumerator)
    {
        this.numerator = newNumerator;

    }
    public  int getDenominator(){
        return this.denominator;

    }
    public  void setDenominator(int newDenominator)
    {
        if(newDenominator == 0) {}
        else
            this.denominator = newDenominator;

    }

    /**
     * declare constructor here
     */
    Fraction(int tu, int mau){
        this.numerator = tu;
        if(mau == 0) mau = 1;
        this.denominator = mau;

    }
    Fraction(){
        this.denominator = 1;
    }

    /**
     * methods
     */
    public int gcd(int a, int b){
        while(a!=b)
        {
            if(a > b) a= a-b;
            if(a < b) b= b-a;
        }
        return a;
    }
    public Fraction reduce(){
        int tu = this.numerator;
        int mau = this.denominator;
        int ucln= gcd(tu, mau);
        
this.numerator = tu/ucln;
        this.denominator = mau/ucln;
        return this;
    }
    // add
    public Fraction add(Fraction other) {
        int tu = this.numerator * other.denominator + this.denominator* other.numerator;
        int mau = this.denominator * other.denominator;
        this.numerator = tu;
        this.denominator = mau;
        return this;
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int tu = this.numerator * other.denominator - this.denominator* other.numerator;
        int mau = this.denominator * other.denominator;
        this.numerator = tu;
        this.denominator = mau;
        return this;
    }

    // multiple
    public Fraction multiple(Fraction other) {
        int tu = this.numerator * other.numerator;
        int mau = this.denominator * other.denominator;
        this.numerator = tu;
        this.denominator = mau;
        return this;
    }

    // divide
    public Fraction divide(Fraction other) {
        if(other.numerator == 0) return  this;
        else {
            int tu = this.numerator * other.denominator;
            int mau = this.denominator * other.numerator;
            this.numerator = tu;
            this.denominator = mau;
            return this;
        }
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;
            
if(this.numerator * other.denominator - this.denominator* other.numerator == 0)
                return true;
            else return false;
            // compare this vs other here

        }
        else return false;
    }
    public void in(){
        System.out.println(this.numerator + "/" + this.denominator);
    }
    public static void main(String args[]){
        Fraction a = new Fraction(3,3);
        Fraction b = new Fraction(9,12);
        a.add(b);
        a.reduce();
        a.in();
    }
}
