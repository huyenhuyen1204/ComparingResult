public class Fraction {

    private int numerator;
    private int denominator;

    public int getNumerator(){
        return numerator;
    }

    public void setNumerator(int numerator){
        this.numerator = numerator;
    }

    public int getDenominator(){
        return denominator;
    }

    public void setDenominator(int denominator){
        if (denominator != 0)
            this.denominator = denominator;
        else
            this.denominator = 1;
    }

    Fraction(){
        this.denominator = 1;
    }

    Fraction(int numerator, int denominator){
        this.setNumerator(numerator);
        if(denominator !=0)
            this.setDenominator(denominator);
        else
            this.denominator = 1;
    }

    public static int gcd(int a, int b){
        while (a != b){
            if (a > b)
                a -= b;
            else
                b -= a;
        }
        return a;
    
}

    public Fraction reduce(){
        int a = Math.abs(this.numerator);
        int b = Math.abs(this.denominator);
        int c = gcd(a,b);
        Fraction f = new Fraction(this.numerator/c, this.denominator/c);
        return f;
    }

    public Fraction add(Fraction other){
        int a = other.numerator*this.denominator + other.denominator*this.numerator;
        int b = other.denominator*this.denominator;
        other = new Fraction(a,b);
        other.reduce();
        return other;
    }

    Fraction subtract(Fraction other){
        int a = this.numerator*other.denominator - this.denominator*other.numerator;
        int b = other.denominator*this.denominator;
        other = new Fraction(a,b);
        other.reduce();
        return other;
    }

    Fraction multiply(Fraction other){
        int a = other.numerator*this.numerator;
        int b = other.denominator*this.denominator;
        other = new Fraction(a,b);
        other.reduce();
        return other;
    }

    Fraction divide(Fraction other){
        if(other.denominator != 0){
            int a = this.numerator * other.denominator;
            int b = this.denominator * other.numerator;
            other = new Fraction(a, b);
            other.reduce();
            return other;
        }
        else
            return null;
    }

    public boolean equals(Object obj){
        if(obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if(other.reduce().numerator*this.reduce().denominator == other.reduce().denominator*this.reduce().numerator){
                return true;
            }
            else {
                return false;
            }
        }
        return false;
    }

    }
}
