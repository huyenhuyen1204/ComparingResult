public class Fraction {
    private int numerator, denominator;

    public void print(){
        System.out.println(numerator + " " + denominator);
    }

    public Fraction(int numerator, int denominator){
        this.numerator = numerator;
        if (denominator != 0)
            this.denominator = denominator;
        else
            this.denominator = 1;
    }

    public void refine(){
        if (denominator < 0){
            denominator = -denominator;
            numerator = -numerator;
        }
    }

    public void setNumerator(int numerator){
        this.numerator = numerator;
    }

    public int getNumerator(){
        return this.numerator;
    }

    public void setDenominator(int denominator){
        if (denominator != 0)
            this.denominator = denominator;
        else
            this.denominator = 1;
    }

    public int getDenominator(){
        return this.denominator;
    }

    public Fraction reduce(){
        int a = getNumerator();
        int b = getDenominator();

        while (b != 0){
            int r = a % b;
            a = b;
            b = r;
        }

        setNumerator(getNumerator() / a);
        setDenominator(getDenominator() / a);

        return this;
    }

    public Fraction add(Fraction other) {
        int num = getNumerator() * other.getDenominator() + getDenominator() * other.getNumerator();
        int den = getDenominator() * other.getDenominator();

        setNumerator(num);
        setDenominator(den);

        return this;
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int num = getNumerator() * other.getDenominator() - getDenominator() * other.getNumerator();
        int den = getDenominator() * other.getDenominator();

        setNumerator(num);
        setDenominator(den);

        return this;
    }

    // multiple
    public Fraction multiply(Fraction other) {
        int num = getNumerator() * other.getNumerator();
        int den = getDenominator() * other.getDenominator();

        setNumerator(num);
        setDenominator(den);

        return this;
    }

    // divide
    public Fraction divide(Fraction other) {
        int num = getNumerator() * other.getDenominator();
        int den = getDenominator() * other.getNumerator();

        setNumerator(num);
        setDenominator(den);

        return this;
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        if (obj instanceof Fraction){
            Fraction This = new Fraction(getNumerator(), getDenominator());
            Fraction That = (Fraction) obj;

            if (This.getDenominator() == 0 || That.getDenominator() == 0) return false;

            This.reduce();
            That.reduce();

            This.refine();
            That.refine();

            return (This.getNumerator() == That.getNumerator() && This.getDenominator() == That.getDenominator());
        }
        else
            return false;
    }
}
