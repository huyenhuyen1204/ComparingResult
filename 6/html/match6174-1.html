public class Fraction
{
	private int numerator,denominator;
	
	public void setNumerator(int numerator)
	{
		this.numerator=numerator;
	}
	
	public void setDenominator(int denominator)
	{
		if (denominator!=0) this.denominator=denominator;
		else this.denominator= 1;
	}
	
	public int getNumerator()
	{
		return numerator;
	}
	
	public int getDenominator()

		return denominator;
	}

		this.denominator = 1;

	public static int gcd(int a,int b)
    {
        int c=1;
        while (c!=0)
        {
            c=a%b;
            a=b;
            b=c;
        }
        return a;

    }
	
	public Fraction reduce()
	{
		int a=Math.abs(this.numerator);

		int c=gcd(a,b);
		Fraction f= new Fraction(this.numerator/c,this.denominator/c);
		return f;
	}

f=new Fraction(a,b);
		f.reduce();
		return f;
	}

		return f;
	}

	{
		int a=f.getNumerator()*this.numerator;
		int b=f.getDenominator()*this.denominator;
		
f=new Fraction(a,b);
		f.reduce();
		return f;
	}

	{
		int a=f.getDenominator()*this.numerator;
		int b=f.getNumerator()*this.denominator;
		
f=new Fraction(a,b);
		f.reduce();
		return f;
