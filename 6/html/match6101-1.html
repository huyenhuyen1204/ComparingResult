public class Fraction {
    private int numerator, denominator;

    //setter, getter cho tu so (numerator)
    public void setNumerator (int n) {
        this.numerator = n;
    }
    public int getNumerator() {
        return numerator;
    }

    //setter, getter cho mau so (denominator)
    public void setDenominator(int n) {
        this.denominator = (n==0?1:n);
    
}
    public int getDenominator() {
        return denominator;
    }

    //khoi tao
    public Fraction() {
        this.denominator = 1;
    }
    public Fraction(int n) {
        this.numerator = n;
        this.denominator = 1;
    }
    public Fraction(int n, int m) {
        this.numerator = n;
        this.denominator = (m==0?1:m);
    }

    //rut gon
    public Fraction reduce() {
        int uoc = gcd(this.numerator, this.denominator);
        this.numerator /= uoc;
        this.denominator /= uoc;
        return this;
    }

    //cac phep tinh phan so
    public Fraction add(Fraction other) {
        
this.numerator = this.numerator*other.denominator + other.numerator*this.denominator;
        this.denominator *= other.denominator;
        return this.reduce();
    }
    public Fraction subtract(Fraction other) {
        this.numerator = this.numerator*other.denominator - other.numerator*this.denominator;
        this.denominator *= other.denominator;
        return this.reduce();
    }
    public Fraction multiply(Fraction other) {
        this.numerator *= other.numerator;
        this.denominator *= other.denominator;
        return this.reduce();
    }
    public Fraction divide(Fraction other) {
        if (other.numerator==0||other.denominator==0) return this;
        this.numerator *= other.denominator;
        this.denominator *= other.numerator;
        return this.reduce();
    }

    //so sanh
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction a = (Fraction) obj;
            a.reduce();
            if (this.numerator==a.numerator&&this.denominator==a.denominator) return true;
            else return false;
        } else return false;
    }

    //tim uoc lon nhat
    public static int gcd(int a, int b) {
        if (a==0|b==0) return 1;
        int result = 1;
        for (int i=1; i<=Math.abs(a); i++) {
            if (a%i==0&&b%i==0) result = i;
        }
        return result;
    }
}
