public class Fraction
{
	private int gcd(int a, int b) {
        while (b > 0) {
            int tmp = b;
            b = a % b;
            a = tmp;
        }
        return a;
    }
    private int numerator;
    private int denominator;
    public void setNumerator(int a){
        this.numerator = a;
    }
    public void setDenominator(int a){
        if(a==0){a=1;}
        this.denominator = a;
    }
    public int getNumerator(){
        return this.numerator;
    }
    public int getDenominator(){
        return this.denominator;
    }
    public Fraction (){
        this.numerator = 1;
        this.denominator = 1;
    }
    public Fraction (int a, int b){
        if(b==0){b=1;};
        this.numerator = a;
        this.denominator = b;
    }
    public Fraction (Fraction a){
        this.numerator = a.getNumerator();
        this.denominator = a.getDenominator();
    }
    public Fraction reduce(){
        int ucln = gcd(this.numerator, this.denominator);
        this.numerator = this.numerator/ucln;
        //System.out.println(this.numerator/(gcd(this.numerator, this.denominator)));
        this.denominator = this.denominator/ucln;
        //.out.println(this.denominator/(gcd(this.numerator, this.denominator)));
        return this;
    }
    public int ucln(int a, int b){
        return gcd(a, b);
    }
    public String toStrings(){
        return this.numerator+"/"+this.denominator;
    }
    // add
    public Fraction add(Fraction other) {

        int numerator = this.numerator * other.denominator + this.denominator * other.numerator;
        int denominator = this.denominator * other.denominator;
        this.numerator = numerator;
        this.denominator = denominator;
        return this;
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int numerator = this.numerator * other.denominator - this.denominator * other.numerator;
        int denominator = this.denominator * other.denominator;
        this.numerator = numerator;
        this.denominator = denominator;
        return this;
    }

    // multiple
    public Fraction multiply(Fraction other) {
        int numerator = this.numerator * other.numerator;
        int denominator = this.denominator * other.denominator;
        this.numerator = numerator;
        this.denominator = denominator;
        return this;
    }

    // divide
    public Fraction divide(Fraction other) {
        if (other.equals(new Fraction(0, 1))) {
            return this;
        }
        int numerator = this.numerator * other.denominator;
        int denominator = this.denominator * other.numerator;
        this.denominator = denominator;
        this.numerator = numerator;
        return this;
    }
    public boolean equals(Object obj) {
        if (super.equals(obj)) {
            return true;
        }
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            other.reduce();
            return this.denominator * other.numerator == other.denominator * this.numerator;
        } else {
            return false;
        }
    }
}
