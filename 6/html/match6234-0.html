public class Fraction {
    private int numerator;
    private int denominator;

    //Constructor
    public Fraction(int num, int de){
        this.denominator = de;
        this.numerator = num;
        if(de == 0){
            this.denominator = 1;
        }
    }


    //Getters and setters
    public void setNumerator(int n){
        this.numerator = n;
    }

    public int getNumerator(){
        return numerator;
    }

    public void setDenominator(int n){
        if(n != 0){
            this.denominator = n;
        }
    }

    public int getDenominator() {
        return denominator;
    }

    public Fraction add(Fraction other) {
        int temp = this.denominator * other.getDenominator() / gcd(this.denominator, other.getDenominator());
        this.numerator = this.numerator*(temp/this.denominator) + other.getNumerator()*(temp/other.getDenominator());
        this.denominator = temp;

        return this;
    }

    public Fraction subtract(Fraction other) {
        int temp = (this.denominator * other.getDenominator()) / gcd(this.denominator, other.getDenominator());
        this.numerator = this.numerator*(temp/this.denominator) - other.getNumerator()*(temp/other.getDenominator());
        this.denominator = temp;

        return this;
    }

    public Fraction multiply(Fraction other) {
        this.numerator = this.numerator * other.getNumerator();
        this.denominator = this.denominator * other.getDenominator();

        return this;
    }

    public Fraction divide(Fraction other) {
        if(other.getNumerator() != 0){
            this.numerator = this.numerator * other.getDenominator();
            this.denominator = this.denominator * other.getNumerator();
        }
        return this;
    }

    public Fraction reduce(){
        int temp = gcd(this.numerator, this.denominator);
        this.numerator /= temp;
        this.denominator /= temp;

        if(this.denominator < 0){
            this.numerator *= -1;
            this.denominator *= -1;
        }
        return this;
    }


    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {

            Fraction other = (Fraction) obj;

            int x = this.denominator * other.numerator;
            int y = this.numerator * other.denominator;

            return (x == y);
        }
        return false;
    }

    public static int gcd(int a, int b){
        if(b == 0) return a;
        return gcd(b, a % b);
    }

}
