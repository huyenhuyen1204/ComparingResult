public class Fraction{

    private int numerator;
    private int denominator;

    public int getNumerator() {
        return this.numerator ;
    }
    public void setNumerator(int _numerator) {
        this.numerator = _numerator;
    }
    public int getDenominator() {
        return this.denominator;
    }
    public void setDenominator(int _denominator){
        if( _denominator != 0) {
            this.denominator = _denominator;
        
}
    }
    
    Fraction() {
        this.denominator = 1;
    }
    Fraction(int num, int deno){
        if(deno != 0){
            this.numerator = num;
            this.denominator = deno;
        } else {
            this.numerator = num;
            this.denominator = 1;
        }
        
    }

    public static int gcd(int x, int y){
        while(y!=0){
            int tmp = y;
            y = x%y;
            x = tmp;
        }
        return x;
    }

    public Fraction reduce(){
        int gcd1 = gcd(this.numerator, this.denominator);
        this.numerator = this.numerator/gcd1;
        this.denominator = this.denominator/gcd1;
        return this;
    }

    public Fraction add(Fraction other){
        int 
add_denominator = this.denominator*other.getDenominator();
        int add_numerator = this.numerator*other.getDenominator() + this.denominator*other.getNumerator();    
        this.numerator = add_numerator;
        this.denominator = add_denominator;
        return this;    
    }

    public Fraction subtract(Fraction other) {
        int sub_denominator = this.denominator*other.getDenominator();
        int sub_numerator = this.numerator*other.getDenominator() - this.denominator*other.getNumerator();
        this.numerator = sub_numerator;
        this.denominator = sub_denominator;
        return this;
    }

    public Fraction multiply(Fraction other) {
        int mul_numerator = this.numerator*other.getNumerator();
        int mul_denominator = this.denominator*other.getDenominator();
        this.numerator = mul_numerator;
        this.denominator = mul_denominator;
        return this;
    }

    public Fraction divide(Fraction other) {
        if(other.getNumerator() == 0){
            return this;
        } else {
            int div_numerator = this.numerator*other.getDenominator();
            int div_denominator = this.denominator*other.getNumerator();
            this.numerator = div_numerator;
            this.denominator = div_denominator;
            return this;
        }
    }

    public boolean equals(Object obj) {
        boolean check = true;
        if( obj instanceof Fraction == false ) {
            check = false;
        } 
        else if( obj instanceof Fraction == true ) {
            Fraction other = (Fraction) obj;
            other.reduce();
            this.reduce();
            if(this.numerator/other.getNumerator() == this.denominator/other.getDenominator()){
                check = true;
            }
            else check = false;
        }
        return check;
    }
    
}
