import java.util.Scanner;

public class Fraction {
    /**
     * declare attributes here
     */ 
    
    private int numerator, denominator;
    /**
     * declare constructor here
     */
    public Fraction()
    {
        denominator=1;
    }
    
    public Fraction(int numerator,int denominator)
    {
        if(denominator==0) return;
        this.numerator=numerator;
        this.denominator=denominator;
    }
    
    public void setNumerator(int numerator)
    {
        this.numerator=numerator;
    }
    public int getNumerator()
    {
        return numerator;
    }
    public void setDenominator(int denominator)
    {
        if(denominator==0) 
        {
            denominator=1;
        }
        else
            
this.denominator=denominator;
    }
    public int getDenominator()
    {
        return denominator;
    }
    
    
    /**
     * methods
     */
    // add
    private int gcd(int a,int b)
    {
        if(a<b) return gcd(b,a);
        
        if(b<0) return gcd(a,-b);
        
        if(b==0) return a;
        
        return gcd(b,a%b);
    }
    private int lcm(int a,int b)
    {
        return (a*b)/gcd(a,b);
    }
    public Fraction reduce()
    {
        
        int tmp=gcd(numerator,denominator);
        
        //System.out.println(tmp);
        
        numerator/=tmp;
     
        denominator/=tmp;
             
        return this;
    }
    public Fraction add(Fraction other) {
         
         int 
lcmdenominator=lcm(this.denominator,other.getDenominator());

         int numeratorTmp=(lcmdenominator/this.denominator)*this.numerator+(lcmdenominator/other.getDenominator())*other.getNumerator();
         
         this.numerator=numeratorTmp;
         this.denominator=lcmdenominator;

         
         return this.reduce();
         
    }

    // subtract
    public Fraction subtract(Fraction other) {
        int lcmdenominator=lcm(this.denominator,other.getDenominator());

         int numeratorTmp=(lcmdenominator/this.denominator)*this.numerator-(lcmdenominator/other.getDenominator())*other.getNumerator();
      
         this.numerator=numeratorTmp;
         this.denominator=lcmdenominator;

         
         return this.reduce();
    }

    // multiple
    public Fraction multiply(Fraction other) {
        
        this.numerator=other.numerator*numerator;
        this.denominator=other.denominator*denominator;
        
        return this.reduce();
    }   

    // divide
    public Fraction divide(Fraction other) {
     
        if(other.getNumerator()==0) return this.reduce();
        
        this.numerator=other.getDenominator()*numerator;
        
        this.denominator=other.getNumerator()*denominator;
        
        
        return this.reduce();
    }

    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        if(obj instanceof Fraction )
        {
            Fraction other= (Fraction) obj;
            other.reduce();
            this.reduce();
            if(this.numerator==other.getNumerator()&&this.denominator==other.getDenominator())
                return true;
        }
        
        return false;
    }
