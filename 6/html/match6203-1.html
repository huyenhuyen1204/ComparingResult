class Fraction{
    private int numerator,denominator;

    public int getDenominator() {
        return denominator;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setDenominator(int denominator) {
        if(denominator !=0) this.denominator = denominator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }
    public Fraction(){
        denominator =1;
    }

    public Fraction(int numerator,int denominator){
       if(denominator != 0){
           this.denominator = denominator;
           this.numerator = numerator;
       }
       else this.denominator=1;
    }

    public  Fraction reduce(){
        int n = Gcd(this.numerator,this.denominator);
        this.numerator/=n;
        this.denominator/=n;
        if(this.denominator < 0){
            this.numerator=-this.numerator;
            this.denominator = -this.denominator;
        }
        return this;
    }

    public int Gcd(int a,int b){
        int gcd=1;
        if(a==b) return a;
        if( a < b){
            for(int i=Math.abs(a); i >=1; i--){
                if (b% i==0 && a % i==0) return i;
            }
        }
        else if (b < a){
            for(int i=Math.abs(b); i >=1; i--){
                if (b% i==0&& a % i==0) return i;
            }
        }
        return gcd;
    
}

    public Fraction add(Fraction f){
        int n = this.numerator* f.denominator + this.denominator*f.numerator;
        int d = this.denominator* f.denominator;
        this.denominator = d;
        this.numerator = n;
        return this;
    }

    public Fraction subtract(Fraction f){
        int n = this.numerator* f.denominator - this.denominator*f.numerator;
        int d = this.denominator* f.denominator;
        this.denominator = d;
        this.numerator = n;
        return this;
    }

    public Fraction multiply(Fraction f){
        int n = this.numerator* f.numerator;
        int d = this.denominator* f.denominator;
        this.denominator = d;
        this.numerator = n;
        return this;
    }

    public Fraction divide(Fraction f){
        if(f.numerator != 0){
            int n = this.numerator* f.denominator ;
            int d = this.denominator* f.numerator;
            this.denominator = d;
            this.numerator = n;
        }
        return this;
    }
    public boolean equals(Object object) {
        if (object instanceof Fraction) {
            Fraction other = (Fraction) object;
            other.reduce();
            this.reduce();
            if (this.numerator == other.numerator&& this.denominator== other.denominator) return true;
            else return false;
        }
        else return false;
    }

    public static void main(String[] args) {
        Fraction a = new Fraction(2,-30);
        a.reduce();
        System.out.printf("%d %d", a.getNumerator(),a.getDenominator());
    }




}
