public class Fraction {
    private int numerator;
    private  int denominator;
    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public int getDenominator() {
        return denominator;
    }

    public void setDenominator(int denominator) {
        if(denominator==0)
        {
            this.denominator=1;
        }
        else this.denominator = denominator;
    }
    Fraction() {
        this.numerator=1;
        this.denominator=1;
    }
    Fraction(int a,int b)
    {
        this.numerator=a;
        if(b!=0)
        {
            this.denominator=b;
        }
        else this.denominator=1;
    }

red(this.numerator,this.denominator);

        {
            this.setNumerator(this.getNumerator()*(-1));
            this.setDenominator(this.getDenominator()*(-1));

    public  Fraction add(Fraction fr)
    {
        if(fr.denominator==0 || this.denominator==0)
        {
            return this;
        }
        else
        {
            this.setNumerator(this.getNumerator()*fr.denominator+fr.numerator*this.getDenominator());
            this.setDenominator(this.getDenominator()*fr.denominator);
            return this;
        }
    }
    public  Fraction subtract(Fraction fr)
    {
        if(fr.denominator==0 || this.denominator==0)
        {
            return this;
        }
        else
        {
            this.setNumerator(this.getNumerator()*fr.denominator - fr.numerator*this.getDenominator());
            this.setDenominator(this.getDenominator()*fr.denominator);
            return this;
        }
    }
    public  Fraction multiply(Fraction fr)
    {
        if(fr.denominator==0 || this.denominator==0)
        {
            return this;
        }
        else
        {
            this.setNumerator(this.getNumerator()* fr.numerator);
            this.setDenominator(this.getDenominator() * fr.denominator);
            return this;
        }
    }
    public  Fraction divide(Fraction fr)
    {
        if(fr.numerator==0 || this.denominator==0|| fr.denominator==0)
        {
            return this;
        }
        else{
            this.setNumerator(this.getNumerator()* fr.denominator);
            this.setDenominator(this.getDenominator() * fr.numerator);
            return this;
        }
    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            if(this.getDenominator()*other.getNumerator() == this.getNumerator()*other.getDenominator())
                return true;
        }
        return false;
    }

}
