public class Fraction {

    private  int numerator;
    private int denominator=1;


    public Fraction(int a, int b) {
        numerator = a;
        if (b==0 ) denominator=1;
        else
        denominator = b;

    }
    public  void setDenominator(int n){
        if(n==0) this.denominator=1;
        else this.denominator = n;
    }
    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int n) {
        this.numerator = n;
    }

    public int getDenominator() {
        return denominator;
    }

    public static int gcd(int a, int b) {
        if(b==0) return a;
        else return gcd(b,a%b);
    }

    public Fraction reduce() {
        int n = gcd(numerator, denominator);
        numerator /=n;
        denominator /= n;
        return this;
    }

    public Fraction add(Fraction b) {
        int numResult = (numerator * b.getDenominator()) +
                (b.getNumerator() * denominator);
        int denoResutl = denominator * b.getDenominator();

        Fraction result= new Fraction(numResult,denoResutl );
        result.reduce();
        return result;
    }

    public Fraction subtract(Fraction b) {
        int numResult = (numerator * b.denominator) -
                (b.numerator * denominator);
        int denoResutl = denominator * b.denominator;
        Fraction result= new Fraction(numResult,denoResutl );
        result.reduce();
        return result;
    }

    public Fraction multiply(Fraction b) {
        int numResult = numerator * b.numerator;
        int denoResutl = denominator * b.denominator;
        Fraction result= new Fraction(numResult,denoResutl );
        result.reduce();
        return result;
    }
    public Fraction divide(Fraction b) {
        if(b.numerator==0) return  null;
        int numResult = numerator * b.getDenominator();
        int denoResutl = denominator * b.numerator;
        Fraction result= new Fraction(numResult,denoResutl );
        result.reduce();
        return result;
    }
    public boolean equals(Object obj) {
        if (obj instanceof Fraction == true) {

            Fraction other = (Fraction) obj;
            int c = this.numerator*other.getDenominator()-this.denominator*other.getNumerator();
            if ((this.denominator == 0) || (other.denominator == 0)) return false;
            if( c != 0) return false;
            else  return  true;

        }
        else return false;
    }
    public String toString() {
        return this.numerator + "/" + this.denominator;
    }

    }
}
