public class Fraction {
    private int numerator, denominator;

    public void setDenominator(int denominator_) {
        if (denominator_ != 0)
            this.denominator = denominator_;
        else this.denominator = 1;
    }

    public int getDenominator() {
        return denominator;
    }

    public int getNumerator() {
        return numerator;
    }

    public void setNumerator(int numerator_) {
        this.numerator = numerator_;
    }

    public Fraction(int numerator_, int denominator_){
        this.numerator = numerator_;
        if (denominator_ != 0)
            this.denominator = denominator_;
        else this.denominator = 1;
    }

    public Fraction(Fraction f_){
        this.numerator = f_.numerator;
        if (f_.denominator != 0)
            this.denominator = f_.denominator;
        else this.denominator = 1;
    }

    public Fraction reduce(){
        int a = this.numerator;
        int b = this.denominator;
        int temp = 1;
        while (b!=0)
        {
            temp = a%b;
            a = b;
            b = temp;
        }
        if (a!=0) {
            this.numerator = this.numerator / a;
            this.denominator = this.denominator / a;
        }
        if (this.denominator < 0)
        {
            this.numerator = - this.numerator;
            this.denominator = - this.denominator;
        }
        return this;
    }

    public Fraction add(Fraction f){
        int a = this.numerator;
        int b = this.denominator;
        this.denominator = b * f.denominator;
        this.numerator = a * f.denominator + f.numerator * b;
        this.reduce();
        return this;
    }

    public Fraction subtract(Fraction f){
        int a = this.numerator;
        int b = this.denominator;
        this.denominator = b * f.denominator;
        this.numerator = a * f.denominator - f.numerator * b;
        this.reduce();
        return this;
    }

    public Fraction multiply(Fraction f){
        this.denominator = this.denominator * f.denominator;
        this.numerator = this.numerator * f.numerator;
        this.reduce();
        return this;
    }

    public Fraction divide(Fraction f){
        this.denominator = this.denominator * f.numerator;
        this.numerator = this.numerator * f.denominator;
        this.reduce();
        return this;
    }

    public boolean equals(Object obj){
        if (obj instanceof Fraction){
            Fraction other = (Fraction) obj;
            Fraction f = new Fraction(this.numerator, this.denominator);
            f.subtract(other);
            if (f.numerator == 0) return true;
            else return false;
        }
        else return false;
    }
