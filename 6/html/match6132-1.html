public 
class Fraction {
    /**
     * declare attributes here
     */

    private  int numerator;
    private  int denominator;
    /**
     * declare constructor here
     */
    public Fraction() {
        //this.numerator = 1;
        denominator = 1;
    }
    public Fraction(Fraction frac) {
        this.numerator = frac.numerator;
        this.denominator = frac.denominator;
    }

    public Fraction (int num, int den) {
        if (den == 0) {
            denominator = 1;
               return;
        }
        numerator = num;
        denominator = den;
    }

    public void setNumerator(int numerator) {
        this.numerator = numerator;
    }

    public  int getNumerator() {
        return numerator;
    }

    public  void setDenominator(int denominator) {
        if (denominator == 0) 
return;
        this.denominator = denominator;
    }

    public  int getDenominator() {
        return denominator;
    }

    public void reduce () {
       int num = Math.abs(this.numerator);
       int den = Math.abs(this.denominator);

       while (num!=0 && den!=0) {
           if (num>den) num %= den;
           else den %= num;
       }

        this.numerator    /= (num + den);
        this.denominator  /= (num + den);
    }

    /****************************
     * methods
     ****************************/

    // add
    public Fraction add(Fraction frac) {
        this.setNumerator(this.getNumerator() * frac.getDenominator() + frac.getNumerator()* this.getDenominator());
        this.setDenominator(frac.getDenominator() * this.getDenominator());

        this.reduce();
        return this;
    }

    // subtract
    public Fraction subtract(Fraction frac) {
        this.setNumerator(this.getNumerator() * frac.getDenominator() - frac.getNumerator()* this.getDenominator());
        this.setDenominator(frac.getDenominator() * this.getDenominator());

        this.reduce();
        return this;
    }

    // multiple
    public Fraction multiply(Fraction frac) {
        if (this.getDenominator()*frac.getDenominator() ==0) return this;
        this.setNumerator(this.getNumerator()*frac.getNumerator());
        this.setDenominator(this.getDenominator()*frac.getDenominator());
        this.reduce();
        return this;
    }

    // devide
    public Fraction divide(Fraction frac) {
        if (this.getDenominator()*frac.getNumerator() ==0) return this;
        this.setNumerator(this.getNumerator()*frac.getDenominator());
        this.setDenominator(this.getDenominator()*frac.getNumerator());
        this.reduce();
        return this;
    }
    /**
     * compare this with other, notice that param is Object type
     */
    public boolean equals(Object obj) {
        if (obj instanceof Fraction) {
            Fraction other = (Fraction) obj;
            return (((double)this.numerator/(double)this.denominator) == ((double)other.numerator/(double)other.denominator));
        }
        return false;
    }

    public static void main(String[] args) {
        Fraction a = new Fraction(-5,3);
        Fraction b = new Fraction(10, -6);
        System.out.println(a.equals(b));
        //System.out.println(a.getNumerator()+"/"+a.getDenominator());
    }
}
