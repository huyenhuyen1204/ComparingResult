class Circle {
    private double radius = 1.0;
    private String color = "red";
    protected final double PI = 3.14;


    /**
     * Phuong thuc khoi tao khong tham so
     */
    public Circle() {
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param radius
     */
    public Circle(double radius) {
        this.radius = radius;
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param radius
     * @param color
     */
    public Circle(double radius, String color) {
        this.radius = radius;
        this.color = color;
    }

    /**
     * Getter cho radius
     * @return ban kinh
     */
    public double getRadius() {
        return radius;
    }

    /**
     * Setter cho radius
     * @param radius
     */
    public 
void setRadius(double radius) {
        this.radius = radius;
    }

    /**
     * Getter cho color
     * @return mau sac
     */
    public String getColor() {
        return color;
    }

    /**
     * Setter cho color
     * @param color
     */
    public void setColor(String color) {
        this.color = color;
    }

    /**
     * Getter cho area
     * @return dien tich hinh tron
     */
    public double getArea(){
        return PI * getRadius() * getRadius();
    }

    @Override
    public String toString() {
        return "Circle[" +
                "radius=" + radius +
                ",color=" + color +
                ']';
    }
}

public class Cylinder extends Circle {
    private double height = 1.0;

    /**
     * Phuong thuc khoi tao khong tham so
     */
    public Cylinder() {
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param height
     */
    public Cylinder(double height) {
        this.height = height;
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param radius
     * @param height
     */
    public Cylinder(double height, double radius) {
        super(radius);
        this.height = height;
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param radius
     * @param height
     * @param color
     */
    public Cylinder(double height, double radius, String color) {
        super(radius, color);
        this.height = height;
    }

    /**
     * Getter cho height
     * @return chieu cao hinh tru
     */
    public double getHeight() {
        return height;
    }

    /**
     * Setter cho height
     * @param height
     */
    public void setHeight(double height) {
        this.height = height;
    }

    /**
     * Getter cho area
     * @return dien tich toan phan hinh tru
     */
    @Override
    public double getArea(){
        return 2 * PI * getRadius() * getHeight() + 2 * super.getArea();
    }

    /**
     * Getter cho volume
     * @return the tich hinh tru
     */
    public 
double getVolume(){
        return super.getArea() * getHeight();
    }

    @Override
    public String toString(){
        return "Cylinder[" +
                super.toString() +
                ",height=" +
                getHeight() +
                "]";
    }
}
