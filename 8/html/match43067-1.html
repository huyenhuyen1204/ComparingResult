public class Circle {
    private double radius=1.0;
    protected final double PI=3.14;
    private String color="red";
    public Circle(){
        this.radius=1.0;
        this.color="red";
    }
    public Circle(double rad){
        this.radius=rad;
        this.color="red";
    }
    public Circle(double rad,String col){
        this.radius=rad;
        this.color=col;
    }
    public double getRadius(){
        return this.radius;
    }
    public 
void setRadius(double rad){
        this.radius=rad;
    }
    public String getColor(){
        return this.color;
    }
    public void setColor(String col){
        this.color=col;
    }
    public double getArea(){
        double S;
        S=PI*this.radius*this.radius*100.0/100.0;
        return S;
    }
    @Override
    public String toString(){
        return "Circle[radius="+this.radius+",color="+this.color+"]";
    }
}

public class Cylinder extends Circle {
    private double height=1.0;
    public Cylinder(){

    }
    public Cylinder(double rad){
        this.setRadius(rad);
        this.height=1.0;
    }
    public Cylinder(double rad,double hei){
        this.setRadius(rad);
        this.height=hei;
    }
    public Cylinder(double rad,double hei,String col){
        this.setRadius(rad);
        this.height=hei;
        this.setColor(col);
    }
    public double getHeight(){
        return this.height;
    }
    public void setHeight(double hei){
        this.height=hei;
    }
    @Override
    public double getArea(){
        return 2*super.getArea()+2*PI*super.getRadius()*this.height;
    }
    public double getVolume(){
        return super.getArea()*this.getHeight()*100.0/100.0;
    }
    
@Override
    public String toString(){
        return "Cylinder["+super.toString()+",height="+this.getHeight()+"]";
    }

    public static void main(String[] args) {
        Cylinder circle=new Cylinder();
        circle.setColor("blue");

        System.out.println(circle.getArea());

        System.out.println(circle.getColor());
        System.out.println(circle.getRadius());

    }
}
