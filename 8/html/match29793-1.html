public class Cylinder extends Circle {
    private double height;
    
    public Cylinder() {
        this.height = 1.0;
    }

    public Cylinder(double radius) {
        super(radius);
        this.height = 1.0;
    }

    public Cylinder(double radius, double height) {
        super(radius);
        this.height = height;
    }

    public Cylinder(double radius, double height, String color) {
        super(radius, color);
        this.height = height;
    }

    public double getHeight() {
        return this.height;
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public double getVolume() {
        double volume = PI * this.getRadius() * this.getRadius() * this.getHeight();
        return volume;
    }

    public 
String toString() {
        String s = "Cylinder[" + super.toString() + ",height=" + this.getHeight() + "]";
        return s;
    }

    public double getArea() {
        double s = super.getArea();
        double c = PI*2*(super.getRadius());
        double area = s*2 + c*this.getHeight();
        return area;
    }

    public static void main(String[] args) {
        Cylinder cyl = new Cylinder(1.2, 2.0);
        //double d = cyl.getArea();
        System.out.println(cyl.toString());
    }
    
}
