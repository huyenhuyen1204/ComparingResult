Circle.java



public class Circle {

	/**
	 * @param args
	 */
	protected final double PI = 3.14 ;
	private double radius = 1.0 ;
	private String color = "red" ;
	public Circle () {
	}
	public Circle(double radius)
	{
		this.radius = radius ; 
	}
	public Circle(double radius, String color)
	{
		this.radius = radius ;
		this.color = color ;
	}
	public double getRadius()
	{
		return this.radius ;
	}
	public void setRadius(double radius)
	{
		this.radius = radius ;
	}
	public String getColor()
	{
		return this.color ;
	}
	public void setColor(String color)
	{
		this.color = color ;
	}
	public double getArea(){
		double result =  (double) Math.round( PI * radius *radius * 100 )/ 100 ;
		return result ;
	}
	public String toString()
	{
		String re = "Circle[radius="+radius+",color="+color+"]" ;
		return re ;
	}
	
		
	
}

public class Cylinder extends Circle {
	private double height = 1.0 ;
 
	public Cylinder()
	{ }
	public Cylinder(double radius)
	{
		super(radius);
	}
	public Cylinder(double radius , double height)
	{
		super.setRadius(radius);
		this.height = height ;
	}
	public Cylinder(double radius , double height , String color)
	{
		super(radius, color) ;
		this.height = height ;
	}
	public double getHeight() {
		return height;
	}
	public void setHeight(double height) {
		this.height = height;
	}
	public double getVolume()
	{
		double volume  =(double) Math.round( super.getArea() *  height * 100 ) / 100  ;
		return volume ;
	}
	public String toString()
	{
		String re1 ="Cylinder["+super.toString()+",height="+this.height+"]";
		return re1 ;
	}
	public double getArea()
	{
		double re2 = (double) Math.round ((2 * super.getArea() + 2 * PI * super.getRadius() * height )*100) / 100  ;
		return re2 ;
	}

	

}
