import java.util.*;

public 
class Circle {
    protected final double PI=3.14;
    private double radius=1;
    private  String color="red";
    public Circle(){
    }
    public Circle(double rad){
        this.setRadius(rad);
    }
    public Circle(double rad, String col){
        this.setRadius(rad);
        this.setColor(col);
    }

    public double getRadius(){
        return this.radius;
    }
    public void setRadius(double rad){
        rad=Math.max(rad,0);
        this.radius=rad;
    
}

    public String getColor(){
        return this.color;
    }
    public void setColor(String col){
        this.color=col;
    }

    public double getArea(){
        double rad=this.getRadius();
        return rad*rad*PI;
    }

    public String toString(){
        String tmp=this.radius + "";
        return "Circle[radius=" + this.getRadius() + ",color=" + this.getColor() + "]";
    }
    public static void main(String[] args){
        Circle test1=new Circle(1.2,"blue");
        System.out.println(test1.toString());
        System.out.println(test1.getArea());

        Cylinder test2=new Cylinder(3.2,1.5,"green");
        System.out.println(test2.toString());
        System.out.println(test2.getArea());
        System.out.println(test2.getVolume());
    }
}

public 
class Cylinder extends Circle {
    private double height=1;
    public Cylinder(){
    }
    public Cylinder(double rad){
        super(rad);
    }
    public Cylinder(double rad, double hei){
        super(rad);
        this.setHeight(hei);
    }
    public Cylinder(double rad, double hei, String col){
        super(rad,col);
        this.setHeight(hei);
    }
    public double getHeight(){
        return this.height;
    }
    public void setHeight(double hei){
        this.height=hei;
    }
    public double getVolume(){
        return super.getArea()*this.height;
    }
    public String toString(){
        return "Cylinder[Circle[radius=" + this.getRadius() + ",color=" + this.getColor() + "],height=" + this.getHeight() + "]";
    }
    public double getArea(){
        double para=PI*2*this.getRadius();
        return para*this.getHeight() + super.getArea()*2;
    }
}
