public 
class Circle {
    private double radius = 1.0;
    private String color = "red";
    final protected double PI = 3.14;

    public Circle(){}

    public Circle(double radius){
        this.setRadius(radius);
    }

    public Circle(double radius, String color){
        this.setRadius(radius);
        this.setColor(color);
    }

    public double getRadius() {
        return this.radius;
    }

    public 
void setRadius(double radius) {
        this.radius = radius;
    }

    public String getColor() {
        return this.color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public double getArea(){
        double ans = PI * this.radius * this.radius;
        return Math.round(ans * 100.0) / 100.0;
    }

    public String toString() {
        StringBuilder stringBuilder = new StringBuilder();
        stringBuilder.append("Circle[radius=").append(this.getRadius()).
                append(",color=").append(this.getColor()).append("]");
        return stringBuilder.toString();
    }

    public static void main(String[] args) {
        Circle cc = new Circle(1);
        System.out.println(cc.getArea());
    }
}

public 
class Cylinder extends Circle{
    private double height = 1.0;

    public Cylinder(){};

    public Cylinder(double radius){
        this.setRadius(radius);
    }

    public Cylinder(double radius, double height){
        this.setRadius(radius);
        this.setHeight(height);
    }

    public Cylinder(double radius, double height, String color){
        this.setRadius(radius);
        this.setHeight(height);
        this.setColor(color);
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public double getHeight() {
        return height;
    }

    public double getVolume(){
        return Math.round(
super.getArea() * this.getHeight() * 100.0) / 100.0;
    }

    @Override
    public double getArea() {
        double ans = super.getArea() * 2 + 2 * PI * this.getRadius() * this.getHeight();
        return Math.round(ans * 100.0) / 100.0;
    }

    public String toString(){
        StringBuilder stringBuilder = new StringBuilder();
        stringBuilder.append("Cylinder[").append(super.toString()).append(",height=").append(this.getHeight()).append("]");
        return stringBuilder.toString();
    }
