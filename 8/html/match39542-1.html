class Cylinder  extends Circle{
    private double height = 1.0;
    public Cylinder(){

    }
    public Cylinder(double radius){
        setRadius(radius);
    }
    public Cylinder(double radius, double height){
        setRadius(radius);
        this.height = height;
    }
    public Cylinder(double radius, double height, String color){
        setRadius(radius);
        this.height = height;
        setColor(color);
    }
    public double getHeight(){
        return height;
    }
    public void setHeight(double height){
        this.height = height;
    }
    public double getVolume(){
        double V = (double)(this.height*PI*getRadius()*getRadius());
        V = (double) Math.round(V * 100) / 100;
        return V;
    }

    @Override
    public String toString(){
        return "Cylinder[Circle[radius=" + getRadius() + ",color="+getColor()+"],height=" + height + "]";
    }
    public double getArea(){
        return 2*PI*getRadius()*getRadius() + 2*PI*height*getRadius();
    }
}
