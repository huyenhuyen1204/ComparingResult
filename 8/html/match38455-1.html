public class Cylinder extends Circle
{

    private double height=1.0;
   public Cylinder()
        {
            super();
        }
   public Cylinder(double radius1)
        {
        super(radius1);
        }
   public Cylinder(double radius1,double height1)
        {
            super(radius1);
            this.height=height1;
        }
   public Cylinder(double radius1,double height1,String color1)
        {
            super(radius1,color1);
            this.height=height1;
        }
   public double getHeight()
        {
            return this.height;
        }
   public void setHeight(double height1)
        {
            this.height=height1;
        }
   public 
double getVolume()
        {
            double answer=super.getArea()*height;
            return answer;
        }
   public String toString()
   {
       String answer="["+super.toString()+",height="+this.height;
       return answer;
   }
   public double getArea()
   {
       double answer=super.getArea()*2+this.getRadius()*2*PI*height;
       return answer;
   }

}
