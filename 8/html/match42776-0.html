public class Circle {
    private double radius = 1.0;
    private String color = "red";
    protected final double PI = 3.14;
    public Circle(){ }
    public Circle(double a,String b)
    {
        this.radius = a;
        this.color = b;
    }
    public Circle(double a)
    {
        this.radius =a;
    }
    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }
    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    public double getArea() {
        double res = (double) this.radius * this.radius * PI;
        return res;
    }
    public String toString()
    {
        return "Circle[radius="+getRadius() +",color="+getColor()+"]";
    }
}

public class Cylinder extends Circle {
    private double height = 1.0;
    public Cylinder(){
        this(1.0);
    }
    public Cylinder (double a, double c,String b)
    {
        super(a,b);
        this.height=c;
    }
    public Cylinder(double a)
    {
        super(a);
        this.height= height;
    }
    public Cylinder(double a,double b)
    {
        super(a);
        this.height=b;
    }
    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        this.height = height;
    }
    public double getVolume()
    {
        double a = super.getArea();
        double res = (double)this.height*a;
        return res;
    }
    public double getArea()
    {
        double a = (double)(2*PI*this.height*super.getRadius()+2*super.getArea());
        return a;
    }
    public String toString()
    {
        return "Cylinder["+super.toString()+",height="+getHeight()+"]";
    }

    public static void main(String[] args) {
        Cylinder a = new Cylinder(2);
        System.out.println(a.getVolume());
    }
}
