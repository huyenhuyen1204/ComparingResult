Circle.java

public class Circle {
    final protected double PI = 3.14;
    private double radius = 1.0;
    private String color = "red";
    
    public Circle(){
        this.radius = 1.0;
        this.color = "red";
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color){
        this.radius = radius;
        this.color = color;
    }
    
    public double getRadius(){
        return radius;
    }
    public 
void setRadius(double radius){
        this.radius = radius;
    }
    public String getColor(){
        return color;
    }
    public void setColor(String color){
        this.color = color;
    }
    
    public double getArea(){
        return PI * radius * radius;
    }
    @Override
    public String toString(){
        String answer ;
        String r = String.valueOf(radius);
        String c = color;
        answer = "Circle[radius=" + r + ",color=" + c + "]";
        return answer;   
    }
}

Cylinder.java

public class Cylinder extends Circle{
    final protected double PI = 3.14;
    private double height = 1.0;
    
    public Cylinder(){
        super();
        this.height = 1.0;
    }
	public Cylinder(double radius) {
		super(radius);
		this.height = 1.0;
	}
    public Cylinder(double radius, double height){
        super(radius);
        this.height = height;
    }
    public Cylinder(double radius, double height, String color){
        super(radius,color);
        this.height = height;
    }
    public double getHeight() {
        return height;
    }
    public void setHeight(double height) {
        this.height = height;
    }
    public double getVolume() {
        return super.getArea() * height;
    
}
    @Override
    public double getArea() {
        double result;
        result = 2 * super.getArea() + 2 * PI * super.getRadius() * height;
        return result;
    }
    @Override
    public String toString() {
        String result;
        result = "Cylinder[" + super.toString() + ",height=" + String.valueOf(this.getHeight()) +"]";
        return result;
    }
    
}
