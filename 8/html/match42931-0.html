public class Cylinder extends Circle{
	private double height;

	public Cylinder(){
		height = 1.0;
		this.setColor("red");
		this.setRadius(1.0);
	}

	public Cylinder(double radius){
		this.height = 1.0;
		this.setColor("red");
		this.setRadius(radius);
	}

	public Cylinder(double radius, double height){
		this.height = height;
		this.setColor("red");
		this.setRadius(radius);
	}

	public Cylinder(double radius, double height, String color){
		this.height = height;
		this.setColor(color);
		this.setRadius(radius);
	}

	public double getHeight(){
		return height;
	}

	public void setHeight(double height){
		this.height = height;
	}

	public double getVolume(){
		return this.PI * this.getRadius() * this.getRadius() * height;
	}

	public String toString(){
		return "Cylinder[Circle[radius=" + this.getRadius() + ",color=" + this.getColor() + "],height=" + height + "]";
	}

	public double getArea(){
		return 2 * this.PI * this.getRadius() * height + 2 * this.PI * this.getRadius() * this.getRadius(); 
	}
}
