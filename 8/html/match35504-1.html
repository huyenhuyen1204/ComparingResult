public class Circle {
    final protected double PI = 3.14;

    private double radius = 1.0;

    private String color = "red";

    public double getRadius() {
        return radius;
    }

    public 
void setRadius(double radius) {
        this.radius = radius;
    }

    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public Circle(){};

    public Circle(double _radius){
        radius = _radius;
    }

    public Circle(double _radius, String _color){
        radius = _radius;
        color = _color;
    }

    public String toString(){
        return "Circle[radius=" + radius + ",color=" + color + "]";
    }

    public double getArea(){
        return PI * radius * radius;
    }

    public static void main(String[] args){
        Cylinder a = new Cylinder(2,2,"2");
        System.out.println(a.getVolume());
    }




}

public class Cylinder extends Circle{
    private double height = 1.0;

    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public 
Cylinder(){};

    public Cylinder(double _radius){
        super.setRadius(_radius);
    }

    public Cylinder(double _radius, double _height){
        super.setRadius(_radius);
        height = _height;
    }

    public Cylinder(double _radius, double _height, String _color){
        super.setRadius(_radius) ;
        height = _height;
        super.setColor(_color);
    }

    public String toString(){
        return "Cylinder[" + super.toString() + ",height=" + height + "]";
    }

    public double getVolume(){
        return height * super.getArea();
    }

    public double getArea(){
        return 2 * super.getArea() + 2 * PI * super.getRadius() * height;
    }


}
