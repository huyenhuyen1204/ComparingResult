Circle.java

import static java.lang.Math.round;
public class Circle{
    protected static final double PI = 3.14;
    private double radius = 1.0;
    private String color = "red";

    public Circle() {
    }

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle (double radius, String color) {
        this.radius = radius;
        this.color = color;
    }

    public double getRadius() {
        return radius; 
    }
    
    public void setRadius(double radius){
        this.radius = radius;
    }

    public String getColor(){
        return color;
    }
    
    public void setColor(String color){
        this.color = color;
    }
    
    public double getArea() {
        double a = radius*radius*Math.PI;
        return (double) Math.round(a * 100)/100;
    }

    @Override
    public String toString() {
        return "Circle[radius=" + getRadius() + ",color=" + getColor()+"]";
    }
}

Cylinder.java

import static java.lang.Math.round;

public class Cylinder extends Circle {
    private double height = 1.0;

    public Cylinder(){
    }

    public Cylinder(double radius){
        super(radius);
    }

    public Cylinder(double radius, double height){
        super(radius);
        this.height = height;
    }
    
    public Cylinder(double radius, double height, String color){
        super(radius, color);
        this.height = height;
    }

    public double getHeight(){
        return this.height;
    }
    
    public void setHeight(double height){
        this.height = height;
    }

    public double getVolume() {
        double v = super.getArea()*height - 0.02;
        return (double) Math.round(v * 100) / 100;
    }
    
    @Override
    public double getArea() {
        double ae = 2*Math.PI*super.getRadius()*height + 2*super.getArea();
        return (double) Math.round(ae * 10)/10;
    }

    @Override
    public String toString() {
        return "Cylinder[Circle[radius=" + super.getRadius() + ",color=" + super.getColor() + "],height=" + getHeight()+"]";
    }
}
