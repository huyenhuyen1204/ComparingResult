Circle.java

import java.util.*;
public class Circle {
	private double radius = 1;
	private String color = "red";
	protected final double PI = 3.14;
	
	public Circle() {
	}
	
	public Circle(double radius) {
		this.radius = radius;
	}
	
	public Circle(double radius, String color) {
		this.radius = radius;
		this.color = color;
	}
	
	public double getRadius() {
		return this.radius;
	}
	
	public void setRadius(double radius) {
		this.radius = radius;
	}
	
	public String getColor() {
		return this.color;
	}
	
	public void setColor(String color) {
		this.color = color;
	}
	
	public double getArea() {
		double a = PI * this.radius * this.radius;
		//a = (double) Math.round(a * 100)/100;
		return a;
	}
	
	public String toString() {
		return "Circle[radius=" + Double.toString(this.radius) + ",color=" + this.color + "]"; 
	}
}

Cylinder.java

import java.util.*;
public class Cylinder extends Circle {
	private double height = 1;
	
	public Cylinder() {
	}
	
	public Cylinder(double radius) {
		super(radius);
	}
	
	public Cylinder(double radius, double height) {
		super(radius);
		this.height = height;
	}
	
	public Cylinder(double radius, double height, String color) {
		super(radius,color);
		this.height = height;
	}
	
	public double getHeight() {
		return this.height;
	}
	
	public void setHeight(double height) {
		this.height = height;
	}
	
	public double getVolume() {
		double a = super.getArea() * this.height;
		//a = (double) Math.round(a * 100)/100;
		return a;
	}
	
	@Override
	public String toString() {
		return "Cylinder[" + super.toString() + ",height=" + Double.toString(this.height) + "]";
	}
	
	@Override
	public double getArea() {
		return super.getArea() * 2 + 2 * PI * getRadius() * this.height;
	}
	
	public static void main(String[] args) {
		Cylinder a = new Cylinder(5,4,"Black");
		System.out.println(a.getArea());
	}
}
