import  java.lang.Math ;
import java.lang.String ;
public class Cylinder extends Circle
{
    private double height  = 1.0 ;

    public Cylinder () { }
    public Cylinder (double radius)
    {
        super (radius) ;
    }
    public Cylinder (double radius, double height)
    {
        super (radius) ;
        this.height = height ;
    }
    public Cylinder (double radius, double height, String color)
    {
        super (radius) ;
        this.height = height;
        super.setColor(color);
    }
    public double getHeight ()
    {
        return this.height ;
    }
    public void setHeight (double height)
    {
        this.height = height ;
    }
    public double getVolume ()
    {
        double totalVolume =  this.height * super.getArea();
        
totalVolume = Math.round(totalVolume * 100) / 100.00 ;
        return totalVolume ;
    }
    public double getArea ()
    {
        double totalArea = super.getArea() * 2 + super.getRadius() * 2 * PI * this.height ;
        totalArea = Math.round(totalArea * 10 ) / 10.0 ;
        return totalArea ;
    }
    public String toString ()
    {
        String height_h = String.valueOf(height) ;
        return "Cylinder[" + super.toString() + ",height=" + height_h + "]" ;
    }
}
