src/Circle.java

import java.util.*;
import java.lang.*;

public 
class Circle {
    private double radius = 1.0;
    protected final double PI = 3.14;
    private String color = "red";
    public Circle(){};
    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle(double radius, String color) {
        this.radius = radius;
        this.color = color;
    }
    public double getRadius() {
        return radius;
    }
    public 
void setRadius(double radius) {
        this.radius = radius;
    }
    public String getColor() {
        return color;
    }
    public void setColor(String color) {
        this.color = color;
    }
    public double getArea() {
        double area = PI * radius * radius;
        return Math.round(area * 100.0) / 100.0;
    }
    public String toString() {
        return "Circle[radius=" + radius + ",color=" + color + "]";
    }
    public static void main (String[] args) {
        Circle newCircle = new Circle();
        newCircle.setRadius(3);
        newCircle.setColor("red");
        System.out.println(newCircle.getRadius());
        System.out.println(newCircle.getColor());
        System.out.println(newCircle.getArea());
        System.out.println(newCircle.toString());
    }
}

public class Cylinder extends Circle {
    private double height = 1.0;
    public Cylinder(){};
    public Cylinder(double radius) {
        super(radius);
    }
    public Cylinder(double radius, double height) {
        super(radius);
        this.height = height;
    }
    public Cylinder(double radius, double height, String color) {
        super(radius, color);
        this.height = height;
    }
    public double getHeight() {
        return height;
    }
    public void setHeight(double height) {
        this.height = height;
    }
    public 
double getVolume() {
        double volume = PI * getRadius() * getRadius() * height;
        return Math.round(volume * 100.0) / 100.0;
    }
    @Override
    public String toString() {
        return "Cylinder[" + super.toString() + ",height=" + height + "]";
    }
    @Override
    public double getArea() {
        return super.getArea() * 10 / 3;
    }
    public static void main (String[] args) {
        Cylinder newCylinder = new Cylinder(3,2, "green");
        System.out.println(newCylinder.getHeight());
        System.out.println(newCylinder.getVolume());
        System.out.println(newCylinder.getColor());
        System.out.println(newCylinder.toString());
        System.out.println(newCylinder.getArea());
    }
}
