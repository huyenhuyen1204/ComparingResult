Circle.java

public class Circle {
    private double radius = 1.0;
    private String color = "red";
    protected final double PI = 3.14;
    public Circle()
    {

    }
    public Circle(double radius)
    {
        if(radius >= 0 )
        {
            this.radius = radius;
        }
    }
    public Circle(double radius, String color)
    {
        if(radius >= 0 && color != null)
        {
            this.radius = radius;
            this.color = color;
        }
    }

    public double getRadius() {
        return radius;
    }

    public String getColor() {
        return color;
    }

    public void setRadius(double radius) {
        if(radius >= 0 )
        {
            this.radius = radius;
        }
    }

    public void setColor(String color) {
        if(color != null)
        {
            this.color = color;
        }
    }

    public double getArea()
    {
        return radius*radius*PI;
    }
    @Override
    public String toString()
    {
        return ("Circle[radius=" + radius + ",color="+ color + "]");
    }
}

Cylinder.java

public class Cylinder extends Circle {
    private double height = 1.0;
    public Cylinder()
    {

    }
    public Cylinder(double radius)
    {
        super(radius);
    }
    public Cylinder(double radius, double height)
    {
        super(radius);
        if(height >= 0)
        {
            this.height = height;
        }
    }
    public Cylinder(double radius, double height, String color)
    {
        super(radius, color);
        if(height >= 0)
        {
            this.height = height;
        }
    }

    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        if(height >= 0)
        {
            this.height = height;
        }
    }

    public double getVolume()
    {
        return getRadius()*getRadius()*PI*height;
    }

    @Override
    public String toString()
    {
        return ("Cylinder[Circle[radius=" + getRadius() + ",color="+ getColor() + "],height=" + height + "]");
    }

    @Override
    public double getArea()
    {
        return super.getArea()*2+height*getRadius()*2*PI;
    }

    public static void main(String[] args)
    {
        Cylinder cyl = new Cylinder(5, 4, "blue");
        Circle cir = new Circle(5, "blue");
        System.out.println(cir.toString());
        System.out.println(cyl.toString());
    }
}
