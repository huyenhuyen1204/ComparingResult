public 
class Circle{
    private double radius = 1.0f;
    private String color = "red";
    protected final double PI = 3.14f;
    
    public Circle(){        
    };
    
    public Circle(double radius){
        this.radius = radius;
    }
    
    public Circle(double radius, String color){
        this.radius = radius;
        this.color = color;
    }
    
    public void setRadius(double radius){
        this.radius = radius;
    }
    
    public double getRadius(){
        return radius;
    }
    
    public void setColor(String color){
        this.color = color;
    }
    
    public String getColor(){
        return color;
    }
    
    public double getArea(){
        return PI*radius*radius;
    }
    
    @Override
    public String toString(){
        String result = "Circle[radius=" + radius + ",color=" + color + "]";
        return result;
    }    
}

Week5.1/Cylinder.java

public class Cylinder extends Circle{
    private double height = 1.0;
    
    public void setHeight(double height){
        this.height = height;
    }
    
    public double getHeight(){
        return height;
    }
    
    public Cylinder(){        
    }
    
    public Cylinder(double radius){
        super(radius);      
    }
    
    public Cylinder(double radius, double height){
        super(radius);
        this.height = height;
    }
    
    public Cylinder(double radius, double height, String color){
        super(radius, color);
        this.height = height;
    }
    
    public double getVolume(){
        return super.getArea()*height;
    
}
    
    @Override
    public double getArea(){
        return 2*super.getArea()+2*PI*super.getRadius()*height;
    }
    
    @Override 
    public String toString(){
        String result = "Cylinder[" +super.toString() + ",height=" + height + "]";
        return result;
    }
    
}
