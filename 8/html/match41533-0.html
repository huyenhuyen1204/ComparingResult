public class Circle {
    protected final double PI = 3.14;
    private double radius = 1.0;
    private String color = "red";
    public Circle(){
    }
    public Circle(double x){
        this.radius = x;
    }
    public Circle(double x, String t){
        this.radius = x;
        this.color = t;
    }
    public double getRadius(){
        return this.radius;
    }
    public void setRadius(double x){
        this.radius = x;
    }
    public String getColor(){
        return this.color;
    }
    public void setColor(String t){
        this.color = t;
    }
    public double getArea(){
        return PI*this.radius*this.radius;
    }
    public String toString(){
        String t="";
        t+="Circle[radius=" + this.radius +",color="+this.color+"]";
        return t ;
    }
  
}

/**

 */
public class Cylinder extends Circle {
    private double height = 1.0;
    public  Cylinder(){
    }
    public  Cylinder(double r){
        super(r);
    }
    public  Cylinder(double r, double h){
        super(r);
        this.height = h;
    }
    public Cylinder(double r, double h, String t){
        super(r,t);
        this.height=h;
    }
    public double getHeight(){
        return this.height;
    }
    public void setHeight(double x){
        this.height = x;
    }
    public double getVolume(){
        return PI*getRadius()*getRadius()*getHeight();
    }
    public String toString(){
        String t = "";
        t = "Cylinder[" + super.toString()+",height=" + this.getHeight() +"]";
        return t;
    }
    public double getArea(){
        return 2*super.getArea() + 2*PI*this.getRadius()*this.height ;
    }
    public static void main(String[] args) {
        Cylinder st = new Cylinder(3,2,"green");
        System.out.println(st.toString());
        double x = st.getArea();
        System.out.println(x);
    }
}
