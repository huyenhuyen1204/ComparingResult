public class Cylinder extends Circle {
    private double height;

    public Cylinder(){
        height = 1.0;
    }

    public Cylinder(double r){
        setRadius(r);
        height = 1.0;
    }

    public Cylinder(double r, double h){
        setRadius(r);
        setHeight(h);
    }

    public Cylinder(double r, double h, String c){
        setRadius(r);
        setHeight(h);
        setColor(c);
    
}

    public double getHeight(){
        return height;
    }

    public void setHeight(double height){
        this.height = height;
    }

    public double getVolume(){
        return getRadius() * getRadius() * height * PI;
    }

    public String toString(){
        return "Cylinder[" + super.toString() +  ",height=" + String.valueOf(height) +  "]";
    }

    public double getArea(){
        return super.getArea() * 2 + 2 * getRadius() * height * PI;
    }

    public static void main(String[] args){
        Cylinder C = new Cylinder(1.9, 90, "purple" );
        System.out.println(C.getArea());
        System.out.println(C.toString());
    }
}
