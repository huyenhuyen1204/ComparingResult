import java.util.List;
import java.util.ArrayList;

public class Layer {

   private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape s) {
        shapes.add(s);
    }

    public void removeCircles()
    {
        for ( int i  = 0 ; i < shapes.size() ; i++)
        {
            if (
shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo()
    {
        String result ="Layer of crazy shapes: ";
        result += '\n';
        for ( int index =0  ; index < shapes.size() ; index ++)
        {
            result += shapes.get(index).toString();
            result += '\n';
        
}
        return result;
    }

    public void removeDuplicates()
    {
        for ( int i  = 0 ; i < shapes.size() -1 ; i++)
        {
            for ( int j  = i + 1 ; j < shapes.size() ; j++)
            {
                if (shapes.get(i).equals(shapes.get(j)) == true)
                {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    

}

public class Point {

    private double x;
    private double y;

    public Point()
    {

    }
    public Point(double x, double y)
    {
        this.x = x;
        this.y =y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance( Point other)
    {
        double 
result = Math.sqrt(Math.pow(getX()- other.x, 2)+ Math.pow(getY()- other.y, 2));
        return result;
    }

    public boolean equals(Object other)
    {
        if (other instanceof Point)
        {
            Point ot = ( Point) other;
            if ( this.x ==ot.getX() && this.y == ot.getY())
            {
                return true;
            }
            else 
return false;
        }
        return false;
    }
