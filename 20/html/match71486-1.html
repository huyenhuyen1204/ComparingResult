//package Polymorphism2;
import java.util.ArrayList;

public class Layer
{
    private ArrayList<Shape> shapes;

    public void addShape(Shape sample)
    {
        shapes.add(sample);
    
}

    public String getInfo()
    {
        String s = "Layer of crazy shapes: \n";
        for (Shape cs: shapes)
        {
            s = s + cs.toString()+"\n";
        }
        return s;
    }

    public void removeCircles()
    {
        int i, n = shapes.size();
        for (i = 0; i < shapes.size(); i++)
        {
            if (shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
            }
        }
    }
    public void removeDuplicates()
    {
        int[] Check = new int[shapes.size()];
        int i,j;
        for (i = 0;i < Check.length;i++)
        {
            Check[i]=0;
        }
        for (i = 0;i < shapes.size();i++)
        {
            for (j = i + 1;j < shapes.size();j++)
            {
                if(shapes.get(i).hashCode() == shapes.get(j).hashCode() && Check[i] == 0)
                {
                    if(shapes.get(i).equals(shapes.get(j)) == true)
                    {
                        Check[j] = 1;
                    }
                }
            }
        }
        for (i = 0;i < shapes.size();i++)
        {
            if(Check[i] == 1)
            {
                shapes.remove(i);
                i--;
            }
        }
    }
}

public class Rectangle extends Shape

    }
    public Rectangle(double width,double length)
    {
        this.width = width;
        this.length = length;
    
}
    public Rectangle(double width,double length,String color,boolean filled)
    {
        this.filled = filled;
        this.color = color;
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft,double width,double length,String color,boolean filled)
    {
        this.filled = filled;
        this.color = color;
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }
    public double getWidth()
    {
        return this.width;
    }
    public void setWidth(double width)
    {
        this.width = width;
    }
    public double getLength()
    {
        return this.length;
    }
    public void setLength(double length)
    {
        this.length = length;
    }
    public double getArea()
    {
        double Area = (this.length) * (this.width);
        return Area;
    }
    public double getPerimeter()
    {
        double Perimeter = 2 * ((this.length) + (this.width));
        return Perimeter;
    }
    public Point getTopLeft()
    {
        return this.topLeft;
    }
    public void setTopLeft(Point topLeft)
    {
        this.topLeft = topLeft;
    }
    public 
boolean equals(Object object)
    {
        if (object instanceof Rectangle)
        {
            Rectangle other = (Rectangle) object;
            if(Math.abs(other.length - length) <= 0.001 && Math.abs(other.width - width) <= 0.001 && Math.abs((other.topLeft).getX() - 
topLeft.getX()) <= 0.001 && Math.abs((other.topLeft).getY() - topLeft.getY()) <= 0.001 )
            {
                return true;
            }
            else
            {
                return false;
            }
        }
        return false;
    }
    public int hashCode()
    {
        return super.hashCode();
    }

    {
