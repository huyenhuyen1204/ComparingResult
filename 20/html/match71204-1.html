import java.util.ArrayList;
import java.util.List;

public class Layer
{
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape e)
    {
        shapes.add(e);
    }

    public 
String getInfo()
    {
        String s = "Layer of crazy shapes: \n";
        for(int i = 0; i < shapes.size(); i++)
        {
            s += shapes.get(i).toString() + "\n";
        }
        return s;
    
}

    public void removeCircles()
    {
        for(int i = 0; i < shapes.size(); i++)
        {
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i = -1;
            }
        }
    
}

    public void removeDuplicates()
    {
        for(int i = 0; i < shapes.size() - 1; i++)
            for(int j = i + 1; j < shapes.size(); j++)
            {
                if(shapes.get(i).equals(shapes.get(j)))
                {
                    shapes.remove(j);
                }
            }
    }
    
    public static void main(String[] args)
    {
        Point t = new Point(5, 8);
        Shape e = new Circle(t, 5, "RED", false);
        
t = new Point(10, 9);
        Point i = new Point(2, 2);
        Shape e1 = new Rectangle(t, 10, 5, "BLUE", true);
        Shape e2 = new Rectangle(t, 12, 6, "BLUE", true);
        Shape e3 = new Rectangle(t, 14, 7, "BLUE", true);
        Shape e4 = new Rectangle(t, 16, 8, "BLUE", true);
        Shape e5 = new Square(t, 7, "YELLOW", false);
        Layer l = new Layer();
        l.shapes.add(e);
        l.shapes.add(e);
        l.shapes.add(e1);
        l.shapes.add(e2);
        l.shapes.add(e3);
        l.shapes.add(e4);
        l.shapes.add(e);
        l.shapes.add(e5);
        l.shapes.add(e5);
        l.removeDuplicates();
        System.out.println(l.getInfo());
        Shape e6 = new Square(i, 7, "RED", true);
        if(e6.equals(e5)) System.out.println("YES");
    }
}
