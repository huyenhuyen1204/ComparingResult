    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea(){
        return this.getRadius()*this.getRadius()*Math.PI;
    }

    public double getPerimeter(){
        return this.getRadius()*2*Math.PI;
    }

    public boolean equals(Object obj){
        if (obj instanceof Circle){
            if (
this.getCenter().distance(((Circle) obj).getCenter()) > 0.001) return false;
            if (Math.abs(this.getRadius() - ((Circle) obj).getRadius()) > 0.001) return false;
            return true;
        }
        else return false;
    }

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape s){
        shapes.add(s);
    }

    public void removeCircles(){
        for (int i=0; i<shapes.size(); i++){
            if (shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            }
        }
    
}

    public void removeDuplicates(){
        for (int i=0; i<shapes.size()-1; i++)
            for (int j=i+1; j<shapes.size(); j++)
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
    }

    public String getInfo(){
        String s = 
"Layer of crazy shapes: \n";
        for (int i=0; i<shapes.size(); i++){
            s = s + shapes.get(i).toString() + "\n";
        }
        return s;
    }

public class Point {
    private double x;
    private double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point p){
        double x_ = this.x - p.x;
        double y_ = this.y - p.y;
        
return Math.sqrt(x_*x_ + y_*y_);
    }

    public boolean equals(Object obj){
        if (obj instanceof Point){
            if (this.getX() != ((Point) obj).getX()) return false;
            if (this.getY() != ((Point) obj).getY()) return false;
            return true;
        }
        else return false;
    }

//    public int hashCode(){
//
//    }

    public String toString(){
        return "(" + this.getX() + "," + this.getY() + ")";
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.length = length;
        this.width = width;
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getArea(){
        return this.getWidth()*this.getLength();
    }

    public double getPerimeter(){
        return 2*(this.getLength()+this.getWidth());
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public boolean equals(Object obj){
        if (obj instanceof Rectangle){
            if (this.getTopLeft().distance((
((Rectangle) obj).getTopLeft())) > 0.001) return false;
            if (Math.abs(this.getWidth() - ((Rectangle) obj).getWidth()) > 0.001) return false;
            if (Math.abs(this.getLength() - ((Rectangle) obj).getLength()) > 0.001) return false;
            return true;
        }
        else return false;
    }

//    public int hashCode(){
//    }

    public String toString() {
        double topLeftX = ((double) Math.round(this.getTopLeft().getX()*10))/10;
        double topLeftY = ((double) Math.round(this.getTopLeft().getY()*10))/10;
        double width = ((double) Math.round(this.getWidth()*10))/10;
        double length = ((double) Math.round(this.getLength()*10))/10;
        return "Rectangle[topLeft=(" + topLeftX + "," + topLeftY + "),width=" + width + ",length=" + length +
                ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }

public class Square extends Rectangle{
    public Square(){}

    public Square (double side){
        super(side, side);
    }

    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled){
        super(topLeft, side, side, color, filled);
    }

    public double getSide() {
        return super.getLength();
    }

    public void setSide(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    public void setWidth(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    public void setLength(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    public boolean equals(Object obj){
        if (obj instanceof Square){
            if (
this.getTopLeft().distance((((Square) obj).getTopLeft())) > 0.001) return false;
            if (Math.abs(this.getSide() - ((Square) obj).getSide()) > 0.001) return false;
            return true;
        }
        else return false;
    }

    public String toString(){

        double topLeftY = ((double) Math.round(this.getTopLeft().getY()*10))/10;
        double side = (double) Math.round(this.getSide()*10)/10;
        return "Square[topLeft=(" + topLeftX + "," + topLeftY + "),side=" +
               side + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
