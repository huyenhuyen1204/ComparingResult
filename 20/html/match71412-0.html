Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){
        center = new Point(0,0);
    }
    public Circle(double radius){
        this.radius=radius;
        center = new Point(0,0);
    }
    public Circle(double radius, String color, boolean filled){
        super(color,filled);
        this.radius=radius;
        center = new Point(0,0);
    }
    public Circle(Point point, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        center = new Point(point.getX(),point.getY());
    }
    public Point getCenter(){
        return center;
    }
    public void setCenter(Point point){
        center.setY(point.getY());
        center.setX(point.getX());
    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius){
        this.radius=radius;
    }
    @Override
    public double getArea(){
        return PI*radius*radius;
    }
    public double getPerimeter(){
        return 2*PI*radius;
    }
    public String toString(){
        return "Circle[center=" + center.toString() + ",radius=" + this.getRadius() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
    //more
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle cir = (Circle) obj;
            if(Math.abs(cir.getArea()-this.getArea())<=0.001 && Math.abs(cir.getPerimeter()-this.getPerimeter())<=0.001){
                return true;
            }
        }
        return false;
    }
    public int hashCode(){
        return (int) ((int) radius*radius+5);
    }
}

Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){
        topLeft = new Point(0,0);
    }
    public Rectangle(double width, double length){
        this.width=width;
        this.length=length;
        topLeft = new Point(0,0);
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color,filled);
        this.width=width;
        this.length=length;
        topLeft = new Point(0,0);
    }
    public Rectangle(Point point, double width, double length, String color, boolean filled){
        super(color,filled);
        topLeft = new Point(point.getX(),point.getY());
        this.width=width;
        this.length=length;
    }
    public double getWidth(){
        return width;
    }
    public double getLength(){
        return length;
    }
    public void setWidth(double width){
        this.width = width;
    }
    public void setLength(double length){
        this.length = length;
    }
    public double getArea(){
        return width*length;
    }
    public double getPerimeter(){
        return 2*(width+length);
    }
    public String toString(){
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + this.getWidth() + ",length=" + this.getLength() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
    public Point getTopLeft(){
        return topLeft;
    }
    public void setTopLeft(Point point){
        topLeft.setX(point.getX());
        topLeft.setY(point.getY());
    }
    public boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle rec = (Rectangle) obj;
            if(Math.abs(this.getArea()-rec.getArea())<=0.001 && Math.abs(this.getPerimeter()-rec.getPerimeter())<=0.001){
                return true;
            }
        }
        return false;
    }
    public int hashCode(){
        return (int) ((int) width*length+1);
    }
    public static void main(String[] args){
        Rectangle rec = new Rectangle(10,5);
        System.out.printf(rec.toString());
        Point x = new Point(1.0,2.0);
        Rectangle rec_1 = new Rectangle(x,10,5,"RED",true);
        System.out.printf(rec_1.toString());
    }
}

public class Square extends Rectangle{

    public Square(){
        topLeft = new Point(0,0);
    }
    public Square(double side){
        topLeft = new Point(0,0);
        this.width=side;
        this.length=side;
    }
    public Square(double side,String color,boolean filled){
        topLeft = new Point(0,0);
        this.color=color;
        this.filled=filled;
        this.width=this.length=side;
    }
    public Square(Point point, double side,String color,boolean filled){
        this.color=color;
        this.filled=filled;
        this.width=this.length=side;
        topLeft = new Point(point.getX(),point.getY());
    }
    public double getSide(){
        return this.length;
    }
    public void setSide(double side){
        this.width=this.length=side;
    }
    public void setWidth(double width){
        this.width=this.length=width;
    }
    public void setLength(double length){
        this.length=this.width=length;
    }
    public String toString(){
        return "Square[topLeft=" + topLeft.toString() + ",side=" + this.getSide() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
    public static void main(String[] args){
        Square sq = new Square(2);
        System.out.printf(sq.toString());
    }
    public boolean equals(Object obj){
        if(obj instanceof Square){
            Square sq = (Square) obj;
            if(Math.abs(sq.getArea()-this.getArea())<=0.001 && Math.abs(sq.getPerimeter()-this.getPerimeter())<=0.001)
                return true;
        }
        return false;
    }
    public int hashCode(){
        return (int) ((int)width*length+6);
    }
}
