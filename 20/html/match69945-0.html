public class Circle extends Shape {
    protected double radius ;
    protected Point center ;

    public Circle(){

    }

    public Circle(double radius){
        this.radius = radius ;
    }

    public Circle(double radius , String color , boolean filled){
        this.radius = radius ;
        this.color = color ;
        this.filled = filled ;
    }

    public Circle(Point center , double radius , String color , boolean filled){
        this.center = center ;
        this.radius = radius ;
        this.color = color ;
        this.filled = filled ;
    }

    public Point getCenter(){
        return this.center ;
    }

    public void setCenter(Point center){
        this.center = center ;
    }

    public double getRadius(){
        return this.radius ;
    }

    public void setRadius(double radius){
        this.radius = radius ;
    }

    public double getArea() {
        return Math.PI*radius*radius;
    }

    public double getPerimeter() {
        return 2*Math.PI*radius;
    }

    public boolean equals(Object obj ){
        if(this == obj ){
            return true ;
        }
        if(obj== null||this.getClass()!=obj.getClass()){
            return false ;
        }
        Circle A = (Circle) obj ;
        return this.radius==A.radius &&
                this.center.equals(A.center) ;

    }

    public int hashCode(){
        return 31 ;
    }


    public String toString(){
        return "Circle[center=("+center.getX()+","+center.getY()+"),radius="+radius+",color="+color+",filled="+filled+"]" ;
    }
}

import java.util.ArrayList ;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape shape){
        shapes.add(shape) ;
    }

    public void removeCircles(){
        Circle x = new Circle();
        for (int i = 0 ; i < shapes.size() ; i++) {
            if (shapes.get(i).getClass() == x.getClass()) {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo(){
        String info = "Layer of crazy shapes: \n" ;
        for(int i=0 ; i<shapes.size() ; i++){
            info = info + shapes.get(i).toString()+"\n" ;
        }
        return info ;
    }

    public void removeDuplicates(){
        for (int i = 0 ; i < shapes.size()-1 ; i++) {
            for (int j = i+1 ; j < shapes.size() ; j++) {
                if (shapes.get(i).getClass() == shapes.get(j).getClass() && shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

public class Point {
    private double x ;
    private double y ;

    public Point(double x, double y){
        this.x = x ;
        this.y = y ;
    }

    public double getX(){
        return this.x ;
    }

    public void setX(double x){
        this.x = x ;
    }

    public double getY(){
        return this.y ;
    }

    public void setY(double y){
        this.y = y ;
    }

    public double distance(Point A){
        return Math.abs(x-y) ;
    }

    public boolean equals(Object obj){
        if(this==obj)
            return true ;
        if(obj==null || this.getClass()!=obj.getClass())
            return false ;
        Point point = (Point) obj ;
        return (this.x +0.001<= point.x && this.x>=point.x+0.001) && (this.y+0.001 <= point.y&& this.y>= point.x+0.001);
    }

    public int hashCode(){
        return 33 ;
    }

    public String toString(){
        return "("+x +","+y +")" ;
    }

}
