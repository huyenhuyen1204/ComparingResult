Circle.java

import java.awt.*;

public class Circle extends Shape
{
    protected Point center;
    protected double radius;

    public Circle () {};

    public Circle (double radius)
    {
        this.radius = radius;
    }

    public Circle (double radius, String color, boolean filled)
    {
        super(color, filled);
        this.radius = radius;
    }

    public Circle (Point center, double radius, String color, boolean filled)
    {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius()
    {
        return radius;
    }

    public void setRadius(double radius)
    {
        this.radius = radius;
    }

    @Override
    public double getArea()
    {
        double Area = Math.PI * radius * radius;
        return Area;
    }

    @Override
    public double getPerimeter()
    {
        double Perimeter = 2 * Math.PI * radius;
        return Perimeter;
    }

    @Override
    public boolean equals(Object obj)
    {
        Circle C = (Circle) obj;
        if(Math.abs(this.radius - C.radius) > 0.001)
            return false;
        if(!this.center.equals(C.center))
            return false;
        return true;
    }

    @Override
    public String toString() {
        return "Circle[center=(" + center.getX() + "," + center.getY() + "),radius=" + getRadius() + ",color=" + super.color + ",filled=" + super.filled + "]";
    }

}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes =new ArrayList<>();
    public void addShape(Shape s)
    {
        shapes.add(s);
    }

    public void removeCircles(){
        int n = shapes.size();
        for (int i = 0;i < n;i++)
            if (shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
                n--;
            }
    }

    public String getInfo(){
        String s="Layer of crazy shapes: \n";
            for (int i = 0; i < shapes.size()-1; i++)
                s = s + shapes.get(i).toString() + "\n";
            s = s + shapes.get(shapes.size()-1).toString();
        return s;
    }
    public void removeDuplicates(){
        for(int i = 0; i < shapes.size() - 1; i++){
            for(int j = i + 1; j < shapes.size(); j++){
                if((shapes.get(i) instanceof Circle) && (shapes.get(j) instanceof Circle)){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if((shapes.get(i) instanceof Rectangle) && (shapes.get(j) instanceof Rectangle)){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if((shapes.get(i) instanceof Square) && (shapes.get(j) instanceof Square)){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
            }

        }
    }
    public static void main(String[] args) {
        Point a1 = new Point(10.5,5.0);
        Point a2 = new Point(8.5,4.0);
        Shape a= new Circle(a1,8.0,"RED",true);
        Shape b= new Circle(a2,8.0,"RED",true);
        Shape c= new Rectangle(a1, 8.0, 6.0,"RED",true);
        Shape d= new Square(a1,8.0,"RED",true);
        List<Shape> list = new ArrayList<Shape>();
        Layer lay= new Layer();
        lay.addShape(a);
        lay.addShape(b);
        lay.addShape(c);
        lay.addShape(d);
        lay.removeDuplicates();
        lay.removeCircles();
        System.out.println(((Circle) a).getCenter().distance(((Circle)b).getCenter()));
        System.out.println(lay.getInfo());

    }
}

import java.awt.*;

public class Square extends Rectangle
{
    public Square ()
    {
        super();
    };

    public Square (double side)
    {
        super(side, side);
    }

    public Square (double side, String color, boolean filled)
    {
        super(side, side, color, filled);
    }

    public Square (Point topLeft, double side, String color, boolean filled)
    {
        super(topLeft, side, side, color, filled);
    }

    public double getSide()
    {
        return super.length;
    }

    public void setSide(double side)
    {
        super.width = side;
        super.length = side;
    }

    @Override
    public void setWidth(double side) {
        super.width = side;
        super.length = side;
    }

    @Override
    public void setLength(double side) {
        super.width = side;
        super.length = side;
    }

    @Override
    public boolean equals(Object obj)
    {
        Square S = (Square) obj;
        if(Math.abs(this.getSide() - S.getSide()) > 0.001)
            return false;
        if(!this.topLeft.equals(S.topLeft))
            return false;
        return true;
    }

    @Override
    public String toString() {
        return "Square[topLeft=(" + topLeft.getX() + "," + topLeft.getY() + "),side=" + super.length + ",color=" + super.color + ",filled=" + super.filled + "]";
    }
}
