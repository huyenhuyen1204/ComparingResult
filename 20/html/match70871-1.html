class Circle extends Shape {

    protected Point center;
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }
    public double getRadius() {
        return this.radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }
    public void setCenter(Point center) {
        this.center = center;
    }
    public Point getCenter() {
        return center;
    }
    public int hashCode() {
        return 1;
    }
    public 
boolean equals(Object ss) {
        if(ss instanceof Circle){
            Circle cc = (Circle) ss;
            if(this.center.equals(cc.center) && this.radius == cc.getRadius()) return true;
        }
        return false;
    }

    public double getArea(){
        return this.radius*this.radius*Math.PI;
    }
    public double getPerimeter(){
        return this.radius*2*Math.PI;
    }
    public String toString(){
        double re1 = (double)Math.round(this.radius*10)/10;
        return "Circle[center="+this.center.toString()+",radius="+re1+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape sp){
        shapes.add(sp);
    }
    public void removeCircles(){
        int vt = shapes.size();
        int i = 0;
        while(i<vt){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
                vt-=1;
            }
            else i+=1;
        }
        return;
    
}
    public String getInfo(){
        String re = "Layer of crazy shapes: \n";
        for(int i=0; i<shapes.size()-1; i++){
            re += shapes.get(i).toString()+"\n";
        }
        re += shapes.get(shapes.size()-1).toString();
        
return re;
    }
    public void removeDuplicates(){
       for(int i= shapes.size()-1; i>=0; i--){
           for(int j = shapes.size()-1; j>i; j--){
               if(shapes.get(i).equals(shapes.get(j))){
                   shapes.remove(j);
               }
           }
        }
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {

    }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }
    public 
double getWidth() {
        return this.width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    
}

    public boolean equals(Object ss){
        if(ss instanceof Rectangle){
            Rectangle rr = (Rectangle) ss;
            if(this.topLeft.equals(rr.topLeft) && this.length == rr.getLength() && this.width == rr.getWidth()) return true;
        }
        return false;
    }

    public int hashCode() {
        return 1;
    }

    public double getArea() {
        return this.width * this.length;
    }

    public double getPerimeter() {
        return 2 * (this.width + this.length);
    }

    public String toString() {
        double re1 = (double)Math.round(this.width*10)/10;
        double re2 = (double)Math.round(this.length*10)/10;
        return "Rectangle[topLeft="+this.topLeft.toString()+",width=" + re1 + ",length=" + re2 + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
}
