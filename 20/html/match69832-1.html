import java.util.Objects;
public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle() {

    }
    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }
    public Point getCenter() {

        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {

        return this.radius;
    }

    public void setRadius(double radius) {

        this.radius = radius;
    }
    @Override
    public double getArea() {
        return Math.PI*this.radius*this.radius;
    }

    @Override
    public double getPerimeter() {

        return 2*this.radius*Math.PI;
    }
    @Override
    public boolean equals(Object p) {
        if(p instanceof Circle == false) return false;
        Circle that = (Circle) p;
        return this.center.equals(that.center) && Math.abs(this.radius - that.radius) <= 0.001;
    }
    @Override
    public int hashCode() {
        return Objects.hash(radius, center);
    }
    @Override
    public String toString() {
        return "Circle[" +
                "center=" + this.center +
                ",radius=" + (double) Math.round(this.radius*10)/10 +
                ",color=" + this.color +
                ",filled=" + this.filled +
                "]";
    }
}

import java.util.ArrayList;
import java.util.List;
import java.util.HashSet;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();
    public void addShape(Shape shape) {

        shapes.add(shape);
    }

    public void removeCircles() {
        for(int i=0;i<shapes.size();i++) {
            if(
shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo() {
        String out = "Layer of crazy shapes: " + "\n";
        for(int i=0;i<shapes.size()-1;i++) {
            out+=shapes.get(i).toString() + "\n";
        }
        out += shapes.get(shapes.size()-1).toString();
        return out;
    }
    public void removeDuplicates() {
        HashSet<Shape> set = new HashSet<>();
        
for(int i=0;i<shapes.size();i++) {
            if(!set.contains(shapes.get(i)))
            set.add(shapes.get(i));
            else {
                shapes.remove(i);
                i--;
            }
        }
    }
}
