import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes= new ArrayList<Shape>();
    public void addShape(Shape _shape){
        this.shapes.add(_shape);
    }
    public void removeCircles(){
        for(int i=0;i<shapes.size();++i){
            if(shapes.get(i) instanceof Circle){
                this.shapes.remove(shapes.get(i));
                i--;
            }
        }
    }
    public String getInfo(){
        if(this.shapes.isEmpty()){
            return "";
        }
        else {
            String s="";
            s="Layer of crazy shapes: \n";
            for(Shape i: shapes){
                s=s+i.toString()+"\n";
            }
            return s;
        }
    
}
    public void removeDuplicates(){
        for(int i=0;i< shapes.size();++i){
            for(int j=i+1;j<shapes.size();++j){
                if(i!=j&&(shapes.get(i) instanceof Circle&& shapes.get(j) instanceof  Circle)){
                    Circle c1= (Circle) shapes.get(i);
                    Circle c2= (Circle) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
                if(i!=j&&(shapes.get(i) instanceof Rectangle&& shapes.get(j) instanceof  Rectangle)){
                    Rectangle c1= (Rectangle) shapes.get(i);
                    Rectangle c2= (Rectangle) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
                if(i!=j&&(shapes.get(i) instanceof Square&& shapes.get(j) instanceof  Square)){
                    Square c1= (Square) shapes.get(i);
                    Square c2= (Square) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
            }
        }
    
}

    public static void main(String[] args) {
        Point a1= new Point(10.0, 5.0);

        Shape a= new Circle(a1,8.0,"RED",true);
        Shape c= new Circle(a1,8.0,"RED",true);
        Shape b= new Rectangle(a1, 8.0, 6.0,"RED",true);
        Shape d= new Square(a1,8.0,"RED",true);
        List<Shape> list = new ArrayList<Shape>();

lay= new Layer();

}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){

    }
    public Rectangle(double _width, double _length){
        this.width=_width;
        this.length= _length;
    }
    public Rectangle( double _width, double _length, String _color, boolean _filled){
        super(_color, _filled);
        this.width=_width;
        this.length=_length;

    }
    public Rectangle(Point _topLeft,double _width, double _length, String _color, boolean _filled){
        super(_color, _filled);
        this.width=_width;
        this.length=_length;
        this.topLeft=_topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }

    public double getLength() {
        return length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public void setLength(double length) {
        this.length = length;
    }
    public double getArea(){
        return this.getWidth()* this.getLength();
    }
    public double getPerimeter(){
        return 2*(this.getWidth()+this.getLength());
    }
    public double round(double a){
        a=Math.round(a*10);
        return a/10;
    }
    public String toString(){
        return "Rectangle[topLeft=("+this.round(this.getTopLeft().getX())+","+this.round(this.getTopLeft().getY())+"),width="+this.round(getWidth())+",length="+this.round(getLength())+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }
    public boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle 
a= (Rectangle) obj;
            if(Math.abs(a.getLength()-this.getLength())<0.001 && a.getTopLeft().equals(this.getTopLeft())&&Math.abs(a.getWidth()-this.getWidth())<0.001&&a.isFilled()==this.isFilled()){
                return true;
            }
        }
        return false;
    }
    public boolean Duplicates(Object obj){
        if(obj instanceof Rectangle){
            Rectangle a= (Rectangle) obj;
            if(Math.abs(a.getLength()-this.getLength())<0.001 && 
a.getTopLeft().equals(this.getTopLeft())&&Math.abs(a.getWidth()-this.getWidth())<0.001){
                return true;
            }
        }
        return false;
    }
}
