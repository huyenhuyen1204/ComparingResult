import java.util.ArrayList;
import java.util.Iterator;
import java.util.LinkedHashSet;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape){
        this.shapes.add(shape);
    }

    public void removeCircles(){
        for (int i=0;i<shapes.size();i++){
            if (shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }

    public String getInfo(){
        String s = new String();
        s += ("Layer of crazy shapes:" +"\n");
        for (Shape sp : shapes){
            if (sp instanceof Circle) {
                sp = (Circle) sp;
                s += (sp.toString() + "\n");
            }
            else if (sp instanceof Rectangle){
                sp = (Rectangle)sp;
                s += (sp.toString()+"\n");
            }
            else if (sp instanceof Square){
                sp = (Square)sp;
                s += (sp.toString()+"\n");
            }
        }
        return s;
    }

    public void removeDuplicates(){
        for (int i =0;i<shapes.size();i++){
            for (int j=i+1;j<shapes.size();j++){
                if (shapes.get(i).equals(shapes.get(j)))
                    if (shapes.get(i).hashCode() == shapes.get(j).hashCode())
                        shapes.remove(j);
            }
        }
    }

    public static void main(String[] args){
        Layer list = new Layer();
        Point p = new Point(1,2);
        Point p1 = new Point(2,3);
        Point p2 = new Point(3,4);
        Circle c1 = new Circle(p,5.,"red",true);
        Circle c3 = new Circle(p,5.,"green",true);
        Circle c2 = new Circle(p1,5,"red",false);
        Square s1 = new Square(p1,5.0,"red",true);
        Square s2 = new Square(p1,5.0,"red",true);
        Rectangle r1 = new Rectangle(p2,3.0,4.0,"red",true);
        list.addShape(c1);
        list.addShape(c3);
        list.addShape(c2);
        list.addShape(s1);
        list.addShape(r1);
        list.addShape(s2);
        String s = new String();
        list.removeDuplicates();
        s = list.getInfo();
        System.out.print(s);
    }
}

Point.java



public class Point {
    private double x;
    private double y;

    public Point(double x,double y){
        this.x = x;
        this.y = y;
    }

    public double getX(){return this.x;}
    public void setX(double x){
        this.x = x;
    }

    public double getY(){return this.y;}
    public void setY(double y){
        this.y = y;
    }

    public double distance(Point other){
        double d;
        d = Math.sqrt(Math.pow(x-other.x,2.) + Math.pow(y-other.y,2.));
        d = Math.round((d*10)/10);
        return d;
    }

    public boolean equals(Object obj){
        if (obj instanceof Point){
            if ((this.x == ((Point) obj).getX()) && (this.y == ((Point) obj).getY())) return true;
            else return false;
        }
        else return false;
    }

    public int hashCode(){
        return (int)(this.x*this.y +1);
    }

    public String toString(){
        String s = new String();
        s += ("("+this.x+","+this.y+")");
        return s;
    }
}

Square.java

public class Square extends Rectangle {
    public Square(){}

    public Square(double side){
        this.length = side;
        this.width = side;
    }

    public Square(double side,String color,boolean filled){
        this.width = side;
        this.length = side;
        this.color = color;
        this.filled = filled;
    }

    public Square(Point p,double side,String color,boolean filled){
        this.topLeft = p;
        this.width = side;
        this.length = side;
        this.color = color;
        this.filled = filled;
    }

    public double getSide(){return this.length;}

    public void setSide(double side){
        this.length = side;
        this.width = side;
    }

    public void setWidth(double side){
        this.width = side;
        this.length = side;
    }

    public void setLength(double side){
        this.length = side;
        this.width = side;
    }

    public String toString(){
        String s = new String();
        s += ("Square[topLeft=("+this.topLeft.getX()+","+this.topLeft.getY()+"),side="+this.length+",color="+this.color+",filled="+this.filled+"]");
        return s;
    }

    public boolean equals(Object obj){
        if (obj instanceof Square){
            if (this.getSide() == ((Square) obj).getSide()) return true;
            else return false;
        }
        else return false;
    }

    public int hashCode(){
        return (int)(this.topLeft.hashCode()*this.getSide()+1);
    }

}
