Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;

    public Circle() { }

    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = new Point(center.getX(), center.getY());
    }

    public Point getCenter() {
        return this.center;
    }
    public void setCenter(Point Center) {
        center.setX(Center.getX());
        center.setY(Center.getY());
    }
    public double getRadius() {
        return this.radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return ((double)Math.round(Math.PI * radius * radius * 100)) / 100;
    }
    @Override
    public double getPerimeter() {
        return ((double)Math.round(2 * Math.PI * radius * 100)) / 100;
    }

    public boolean equals(Object obj) {
        if (obj == null || !(obj instanceof Circle))
            return false;
        if (obj == this) return true;
        Circle temp = (Circle)obj;
        boolean checkR = (Math.abs(radius - temp.radius) < 0.001);
        return (checkR && this.center.equals(temp.center));
    }
    public int hasdCode() {
        int res = Double.hashCode(radius);
        res = 31 * res + (center == null ? 0 : center.hashCode());
        return res;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + (double)Math.round(this.radius * 10) / 10
                + ",color=" + super.color + ",filled=" + super.filled + "]";
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();
    public void addShape(Shape shape) {
        shapes.add(shape);
    }
    public void removeCircles() {
        for(int i = 0 ; i < shapes.size(); i++) {
            if(shapes.get(i) instanceof Circle)
                shapes.remove(i--);
        }
    }
    public String getInfo() {
        String res = "Layer of crazy shapes: \n";
        int n = shapes.size();
        for (int i = 0; i < n; i++) {
            res += shapes.get(i).toString();
            if (i != n-1) res += "\n";
        }
        return res;
    }
    public void removeDuplicates() {
        for (int i = 0; i < shapes.size() - 1; i++)
            for (int j = i + 1; j < shapes.size(); j++)
                if (shapes.get(i).equals(shapes.get(j)))
                    shapes.remove(j--);
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }
    public void setX(double x) {
        this.x = x;
    }
    public double getY() {
        return this.y;
    }
    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point a) {
        return Math.sqrt((this.x - a.getX()) * (this.x - a.getX()) + (this.y - a.getY()) * (this.y - a.getY()));
    }
    public boolean equals(Object obj) {
        if (obj == null || !(obj instanceof Point))
            return false;
        if (obj == this) return true;
        Point temp = (Point)obj;
        boolean checkX = (Math.abs(x - temp.getX()) < 0.001);
        boolean checkY = (Math.abs(y - temp.getY()) < 0.001);
        return (checkX && checkY);
    }
    public int hashCode() {
        int val = Double.hashCode(x);
        val = 31 * val + Double.hashCode(y);
        return val;
    }
    public String toString() {
        return "(" + this.x + "," + this.y + ")";
    }
}

Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() { }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = new Point (topLeft.getX(), topLeft.getY());
    }

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return ((double)Math.round(width * length * 100)) / 100;
    }

    @Override
    public double getPerimeter() {
        return ((double)Math.round(2 * (width + length) * 100)) / 100;
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topleft) {
        topLeft.setX(topleft.getX());
        topLeft.setY(topleft.getY());
    }

    public boolean equals(Object obj) {
        if (obj == null || !(obj instanceof Rectangle))
            return false;
        if (obj == this) return true;
        Rectangle temp = (Rectangle)obj;
        boolean checkW = (Math.abs(width - temp.width) < 0.001);
        boolean checkL = (Math.abs(length - temp.length) < 0.001);
        return (checkW && checkL && topLeft.equals(temp.getTopLeft()));
    }

    public int hashCode() {
        int res = Double.hashCode(width);
        res = 31 * res + Double.hashCode(length);
        res = 31 * res + (topLeft == null ? 0 : topLeft.hashCode());
        return res;
    }

    @Override
    public String toString(){
        double Width = (double)(Math.round(width * 10)) / 10;
        double Length = (double)(Math.round(length * 10)) / 10;

        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + Width + ",length="
                + Length + ",color=" + color + ",filled=" + filled + "]";
    }
}

public class Square extends Rectangle{
    public Square(){ }

    public Square(double side){
        super(side, side);
    }

    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled) {
        super(topLeft, side, side, color, filled);
    }

    public double getSide(){
        return super.getLength();
    }

    public void setSide(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public void setWidth(double side){
        super.setWidth(side);
        super.setLength(side);
    }

    @Override
    public void setLength(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    public boolean equals(Object obj) {
        if (obj == null || !(obj instanceof Square))
            return false;
        if (obj == this) return true;
        Square temp = (Square) obj;
        boolean checkW = (Math.abs(width - temp.width) < 0.001);
        boolean checkL = (Math.abs(length - temp.length) < 0.001);
        return (checkW && checkL && topLeft.equals(temp.getTopLeft()));
    }
    public int hashCode() {
        return (super.hashCode());
    }

    @Override
    public String toString(){
        return "Square[topLeft=" + super.topLeft.toString() + ",side=" + getSide()
                + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
}
