class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle(){}

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea(){
        return Math.pow(this.radius, 2) * Math.PI;
    }

    public double getPerimeter(){
        return Math.PI * 2 * this.radius;
    }

    @Override
    public 
boolean equals(Object o){
        if(o instanceof Circle){
            Circle temp = (Circle) o;
            if(this.center.equals(temp.center)){
                return (this.radius == temp.radius);
            }
            else return false;
        }
        else return false;
    }

    @Override
    public String toString(){
        String circleString = "Circle[center=" + this.center.toString() + ",radius=" + Double.toString(Math.round(this.radius * 10) / 10) + ",color=" + this.color + ",filled=" + this.filled + "]";
        return circleString;
    }
}

import java.util.HashSet;
import java.util.List;

public class Layer {
    private List<Shape> shapes;

    public void addShape(Shape shape){
        shapes.add(shape);
    }

    public void removeCircles(){
        for(Shape shape : shapes){
            
if(shape.getClass() == Circle.class){
                shapes.remove(shape);
            }
        }
    }

    public String getInfo(){
        String info = "Layer of crazy shapes:\n";
        for(Shape shape : shapes){
            info += shape.toString() + "\n";
        }
        return info;
    }

    public void removeDuplicates(){
        HashSet<Shape> set = new HashSet<Shape>();

        for(Shape shape : shapes){
            set.add(shape);
        }

        this.shapes = (List<Shape>) set;
    }
}

public class Point {
    private double x;
    private double y;

    public 
Point( double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public double getY() {
        return this.y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point point){
        double distance = Math.sqrt(Math.pow(this.x - point.getX(), 2) +Math.pow(this.y - point.getY(), 2));
        return distance;
    }

    @Override
    public boolean equals(Object o){
        if(o instanceof Point){
            Point temp = (Point) o;
            return (this.x == temp.getX() && this.y == temp.getY());
        }
        else return false;
    }


    @Override
    public String toString(){
        String ptString = "(" +Double.toString(Math.round(this.x *10) / 10) + "," + Double.toString(Math.round(this.y * 10) / 10) + ")";
        return ptString;
    }
}

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.length = length;
        this.width = width;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.length = length;
        this.width = width;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.length = length;
        this.width = width;
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getLength() {
        return this.length;
    }

    public double getWidth() {
        return this.width;
    }

    public void setLength(double length) {
        this.length  = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getArea(){
        return this.width * this.length;
    }

    public double getPerimeter(){
        return (this.length + this.width) * 2;
    }

    @Override
    public boolean equals(Object o){
        if(o instanceof Rectangle){
            Rectangle temp = (Rectangle) o;
            if(
this.topLeft.equals(temp.topLeft)){
                return (this.width == temp.width && this.length == temp.length);
            }
            else return false;
        }
        else return false;
    }

    @Override
    public String toString(){
        String rectString = "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + Double.toString(Math.round(this.width * 10) / 10) + ",length=" + Double.toString(Math.round(this.length * 10) / 10) + ",color=" + this.color + ",filled=" +Boolean.toString(this.filled) + "]";
        return rectString;
    }
}
