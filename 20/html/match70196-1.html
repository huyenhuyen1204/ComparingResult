Circle.java

public class Circle extends Shape{
    protected Point center;
    protected double radius;
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }
    public Point getCenter() {
        return center;
    }
    public void setCenter(Point center) {
        this.center = center;
    }
    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle() {
    }
    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }
    //Override
    public double getArea() {
        return ( Math.PI * radius * radius );
    }
    //Override
    public double getPerimeter() {
        return ( 2 * Math.PI * radius );
    }
    //Override
    public String toString() {
        return "Circle[center=" + this.center.toString() + ",radius=" + this.radius + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
    public int hashCode(){
        return -1;
    }
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle s = (Circle) obj;
            if(
this.center.getX() == s.getCenter().getX() && this.center.getY() == s.getCenter().getY()){
                return true;
            }
        }
        return false;
    }
}

import java.util.ArrayList;
import java.util.List;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();
    public void addShape(Shape shape){
        shapes.add(shape);
    }
    public void removeCircles(){
        for (int i = 0; i < this.shapes.size(); i++) {
            
if(this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
            }
        }
    }
    public String getInfo(){
        String infro = "Layer of crazy shapes:\n";
        for (Shape shape : shapes) {
            if(shape instanceof Circle){
                Circle s = (Circle) shape;
                infro += s.toString() + "\n";
            }else if(shape instanceof Rectangle){
                Rectangle s = (Rectangle) shape;
                infro += s.toString() + "\n";
            }else if(shape instanceof Square){
                Square s = (Square) shape;
                infro += s.toString() + "\n";
            }
        
}
        return infro;
    }
    public void removeDuplicates(){
        for (int i = 0; i < this.shapes.size() - 1; i++) {
            for (int j = i + 1; j < this.shapes.size(); j++) {
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle){
                    Circle s1 = (Circle) shapes.get(i);
                    Circle s2 = (Circle) shapes.get(j);
                    if(s1.getRadius() == s2.getRadius() && s1.getCenter().equals(s2)){
                        shapes.remove(j);
                    }
                }else if(shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle){
                    Rectangle s1 = (Rectangle) shapes.get(i);
                    Rectangle s2 = (Rectangle) shapes.get(j);
                    if(s1.getWidth()== s2.getWidth() && s1.getLength()== s2.getLength() && s1.getTopLeft().equals(s2)){
                        shapes.remove(j);
                    }
                }else if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Square){
                    Square s1 = (Square) shapes.get(i);
                    Square s2 = (Square) shapes.get(j);
                    if(s1.getWidth()== s2.getWidth() && s1.getLength()== s2.getLength() && s1.getTopLeft().equals(s2)){
                        shapes.remove(j);
                    }
                }
            }
        }
    }
}

public class Point {
    private double x = 1.0;
    private double y = 1.0;
    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return x;
    }
    public void setX(double x) {
        this.x = x;
    }
    public double getY() {
        return y;
    }
    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p){
        double a = Math.abs( this.x - p.getX() );
        double b = Math.abs( this.y - p.getY() );
        return Math.pow( a*a + b*b, 1/2 );
    }
    public 
boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle s = (Circle) obj;
            if( this.getX() == s.getCenter().getX() && this.getY() == s.getCenter().getY() ){
                return true;
            }
        }else if(obj instanceof Rectangle){
            Rectangle s = (Rectangle) obj;
            if( this.getX() == s.getTopLeft().getX() && this.getY() == s.getTopLeft().getY() ){
                return true;
            }
        }else if(obj instanceof Square){
            Square s = (Square) obj;
            if( this.getX() == s.getTopLeft().getX() && this.getY() == s.getTopLeft().getY() ){
                return true;
            }
        }
        return false;
    }
    public int hashCode(){
        return -1;
    }
    public String toString(){
        return "(" + this.x + "," + this.y + ")";
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){}
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }
    public Point getTopLeft() {  return topLeft; }

    public void setTopLeft(Point topLeft) {  this.topLeft = topLeft; }
    public double getWidth() {
        return width;
    }
    public double getLength() {
        return length;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public void setLength(double length) {
        this.length = length;
    }
    //Override
    public double getArea() {
        return ( width * length );
    }
    //Override
    public double getPerimeter() {
        return ( ( width + length ) * 2 );
    }
    //Override
    public String toString() {
        return "Rectangle[topLeft=" + this.topLeft.toString() +  "," + "width=" + this.width + ",length=" + this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
    public 
int hashCode(){
        return -1;
    }
    public boolean equals(Object obj){
        if(obj instanceof Rectangle || obj instanceof Square){
            Rectangle s = (Rectangle) obj;
            if(this.topLeft.getX() == s.getTopLeft().getX() && this.topLeft.getY() == s.getTopLeft().getY()){
                return true;
            }
        }
        return false;
    }
}
