import static java.lang.Math.PI;

public class Circle extends Shape {
    protected Point center;
    protected double radius;

    public Circle() {
        super();
        
this.center = new Point();
        this.radius = 0.0;
    }

    public Circle(double radius) {
        super();
        this.center = new Point();
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.center = new Point();
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public void setCenter(Point center) { this.center = center; }
    public void setRadius(double radius) { this.radius = radius; }

    public Point getCenter() { return center; }
    public double getRadius() { return radius; }

    @Override
    public double getArea() { 
return PI * PI * radius; }

    @Override
    public double getPerimeter() { return 2 * PI * radius; }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Circle) {
            final double EPS = 0.001;

            Circle c = (Circle) obj;

            return center.equals(c.center) && Math.abs(radius - c.radius) < EPS;
        }

        return false;
    
}

    @Override
    public int hashCode() {
        return (center.hashCode() + (int) radius) % 420;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center.toString()
                    + ",radius=" + String.format("%.1f", radius)
                    + ",color=" + color
                    + ",filled=" + Boolean.toString(filled) + "]";
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes;

    public Layer() { shapes = new ArrayList<>(); }

    public void addShape(Shape shape) { shapes.add(shape); }

    public void removeCircles() {
        for (int i = shapes.size() - 1; i >= 0; --i) {
            if (shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    
}

    public void removeDuplicates() {
        Set<Shape> set = new LinkedHashSet<>(shapes);

        shapes = null;
        shapes = new ArrayList<>(set);
    }

    public String getInfo() {
        StringBuilder res = new StringBuilder("Layer of crazy shapes: \n");

        for (Shape shape : shapes) {
            res.append(shape.toString() + "\n");
        }

        return res.toString();
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle() {
        super();
        this.topLeft = new Point();
        
this.width = 0.0;
        this.length = 0.0;
    }

    public Rectangle(double width, double length) {
        super();
        this.topLeft = new Point();
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = new Point();
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public void setWidth(double width) { this.width = width; 
}
    public void setLength(double length) { this.length = length; }
    public void setTopLeft(Point topLeft) { this.topLeft = topLeft; }

    public double getWidth() { return width; 
}
    public double getLength() { return length; }
    public Point getTopLeft() { return topLeft; }

    @Override
    public double getArea() { return width * length; }

    @Override
    public double getPerimeter() { return 2 * (width + length); }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle) {
            final double EPS = 0.001;

            Rectangle r = (Rectangle) obj;

            return topLeft.equals(r.topLeft) && Math.abs(width - r.width) < EPS && Math.abs(length - r.length) < EPS;
        }

        return false;
    
}

    @Override
    public int hashCode() {
        return topLeft.hashCode() * (int) (width * length) % 420;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString()
                        + ",width=" + String.format("%.1f", width)
                        + ",length=" + String.format("%.1f", length)
                        + ",color=" + color
                        + ",filled=" + Boolean.toString(filled) + "]";
    }
}

Shape.java

public abstract class Shape {
    protected String color;
    protected boolean filled;

    public Shape() {
        this.color = null;
        this.filled = false;
    
}

    public Shape(String color, boolean fiiled) {
        this.color = color;
        this.filled = fiiled;
    }

    public void setColor(String color) { this.color = color; }
    public void setFilled(boolean filled) { this.filled = filled; }

    public String getColor() { return color; }
    public boolean isFilled() { return filled; }

    public abstract double getArea();
    public abstract double getPerimeter();

    @Override
    public String toString() {
        return "Shape[color=" + color + ",filled=" + Boolean.toString(filled) + "]";
    }
}
