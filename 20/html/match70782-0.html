public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle(){
        this.radius = 1.0;
    }

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center,double radius, String color, boolean filled){
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea(){
        return this.radius*this.radius*Math.PI;
    }

    @Override
    public double getPerimeter(){
        return this.radius*2*Math.PI;
    }

    @Override
    public boolean equals(Object O) {
        if (O instanceof Circle){
            if ((this.center == ((Circle) O).center) && (this.radius == ((Circle) O).radius))
                return true;
            else return false;
        }
        else return false;
    }

    @Override
    public String toString(){
        return "Circle[center=(" + this.getCenter().getX() + "," + this.getCenter().getY() +  "),radius="
                + this.radius + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes;

    public Layer(){
        this.shapes =  new ArrayList<Shape>();
    }

    public void addShape(Shape S){
        this.shapes.add(S);
    }

    public String getInfo(){
        String Out = "Layer of crazy shapes: \n" ;
        for (int i=0; i<this.shapes.size(); i++){
            if (this.shapes.get(i) instanceof Circle){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }

            else if (this.shapes.get(i) instanceof Rectangle){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }

            else if (this.shapes.get(i) instanceof Square){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }
        }
        return Out;
    }

    public void removeCircles(){
        for(int i=0; i<this.shapes.size(); i++){
            if (this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);

                if (i==this.shapes.size()-1)
                    this.shapes.remove(i);
            }
        }
    }

    public void removeDuplicates() {
        for (int i = 0; i < this.shapes.size() - 1; i++)
            for (int j = i + 1; j < this.shapes.size(); j++) {
                if (this.shapes.get(i).equals(this.shapes.get(j)))
                        this.shapes.remove(j);
            }
    }
