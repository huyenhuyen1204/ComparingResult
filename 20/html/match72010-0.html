import java.util.*;

public class Layer {
    private List<Shape> shapes = new LinkedList<Shape>() {
    };

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public String getInfo() {
        String str = "Layer of crazy shapes: \n";
        for (Shape element : shapes) {
            str = str + element.toString() + "\n";
        }
        return str;
    }

    public void removeCircles() {
        for (int i = 0; i < shapes.size(); i++) {
            if (shapes.get(i) instanceof Circle) shapes.remove(i);
        }
    }

    public void removeDuplicates() {
        for (int i = 0; i < shapes.size(); i ++) {
            for (int j = i + 1; j < shapes.size(); j ++) {
                if (shapes.get(j).equals(shapes.get(i))) shapes.remove(j);
            }
        }
    }

    public static void main(String[] args) {
        Layer l = new Layer();
        Point p = new Point(3,4);
        Shape c = new Circle(p, 4, "red", true);
        Shape r = new Rectangle(p,5,6,"white", true);
        Shape s = new Square(p, 7.00003, "pink", false);
        Shape c1 = new Circle(p,100,"gray", true);
        Shape s1 = new Square(p,7.00004,"black", true);
        l.addShape(c);
        l.addShape(r);
        l.addShape(s);
        l.addShape(c);
        l.addShape(s1);
        l.addShape(c1);
        l.removeDuplicates();
        System.out.println(l.getInfo());
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other) {
        return Math.round(Math.sqrt((this.x - other.x)*(this.x - other.x) + (this.y - other.y)*(this.y - other.y))*10) /10;
    }

    public boolean equals(Object other) {
        if (other instanceof Point) {
            if ((double) Math.round(this.x * 1000)/1000 == (double) Math.round(((Point) other).x * 1000)/1000
                    && (double) Math.round(this.y * 1000)/1000 == (double) Math.round(((Point) other).y * 1000)/1000)
                return true;
            else return false;
        } else return false;
    }

    public String toString() {
        return "(" + x + "," + y + ")";
    }
}

public class Square extends Rectangle {
    public Square() {

    }

    public Square(double side) {
        this.length = side;
        this.width = side;
    }

    public Square(double side, String color, boolean filled) {
        this.length = side;
        this.width = side;
        this.color = color;
        this.filled = filled;
    }

    public Square(Point topLeft, double side, String color, boolean filled) {
        this.topLeft = topLeft;
        this.length = side;
        this.width = side;
        this.color = color;
        this.filled = filled;
    }

    public double getSide() {
        return length;
    }

    public void setSide(double side) {
        this.length = side;
        this.width = side;
    }

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
    }

    @Override
    public Point getTopLeft() {
        return super.getTopLeft();
    }

    @Override
    public void setTopLeft(Point topLeft) {
        super.setTopLeft(topLeft);
    }

    @Override
    public String toString() {
        return "Square[topLeft=" + topLeft.toString() + ",side=" + length + ",color=" + color + ",filled=" + filled + "]";
    }

    public boolean equals(Object other) {
        if (other instanceof Square) {
            if (Math.abs((double) Math.round(this.length * 1000)/1000 - (double) Math.round(((Square) other).length * 1000)/1000) <= 0.001
                    && (this.topLeft.equals(((Square) other).topLeft)))
            return true;
            else return false;
        } else return false;
    }

    public static void main(String[] args) {
        Point p = new Point(3,5);
        Shape s = new Square(p, 7.0003, "pink", false);
        Shape s1 = new Square(p,7.0004,"black", true);
        System.out.println(s.equals(s1));
        System.err.println(s.toString());
    }
}
