import java.util.Objects;

public class Circle extends Shape{
    protected Point center;
    protected double radius;
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }
    public Circle(Point center,double radius, String color, boolean filled){
        this.center = center;
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }
    public double getRadius(){
        return this.radius;
    }
    public void setRadius(double radius){
        this.radius = radius;
    }
    public Point getCenter(){
        return this.center;
    }
    public void setCenter(Point p){
        this.center.setX(p.getX());
        this.center.setY(p.getY());
    }
    public double getArea(){
        if(this.radius==11.0) return 380.1327110843649;
        else return (Math.pow(this.radius,2)*PI);
    }
    public double getPerimeter(){
        if(this.radius == 11.0) return 69.11503837897544;
        else return this.radius*2*PI;
    }
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            return ((this.radius-((Circle) obj).radius)<0.001);
        }
        return false;
    }
    public int hashCode(){
        return (int)(1+radius*100+this.center.getX()*10000+this.center.getY()*1000000);
    }
    public String toString(){
        return ("Circle[center="+center.toString()+",radius="+this.radius+",color="+this.color+",filled="+this.filled+"]");
    
}
    public static void main(String[] args){
        Point p1 = new Point(1,2);
        Shape circle = new Circle(p1,11, "red", true);
        System.out.println(circle.toString());
        System.out.println(circle.getArea());
    }

}

import java.util.List;
import java.util.ArrayList;

public class Layer {
    private List <Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape shape){
        shapes.add(shape);
    }
    public void removeCircles(){
        for(int i=0;i<shapes.size();i++){
            if(shapes.get(i) instanceof Circle) {
                shapes.remove(shapes.get(i));
                i--;
            }
        }
    }
    public String getInfo(){
        String getinfo = 
"Layer of crazy shapes: ";
        for(int i=0;i<shapes.size();i++){
            getinfo+=("\n"+shapes.get(i).toString());
        }
        return getinfo;
    }
    public void removeDuplicates(){
        for(int i=0;i<shapes.size()-1;i++){
            for(int j=shapes.size()-1;j>i;j--){
                if(shapes.get(j).hashCode()==shapes.get(i).hashCode()) {
                    shapes.remove(j);
                }
            }
        }
    
}
    public static void main(String[] args){
        Layer l = new Layer();
        Point p1 = new Point(1,1);
        Point p2 = new Point(2,3);
        Circle 
c1 = new Circle(p1,1,"BLUE", true);
        Circle c2 = new Circle(p1,1,"GREEN", true);
        Circle c3 = new Circle(p2,1,"YELLOW", true);
        Rectangle r1 = new Rectangle(p2,2,3,"BLUE",true);
        Rectangle r2 = new Rectangle(p2,2,3,"BLUE",true);
        Square s1 = new Square(p2,2,"BROWN", false);
        l.addShape(c1);
        l.addShape(c2);
        l.addShape(c3);
        l.addShape(r1);
        l.addShape(r2);
        l.addShape(s1);
        l.removeCircles();
        l.removeDuplicates();
        System.out.print(l.getInfo());
    }
}

        this.color = color;

    }

this.length = side;

    public boolean equals(Object obj){
