/**
 * Created by CCNE on 21/10/2019.
 */
public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle() {
    }

    public Circle(double radius) {
        super();
        this.radius = radius;
    }

    public Circle(double radius, Point center) {
        super();
        this.radius = radius;
        this.center = center;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return round(radius);
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return round(radius * radius * Math.PI);
    }

    @Override
    public double getPerimeter() {
        return round(2 * radius * Math.PI);
    }

    @Override
    public String infoString() {
        return "Circle[center=" + center.toString() + ",radius=" + getRadius() +
                ",color=" + getColor() + ",filled=" + isFilled() + "]\n";
    }

    @Override
    public String toString() {
        return super.toString();
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof Circle)) return false;
        Circle other = (Circle) obj;
        if (this.center.equals(other.getCenter()) && Math.abs(this.radius - other.getRadius()) < 0.001) return true;
        return false;
    }
}

Layer.java

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        shapes.removeIf(shape -> shape instanceof Circle);
    }

    public String getInfo() {
        String info = "Layer of crazy shapes: \n";
        for (Shape shape : shapes) {
            info += shape.toString();
        }
        return info.trim();
    }

    public void removeDuplicates() {
        for (int i = shapes.size() - 1; i >= 0; i--) {
            for (int j = shapes.size() - 1; j > i; j--) {
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.get(j).remove();
                }
            }
        }
    }
}

Point.java

/**
 * Created by CCNE on 21/10/2019.
 */
public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other) {
        return Math.sqrt((this.x - other.getX()) * (this.x - other.getX()) +
                (this.y - other.getY()) * (this.y - other.getY()));
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof Point)) return false;
        Point other = (Point) obj;
        if (distance(other) >= 0.001) return false;
        return true;
    }

    private double round(double number) {
        return (Math.round(number * 10)) / 10.0;
    }

    @Override
    public String toString() {
        return "(" + round(x) + "," + round(y) + ")";
    }
}

/**
 * Created by CCNE on 21/10/2019.
 */
public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {
    }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return round(width);
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return round(length);
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return round(width * length);
    }

    @Override
    public double getPerimeter() {
        return round((width + length) * 2);
    }

    @Override
    public String infoString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + getWidth() + ",length=" + getLength() +
                ",color=" + getColor() + ",filled=" + isFilled() + "]\n";
    }

    @Override
    public String toString() {
        return super.toString();
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof Rectangle)) return false;

        Rectangle other = (Rectangle) obj;
        boolean isSizeIdentical = (Math.abs(this.width - other.getWidth()) < 0.001 &&
                Math.abs(this.length - other.getLength()) < 0.001);

        if (this.topLeft.equals(other.getTopLeft()) && isSizeIdentical) return true;
        return false;
    }
}

Shape.java

/**
 * Created by CCNE on 21/10/2019.
 */
public abstract class Shape {
    protected String color;
    protected boolean filled;
    protected boolean removed = false;

    public Shape() {
    }

    public double round(double number) {
        return Math.round(number * 10) / 10.0;
    }

    public Shape(String color, boolean filled) {
        this.color = color;
        this.filled = filled;
    }

    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public boolean isFilled() {
        return filled;
    }

    public void setFilled(boolean filled) {
        this.filled = filled;
    }

    public abstract double getArea();


    public abstract double getPerimeter();

    protected String nullString() {
        return "";
    }

    protected String infoString() {
        return "Shape[color=" + color + ",filled=" + filled + "]";
    }

    public String toString() {
        if (removed) return nullString();
        return infoString();
    }

    public void remove() {
        removed = true;
    }
}

public class Square extends Rectangle {
    public Square() {
    }

    public Square(double side) {
        super(side, side);
    }

    public Square(double side, String color, boolean filled) {
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled) {
        super(topLeft, side, side, color, filled);
    }

    public double getSide() {
        return round(this.width);
    }

    public void setSide(double side) {
        this.width = side;
        this.length = side;
    }

    @Override
    public void setLength(double side) {
        this.length = side;
        this.width = side;

    }

    @Override
    public void setWidth(double side) {
        this.width = side;
        this.length = side;
    }

    @Override
    public String infoString() {
        return "Square[topLeft=" + topLeft.toString() + ",side=" + getSide() +
                ",color=" + getColor() + ",filled=" + isFilled() + "]\n";
    }

    @Override
    public String toString() {
        return super.toString();
    }

    @Override
    public boolean equals(Object obj) {
        if (!(obj instanceof Square)) return false;
        Rectangle other = (Rectangle) obj;
        return super.equals(other);
    }
}
