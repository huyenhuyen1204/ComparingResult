import java.util.Objects;

public class Circle extends Shape {
    protected double radius;
    protected final double PI = Math.PI;
    protected Point center;

    public Circle() {}
    public Circle(double _radius) {
        radius = _radius;
    }
    public Circle(double _radius, String _color, boolean _filled) {
        super(_color, _filled);
        radius = _radius;
    }
    public Circle(Point _center, double _radius, String _color, boolean _filled) {
        super(_color, _filled);
        radius = _radius;
        center = _center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    @Override
    public double getArea() {
        return PI * radius * radius;
    }

    @Override
    public double getPerimeter() {
        return 2 * PI *radius;
    }

    @Override
    public 
boolean equals(Object obj) {
        if (obj instanceof Circle) {
            Circle other = (Circle) obj;
            if (Math.abs(other.radius - radius) <= 0.001 && Math.abs(other.center.getX() - center.getX()) <= 0.001
                    && Math.abs(other.center.getY() - center.getY()) <= 0.001) {
                return true;
            }
        }
        return false;

    }

    @Override
    public int hashCode() {
        return Objects.hash(radius, PI);
    
}

    @Override
    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + getRadius() + ",color=" + getColor() + ",filled=" + isFilled() + "]";

    }

}

import java.util.ArrayList;
import java.util.List;

public class Layer {

    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape newShape){
        this.shapes.add(newShape);
    }

    public 
String getInfo(){
        String result = "Layer of crazy shapes: \n";
        for (Shape s : shapes){
            result = result + s.toString() + "\n";
        }
        return result;
    }

    public void removeCircles(){
        for (int i = 0 ; i < shapes.size(); i ++){
            if (shapes.get(i) instanceof  Circle){
                shapes.remove(i);
                i--;
            }
        }
    }

    public void removeDuplicates(){
        for (int i = 1; i < shapes.size(); i++){
            for (int j = 0; j < i; j++){
                if (shapes.get(i).equals(shapes.get(j)) == true){
                    shapes.remove(i);
                    i--;
                    break;
                }
            }
        }
    }
}

import java.util.Objects;

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {}
    public Rectangle(double _width, double _length) {
        width = _width;
        length = _length;
    }
    public Rectangle(double _width, double _length, String _color, boolean _filled) {
        super(_color, _filled);
        width = _width;
        length = _length;
    }
    public Rectangle(Point _topLeft, double _width, double _length, String _color, boolean _filled) {
        super(_color, _filled);
        width = _width;
        length = _length;
        topLeft = _topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return width * length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }

    @Override
    public 
boolean equals(Object obj) {
        if (obj instanceof Rectangle){
            Rectangle other = (Rectangle) obj;
            if (Math.abs(other.width - width) <= 0.001 && Math.abs(other.length - length) <= 0.001
                    && Math.abs(other.topLeft.getX() - topLeft.getX()) <= 0.001 && Math.abs(other.topLeft.getY() - topLeft.getY()) <= 0.001) {
                return true;
            }
        }
        return false;

    }

    @Override
    public int hashCode() {
        return Objects.hash(width, length, topLeft);
    
}

    @Override
    public String toString(){
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + getWidth() + ",length=" + getLength() + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }


}
