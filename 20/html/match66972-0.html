public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){}
    public Circle(double radius){
        this.radius = radius;
    }
    public  Circle(double radius,String color,boolean filled){
        super(color,filled);
        this.radius = radius;
    }
    public Circle(Point point,double radius,String color,boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = point;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        double PI = Math.PI;
        return Math.pow(getRadius(),2)*PI;
    }

    @Override
    public double getPerimeter() {
        double PI = Math.PI;
        return getRadius()*2*PI;
    }

    @Override
    public boolean equals(Object obj) {
        if(obj instanceof Circle){
            if(this.center.equals(((Circle) obj).center) &&
            Math.abs(this.radius - ((Circle) obj).radius) <= 0.001)return true;
            else return false;
        }else return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        String s;
        if(super.filled == true){
            s ="true";
        }
        else s = "false";
        return "Circle[center=" + this.center.toString() +",radius=" +  (double)Math.round(this.radius*10)/10 + ",color=" + super.color + ",filled=" + s + "]";
    }
}

import java.util.LinkedList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new LinkedList<>();
    public void addShape(Shape shape){
        this.shapes.add(shape);
    }
    public void removeCircles(){
        for(int i = 0;i < shapes.size();i++){
            if(shapes.get(i) instanceof Circle){
                this.shapes.remove(shapes.get(i));
            }
        }
    }
    public String getInfo(){

       String a = "Layer of crazy shapes: " + "\n";
       for(Shape x : this.shapes){
          a += x.toString() + "\n";
       }
        return a;
    }
    public void removeDuplicates(){
        for(int i = 0;i < shapes.size();i++){
            Shape a = shapes.get(i);
            for(int j = i + 1;j < shapes.size();j++){
                if(a.equals(shapes.get(j))){
                    shapes.remove(j);
                    j--;
                }
            }

        }
    }
}

public class Point {
    private double x;
    private double y;
    public Point(double x,double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point point){
        double x = getX() - point.getX();
        double y = getY() - point.getY();
        return Math.sqrt(x*x + y*y);


    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public boolean equals(Object obj) {
        if(obj instanceof Point){
            if(Math.abs(((Point) obj).getX() - getX()) <= 0.001 && Math.abs(((Point) obj).getY() - getY()) <= 0.001)return true;
            else return false;
        }
        else return false;

    }

    @Override
    public String toString() {
        return "(" + (double)Math.round(getX()*10)/10 + "," + (double)Math.round(getY()*10)/10 + ")";
    }
}

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){

    }
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color,filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point point,double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = point;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    @Override
    public double getArea() {
        return getLength()*getWidth();
    }

    @Override
    public double getPerimeter() {
        return getLength()*2 + getWidth()*2;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public boolean equals(Object obj) {
        if(obj instanceof Rectangle){
            if(this.topLeft.equals(((Rectangle) obj).topLeft) &&
                    Math.abs(this.width - ((Rectangle) obj).width) <= 0.001 &&
                    Math.abs(this.length - ((Rectangle) obj).length) <= 0.001)return true;
            else return false;
        }else return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        String s;
        if(super.filled == true){
            s ="true";
        }
        else s = "false";
        return "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + (double)Math.round(this.width*10)/10 + ",length=" + (double)Math.round(this.length*10)/10 + ",color=" + super.color + ",filled=" + s + "]";
    }
}
