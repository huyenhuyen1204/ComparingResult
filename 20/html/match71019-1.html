import java.util.Objects;

public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle() {}

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center,double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Circle)) return false;
        Circle circle = (Circle) o;
        return Double.compare(circle.radius, radius) == 0 &&
                center.equals(circle.center);
    }

    @Override
    public double getArea() {
        return radius*radius*Math.PI;
    }

    @Override
    public double getPerimeter() {
        return 2*radius*Math.PI;
    }

    public String toString() {
        String s = center.toString();
        return "Circle[center=" + s + ",radius=" + radius + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes;
     public Layer () {
         shapes = new ArrayList<Shape>();
     }

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
         for (int i = shapes.size() - 1; i >= 0; i--) {
             
if (shapes.get(i) instanceof Circle) {
                 shapes.remove(i);
             }
         }
    }

    public String getInfo() {
         String s = "Layer of crazy shapes: \n";
         for (int i = 0 ; i < shapes.size(); i++) {
             s = s + shapes.get(i).toString() + "\n";
         }
         return s;
    }

    public void removeDuplicates() {
         for (int i = 0; i < shapes.size() - 1; i++)
         {
             for (int j = i + 1; j < shapes.size(); j++)
             {
                 if (shapes.get(i) instanceof Circle) {
                     if (((Circle) shapes.get(i)).equals(shapes.get(j))) {
                         shapes.remove(j);
                     }
                 } else if (shapes.get(i) instanceof Rectangle) {
                     if (((Rectangle) shapes.get(i)).equals(shapes.get(j))) {
                         shapes.remove(j);
                     }
                 } else if (shapes.get(i) instanceof Square) {
                     if (((Square) shapes.get(i)).equals(shapes.get(j))) {
                         shapes.remove(j);
                     }
                 }
             }
         }
    }
}

public class Square extends Rectangle {
    public Square() {}
    public Square(double side) {
        setWidth(side);
    }

    public Square(double side, String color, boolean filled) {
        super(side,side,color,filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled) {
        super(topLeft, side, side,color,filled);
    }

    public double getSide() {
        return getLength();
    }

    public void setSide(double side) {
        this.length = side;
        this.width = side;
    }

    @Override
    public boolean equals(Object o) {
        return super.equals(o);
    
}

    @Override
    public void setLength(double side) {
        this.length = side;
        this.width = side;
    }

    @Override
    public void setWidth(double side) {
        this.width = side;
        this.length = side;
    }

    public String toString() {
        String s = topLeft.toString();
        return "Square[topLeft=" + s + ",side=" + this.length +",color=" + this.color + ",filled=" + this.filled + "]";
    }
}
