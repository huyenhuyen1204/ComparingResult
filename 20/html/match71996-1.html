import java.util.Objects;

/**
 * Created by CCNE on 02/10/2019.
 */
public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public Circle(){};

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        this.radius = radius;
        super.setColor(color);
        super.setFilled(filled);
    }

    public Circle(Point a, double radius, String color, boolean filled){
        this.center = a;
        this.radius = radius;
        super.setColor(color);
        super.setFilled(filled);
    }
    public double getArea(){
        return Math.PI * this.getRadius() * this.getRadius();
    }

    public double getPerimeter(){
        return Math.PI * 2 * this.getRadius();
    }

    public String toString(){
        String res = "Circle[center=" + this.center.toString() + ",radius=" + radius + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
        return res;
    }

    
@Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Circle circle = (Circle) o;
        return circle.radius - radius <= 0.001 && circle.radius - radius >= -0.001 &&
                center.equals(circle.center);
    }

    @Override
    public int hashCode() {
        return Objects.hash(radius, center);
    }

    public static void main(String[] args){
        Point p1 = new Point(1,2);
        Point p2 = new Point(2,3);
        Circle a1 = new Circle(p1, 3,"red",true);
        Circle a4 = new Circle(p1, 3,"green", false);
        Rectangle a2 = new Rectangle(p2, 2, 3, "blue", true);
        Square a3 = new Square(p1,2,"green",true);
        Layer l = new Layer();
        l.addShape(a1);
        l.addShape(a2);
//        System.out.println(l.shapes.size());
        l.addShape(a1);
        l.addShape(a3);
        l.addShape(a4);

//        l.removeCircles();
        l.removeDuplicates();
        System.out.println(l.getInfo());
    }


}

src/Layer.java

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape a){
        shapes.add(a);
    }

    public void removeCircles(){
        for (int i = 0; i < shapes.size(); i++){
            
if (shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    }

    public String getInfo(){
        String res = "Layer of crazy shapes: " + "\n";
        for (int i = 0; i < shapes.size(); i++){
            res = res + shapes.get(i).toString() + "\n";
        }
        return res;
    }

    public void removeDuplicates(){
        for (int i = 0; i < shapes.size() - 1; i++){
            for(int j = i + 1; j < shapes.size(); j++){
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                }
            }
        }
    }


}

import java.util.Objects;

/**
 * Created by CCNE on 02/10/2019.
 */
public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Rectangle(){};

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        this.width = width;
        this.length = length;
        super.setColor(color);
        super.setFilled(filled);
    }

    public Rectangle(Point a, double width, double length, String color, boolean filled){
        this.topLeft = a;
        this.width = width;
        this.length = length;
        super.setColor(color);
        super.setFilled(filled);
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getArea(){
        return this.getLength() * this.getWidth();
    }

    public double getPerimeter(){
        return 2 * (this.getWidth() + this.getLength());
    }

    public String toString(){
        String res = "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";

        return res;
    }

    @Override
    public boolean equals(Object o) {
        
if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Rectangle rectangle = (Rectangle) o;
        return rectangle.width - width <= 0.001 && rectangle.width - width >= -0.001 &&
                rectangle.length - length <= 0.001 && rectangle.length - length >= -0.001 &&
                topLeft.equals(rectangle.topLeft);
    }

    @Override
    public int hashCode() {
        return Objects.hash(width, length, topLeft);
    }
}
