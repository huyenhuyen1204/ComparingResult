Circle.java

import java.lang.*;
public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){};
    public Circle(double r){
        this.radius=r;
    }
    public Circle(double r,String c,boolean f){
        super(c,f);
        this.radius=r;
    }
    public Circle(Point p,double r,String c,boolean f){
        super(c,f);
        this.radius=r;
        this.center=new Point(p.getX(),p.getY());
    }
    public double getRadius(){
        return this.radius;
    }
    public void setRadius(double r){
        this.radius=r;
    }
    public Point getCenter(){
        return this.center;
    }
    public void setCenter(Point c){
        this.center.setX(c.getX());
        this.center.setY(c.getY());
    }
    public boolean equals(Object m){
        if (m instanceof Circle){
            Circle other = (Circle) m;
            if (this.center.equals(other.getCenter())==true && Math.abs(this.radius-other.getRadius())<0.001)
                return true; else return false;
        }
        else return false;
    }

    @Override
    public double getArea(){
        return Math.PI*this.radius*this.radius;
    }
    @Override
    public double getPerimeter(){
        return Math.PI*2*this.radius;
    }
    public String toString(){
        String s="Circle[center=" + this.center.toString() +
                ",radius="+String.valueOf(((double)Math.round(this.radius*10))/10)+
                ",color="+this.color+",filled="+String.valueOf(this.filled)+"]";
        return s;
    }
}

public class Point {
    private double x;
    private double y;
    public Point(double x1, double y1){
        this.x=x1;
        this.y=y1;
    }
    public double getX(){
        return this.x;
    }
    public double getY(){
        return this.y;
    }
    public void setX(double x1){
        x=x1;
    }
    public void setY(double y1){
        y=y1;
    }
    public double distance(Point other){
        double subx=this.x-other.getX();
        double suby=this.y-other.getY();
        double m=Math.sqrt(subx*subx+suby*suby);
        return m;
    }
    public String toString(){
        String s="("+String.valueOf(((double)Math.round(this.x*10))/10)
                +","+String.valueOf(((double)Math.round(this.y*10))/10)+")";
        return s;
    }
    public boolean equals(Object m){
        if (m instanceof Point ){
            Point other = (Point) m;
            if (this.distance(other)<0.001) return true; else return false;
        }
        else return false;
    }
}

Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){};
    public Rectangle(double w,double l){
        this.width=w;
        this.length=l;
    }
    public Rectangle(double w,double l,String c,boolean f){
        super(c,f);
        this.width=w;
        this.length=l;
    }
    public Rectangle(Point t,double w,double l,String c,boolean f){
        super(c,f);
        this.width=w;
        this.length=l;
        this.topLeft = new Point(t.getX(),t.getY());
    }
    public double getWidth(){
        return this.width;
    }
    public void setWidth(double w){
        this.width=w;
    }
    public double getLength(){
        return this.length;
    }
    public void setLength(double l){
        this.length=l;
    }
    public Point getTopLeft(){
        return this.topLeft;
    }
    public void setTopLeft(Point t){
        this.topLeft.setX(t.getX());
        this.topLeft.setY(t.getY());
    }
    public boolean equals(Object m){
        if (m instanceof Rectangle){
            Rectangle other = (Rectangle) m;
            if (this.topLeft.equals(other.getTopLeft())==true
                    && Math.abs(this.width-other.getWidth())<0.001
                    && Math.abs(this.length-other.getLength())<0.001)
                return true; else return false;
        }
        else return false;
    }
    public String toString(){
        String s="Rectangle[topLeft=" + this.topLeft.toString()+
                ",width="+String.valueOf(((double)Math.round(this.width*10))/10)+
                ",length="+String.valueOf(((double)Math.round(this.length*10))/10)+
                ",color="+this.color+
                ",filled="+String.valueOf(this.filled)+"]";
        return s;
    }

    @Override
    public double getArea(){
        return this.length*this.width;
    }
    @Override
    public double getPerimeter(){
        return 2*(this.length+this.width);
    }

}

Square.java

public class Square extends Rectangle{
    public Square(){};
    public Square(double s){
        super(s,s);
    }
    public Square(double s,String c,boolean f){
        super(s,s,c,f);
    }
    public Square(Point t,double s,String c,boolean f){
        super(s,s,c,f);
        this.topLeft = new Point(t.getX(),t.getY());
    }
    public double getSide(){
        return this.width;
    }
    public void setSide(double s){
        this.width=s;
        this.length=s;
    }
    public boolean equals(Object m){
        if (m instanceof Square){
            Square other = (Square) m;
            if (this.topLeft.equals(other.getTopLeft())==true
                    && Math.abs(this.length-other.getLength())<0.001)
                return true; else return false;
        }
        else return false;
    }
    public String toString(){
        String s="Square[topLeft=" + this.topLeft.toString()
                +",side="+String.valueOf(((double)Math.round(this.width*10))/10)
                +",color="+this.color
                +",filled="+String.valueOf(this.filled)+"]";
        return s;
    }
    public void setWidth(double s){
        this.length=s;
        this.width=s;
    }
    public void setLength(double s){
        this.length=s;
        this.width=s;
    }
}
