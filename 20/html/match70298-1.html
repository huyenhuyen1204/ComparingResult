class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){};

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point point, double radius, String color, boolean filled) {
        super(color, filled);
        center = point;
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }
    @Override
    public double getArea(){
        return 3.14159265359*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return 2*radius*3.14159265359;
    }


    public String toString() {
        radius = (double)Math.round(radius*10)/10;
        return "Circle[center=" + center.toString() + ",radius=" + radius  +",color=" + color + ",filled=" +  filled + "]";
    }

    @Override
    public boolean equals(Object obj) {
        if(this.getClass() != obj.getClass()) return false;
        Circle c = (Circle)obj;
        if(Math.abs(radius - c.radius) <= 0.001 && center.equals(c.getCenter())) return true;
        return false;
    }
}

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape s) {
        shapes.add(s);
    }

    public void removeCircles() {
        for(int i = 0; i < shapes.size(); i++){
            if(
shapes.get(i).getClass() == Circle.class) shapes.remove(i--);
        }
    }

    public String getInfo() {
        String result = "Layer of crazy shapes: ";
        for(Shape s : shapes){
            result += "\n" + s.toString();
        }

        return result;
    }

    public void removeDuplicates() {
        // Create a new ArrayList
        List<Shape> newList = new ArrayList<Shape>();

        // Traverse through the first list
        for (Shape element : shapes) {

            // If this element is not present in newList
            // then add it
            if (!newList.contains(element)) {

                newList.add(element);
            }
        }

        shapes = newList;
    }

}

Point.java

public class Point {
    private double x;
    private double y;
    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point p) {
        double px = p.getX();
        double py = p.getY();
        return Math.sqrt((x - px)*(x - px) + (y - py)*(y - py));
    }

    @Override
    public boolean equals(Object obj) {
        if(obj.getClass() != this.getClass()) return false;
        Point p = (Point)obj;
        if(Math.abs(
p.getX() - this.getX()) <= 0.001 && Math.abs(p.getY() - this.getY()) <= 0.001) return true;
        return false;
    }

    @Override
    public String toString() {
        x = (double)Math.round(x*10)/10;
        y = (double)Math.round(y*10)/10;
        return "(" + x + "," + y + ")";
    }

}

public class Rectangle extends Shape {

    }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.length = length;
        this.width = width;
    }

    public Rectangle(Point point, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.length = length;
        this.width = width;
        topLeft = point;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public 
void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return (width + length)*2;
    }

    public String toString() {
        width = (double)Math.round(width*10) / 10;
        length = (double)Math.round(length*10) / 10;
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + color +",filled=" + filled +"]";
    }

    @Override
    public boolean equals(Object obj) {
        if(this.getClass() != obj.getClass()) return false;
        Rectangle 
r = (Rectangle) obj;
        if(Math.abs(width - r.width) <= 0.001 && Math.abs(length - r.length) <= 0.001 && this.topLeft.equals(r.getTopLeft())) return true;
        return false;
    }
}
