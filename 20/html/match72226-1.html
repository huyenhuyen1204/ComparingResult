Circle.java

public class Circle extends Shape {

    protected double radius;
    protected Point center;

    public Circle()
    {
        super();
    }
    public Circle(double radius)
    {
        super();
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color, filled);
        this.radius=radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius=radius;
        this.center=center;
    }

    public Point getCenter(){
        return center;
    }
    public void setCenter(Point center){
        this.center=center;
    }

    public double getRadius()
    {
        return radius;
    }
    public void setRadius(double radius)
    {
        this.radius=radius;
    }

    @Override
    public double getArea(){
        return radius*radius*PI;
    }

    @Override
    public double getPerimeter() {
        return 2*radius*PI;
    
}

    public boolean equals(Object object){

        if(object instanceof Circle)
        {
            Circle obj = (Circle) object;
            if(this.getCenter().equals(obj.getCenter()))
                if(this.getRadius()==obj.getRadius())
                    return true;
        
}
        return false;
    }
    @Override
    public String toString() {
        return "Circle[center="+this.getCenter().toString()+",radius="+radius+",color="+color+",filled="+filled+"]";
    }

    public static void main(String[] args)
    {
        Circle circle=new Circle(9.8, "blue",true);
        System.out.println(circle.getPerimeter());
    }
}
