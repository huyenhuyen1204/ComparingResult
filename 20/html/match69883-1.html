import java.lang.Math;
public class Circle extends Shape{
    protected double radius = 0;
    protected Point center;

    public Circle(){

    }

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        this.radius = radius;
        super.setColor(color);
        super.setFilled(filled);
    }

    public Circle(Point center, double radius, String color, boolean filled){
        this.center = center;
        this.radius = radius;
        super.setColor(color);
        super.setFilled(filled);
    }

    public double getRadius(){
        return this.radius;
    }

    public void setRadius(double radius){
        this.radius = radius;
    }

    public void setCenter(Point center){
        this.center.setX(center.getX());
        this.center.setY(center.getY());
    }

    public Point getCenter(){
        return this.center;
    }

    @Override
    public double getArea(){
        double result = Math.pow(this.radius, 2) * Math.PI;
        return result;
    }

    @Override
    public double getPerimeter(){
        double result = 2 * Math.PI * this.radius;
        return result;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null) return false;
        
if (this.getClass() != o.getClass()) return false;
        Circle user = (Circle) o;
        return Math.abs(radius - user.radius) <= 0.001;
    }

    @Override
    public String toString(){
        //Circle[center=(10.0,5.0),radius=8.0,color=RED,filled=true]
        String result = "Circle[center=(" + this.center.getX() + "," + this.center.getY() +
                "),radius=" + this.radius + ",color=" + super.getColor() + ",filled=" +
                super.isFilled() + "]";
        return result;
    }

    @Override
    public int hashCode() {
        return Double.toString(this.radius).hashCode() * this.center.hashCode();
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape shape){
        this.shapes.add(shape);
    }

    public Layer(){

    }

    public void removeCircles(){
        for (int i = 0; i < this.shapes.size(); i++){
            if (
this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
            }
        }
    }

    public String getInfo(){
        String result = "Layer of crazy shapes: ";
        for (int i = 0; i < this.shapes.size(); i++){
            result = result + "\n" + 
this.shapes.get(i).toString();
        }
        return result;
    }

    public void removeDuplicates(){
        for (int i = 0; i < this.shapes.size()-1; i++)
        {
            for (int j = i+1; j <this.shapes.size(); j++)
            {
                if (this.shapes.get(i).hashCode() == this.shapes.get(j).hashCode()){
                    this.shapes.remove(j);
                    j--;
                }
            }
        }
    }

    public static void main(String[] args){

        Square sqr = new Square(new Point(3,1),2,"BLUE",true);
        System.out.println(sqr.toString());
        Square sqr1 = new Square(6.6);
        System.out.println(sqr1.toString());

        Layer layer = new Layer();
        layer.addShape(new Rectangle(new Point(3,5) ,1.0, 2.0, "RED", true));
        layer.addShape(new Circle(new Point(2,3), 3,"RED", false));
        layer.addShape(new Circle(new Point(2,3), 3,"BLUE", false));
        layer.addShape(new Rectangle(new Point(3,5) ,1.0, 2.0, "RED", false));
        layer.addShape(new Square(new Point(3,1),2,"BLUE",true));
        layer.addShape(new Circle(new Point(2,3), 3,"GREEN", false));
        layer.addShape(new Circle(new Point(2,3), 3,"PINK", false));
        layer.addShape((Rectangle) new Square(new Point(3,1),2,"RED",true));

        layer.removeDuplicates();

    }
}

import java.util.HashMap;

public class Point {
    private double x = 0.0;
    private double y = 0.0;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX(){
        return this.x;
    }

    public void setX(double x){
        this.x = x;
    }

    public double getY(){
        return this.y;
    }

    public void setY(double y){
        this.y = y;
    }

    public double distance(Point p){
        double result = Math.sqrt((p.x - this.x) * (p.x - this.x) + (p.y - this.y) * (p.y - this.y));
        return result;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null) return false;
        if (this.getClass() != o.getClass()) return false;
        Point 
user = (Point) o;
        return (x - user.x) <= 0.001
                && (y - user.y) <= 0.001;
    }

    public String toString(){
        String result = "(" + x + "," + y + ")";
        return result;
    }

    @Override
    public int hashCode() {
        return Double.toString(this.x).hashCode() * Double.toString(this.y).hashCode();
    }

}

public class Square extends Rectangle {
    public 
Square() {

    }

    public Square(double side) {
        super.width = side;
        super.length = side;
    }

    public Square(double side, String color, boolean filled) {
        super.width = side;
        super.length = side;
        super.setColor(color);
        super.setFilled(filled);
    }

    public Square(Point topLeft,double side, String color, boolean filled) {
        super.topLeft = topLeft;
        super.width = side;
        super.length = side;
        super.setColor(color);
        super.setFilled(filled);
    
}

    public double getSide() {
        return super.width;
    }

    public void setSide(double side){
        super.width = side;
        super.length = side;
    }

    @Override
    public void setWidth(double side){
        this.setSide(side);
    }

    @Override
    public void setLength(double side){
        this.setSide(side);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        
if (o == null) return false;
        if (this.getClass() != o.getClass()) return false;
        Rectangle user = (Rectangle) o;
        return Math.abs(width - user.width) <= 0.001
                && Math.abs(length - user.length) <= 0.001
                && Math.abs(length - user.width) <= 0.001;
    }

    @Override
    public String toString(){
        //Square[topLeft=(10.0,5.0),side=8.0,color=RED,filled=true]
        double get_x = this.topLeft.getX();
        double get_y = this.topLeft.getY();
        String result = String.format("Square[topLeft=(%s,%s),side=%s,color=%s,filled=%s]",
                Double.toString(get_x),
                Double.toString(get_y),
                Double.toString(super.getLength()), super.getColor(), super.isFilled());
        return result;
    }
