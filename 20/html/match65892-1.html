class Circle extends Shape {
    protected double radius=1.0;
    protected Point center= new Point(0,0);
    public Circle() {}
    public Circle(double radius)    {
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled)  {
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }
    public Circle(Point center, double radius, String color, boolean filled)    {
        this.center.setX(center.getX());
        this.center.setY(center.getY());
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }

    public Point getCenter()    {
        Point center = new Point(this.center.getX(), this.center.getY());
        return center;
    }

    public void setCenter(Point center) {
        this.center.setX(center.getX());
        this.center.setY(center.getY());
    }

    public double getRadius()   {
        double r = this.radius;
        return r;
    }
    public void setRadius(double radius)    {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        double r = this.radius;
        return PI*r*r;
    }

    @Override
    public double getPerimeter()    {
        return 2*PI*this.radius;
    }
    public boolean equals (Object object)   {
        if(object instanceof Circle) {
            Circle circle = (Circle) object;
            if((this.center).equals(circle.getCenter()) && Math.abs(this.radius-circle.getRadius()) <=0.001)
                return true;
        }
        return false;
    }
    public String toString ()   {
        String s= new String();
        s="Circle[center=("+Lamtron(this.center.getX())+","+Lamtron(this.center.getY())+"),radius="+Lamtron(this.getRadius())+",color="+this.color+",filled="+Boolean.toString(this.filled)+"]";
        return s;
    }

}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public Layer()  {}
    public void addShape(Shape shape)   {
        shapes.add(shape);
    }
    public void removeCircles()   {
        int i = 0;
        while (i<shapes.size()) {
            if(shapes.get(i) instanceof Circle)
                shapes.remove(i);
            else    i++;
        }
        return;
    }
    public String getInfo ()    {
        String Sum = "Layer of crazy shapes: ";
        for(int i = 0; i<shapes.size(); i++)    {
            String s = (shapes.get(i)).toString();
            Sum = Sum + "\n" + s;
        }
        return Sum;
    }

    public void removeDuplicates () {
        int i,j;
        for(i=0; i<shapes.size()-1; i++)
            for (j=i+1; j<shapes.size(); j++)
                if(shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                }
        return;
    }
}

public 
class Point {
    private double x;
    private double y;
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }
    public double getX ()   {
        double x = this.x;
        return x;
    }
    public double getY ()   {
        double y = this.y;
        return y;
    }
    public void setX(double x)  {
        this.x = x;
    }
    public void setY(double y)  {
        this.y = y;
    }
    public double distance (Point point)    {
        double dentaX = Math.abs(this.x - point.getX());
        double dentaY = Math.abs(this.y - point.getY());
        return Math.sqrt(dentaX*dentaX + dentaY*dentaY);
    }
    public boolean equals (Object object)   {
        if(object instanceof Point) {
            Point point = (Point) object;
            if(point.getX() == this.x && point.getY() == this.y)
                return true;
        }
        return false;
    }

    public String toString ()   {
        String s ="("+Double.toString(this.x)+","+Double.toString(this.y)+")";
        return s;
    }
}

class Rectangle extends Shape {
    protected Point topLeft= new Point(0,0);
    protected double width;
    protected double length;
    public Rectangle() {}
    public Rectangle(double width, double length)   {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }
    public Rectangle(Point topLeft,double width, double length, String color, boolean filled) {
        this.topLeft.setX(topLeft.getX());
        this.topLeft.setY(topLeft.getY());
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }
    public double getWidth()    {
        double _width = this.width;
        return _width;
    }
    public void setWidth(double width)  {
        this.width = width;
    }
    public double getLength()   {
        double _length = this.length;
        return _length;
    }
    public void setLength(double length)    {
        this.length = length;
    }
    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return 2*(width+length);
    }
    public Point getTopLeft()   {
        Point topLeft = new Point(this.topLeft.getX(), this.topLeft.getY());
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft.setX(topLeft.getX());
        this.topLeft.setY(topLeft.getY());
    }
    public boolean equals (Object object)   {
        if(object instanceof Rectangle) {
            Rectangle rectangle = (Rectangle) object;
            if(rectangle.topLeft.equals(this.topLeft) && Math.abs(rectangle.getWidth() - this.width)<=0.001 && Math.abs(rectangle.getLength() - this.length)<=0.001)
                return true;
        }
        return false;
    }
    public String toString() {
        String s="Rectangle[topLeft=("+super.Lamtron(this.topLeft.getX())+","+super.Lamtron(this.topLeft.getY())+"),width="+super.Lamtron(this.getWidth())+",length="+super.Lamtron(this.getLength())+",color="+this.color+",filled="+Boolean.toString(this.filled)+"]";
        return s;
    }
}

import java.text.DecimalFormat;

public abstract class Shape {
    protected String color;
    protected boolean filled;
    protected final double PI = 3.14;
    public Shape()  {}
    public Shape(String color, boolean filled)  {
        this.color =color;
        this.filled = filled;
    }
    public String getColor()  {
        String _color = this.color;
        return color;
    }
    public void setColor(String color)  {
        this.color = color;
    }
    public boolean isFilled()   {
        boolean _filled = this.filled;
        return _filled;
    }
    public void setFilled(boolean filled)   {
        this.filled = filled;
    }
    public String toString() {
        String s="Shape[color="+this.color+",filled="+Boolean.toString(filled)+"]";
        return s;
    }
    public String Lamtron (double x)    {
        double s = (double) Math.floor(x*10+0.5)/10;
        return Double.toString(s);
    }
    public abstract double getArea();
    public abstract double getPerimeter();
}

    public void setWidth(double side) {

        super.setLength(side);
