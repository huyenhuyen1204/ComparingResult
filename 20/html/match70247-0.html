import java.util.ArrayList;
import java.util.List;

public class Circle extends Shape {

    protected double radius;
    protected Point center;


    public Circle(){}

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public double getRadius(){
        return radius;
    }

    public void setRadius(double radius){
        this.radius = radius;
    }

    public double getArea(){
        return pi * radius * radius;
    }

    public double getPerimeter(){
        return 2 * pi * radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Circle){
            Circle circle = (Circle) obj;
            if (Math.abs(circle.getRadius() - this.getRadius()) <= 0.001
                    && Math.abs(circle.getCenter().getX() - this.getCenter().getX()) <= 0.001
                    && Math.abs(circle.getCenter().getY() - this.getCenter().getY()) <= 0.001){
                return true;
            }
        }
        return false;
    }

    public String toString(){

        return "Circle[center=" + this.center.toString() + ",radius=" + (double)Math.round(this.radius *10)/10
                + ",color=" + this.color + ",filled=" + this.filled + "]";
    }



}

import java.util.ArrayList;
import java.util.List;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape s){
        shapes.add(s);
    }

    public void removeCircles(){
        for (int i = 0; i < shapes.size(); i++){
            if (shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo(){
        String str = "Layer of crazy shapes: \n";
        for(Shape s : shapes){
            str = str + s.toString() + "\n";
        }
        return str;
    }

    public void removeDuplicates(){
        for (int i = 1; i < shapes.size(); i++){
            for (int j = 0; j < i; j++){
                if (shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(i);
                    i--;
                    break;
                }
            }
        }
    }

}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point p1){
        double dis = (p1.getY() - this.getY())*(p1.getY() - this.getY()) + (p1.getX() - this.getX())*(p1.getX() - this.getX());
        return Math.sqrt(dis);
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Point) {
            Point point = (Point) obj;
            if (this.getX() == point.getX() && this.getY() == point.getY()){
                return true;
            }
        }
        return false;
    }

    public int hashCode(){
        return 1;
    }

    public String toString(){
        return "(" + (double)Math.round(this.x * 10)/10 + "," + (double)Math.round(this.y * 10)/10 + ")";
    }
}

public class Rectangle extends Shape {

    protected double width;
    protected double length;
    protected  Point topLeft;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public double getWidth(){
        return width;
    }

    public void setWidth(double width){
        this.width = width;
    }

    public double getLength(){
        return length;
    }

    public void setLength(double length){
        this.length = length;
    }

    public double getArea(){
        return width * length;
    }

    public double getPerimeter(){
        return 2 * (width + length);
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle){
            Rectangle rectangle = (Rectangle) obj;
            if (Math.abs(rectangle.getLength() - this.getLength()) <= 0.001
                    && Math.abs(rectangle.getWidth() - this.getWidth()) <= 0.001
                    && Math.abs(rectangle.getTopLeft().getX() - this.getTopLeft().getX()) <= 0.001
                    && Math.abs(rectangle.getTopLeft().getY() - this.getTopLeft().getY()) <= 0.001){
                return true;
            }
        }
        return false;
    }


    public String toString(){
        return "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + (double)Math.round(this.width * 10)/10 +
                ",length=" + (double)Math.round(this.length *10)/10 + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
}

public class Square extends Rectangle {

    public Square(){}

    public Square(double side){
        setSide(side);
    }

    public Square(double side, String color, boolean filled){
        setSide(side);
        this.color = color;
        this.filled = filled;
    }

    public Square(Point topLeft, double side, String color, boolean filled){
        this(side, color, filled);
        this.setTopLeft(topLeft);
    }

    public double getSide(){
        return this.width;
    }

    public void setSide(double side){
        this.width = side;
        this.length = side;
    }

    @Override
    public void setWidth(double side){
        this.width = side;
        this.length = side;
    }

    @Override
    public void setLength(double side){
        this.length = side;
        this.width = side;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Square){
            Square square = (Square) obj;
            if (Math.abs(square.getSide() - this.getSide()) <= 0.001
                    && Math.abs(square.getTopLeft().getX() - this.getTopLeft().getX()) <= 0.001
                    && Math.abs(square.getTopLeft().getY() - this.getTopLeft().getY()) <= 0.001)
                return true;
        }
        return false;
    }

    public String toString(){
        return "Square[topLeft=" + topLeft.toString() + ",side=" +
                (double)Math.round(getSide() *10)/10 + ",color=" + this.color + ",filled=" + this.filled + "]";
    }


}
