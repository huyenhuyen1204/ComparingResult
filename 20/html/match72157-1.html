import java.util.Objects;

public class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){};
    public Circle(double radius )
    {
        this.radius=radius;
    }
    public Circle(double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.radius=radius;
    }
    public Circle(Point a,double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.center=a;
        this.radius=radius;

    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }


    @Override
    public double getArea() {
        double a= this.radius*this.radius;
        return a*3.14;
    }

    @Override
    public double getPerimeter() {
        return 2*3.14*this.getRadius();
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Circle circle = (Circle) o;
        return Double.compare(circle.radius, radius) == 0 &&
                center.equals(circle.center);
    }

    @Override
    public int hashCode() {
        return Objects.hash(center, radius);
    }

    @Override
    public String toString() {
        String s = "Circle[center="+this.center.toString()+",radius="+this.radius+",color="+this.color+",filled="+this.filled+"]";

        return s;
    }
}
