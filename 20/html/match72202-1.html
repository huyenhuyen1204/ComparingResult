class Circle extends Shape{
    protected Point center;
    protected double radius;

    public Circle(){};

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.color = color;
        this.filled = filled;
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.center = center;
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }
    public Point getCenter(){
        return this.center;
    }
    public void setCenter(Point p){
        this.center = p;
    }

    public double getRadius(){
        return this.radius;
    }

    public void setRadius(double radius){
        this.radius = radius;
    }
    public double getArea(){
        return Math.PI * this.radius * this.radius;
    }
    public double getPerimeter(){
        return Math.PI * 2 * this.radius;
    }

    public 
boolean equals(Object o){
        if(o instanceof Circle){
            Circle other = (Circle)o;
            if(this.getRadius() == other.getRadius() && this.getCenter().equals(other.getCenter())){
                return true; 
            }
            else return false;
        }
        else return false;
    }
    @Override
    public String toString(){
        return "Circle[center=" + this.center.toString() + ",radius="+this.radius+",color="+this.color+",filled="+this.filled+"]";
    }

}

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class Layer{
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape s){
        this.shapes.add(s);
    }
    public void removeCircles(){
        for(int i = 0; i < shapes.size(); i++){
            
if(this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
            }
        }
        
    }
    public String getInfo(){
        String res = "Layer of crazy shapes:\n";
        for(int i = 0; i < shapes.size(); i++){
            res = res + this.shapes.get(i).toString() + "\n";
        }

        return res;
    
}
    public void removeDuplicates(){
        Set<Shape> set = new HashSet<Shape>();
        for(int i = 0; i < shapes.size(); i++){
            set.add(shapes.get(i));
        }
        List<Shape> res = new ArrayList<Shape>();

        for(Shape s : set){
            res.add(s);
        }
    }
}

Rectangle.java

public class Rectangle extends Shape{
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(String color, boolean filled){
        super(color, filled);
        this.color = color;
        this.filled = filled;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);        
        this.color = color;
        this.filled = filled;
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.topLeft = topLeft;       
        this.color = color;
        this.filled = filled;
        this.width = width;
        this.length = length;
    }
    public Point getTopLeft(){
        return this.topLeft;
    }
    public void setTopLeft(Point t){
        this.topLeft = t;
    }

    public double getWidth(){
        return width;
    }
    public void setWidth(double width){
        this.width = width;
    }
    public double getLength(){
        return length;
    }
    public void setLength(double length){
        this.length = length;
    }
    public double getArea(){
        return this.width * this.length;
    }
    public 
double getPerimeter(){
        return (this.width + this.length) * 2;
    }
    public boolean equals(Object o){
        if(o instanceof Rectangle){
            Rectangle other = (Rectangle)o;
            if(this.topLeft.equals(other.topLeft) && this.length == other.length && this.width == other.width){
                return true;
            }
            else return false;
        }
        else return false;
    }
    
    @Override
    public String toString(){
        return "Rectangle[topLeft=" + this.topLeft.toString() + ",width="+this.width+",length="+this.length+",color="+this.color+",filled="+this.filled+"]";
    }

}
