Circle.java

public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){}
    public Circle(double radius) {
        this.radius = radius;
    }
    public Circle(double radius, Point center) {
        this.radius = radius;
        this.center = center;
    }
    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }
    public void setCenter(Point center) {
        this.center = center;
    }
    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return Math.PI * radius * radius;
    }

    @Override
    public double getPerimeter() {
        return 2 * Math.PI * radius;
    }

    public boolean equals(Object o) {
        if (o instanceof Circle) {
            if ( (this.center.equals(((Circle) o).center)) && (this.radius - ((Circle) o).radius <= 0.001) )
                return true;
            else
                return false;
        }
        else return false;
    }

    @Override
    public String toString() {
        double r= (double) Math.round(radius*10)/10;
        return "Circle[center=" + center.toString() + ",radius=" + r + ",color=" + color + ",filled=" + filled + "]";
    }

    @Override
    public int hashCode() {
        return (int) radius;
    }
}

import java.util.*;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }
    public String getInfo() {
        String str = "Layer of crazy shapes: ";
        for ( Shape i : shapes) {
            str +="\n"+ i.toString() ;
        }
        return str;
    }
    public void removeCircles() {
        for ( int i = 0; i < shapes.size(); i++) {
            if ( shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }
    public void removeDuplicates() {
        Set<Shape> remove_shape = new LinkedHashSet<>(shapes);
        shapes = new ArrayList<Shape>(remove_shape);
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle(){}
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, Point topLeft) {
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }


    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return width * length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }

    public boolean equals(Object o) {
        if (o instanceof Rectangle) {
            if ( (this.topLeft.equals(((Rectangle) o).topLeft)) && (this.width - ((Rectangle) o).width <= 0.001) && (this.length - ((Rectangle) o).length <= 0.001) )
                return true;
            else
                return false;
        }
        else return false;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width="+ width + ",length=" + length + ",color=" + color +",filled=" + filled + "]";
    }

    @Override
    public int hashCode() {
        return (int)(width + length);
    }
}
