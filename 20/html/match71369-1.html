//package Week6;
import java.util.Objects;
import java.util.*;

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){}

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }


    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public double getRadius(){
        return this.radius;
    }

    public void setRadius(double radius){
        this.radius = radius;
    }

    public double getArea(){
        return this.radius * this.radius * Math.PI;
    }

    public double getPerimeter(){
        return 2 * Math.PI * this.radius;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center){
        this.center.setX(center.getX());
        this.center.setY(center.getY());
    }

    public 
boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle other = (Circle) obj;
            if(this.radius == other.radius && this.center.equals(other.center) && this.isFilled() == other.isFilled() && this.getColor().equals(other.getColor())) return true;
            else return false;
        }else return false;
    }

    public int hashCode() {
        return Objects.hash(radius, center);
    }

    @Override
    public String toString(){
        return "Circle[center="+ getCenter()+",radius="+radius+",color="+getColor()+",filled="+isFilled()+"]";
    }
}

//package Week6;

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape s){
        this.shapes.add(s);
    }

    public void removeCircles(){
        for(int i = 0; i < this.shapes.size(); i++){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
            }
        }
    }

    public String getInfo(){
        String 
s = "Layer of crazy shapes:\n";
        for(int i = 0; i < this.shapes.size(); i++){
            s += shapes.get(i).toString() + "\n";
        }
        return s;
    }

    public void removeDuplicates(){
        for(int i = 0; i < shapes.size() - 1; i++){
            for(int j = i + 1; j < shapes.size(); j++){
                
if(shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                }
            }
        }
    }
}

//package Week6;

import java.util.Objects;

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }

    public double getWidth(){
        return this.width;
    }

    public void setWidth(double width){
        this.width = width;
    }

    public double getLength(){
        return this.length;
    }

    public void setLength(double length){
        this.length = length;
    }

    public Point getTopLeft(){
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft){
        this.topLeft = topLeft;
    }

    public double getArea(){
        return this.length * this.width;
    }

    public double getPerimeter(){
        return 2 * (this.length + this.width);
    }

    @Override
    public String toString(){
        return "Rectangle[topLeft="+ this.getTopLeft()+",width="+width+",length="+length+",color="+getColor()+",filled="+isFilled()+"]";
    }

    public 
boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle other = (Rectangle) obj;
            if(this.topLeft.equals(other.topLeft) && this.width == other.width && this.length == other.length && this.isFilled() == other.isFilled() && this.getColor().equals(other.getColor())) return true;
            else return false;
        }else return false;
    }
}
