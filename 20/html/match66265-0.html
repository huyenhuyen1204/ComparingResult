Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public Circle(Point P,double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
        this.center = P;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    @Override
    public double getArea() {
        return 3.14159*radius*radius;
    }
    public double getPerimeter(){
        return 3.14159*radius*2;
    }

    @Override
    public boolean equals(Object obj){
        if (obj instanceof Circle){
            Circle other = (Circle)obj;
            {
                if (Math.abs(this.radius-other.radius)<=0.001 && Math.abs(this.center.getX()-other.center.getX())<=0.001 && Math.abs(this.center.getY()-other.center.getY())<=0.001){
                    return true;
                }
            }
        }
        return false;
    }
    @Override
    public int hashCode(){
        return super.hashCode();
    }
    @Override
    public String toString(){
        radius = (double)Math.round(radius*10)/10;
        return "Circle[center="+getCenter().toString()+",radius="+radius+",color="+color+",filled="+filled+"]";
    }

}

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(double width ,double length){
        this.width = width;
        this.length = length;
    }
    public  Rectangle(double width, double length ,String color ,boolean filled)
    {
        super(color,filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle() {

    }

    public  Rectangle(Point P,double width, double length ,String color ,boolean filled)
    {
        super(color,filled);
        this.width = width;
        this.length = length;
        this.topLeft = P;
    }

    public void setWidth(double width)
    {
        this.width = width;
    }
    public double getWidth()
    {
        return  width;
    }

    public void setLength(double lenght) {
        this.length = length;
    }

    public double getLength() {
        return length;
    }

    public void setTopLeft(Point P)
    {
        this.topLeft = P;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return 2*width+2*length;
    }

    public boolean equals(Object obj)
    {
        if(obj instanceof Rectangle)
        {
            Rectangle other = (Rectangle)obj;
            if(Math.abs(this.width - other.width)<=0.001 && Math.abs(this.length - other.length)<=0.001
                    && Math.abs(this.topLeft.getX()- other.topLeft.getX() )<=0.001 && Math.abs(this.topLeft.getY()- other.topLeft.getY() )<=0.00)
            {
                return true;
            }
        }
        return false;
    }

    @Override
    public int hashCode(){
        return super.hashCode();
    }

    @Override
    public String toString() {
        width = (double)Math.round(width*10)/10;
        length = (double)Math.round(length*10)/10;
        String a = "Rectangle[topLeft="+ topLeft.toString() +",width="+width+",length="+length+",color="+color+",filled="+filled+"]";
        return a;
    }
}
