class Circle extends Shape {
    protected double radius;
    protected Point center;


    public Circle(){

    }
    public Circle(double radius){
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled){
        super(color,filled);
        this.radius=radius;
    }
    public Circle(Point center,double radius,String color,boolean filled){
        super(color,filled);
        this.radius=radius;
        this.center=center;

    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius){
        this.radius=radius;
    }
    public Point getCenter(){
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }


    @Override
    public double getArea(){
        return Math.PI*radius*radius;
    }
    @Override
    public double getPerimeter(){
        return 2*Math.PI*radius;
    }
    @Override
    public String toString() {
        return "Circle[center=("+center.getX()+","+center.getY()+")"+",radius="+radius+",color="+color+",filled="+filled+"]";
    
}

    @Override
    public boolean equals(Object obj) {
        return super.equals(obj);
    }
    @Override
    public int hashCode() {
        return super.hashCode();
    }
}

public class Layer {
    private List<Shape>shapes= new ArrayList<>();

    public void addShape(Shape temp){
        shapes.add(temp);

    }
    public 
String getInfo(){
        String result    = "Layer of crazy shapes: "+"\n";
        for (int i =0 ;i<shapes.size(); i++){
            result += shapes.get(i).toString() +"\n";
        }
        return result;
    }
    public void removeCircles(){
        int count = 0;
        while ( count < shapes.size()){
            if (shapes.get(count) instanceof Circle){
                shapes.remove(count);
            } else{
                count++;
            }
        }
    }
    public void removeDuplicates(){
        int i,j;
        for(i=0;i<shapes.size();i++){
            for(
j=i+1;j<shapes.size();j++){
                if(shapes.get(i).equals(shapes.get(j))){
                    hashCode();
                }
            }
        }
    }


}
