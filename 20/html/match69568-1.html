class Circle extends Shape {
    protected double radius;
    protected Point center;
    //Constructer
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
        this.center = center;
    }
    //Setter,getter

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    //Another method
    public double getArea(){
        return Math.PI*this.radius*this.radius;
    }
    public double getPerimeter(){
        return 2.0*Math.PI*this.radius;
    }
    public String toString(){
        return "Circle[center=" + this.center.toString() + ",radius=" + this.radius + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    
}

    public boolean equals(Object obj) {
        if(obj instanceof Circle)
        {
            Circle other = (Circle)obj;
            if((other.getRadius()-this.getRadius())*(other.getRadius()-this.getRadius()) > 0.001*0.001) return false;
            if(other.getCenter().equals(this.center)) return true;
            return false;
        }
        return false;
    }

    public int hashCode() {
        return this.hashCode();
    }
}

import org.w3c.dom.css.Rect;

import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Set;

public class Layer {
    private List<Shape> shapes = new LinkedList<Shape>();

    public void addShape(Shape other)
    {
        this.shapes.add(other);
    }
    public void removeCircles()
    {
        for(int i=0;i<shapes.size();i++)
        {
            
if(shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo()
    {
        String temp = "Layer of crazy shapes: ";
        temp = temp + '\n';
        for(int i=0;i<shapes.size();i++)
        {
            temp = temp + shapes.get(i).toString() + '\n';
        
}
        return temp;
    }
    public void removeDuplicates()
    {
        for(int i=0;i<shapes.size()-1;i++)
        {
            for(int j=i+1;j<shapes.size();j++)
            {
                if(((shapes.get(i) instanceof Circle) && (shapes.get(j) instanceof Circle)))
                {
                    Circle c1 = (Circle)shapes.get(i);
                    Circle c2 = (Circle)shapes.get(j);
                    if(c1.getCenter().equals(c2.getCenter())&&c1.getRadius()==c2.getRadius())
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
                
if(((shapes.get(i) instanceof Rectangle) && (shapes.get(j) instanceof Rectangle)))
                {
                    Rectangle r1 = (Rectangle)shapes.get(i);
                    Rectangle r2 = (Rectangle)shapes.get(j);
                    if(r1.getTopLeft().equals(r2.getTopLeft()) && r1.getWidth() == r2.getWidth() && r1.getLength() == r2.getLength())
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
                
if(((shapes.get(i) instanceof Square) && (shapes.get(j) instanceof Square)))
                {
                    Square s1 = (Square)shapes.get(i);
                    Square s2 = (Square)shapes.get(j);
                    if(s1.getSide()==s2.getSide()&&s1.getTopLeft().equals(s2.getTopLeft()))
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
            }
        }
    }
}

import java.util.Objects;

public class Point {
    private double x;
    private double y;

    //Constructer
    public Point(double x,double y)
    {
        this.x = x;
        this.y = y;
    }
    //Setter,Getter

    public double getX() {
        return x;
    }

    public 
double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }
    //Another method

    public double distance(Point other)
    {
        return Math.sqrt((this.x-other.x)*(this.x-other.x)+(this.y-other.y)*(this.y-other.y));
    }
    public boolean equals(Object o)
    {
        if(o instanceof Point)
        {
            Point other = (Point)o;
            if((other.getX()-this.x)*(other.getX()-this.x)>(0.001*0.001)) return false;
            if ((other.getY() - this.y) * (other.getY() - this.y) > (0.001 * 0.001)) return false;
            return true;
        }
        return false;
    }

    public int hashCode() {
        return this.hashCode();
    }

    public String toString() {
        return "(" + this.x + "," + this.y + ")";
    }
}

public class Square extends Rectangle {
    public Square(){

    }
    public Square(double side)
    {
        this.width = side;
        this.length = side;
    }
    public Square(double side,String color,boolean filled)
    {
        super(side,side,color,filled);

    }
    public Square(Point topleft,double side,String color,boolean filled)
    {
        super(side,side,color,filled);
        super.topLeft = topleft;
    }
    //Setter,getter
    public double getSide()
    {
        return super.width;
    }
    public void setSide(double side)
    {
        this.width = side;
        this.length = side;
    }
    public void setWidth(double side)
    {
        this.width = side;
        this.length = side;
    }
    public void setLength(double side)
    {
        this.length = side;
        this.width = side;
    }
    public String toString(){
        return "Square[topLeft=" + this.topLeft.toString() + ",side=" + this.length + ",color=" + super.color + ",filled=" + this.filled + "]";
    
}

    public boolean equals(Object obj) {
        if(obj instanceof Square)
        {
            Square other = (Square)obj;
            if((other.getSide()-this.getSide())*(other.getSide()-this.getSide())>0.001*0.001) return false; // Sai Số Quá 0.01
            if(other.getTopLeft().equals(this.topLeft)) return true; // Cùng topLeft
            return false;
        }
        return false;
    }

    public int hashCode() {
        return this.hashCode();
    }
}
