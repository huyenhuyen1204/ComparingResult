Point.java

import java.text.DecimalFormat;

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point orther) {
        double distance = Math.sqrt(Math.pow(x - orther.getX(), 2) + Math.pow(y - orther.getY(), 2));
        return distance;
    
}

    public boolean equals(Object obj) {
        if (obj instanceof Point) {
            Point other = (Point) obj;
            if (Math.abs(other.getX() - x) <= 0.001
                    && Math.abs(other.getY() - y) <= 0.001)
                return true;
            else
                return false;
        }
        return false;
    }

    @Override
    public String toString() {
        DecimalFormat df = new DecimalFormat("#.0");
        String s="("+df.format(x)+","+df.format(y)+")";
        return s;
    }
}

Rectangle.java

import java.text.DecimalFormat;

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {
    }

    public Rectangle(double width, double length) {
        this.length = length;
        this.width = width;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        this.setColor(color);
        this.setFilled(filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point point, double width, double length, String color, boolean filled) {
        this.topLeft = point;
        this.width = width;
        this.length = length;
        this.setColor(color);
        this.setFilled(filled);
    }


    public 
double getWidth() {
        return width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point toLeft) {
        this.topLeft = toLeft;
    }

    @Override
    public double getArea() {
        return width * length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle) {
            Rectangle other = (Rectangle) obj;
            if(
Math.abs(other.width-width)<=0.001&&Math.abs(other.length-length)<=0.001) return true;
            else return false;
        } else return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        DecimalFormat df = new DecimalFormat("#.0");
        String s = "Rectangle[topLeft=(" + df.format(topLeft.getX()) + "," + df.format(topLeft.getY()) +
                "),width=" + df.format(width) + ",length=" + df.format(length) +
                ",color=" + getColor() + ",filled=" + isFilled() + "]";
        return s;
    }
}
