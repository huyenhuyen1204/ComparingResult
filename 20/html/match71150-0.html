public class Circle extends  Shape{
    protected double radius;
    protected Point center;
    public Circle() {}
    public Circle(double radius)
    {
        this.radius = radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
    }
    public Circle(Point p,double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
        this.center = p;
    }
    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return radius * radius * Math.PI;
    }

    @Override
    public double getPerimeter() {
        return radius * 2 * Math.PI;
    }

    @Override
    public String toString() {
        return "Circle[center="+ center.toString() + ",radius=" + lamtron(this.radius) + ",color="+this.color+",filled=" + this.filled + "]";
    }

    public double HashCode() {
        return getArea() * getPerimeter() * radius;
    }

    public boolean equals(Object obj){
        if (obj instanceof Circle){
            if (this.getCenter().distance(((Circle) obj).getCenter()) > 0.001) return false;
            if (Math.abs(this.getRadius() - ((Circle) obj).getRadius()) > 0.001) return false;
            return true;
        }
        else return false;
    }
}

Point.java

public class Point {
    private double x;
    private double y;
    public Point(double x,double y)
    {
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double sqr(double x)
    {
        return (double) x * x;
    }
    public double distance(Point p)
    {
        return lamtron(Math.sqrt((sqr(p.x-this.x) + sqr(p.y-this.y))));
    }
    public boolean equals(Object obj)
    {
        if (obj instanceof Point) {
            if (Math.abs(((Point) obj).getX()-this.getX()) > 0.001) return false;
            if (Math.abs(((Point) obj).getY()-this.getY()) > 0.001) return false;
            return true;
        }
        return false;
    }

    public double lamtron(double t)
    {
        return (double) (Math.round(t * 10)  / 10);
    }

    public String toString() {
        return "(" + this.x + "," + this.y + ")";
    }

}

public class Square extends Rectangle {
    public Square() {}
    public Square(double side)
    {
        this.length = side;
        this.width = side;
    }
    public Square(double side,String color,boolean filled)
    {
        super(side,side,color,filled);
    }
    public Square(Point p,double side,String color,boolean filled)
    {
        super(p,side,side,color,filled);
    }
    public double getSide()
    {
        return super.getWidth();
    }
    public void setSide(double side)
    {
        this.length = side;
        this.width = side;
    }

    @Override
    public void setWidth(double side) {
        super.setWidth(side);
        super.setLength(side);
    }

    @Override
    public void setLength(double side) {
        super.setWidth(side);
        super.setLength(side);
    }

    @Override
    public String toString() {
        return "Square[topLeft=" + this.topLeft.toString() + ",side=" + lamtron(this.getSide()) + ",color=" + this.color + ",filled=" + this.filled+ "]";
    }

    public boolean equals(Object obj) {
        if (obj instanceof  Square) {
            if (Math.abs(((Square) obj).getSide()-this.getSide()) > 0.001) return false;
            if (this.getTopLeft().distance(((Square) obj).getTopLeft()) > 0.001) return false;
            return true;
        }
        return false;
    }

    public double HashCode() {
        return getArea() * getPerimeter() ;
    }

}
