public class Circle extends Shape {
    protected double radius;
    protected final double PI = Math.PI;
    protected Point 
center;

    public Circle() {}

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    @Override
    public String getColor() {
        return super.getColor();
    }

    @Override
    public void setFilled(boolean filled) {
        super.setFilled(filled);
    }

    
@Override
    public double getArea() {
        return radius*radius*PI;
    }

    @Override
    public double getPerimeter() {
        return 2*PI*radius;
    }

    @Override
    public String toString() {
        return "Circle[center=("+ center.getX() + "," + center.getY() + "),radius="+ radius +",color="+ color +",filled="+ filled +"]";
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Circle && (this.radius - ((Circle) obj).getRadius() < 0.001) && this.center.equals(((Circle) obj).getCenter())) {
            return true;
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape newShape)
    {
        shapes.add(newShape);
    }

    public String getInfo()
    {
        String s = "Layer of crazy shapes: \n";
        for (Shape shape :
                shapes) {
            s += shape.toString() + "\n";
        }
        return s;
    }

    public void removeCircles()
    {
        for (int i = 0; i < shapes.size(); i++) {
            if (shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
            }
        }
    }

    public void removeDuplicates()
    {
        
for (int i = 0; i < shapes.size(); i++) {
            for (int j = 0; j < shapes.size(); j++) {
                if (i != j && shapes.get(i).equals(shapes.get(j)))
                {
                    shapes.remove(j);
                }
            }
        }
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other)
    {
        return Math.sqrt((x - other.getX())*(x - other.getX()) + (y - other.getY())*(y - other.getY()));
    }

    @Override
    public boolean equals(Object obj) {
        if (distance((Point) obj) < 0.001) return true;
        return false;
    
}

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "(" + x + "," + y + ")";
    }

}

public class Rectangle extends Shape {

        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }


    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public 
void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return 2*(width + length);
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle && (this.width == ((Rectangle) obj).getWidth()) && (this.length == ((Rectangle) obj).getLength()) && this.topLeft.equals(((Rectangle) obj).getTopLeft())) {
            return true;
        }
        return false;
    
}

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=("+ topLeft.getX() + "," + topLeft.getY() + "),width=" + width + ",length=" + length + ",color=" + super.getColor() + ",filled="+ super.isFilled() + "]";
    }
}
