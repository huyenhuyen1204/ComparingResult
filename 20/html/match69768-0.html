import java.util.*;
import java.util.List;
public class Layer {
    private List<Shape > shapes;
    public void addShape(Shape a){
        this.shapes.add(a);
    }
    public void removeCircles(){
        for (int i = 0; i < this.shapes.size(); i ++)
        if (this.shapes.get( i ) instanceof Circle)
        {
            this.shapes.remove( i );
            i--;
        }
    }
    public String getInfo (){
        String s = null;
        for (int i = 0; i < this.shapes.size(); i ++){
            if (this.shapes.get(i) instanceof Circle)
                s = s + this.shapes.get(i).toString()+"/n";
            if (this.shapes.get(i) instanceof Rectangle)
                s = s + this.shapes.get(i).toString()+"/n";
            if (this.shapes.get(i) instanceof Square)
                s = s + this.shapes.get(i).toString()+"/n";
        }
        return s;
    }
    public void removeDuplicates(){
        for (int i = 0; i< shapes.size()-1; i ++)
            for (int j = i+1;j < shapes.size();j ++)
                if ( shapes.get(i).equals(shapes.get(j)))
                {
                    shapes.remove(j);
                    j--;
                }
    }
}

public class Point {
    private double x;
    private double y;

    public Point(double x,double y) {
        this.x = x;
        this.y = y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public double getX() {
        return x;
    }
    public double distance(Point a){
         return Math.sqrt ((this.x - a.getX()) * (this.x - a.getX()) + (this.y - a.getY()) * (this.y - a.getY()));

    }
    public String toString(){
        return "(" + Double.toString((double) Math.round(x *10)/10) +","+ Double.toString((double) Math.round(y *10)/10)+")";
    }
    public boolean equals(Object o){
        if (o instanceof Point) {
        Point other= (Point) o;
        if( Math.abs(other.getX()-this.getX()) <= 0.001)
            if( Math.abs(other.getY()-this.getY()) <= 0.001)
                return true;
        return false;
        }
        return false;
    }
    public int hashCode(){
        return this.hashCode();
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle() {
    }
   public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    

    public void setLength(double length) {
        this.length = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public double getWidth() {
        return width;
    }
    public double getArea(){
        return width*length;
    }
    public double getPerimeter(){
        return 2*(width+length);
    }
    public String toString(){
        return "Rectangle[topLeft=" + topLeft.toString()+",width=" +Double.toString((double) Math.round(width *10)/10)+",length=" + Double.toString((double) Math.round(length *10)/10) + ",color=" +super.getColor() + ",filled=" +Boolean.toString(super.isFilled())+ "]";
    }
     public boolean equals( Object o){
        if (o instanceof Rectangle){
            Rectangle other = (Rectangle) o;
            if ( Math.abs(this.getLength()-other.getLength()) <=0.001)
                if ( Math.abs (this.getWidth()-other.getWidth()) <= 0.001)
                    if (this.getTopLeft().equals(other.getTopLeft())) 
                        return true;
        return false;
        }
        return false;
    }

    public int hashCode(){
        return this.hashCode();
    }

}

public class Square extends Rectangle {

        super(side,side,color,filled);

    }
    public Square (Point topleft, double side, String color, boolean filled){
        super (topleft,side,side,color,filled);
    }

    }

        super.setWidth(side);

    }
    public int hashCode(){
        return this.hashCode();
    }
    public boolean equals(Object o) {
        if ( o instanceof Square){
            Square other = (Square ) o;
            if (Math.abs(this.getSide()- other.getSide()) <=0.001)

                    return true;
        return false;

    }
