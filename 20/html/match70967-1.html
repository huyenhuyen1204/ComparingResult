import java.lang.Math;
public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle() {}

    public Circle (double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea() {
        return Math.PI * this.radius * this.radius;
    }

    public double getPerimeter() {
        return 2.0*Math.PI*this.radius;
    }

    public boolean equals(Object obj) {
        if(obj instanceof Circle) {
            Circle other = (Circle) obj;
            if(this.getRadius() == other.getRadius() && this.getCenter().equals(other.getCenter())) {
                return true;
            } else return false;
        }
        else return false;
    
}

    public String toString() {
        return String.format("Circle[center=(%.1f,%.1f),radius=%.1f,color=%s,filled=%b]" , this.center.getX(), this.center.getY() ,this.radius, this.getColor(), this.isFilled() );
    }

}

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        for(int i = 0; i < shapes.size(); i++) {
            
if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo() {
        String ans = "Layer of crazy shapes: \n";
        for(Shape s : shapes) {
            ans += s.toString() + "\n";
        }
        return ans;
    }

    public void removeDuplicates() {
        for(int i=0; i < shapes.size(); i++) {
            for(int j = i+1; j < shapes.size(); j++) {
                if(shapes.get(i).equals(shapes.get(j)) ) {
                    this.shapes.remove(j);
                    j--;
                }
            }
        }
    }

}

src/Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {}

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft ,double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getArea() {
        return this.length * this.width;
    }

    public double getPerimeter() {
        return 2.0*(this.width + this.length);
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    
}

    public boolean equals(Object obj) {
        if( obj instanceof Rectangle ) {
            Rectangle other = (Rectangle) obj;
            
            return (this.getLength() == other.getLength() && this.getWidth() == other.getWidth() && this.topLeft.equals(other.getTopLeft()));
        }
        else return false;
    
}

    public String toString() {
        return String.format("Rectangle[topLeft=(%.1f,%.1f),width=%.1f,length=%.1f,color=%s,filled=%b]", this.topLeft.getX(), this.getTopLeft().getY() ,this.width, this.length, this.getColor(), this.isFilled());
    }

}
