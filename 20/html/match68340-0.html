Circle.java

public class Circle extends Shape {
    public final double PI = Math.PI;
    protected Point center;
    protected double radius;

    public Circle() {

    }

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public double getArea() {
        return PI * this.radius * this.radius;
    }

    @Override
    public double getPerimeter() {
        return 2.0 * PI * this.radius;
    }

    public boolean equals(Object obj){
        if(obj instanceof Circle) {
            Circle other = (Circle) obj;
            if(Math.abs(other.center.getX() - center.getX()) < 0.001
                    && Math.abs(other.center.getY() - center.getY()) < 0.001
                    && other.getRadius() - this.radius < 0.001 ) {
                return  true;
            }
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {

        return "Circle[center=(" + center.getX() + "," + center.getY()
                + "),radius=" + radius
                + ",color=" + this.getColor()
                + ",filled=" + this.isFilled() + "]";
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other) {
        double answer = Math.sqrt(Math.pow(this.x - other.getX(),2.0) + Math.pow(this.y - other.getY(),2.0));
        return answer;
    }

    @Override
    public boolean equals(Object obj) {
        return super.equals(obj);
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "(" + this.x + "," + this.y + ")";
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle() {

    }

    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return this.width * this.length;
    }
    @Override
    public double getPerimeter() {
        return 2.0 * (this.width + this.length);
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public boolean equals(Object obj){
        if (obj instanceof Rectangle) {

            Rectangle other = (Rectangle) obj;

            if(Math.abs(other.topLeft.getX() - topLeft.getX()) < 0.001
                    &&  Math.abs(other.topLeft.getY() - topLeft.getY()) < 0.001
                    && other.getLength() - this.length < 0.001
                    && other.getWidth() - this.width < 0.001) {
                return true;
            }
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {

        return "Rectangle[topLeft=(" + topLeft.getX() + "," + topLeft.getY()
                + "),width=" + this.width
                + ",length=" + this.length
                + ",color=" + super.getColor()
                + ",filled=" + super.isFilled() + "]";
    }

}

Square.java

public class Square extends Rectangle{

    public Square() {

    }

    public Square(double side) {
        super(side,side);
    }

    public Square(double side, String color, boolean filled) {
        super(side, side, color, filled);
    }

    public  Square(Point topLeft, double side, String color, boolean filled) {
        super(topLeft, side, side, color, filled);
    }

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
        super.setLength(width);
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
        super.setWidth(length);
    }

    public double getSide() {
        return super.getLength();
    }

    public void setSide(double side) {
        setLength(side);
        setWidth(side);
    }

    public boolean equals(Object obj){
        if (obj instanceof Square) {

            Square other = (Square) obj;

            if(Math.abs(other.getSide() - this.getSide()) < 0.001) {
                return true;
            }
        }
        return false;
    }

    @Override
    public String toString() {


        return "Square[topLeft=(" + super.getTopLeft().getX() + "," + super.getTopLeft().getY()
                + "),side=" + this.getSide()
                + ",color=" + this.getColor()
                + ",filled=" + this.isFilled() + "]";
    }
}
