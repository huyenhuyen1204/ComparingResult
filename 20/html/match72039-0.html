import javax.swing.*;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Layer {
    private List<Shape> shapes=new ArrayList<>();
    public void addShape(Shape shape1){
    shapes.add(shape1);
    }
    public void removeCircles(){
     for(int i=shapes.size()-1;i>=0;i--){
         if(shapes.get(i) instanceof Circle){
             shapes.remove(i);
         }
     }
    }
    public String getInfo(){
        String a="";
        a+="Layer of crazy shapes: "+"\n";
        for (int i = 0; i < shapes.size(); i++)
        {
            a=a+shapes.get(i).toString()+"\n";

        }
    return a;
    }
    public void removeDuplicates(){
        for(int i=0;i<shapes.size()-1;i++){
            for(int j=i+1;j<shapes.size();j++){
                if(shapes.get(i) instanceof Circle&&shapes.get(j) instanceof Circle){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Rectangle&&shapes.get(j) instanceof Rectangle){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Square&&shapes.get(j) instanceof Square){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
            }

        }
    }

}

Rectangle.java

import java.awt.*;
import java.util.Objects;

public class Rectangle extends Shape{
protected double width;
protected double length;
protected Point topLeft;
public Rectangle(){
}
public Rectangle(double width, double length){
    super();
    this.width=width;
    this.length=length;
}
public Rectangle(double width,double length,String color, boolean filled){
    super(color,filled);
    this.width=width;
    this.length=length;
}
public Rectangle(Point topLeft,double width,double length,String color,boolean filled){
    super(color,filled);
    this.topLeft=topLeft;
    this.width=width;
    this.length=length;

}
    public double getWidth() {
       return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    @Override
    public double getArea(){
       return getWidth()*getLength();
    }
    @Override
    public double getPerimeter(){
       return 2*(getLength()+getWidth());
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Rectangle rectangle = (Rectangle) o;
        return rectangle.width==this.width&&rectangle.length==this.length&&rectangle.topLeft==this.topLeft
                &&rectangle.color==this.color&&rectangle.filled==this.filled;
    }

    @Override
    public int hashCode() {
        return Objects.hash(width, length, topLeft,color,filled);
    }

    public String toString(){
        return "Rectangle[topLeft="+topLeft.toString()+",width="+Shape.Round(getWidth())+",length="+Shape.Round(getLength())+",color="+getColor()+",filled="+isFilled()+"]";
    }

        System.out.println(s.getPerimeter());
    }*/
}
