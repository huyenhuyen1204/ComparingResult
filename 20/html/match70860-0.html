src/Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    final private double PI = Math.PI;

    public Circle() {this.radius = 0.0;}
    public Circle(double r) {this.radius = r;}
    public Circle(double r, String c, boolean f) {super(c, f); this.radius = r;}
    public Circle(Point p, double r, String c, boolean f) {
        super(c, f);
        this.radius = r;
        this.center = p;
    }

    public void setRadius(double x) {this.radius = x;}
    public double getRadius() {return this.radius;}

    @Override
    public double getArea(){
        return this.radius * this.radius * PI;
    }
    @Override
    public double getPerimeter(){
        return this.radius * 2.0 * PI;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point p) {
        this.center = p;
    }

    public String toString(){
        return ("Circle[center=" + center.toString()
                +",radius=" + String.format("%.1f",this.radius)
                + ",color=" + this.color + ",filled=" + this.filled + "]");
    }

    public boolean equals(Object o) {
        if(o instanceof Circle) {
            Circle cir = (Circle) o;
            if(Math.abs(getRadius()- cir.getRadius()) <= 0.001
                    && Math.abs(getRadius()- cir.getRadius()) <= 0.001
                    && center.equals(cir.center))
                return true;
        }
        return false;
    }
}

src/Layer.java

import java.util.List;
import java.util.LinkedList;

public class Layer {
    private List<Shape> shapes = new LinkedList<Shape>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        for(int i = 0; i < shapes.size(); ++i) {
            if(shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }

    public String getInfo() {
        String rs = "Layer of crazy shapes: \n";
        for (int i = 0; i < shapes.size(); ++i) {
            rs += shapes.get(i).toString();
            rs  += "\n";
        }
        return rs;
    }

    public void removeDuplicates() {
        Shape shape;
        for (int i = 0; i < shapes.size(); ++i) {
            shape = shapes.get(i);
            for(int j = i + 1; j < shapes.size(); ++j) {
                if(shape.equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point p) {
        double dX = getX() - p.getX();
        double dY = getY() - p.getY();
        return Math.sqrt(dX * dX + dY * dY);
    }

    public String toString() {
        return "(" + String.format("%.1f",getX()) + "," + String.format("%.1f",getY()) + ")";
    }

    public boolean equals(Object o) {
        if(o instanceof Point) {
            Point p = (Point) o;
            if(Math.abs(getX()- p.getX()) <= 0.001 && Math.abs(getY()- p.getY()) <= 0.001)
                return true;
        }
        return false;
    }
}

src/Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {
        this.width = 0.0;
        this.length = 0.0;
    }
    public Rectangle(double w, double l){
        this.width = w;
        this.length = l;
    }
    public Rectangle(double w, double l, String c, boolean f){
        super(c, f);
        this.width = w;
        this.length = l;
    }

    public Rectangle(Point p, double w, double l, String c, boolean f){
        super(c, f);
        this.width = w;
        this.length = l;
        this.topLeft = p;
    }

    public double getWidth() {return this.width;}
    public void setWidth(double x) {this.width = x;}

    public double getLength() {return this.length;}
    public void setLength(double x) {this.length = x;}

    @Override
    public double getArea() {return (this.width * this.length);}
    @Override
    public double getPerimeter() {return (this.width + this.length) * 2.0;}

    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point p) {
        this.topLeft = p;
    }

    public String toString(){
        return ("Rectangle[topLeft=" + topLeft.toString()
                + ",width=" + String.format("%.1f",this.width)
                + ",length=" + String.format("%.1f",this.length)
                + ",color=" + this.color + ",filled=" + this.filled + "]");
    }

    public boolean equals(Object o) {
        if(o instanceof Rectangle) {
            Rectangle rec = (Rectangle) o;
            if(Math.abs(getLength()- rec.getLength()) <= 0.001
                    && Math.abs(getWidth()- rec.getWidth()) <= 0.001
                    && topLeft.equals(rec.topLeft))
                return true;
        }
        return false;
    }
}

public class Square extends Rectangle{

    public Square() {super(0.0, 0.0);}
    public Square(double x) {super(x, x);}
    public Square(double x, String c, boolean f) {super(x, x, c, f);}

    public void setSide(double x){
        super.setWidth(x);

    }

        super.setLength(x);

    public double getWidth(){
        return super.getLength();
    }

    @Override
    public void setLength(double x){
        super.setWidth(x);

    public double getLength(){

        return super.equals(o);

}
