Circle.java

public class Circle extends Shape {
    protected Point center = new Point(0.0, 0.0);
    protected double radius;

    public Circle(){}

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        this.radius = radius;
        this.color  = color;
        this.filled = filled;
    }

    public Circle(Point point, double radius, String color, boolean filled) {
        this.center = point;
        this.radius = radius;
        this.color  = color;
        this.filled = filled;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        double result;
        result = Math.PI*radius*radius;
        result = Rounding(result);
        return result;
    }

    @Override
    public double getPerimeter() {
        double result;
        result = 2*Math.PI*radius;
        result = Rounding(result);
        return result;
    }

    public boolean equals(Object obj) {
        if ( obj instanceof Circle) {
            Circle circle = (Circle) obj;
            return (radius - circle.radius <= 0.001);
        }
        return false;
    }


    public int hashCode() {
        return (center.hashCode() + (int) radius);
    }

    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + color + ",filled=" + filled + "]";
    }

}

import java.util.LinkedList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new LinkedList<>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        for ( int i = 0; i < shapes.size(); i++ ) {
            if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    }

    public String getInfo() {
        String result = "Layer of crazy shapes:\n";
        for ( int i = 0; i < shapes.size(); i++ ) {
            result = result + shapes.get(i).toString() + "\n";
        }
        return result;
    }

    public void removeDuplicates() {
        for ( int i = 0; i < shapes.size() - 1; i++) {
            for ( int  j = i + 1; j < shapes.size(); j++ ) {
                if (shapes.get(i).hashCode() ==  shapes.get(j).hashCode())
                    shapes.remove(j);
            }
        }
    }

    public static void main(String[] args) {
        Rectangle rectangle = new Rectangle(1, 2);
        Rectangle rectangle1 = new Rectangle(1,2);
        Rectangle rectangle2 = new Rectangle(2,3);
        Layer layer = new Layer();
        layer.shapes.add(rectangle);
        layer.shapes.add(rectangle1);
        layer.shapes.add(rectangle2);
        layer.removeDuplicates();
        System.out.println(layer.getInfo());
    }

}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft = new Point(0,0);
    protected double width;
    protected double length;

    public Rectangle(){};

    public Rectangle(double width, double length) {
        this.width = width;
        this.length= length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        this.width = width;
        this.length= length;
        this.color = color;
        this.filled= filled;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        this.topLeft = topLeft;
        this.width   = width;
        this.length  = length;
        this.color   = color;
        this.filled  = filled;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        double result = width*length;
        result = Rounding(result);
        return result;
    }

    @Override
    public double getPerimeter() {
        double result = (width+length)*2;
        result = Rounding(result);
        return result;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle) {
            Rectangle rec = (Rectangle) obj;
            return rec.width == width && rec.length == length;
        }
        return false;
    }

    public int hashCode(){
        return (topLeft.hashCode() + (int) width + (int) length);
    }

    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() +",width="+ this.width + ",length=" + this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
}

public class Square extends Rectangle {
    public Square() {
    }

    public Square(double side) {
        setLength(side);
        setWidth(side);
    }

    public Square(double side, String color, boolean filled) {
        setLength(side);
        setWidth(side);
        setColor(color);
        setFilled(filled);
    }

    public Square(Point point, double side, String color, boolean filled) {
        this.topLeft =  point;
        this.width   =  side;
        this.length  =  side;
        this.color   = color;
        this.filled  = filled;
    }
    public double getSide() {
        return length;
    }

    public void setSide(double side) {
        super.length = side;
        super.width  = side;
    }

    @Override
    public void setWidth(double side) {
        width = side;
        length =side;
    }

    @Override
    public void setLength(double side) {
        length = side;
        width  = side;
    }

    public String toString (){
        return "Square[topLeft=" + topLeft.toString() +",side="+ this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

    public boolean equals(Object obj) {
        if ( obj instanceof  Square) {
            Square square = (Square) obj;
            return getSide() == square.getSide();
        }

        return false;
    }

    public int hashCode() {
        return (topLeft.hashCode() + (int) getSide());
    }
}
