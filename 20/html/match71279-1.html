class Circle extends Shape{
    protected double radius;
    protected Point center;

    public Circle(){
    }

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    
@Override
    public double getArea() {
        return PI*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return 2*PI*radius;
    }

    @Override
    public boolean equals(Object obj) {
        boolean check = false;
        if(obj instanceof Circle){
            Circle other = (Circle) obj;
            if(this.center.equals(other.center) && Math.abs(other.getRadius() - this.radius) <= 0.001) check = true;
        }
        return check;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + (double)Math.round(this.radius * 10)/10 + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
}

import java.util.ArrayList;
import java.util.List;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public Layer(){
    }

    public void addShape(Shape shape){
        this.shapes.add(shape);
    }

    public void removeCircles(){
        for(int i = 0; i < this.shapes.size(); i++){
            
if(this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
            }
        }
    }

    public String getInfo(){
        String result = "Layer of crazy shapes: \n";
        for(int i = 0; i < this.shapes.size(); i++){
            result += 
this.shapes.get(i).toString() + "\n";
        }
        return result;
    }

    public void removeDuplicates(){
        for(int i = 0; i < this.shapes.size(); i++){
            for (int j = i+1; j < this.shapes.size(); j++) {
                if(this.shapes.get(j).equals(this.shapes.get(i))){
                    this.shapes.remove(j);
                }
            }
        }
    }


}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other){
        return Math.hypot(this.getX() - other.getX(), this.getY() - other.getY());
    }

    @Override
    public boolean equals(Object obj) {
        boolean check = false;
        if(obj instanceof Point){
            Point other = (Point)obj;
            if(this.distance(other) <= 0.001) check = true;
        }
        
return check;
    }

    @Override
    public String toString() {
        return ("(" + (double)Math.round(this.x * 10)/10 + "," + (double)Math.round(this.y * 10)/10 + ")");
    }
}

public class Rectangle extends Shape {

    }

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return this.length * this.width;
    }

    @Override
    public double getPerimeter() {
        return (this.length + this.width) * 2;
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public boolean equals(Object obj) {
        boolean check = false;
        
if(obj instanceof Rectangle){
            Rectangle other = (Rectangle) obj;
            if(this.topLeft.equals(other.topLeft) && Math.abs(other.getWidth() - this.getWidth()) <= 0.001 && Math.abs(other.getLength() - this.getLength()) <= 0.001) check = true;
        }
        return check;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + (double)Math.round(this.width * 10)/10 + ",length=" + (double)Math.round(this.length * 10)/10 + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
