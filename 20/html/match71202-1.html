src/Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    final private double PI = Math.PI;

    public Circle() {this.radius = 0.0;}
    public Circle(double r) {this.radius = r;}
    public Circle(double r, String c, boolean f) {super(c, f); this.radius = r;}
    public Circle(Point p, double r, String c, boolean f) {
        super(c, f);
        this.radius = r;
        this.center = p;
    }

    public void setRadius(double x) {this.radius = x;}
    public 
double getRadius() {return this.radius;}

    @Override
    public double getArea(){
        return this.radius * this.radius * PI;
    }
    @Override
    public double getPerimeter(){
        return this.radius * 2.0 * PI;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point p) {
        this.center = p;
    }

    public String toString(){
        return ("Circle[center=" + center.toString()
                +",radius=" + String.format("%.1f",this.radius)
                + ",color=" + this.color + ",filled=" + this.filled + "]");
    }

    public boolean equals(Object o) {
        if(o instanceof Circle) {
            Circle cir = (Circle) o;
            if(Math.abs(getRadius()- cir.getRadius()) <= 0.001
                    && Math.abs(getRadius()- cir.getRadius()) <= 0.001
                    && center.equals(cir.center))
                return true;
        }
        return false;
    }
}

src/Layer.java

import java.util.List;
import java.util.LinkedList;

public class Layer {
    private List<Shape> shapes = new LinkedList<Shape>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        for(int i = 0; i < shapes.size(); ++i) {
            
if(shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }

    public String getInfo() {
        String rs = "Layer of crazy shapes: \n";
        for (int i = 0; i < shapes.size(); ++i) {
            rs += shapes.get(i).toString();
            rs  += "\n";
        }
        return rs;
    }

    public void removeDuplicates() {
        Shape shape;
        for (int i = 0; i < shapes.size(); ++i) {
            
shape = shapes.get(i);
            for(int j = i + 1; j < shapes.size(); ++j) {
                if(shape.equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

public class Square extends Rectangle{

    public Square(Point p, double x, String c, boolean f) {super(p, x, x, c, f);}

    public void setSide(double x){
        super.setWidth(x);

    }

void setWidth(double x){

}
