import java.util.List;
import java.util.ArrayList;
public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape){
        shapes.add(shape);
    }
    public void removeCircles(){
        for (int i = 0;i < shapes.size();i++){
            if (shapes.get(i) instanceof Circle){
                shapes.remove(i);
            }
        }
    }
    public String getInfo(){
        String s = "Layer of crazy shapes: "+"\n";
        for (Shape i : shapes){
            s+=i.toString()+"\n";
        }
        return s;
    }
    public void removeDuplicates(){
    	for (int i = 0;i < shapes.size();i++) {
    		for (int j = i+1;j < shapes.size();j++) {
    			if (shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle) {
    				Circle obj1 = (Circle) shapes.get(i);
    				Circle obj2 = (Circle) shapes.get(j);
    				if (obj1.center.equals(obj2.center) && obj1.radius == obj2.radius) shapes.remove(j);
    			}
    			if (shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle) {
    				Rectangle obj1 = (Rectangle) shapes.get(i);
    				Rectangle obj2 = (Rectangle) shapes.get(j);
    				if (obj1.topLeft.equals(obj2.topLeft) && obj1.width == obj2.width && obj1.length == obj2.length) shapes.remove(j);
    			}
    			if (shapes.get(i) instanceof Square && shapes.get(j) instanceof Square) {
    				Square obj1 = (Square) shapes.get(i);
    				Square obj2 = (Square) shapes.get(j);
    				if (obj1.topLeft.equals(obj2.topLeft) && obj1.getSide() == obj2.getSide()) shapes.remove(j);
    			}
    		}
    	}
    }
}

import java.lang.Math;

public class Point {
    private double x;
    private double y;

    public Point(){};
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point other){
        double a = this.x - other.x;
        double b = this.y - other.y;
        return Math.sqrt(a*a + b*b);
    }
    public boolean equals(Object other){
        if (other instanceof Point){
            Point obj = (Point) other;
            if (this.x == obj.x && this.y == obj.y) return true;
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "("+ this.x + ","+ this.y +")";
    }
}

src/Rectangle.java


public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){}
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        this.width = width;
        this.length = length;
        setColor(color);
        setFilled(filled);
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
        setColor(color);
        setFilled(filled);
    }

    public double getWidth() {
        return width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public void setWidth(double width) {
        this.width = width;
    }
    public double getArea(){
        return this.length*this.width;
    }
    public double getPerimeter(){
        return (this.width + this.length)*2;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }
    public boolean equals(Object obj){
        if (obj instanceof Rectangle){
            Rectangle other = (Rectangle) obj;
            if (this.width == other.width && this.length == other.length) return true;
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=(" + topLeft.getX()+","+topLeft.getY()+"),width=" + this.width + ",length=" + this.length +",color=" + getColor() +",filled=" + isFilled() +"]";
    }
}

public class Square extends Rectangle{
    public Square(){}
    public Square(double side){
        setSide(side);
    }
    public Square(double side, String color, boolean filled){
        setSide(side);
        setColor(color);
        setFilled(filled);
    }
    public Square(Point point, double side, String color, boolean filled){
        setTopLeft(point);
        setSide(side);
        setColor(color);
        setFilled(filled);
    }
    public double getSide(){
        return getLength();
    }
    public void setSide(double side){
        setLength(side);
        setWidth(side);
    }
    public void setWidth(double side){
        this.width = side;
        this.length = side;
    }
    public void setLength(double side){
        this.length = side;
        this.width = side;
    }
    public boolean equals(Object obj){
        if (obj instanceof Square){
            Square other = (Square) obj;
            if (this.getSide() == other.getSide()) return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "Square[topLeft=(" + topLeft.getX()+","+topLeft.getY()+"),side=" + getSide() + ",color=" + getColor() + ",filled=" + isFilled() +"]";
    }
}
