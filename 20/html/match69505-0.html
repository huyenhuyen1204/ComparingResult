Circle.java

public class Circle extends  Shape{
       protected  Point  center = new Point();
       protected  double radius;
       final      double PI = Math.PI;
       public Circle(){}
       public Circle(double radius){
           this.radius = radius;
       }
       public Circle(double radius,String color,boolean filled){
           super(color,filled);
           this.radius = radius;
       }
       public Circle(Point point,double radius,String color,boolean filled){
               super(color, filled);
               this.radius = radius;
               setCenter(point);
       }
       public Point getCenter() {
         return this.center;
       }
       public void setCenter(Point point) {
         this.center.setX(point.getX());
         this.center.setY(point.getY());
       }
       public double getRadius(){
           return this.radius;
       }

       public void setRadius(double radius) {
         this.radius = radius;
       }
       @Override
       public double getArea() {
         return PI*getRadius()*getRadius();
       }

       @Override
       public double getPerimeter() {
          return 2*PI*getRadius();
       }
       public boolean equals(Object obj) {
           if(obj instanceof  Circle){
               Circle temp = (Circle)obj;
               if(temp.getRadius()==getRadius()&&temp.hashCode()==hashCode()&&temp.getCenter().equals(getCenter()))
                   return true;
           }
           return false;
       }
       public int hashCode() {
           String str =  "Circle[center=" + getCenter().toString() + ",radius=" + getRadius()  +",filled=" + isFilled() + "]";
           return str.hashCode();
       }
       public String toString() {
           return  "Circle[center=" + getCenter().toString() + ",radius=" + getRadius() + ",color=" + getColor() +",filled=" + isFilled() + "]";
       }
}

import   java.util.*;
public class Layer {
    private List<Shape> shapes= new ArrayList<>();
    public void addShape(Shape other) {
        shapes.add(other);
    }
    public void removeCircles() {
        for(int i=0;i<shapes.size();i++) {
            if(shapes.get(i) instanceof Circle)
                shapes.remove(i);

        }
    }
    public void removeDuplicates() {

        for(int i=0;i<shapes.size()-1;i++) {
            Object a=shapes.get(i);
            for(int j=shapes.size()-1;j>i;j--) {
                Object b=shapes.get(j);
                if(a.equals(b)) {
                    shapes.remove(j);
                }
            }
        }
    }
    public String getInfo() {
        String str="Layer of crazy shapes: \n";
        for (Object shape : shapes)
            str += shape.toString() + "\n";

        return str;
    }
}

Point.java

public class Point{
    private   double x;
    private   double y;
    public Point(){}
    public Point(double x,double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point o){
        return 1.0*((int) (Math.sqrt((x-o.x)*(x-o.x)+((y-o.y)*(y-o.y)))*10000+0.5))/10000;
    }
    public boolean equals(Object obj) {
        if(obj instanceof  Point){
            Point temp = (Point)obj;
            if(temp.hashCode()==hashCode()&&temp.getX()==getX()&&temp.getY()==getY()) return true;
        }
        return false;
    }
    public int hashCode() {
        return toString().hashCode();
    }
    public String toString() {
        return "("+getX() +"," +getY() +")";
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft = new Point();
    protected double width;
    protected double length;

    public Rectangle() {}
    public Rectangle(double width,double length){
        this.width  = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point point,double width,double length,String color,boolean filled){
          super(color, filled);
          this.width  = width;
          this.length = length;
          setTopLeft(point);
    }
    public double getWidth(){
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }
    public Point getTopLeft(){
        return this.topLeft;
    }

    public void setTopLeft(Point top) {
         this.topLeft.setX(top.getX());
         this.topLeft.setY(top.getY());
    }

    @Override
    public double getArea() {
        return getWidth()*getLength();
    }

    @Override
    public double getPerimeter() {
        return 2*(getWidth()+getLength());
    }
    public boolean equals(Object obj) {
       if(obj instanceof  Rectangle){
           Rectangle temp = (Rectangle)obj;
           if(temp.getTopLeft().equals(getTopLeft())&&temp.hashCode()==hashCode()&&temp.getWidth()==getWidth()&&temp.getLength()==getLength()){
               return true;
           }
       }
       return false;
    }
    public int hashCode() {
        String str = "Rectangle[topLeft=" + getTopLeft().toString() + ",width=" + getWidth() +
                ",length=" + getLength() + ",filled=" + isFilled() + "]";
        return str.hashCode();
    }
    public String toString() {
        return "Rectangle[topLeft=" + getTopLeft().toString() + ",width=" + getWidth() +
                ",length=" + getLength() + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }
}

public class Square extends  Rectangle{
     public Square(){}
     public Square(double side){
         super(side,side);
         width = side;
     }
     public Square(double side,String color,boolean filled){
             super(side,side,color,filled);
     }
     public Square(Point point,double side,String color,boolean filled){
             super(side,side,color,filled);
             setTopLeft(point);
     }
     public double getSide(){
         return getLength();
     }
     public void setSide(double side){
         super.setLength(side);
         super.setWidth(side);
     }
    public void setLength(double side) {
        setSide(side);
    }
    public void setWidth(double side) {
        setSide(side);
    }
    public boolean equals(Object obj) {
        if(obj instanceof  Square){
            Square temp = (Square)obj;
            if(temp.getSide()==getSide()&&temp.getTopLeft().equals(getTopLeft())&&temp.hashCode()==hashCode())
                return true;
        }
        return false;
    }

        return "Square[topLeft=" + getTopLeft().toString() + ",side=" + getSide() + ",color=" + super.getColor() + ",filled=" + isFilled() + "]";
    }
}
