Circle.java

import java.util.Objects;

public class Circle extends Shape {
    private double PI = Math.PI;
    protected double radius;
    protected Point center;
    public Circle(){};
    public Circle(double radius)
    {
        this.radius = radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
    }
    public Circle(Point p,double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
        this.center = p;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return getRadius()*getRadius()*PI;
    }

    @Override
    public double getPerimeter() {
        return 2*getRadius()*PI;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Circle)) return false;
        Circle circle = (Circle) o;
        return Double.compare(circle.getRadius(), getRadius()) == 0 &&
                getCenter().equals(circle.getCenter());
    }

    @Override
    public int hashCode() {
        return Objects.hash(getRadius(), getCenter());
    
}

    public String toString()
    {
        return "Circle[center="+this.center.toString()+",radius="+getRadius()+",color="+getColor()+",filled="+isFilled()+"]";
    }
}

import java.util.ArrayList;
import java.util.List;
public class Layer {
    private List<Shape> shapes;
    public Layer()
    {
        this.shapes = new ArrayList<Shape>();
    }
    public void addShape(Shape S)
    {
        this.shapes.add(S);
    }
    public void removeCircles()
    {
        for(int i = 0;i<this.shapes.size();i++)
        {
            
if(this.shapes.get(i) instanceof Circle)
            {
                this.shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo()
    {
        String result = "Layer of crazy shapes: \n";
        for(int i = 0;i<shapes.size();i++)
        {
            result = result + 
shapes.get(i).toString()+"\n";
        }
        return result;
    }
    public void removeDuplicates()
    {
        for(int i=0;i<shapes.size();i++)
        {
            for(int j=i+1;j<shapes.size();j++)
            {
                if(shapes.get(j).hashCode() == shapes.get(i).hashCode())
                {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

Point.java

import java.util.Objects;

public class Point {
    private double x;
    private double y;
    public Point(double x,double y)
    {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p)
    {
        double dis = Math.pow((this.x-p.x),2)+ Math.pow((this.y-p.y),2);
        return Math.sqrt(dis);
    }
    public boolean equals(Object j)
    {
        if( j instanceof Point)
        {
            Point p = (Point) j;
            if(p.x == this.x && p.y == this.y) 
return true;
        }
        return false;
    }

    @Override
    public int hashCode() {
        return Objects.hash(getX(), getY());
    }

    public String toString()
    {
        return "("+this.x+","+this.y+")";
    }
}

Rectangle.java

import java.util.Objects;

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){};
    public Rectangle(double width,double length)
    {
        this.length = length;
        this.width = width;
    }
    public Rectangle(double width,double length,String color,boolean filler)
    {
        super(color, filler);
        this.length = length;
        this.width = width;
    }
    public Rectangle(Point p,double width,double length,String color,boolean filler)
    {
        super(color, filler);
        this.length = length;
        this.width = width;
        this.topLeft = p;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Rectangle)) return false;
        Rectangle rectangle = (Rectangle) o;
        return Double.compare(rectangle.getWidth(), getWidth()) == 0 &&
                Double.compare(rectangle.getLength(), getLength()) == 0 &&
                getTopLeft().equals(rectangle.getTopLeft());
    
}

    @Override
    public int hashCode() {
        return Objects.hash(getWidth(), getLength(), getTopLeft());
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getLength() {
        return length;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getWidth() {
        return width;
    }

    @Override
    public double getArea() {
        return getLength()*getWidth();
    }

    @Override
    public double getPerimeter() {
        return 2*(getLength()+getWidth());
    }
    public String toString()
    {
        return "Rectangle[topLeft="+this.topLeft.toString()+",width="+getWidth()+",length="+getLength()+",color="+getColor()+",filled="+isFilled()+"]";
    }
}
