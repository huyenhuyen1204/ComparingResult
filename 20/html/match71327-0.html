import java.util.Scanner;

public class Circle extends Shape {
    protected double radius;
    protected Point center = new Point();
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.center =  center;
        this.radius = radius;
    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius){
        this.radius = radius;
    }
    public Point getCenter(){ return this.center;}
    public void setCenter(Point center){
        this.center = center;
    }
    public double getArea() {
        return radius*radius*Math.PI;
    }
    public double getPerimeter(){
        return 2*radius*Math.PI;
    }
    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + getColor() +
                ",filled=" + isFilled() + "]";
    }
    public boolean equals(Object a){
        if(a instanceof Circle) {
            Circle another = (Circle)a;
            int r1 = (int) Math.ceil(another.radius * 1000);
            int r2 = (int) Math.ceil(this.radius * 1000);
            if (r1==r2 && this.center.equals(another.center))
                return true;
        }
        return false;
    }
    public int hashCode(){
        return 31 + (int)radius + center.hashCode();
    }
}

import java.util.*;

public class Layer {
    private List<Shape> shapes = new LinkedList<>();
    public void addShape(Shape a){
        shapes.add(a);
    }
    public void removeCircles() {
        for (int i = 0; i < shapes.size(); i++) {
            if (shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        int n = shapes.size();
        String s = "Layer of crazy shapes: " + "\n";
        for(int i=0; i<n; i++){
            s += shapes.get(i).toString() + "\n";
        }
        return s;
    }

    public void removeDuplicates(){
        for(int i=0; i<shapes.size()-1; i++){
            for(int j=i+1; j<shapes.size(); j++){
                if(shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    public static void main(String [] args)
    {
        Shape x1 = new Circle(3.1423, "RED", false);
        Shape x2 = new Circle(3.1424, "BLUE", true);
        Shape x3 = new Rectangle(5.5, 7.8, "PINK", false);
        Shape x4 = new Square(9.5, "BLACK", true);
        Layer A = new Layer();
        A.addShape(x1);
        A.addShape(x2);
        A.addShape(x1);
        A.addShape(x3);
        A.addShape(x4);
        A.addShape(x2);
        A.addShape(x3);
        A.addShape(x3);
        A.addShape(x4);
        A.removeDuplicates();
        //A.removeCircles();
        System.out.print(A.getInfo());
    }
}

import java.lang.reflect.WildcardType;

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft = new Point();
    public Rectangle(){
    }
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public  Rectangle(double width, double length, String color, boolean filled){
        super(color,filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }
    public double getWidth(){
        return width;
    }
    public void setWidth(double width){
        this.width = width;
    }
    public double getLength(){
        return length;
    }
    public void setLength(double length){
        this.length = length;
    }
    public Point getTopLeft(){
        return topLeft;
    }
    public void setTopLeft(Point topLeft){
        this.topLeft = topLeft;
    }
    public double getArea(){
        return getWidth()*getLength();
    }
    public double getPerimeter(){
        return (getWidth() + getLength())*2;
    }
    @Override
    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + getWidth() + ",length=" + getLength()+
                ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }

    public boolean equals(Object a){
        if(a instanceof Rectangle) {
            Rectangle another = (Rectangle)a;
            int w1 = (int) Math.ceil(another.width * 1000);
            int w2 = (int) Math.ceil(this.width * 1000);
            int l1 = (int) Math.ceil(another.length * 1000);
            int l2 = (int) Math.ceil(this.length * 1000);
            if (w1 == w2 && l1 == l2 && another.topLeft.equals(this.topLeft)) return true;
        }
        return false;
    }
    public int hashCode(){
        return 31 + (int)width + (int)length + topLeft.hashCode();
    }
}
