Circle.java

public class Circle extends Shape{
    protected Point center;
    protected double radius;
    public Circle(){
    }
    public Circle(double _radius){
        this.radius=_radius;
    }
    public Circle(double _radius, String color, boolean filled){
        super(color,filled);
        this.radius=_radius;
    }
    public Circle(Point _center,double _radius, String color, boolean filled){
        super(color,filled);
        this.radius=_radius;
        this.center=_center;
    }
    public Point getCenter(){
        return center;
    }
    public void setCenter(Point center){
        this.center=center;
    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius){
        this.radius=radius;
    }
    public double getArea(){
        return Math.PI*radius*radius;
    }
    public double getPerimeter(){
        return 2*Math.PI*radius;
    }
    public boolean equals(Object obj){
        if (obj instanceof Circle) {
            Circle another = (Circle) obj;
            if (Math.abs(this.getCenter().getX()-another.getCenter().getX()) <= 0.001 && Math.abs(this.getCenter().getY()-another.getCenter().getY()) <= 0.001 &&Math.abs(this.getRadius()-this.getRadius()) <= 0.001) return true;
        }
        return false;
    }
    public int hashCode(){
        return this.hashCode();
    }
    public String toString(){
        return "Circle[center=("+this.center.getX()+","+this.center.getY()+"),radius="+this.getRadius()+",color="+super.getColor()+",filled="+super.isFilled()+"]";
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {

    private List<Shape> shapes =  new ArrayList<>();

    public void addShape(Shape s) {
        shapes.add(s);
    }
    public String getInfo() {
        String list = "Layer of crazy shapes: ";
        for(int i=0; i<shapes.size(); ++i) {
            list +="\n"+ shapes.get(i).toString() ;
        }
        return list;
    }
    public void removeCircles() {
        for(int i=0; i<shapes.size(); ++i) {
            if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    }
    public void removeDuplicates() {
        for(int i=0; i<shapes.size()-1; ++i) {

            for(int j=i+1; j<shapes.size(); ++j) {
                if(shapes.get(i)  instanceof Circle && shapes.get(j)  instanceof Circle) {
                    Circle c1 = (Circle) shapes.get(i);
                    Circle c2 = (Circle) shapes.get(j);
                    if(Math.abs(c1.getRadius()-c2.getRadius()) <= 0.001 && c1.getCenter().equals(c2.getCenter())) shapes.remove(j);
                }
                if(shapes.get(i)  instanceof Rectangle && shapes.get(j)  instanceof Rectangle ) {
                    Rectangle c1 = (Rectangle) shapes.get(i);
                    Rectangle c2 = (Rectangle) shapes.get(j);
                    if(Math.abs(c1.getLength()-c2.getLength()) <= 0.001 && Math.abs(c1.getWidth()-c2.getWidth()) <= 0.001 && c1.getTopLeft().equals(c2.getTopLeft())) shapes.remove(j);
                }
                if(shapes.get(i)  instanceof Square && shapes.get(j)  instanceof Square ) {
                    Square c1 = (Square) shapes.get(i);
                    Square c2 = (Square) shapes.get(j);
                    if(Math.abs(c1.getSide()-c2.getSide()) <= 0.001 && c1.getTopLeft().equals(c2.getTopLeft())) shapes.remove(j);
                }
            }
        }
    }

}

Point.java


public class Point {
    private double x;
    private double y;
    public Point(double x,double y){
        this.x=x;
        this.y=y;
    }
    public double getX(){
        return x;
    }
    public void setX(double x){
        this.x=x;
    }
    public double getY(){
        return y;
    }
    public void setY(double y){
        this.y=y;
    }
    public double distance(Point t){
        double k=Math.pow(t.getX()-this.getX(),2)+Math.pow(t.getY()-this.getY(),2);
        return Math.sqrt(k);
    }
    public boolean equals(Object obj){
        if (obj instanceof Point) {
            Point another = (Point) obj;
            if (this.x==another.x&&this.y==another.y) return true;
        }
        return false;
    }
    public int hashCode(){
        return this.hashCode();
    }
    public String toString(){
        return "("+this.getX()+","+this.getY()+")";
    }
}

public class Square extends Rectangle{
    public Square(){
        super();
    }
    public Square(double side){
        super(side,side);
    }
    public Square(double side,String color, boolean filled){
        super(side,side,color,filled);
    }
    public Square(Point topLeft,double side,String color, boolean filled){
        super(topLeft,side,side,color,filled);
    }
    public double getSide(){
        return super.getWidth();
    }
    public void setSide(double side){
        super.width=side;
        super.length=side;
    }
    public void setWidth(double side){
        super.width=side;
        super.length=side;
    }
    public void setLength(double side){
        super.length=side;
        super.width=side;
    }
    public boolean equals(Object obj){
        return super.equals(obj);
    }
    public String toString(){
        return "Square[topLeft=("+super.getTopLeft().getX()+","+super.getTopLeft().getY()+"),side="+this.getSide()+",color="+super.getColor()+",filled="+super.isFilled()+"]";
    }
}
