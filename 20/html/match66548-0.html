public class Layer {
    List<Shape> shapes=new LinkedList<Shape>();
    public void addShape(Shape x){
        shapes.add(x);
    }
    public void removeCircles(){
        for (int i=0;i<shapes.size();i++){
            if (shapes.get(i) instanceof Circle ){
                shapes.remove(i);
            }
        }
    }
    public String getInfo(){
        String s="Layer of crazy shapes: \n";
        for (Shape shape: shapes){
            s=s+shape.toString()+'\n';
        }
        return s;
    }
    public void removeDuplicates(){
        for (int i=0;i<shapes.size();i++){
            for (int j=0;j<shapes.size();j++){
                if (i!=j &&shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                }
            }
        }
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){}
    public Point getTopLeft(){
        return this.topLeft;
    }
    public void setTopLeft(Point x){
        this.topLeft=x;
    }
    public Rectangle(double width,double length)
    {
        this.width=width;
        this.length=length;
    }
    public Rectangle(double width,double length,String color, boolean filled)
    {
        super(color, filled);
        this.width=width;
        this.length=length;
    }
    public Rectangle(Point topLeft,double width,double length,String color,boolean _filled){
        this.topLeft=topLeft;
        this.length=length;
        this.width=width;
        this.color=color;
        this.filled=_filled;
    }
    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        return (double) Math.round(width*length*100)/100;
    }
    @Override
    public double getPerimeter() {
        return (width+length)*2;
    }
    public boolean equals(Object c){
        if (c instanceof Rectangle){
            Rectangle rec=(Rectangle) c ;
            if (this.width==rec.width&&this.length==rec.length&&this.getTopLeft().equals(rec.getTopLeft()));
                return true;
        }
        return false;
    }
    public int hashCode(){
        return this.hashCode();
    }
    @Override
    public String toString() {
        return "Rectangle[topLeft=("+this.topLeft.x+","+this.topLeft.y+"),width="+this.getWidth()+",length="+this.getLength()+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }
}

public class Square extends Rectangle {
    public Square() {
    }

    public Square(double side) {
        super(side,side);
    }

    public Square(double side, String color, boolean filled) {
        super(side,side,color,filled);
    }
    public Square(Point point,double side,String color,boolean filled){
        super(point,side,side,color,filled);
    }
    public double getSide() {
        return length;
    }

    public void setSide(double side) {
        this.length=side;
        this.width=side;
    }

    @Override
    public void setWidth( double side) {
        this.width=side;
        this.length=side;
    }
    public void setLength(double side){
        this.length=side;
        this.width=side;
    }

    @Override
    public String toString() {
        return "Square[topLeft=("+this.topLeft.x+","+this.topLeft.y+"),side=" +this.getSide() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
    public boolean equals(Object c){
        if (c instanceof Square){
            Square square=(Square) c;
            if (this.width==square.getSide()&&this.getTopLeft().equals(square.getTopLeft())){
                return true;
            }
        }
        return false;
        
    }
}
