src/Circle.java

public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){
    };
    public Circle(double radius)
    {
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius=radius;
    }

    public Circle(Point center, double radius,String color,boolean filled){
        super(color,filled);
        this.center=center;
        this.radius=radius;
    }

    public double getRadius() {
        return radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return radius*radius*Math.PI;
    }
    public double getPerimeter(){
        return Math.PI*radius*2;
    }

    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle other = (Circle) obj;
            if(other.getCenter()==this.getCenter() && other.radius==this.getRadius()){
                return true;
            }
            return false;
        }
        return false;
    }

    @Override
    public String toString() {
        return "Circle[center=" +getCenter() + ",radius="+getRadius()+",color="+super.getColor()+",filled="+super.isFilled()+"]";
    }
}

src/Layer.java

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape tmp){
        shapes.add(tmp);
    }
    public void removeCircles(){
        for(int i=0;i<shapes.size();i++){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
            }
        }
    }
    public String getInfo(){
        String result="";
        for(int i=0;i<shapes.size();i++){
            result=result+shapes.get(i).toString()+"\n";
        }
        return "Layer of crazy shapes: "+"\n"+result;
    }
    public void removeDuplicates(){
        for(int i=0;i<=shapes.size()-1;i++){
            for(int j=i+1;j<shapes.size();j++){
                if(shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                }
            }
        }
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;
    public Point(double x, double y){
        this.x=x;
        this.y=y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p){
        return Math.sqrt((this.x - p.x)*(this.x - p.x) + (this.y - p.y)*(this.y -p.y));
    }
    public boolean equals(Object obj){
        if(obj instanceof Point) {
            Point other = (Point) obj;
            if(other.x==this.x && other.y==this.y){
                return true;
            }
            else {
                return false;
            }
        }
        return false;
    }
    public String toString(){
        return "("+ x +","+y+")";
    }
}

src/Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){}
    public Rectangle(double width,double length)
    {
        this.width=width;
        this.length=length;
    }
    public Rectangle(double width,double length,String color, boolean filled)
    {
        super(color, filled);
        this.width=width;
        this.length=length;
    }

    public Rectangle(Point topLeft,double width,double length,String color, boolean filled ){
        super(color,filled);
        this.topLeft = topLeft;
        this.width=width;
        this.length=length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return (double) Math.round(width*length*100)/100;
    }
    @Override
    public double getPerimeter() {
        return (double)(width+length)*2;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=" +getTopLeft()+ ",width="+getWidth()+",length="+getLength()+",color="+getColor()+",filled="+isFilled()+"]";
    }

    public boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle other = (Rectangle) obj;
            if(other.getLength()==this.getLength() && other.getWidth()==this.getWidth()){
                return true;
            }
            return false;
        }
        return false;
    }
}
