class Circle extends Shape {
	protected double radius;
	protected Point center;
	public Circle(){
		
	}
	public Circle(double radius){
		this.radius = radius;
	}
	public Circle(double radius, String color, boolean filled){
		super(color, filled);
		this.radius = radius;
	}
	public Circle(Point center, double radius, String color, boolean filled){
		super(color, filled);
		this.radius = radius;
		this.center = center;
	}
	
	public double getRadius() {
		return this.radius;
	}
	public void setRadius(double radius) {
		this.radius = radius;
	}
	
	public Point getCenter() {
		return center;
	}
	public void setCenter(Point center) {
		this.center = center;
	}
	public boolean equals(Object obj){
		if(obj instanceof Circle){
			Circle c = (Circle)obj;
			if(Math.abs(this.radius - c.radius) >=0 && Math.abs(this.radius - c.radius) <= 0.001)
				return true;
			else 
return false;
		}
		return false;
	}
	public int hashCode(){
		return 0;
		
	}
	@Override
	public double getArea() {
		return Math.PI*this.getRadius()*this.getRadius();
	}

	@Override
	public double getPerimeter() {
		return 2*Math.PI*this.getRadius();
	}
	@Override
	public String toString(){
		Point c = this.getCenter();
		double x1 = c.getX();
		double y1 = c.getY();
		y1 = Math.round( (y1*10)/10 );
		x1 = Math.round( (x1*10)/10 );
		return "Circle[center=("+ x1+","+ y1 + ")" + ",radius=" + this.getRadius() + ",color="
		+ this.getColor() + ",filled=" + this.isFilled() + "]";
		
	}
	

	
}

import java.util.List;

public class Layer {
	private List<Shape> shapes;
	public void addShape(Shape shape){
		shapes.add(shape);
	}
	public void removeCircles(){
		if(shapes != null) {
		for(int i = 0; i < shapes.size(); i++){
			if(shapes.get(i) instanceof Circle){
				shapes.remove(i);
			}
		}
	}
}
	public String getInfo(){
		String s = "";
		for(int i = 0; i < shapes.size(); i++){
			s += shapes.get(i).toString() +"\n";
		}
		return s;
	}
	public void removeDuplicates(){
		if(shapes.size() != 0){
		
		
for(int i = 0; i < shapes.size() - 1; i++){
			for(int j = i + 1; j < shapes.size() ; j++){
				if(shapes.get(i).equals(shapes.get(j))){
					shapes.remove(j);
					
				}
			}
		}
	}
	}	
}

Point.java




public class Point {
	private double x;
	private double y;
	public Point(double x, double y){
		this.x = x;
		this.y = y;
	}
	public double getX() {
		return x;
	}
	public void setX(double x) {
		this.x = x;
	}
	public double getY() {
		return y;
	}
	public void setY(double y) {
		this.y = y;
	}
	public double distance(Point p){
		return Math.sqrt( Math.pow(this.x - p.x, 2) + Math.pow(this.y - p.y, 2) );
	
}
	public boolean equals(Object obj){
		if(obj instanceof Point){
			Point p = (Point) obj;
			if( Math.abs(this.x - p.x) >= 0 &&  Math.abs(this.x - p.x) <= 0.001
				&& Math.abs(this.y - p.y) >= 0 &&  Math.abs(this.y - p.y) <= 0.001	) return true;
		}
		
		return false;
		
	}
	public String toString(){
		y = Math.round( (y*10)/10 );
		x = Math.round( (x*10)/10 );
		return "(" + x + "," + y + ")";
	}
	
}

public class Rectangle extends Shape {
	protected double width;
	protected double length;
	protected Point topLeft;
	public Rectangle() {
	}
	public Rectangle(double width, double length){
		this.width  = width;
		this.length = length;
	}
	public Rectangle(double width, double length, String color, boolean filled){
		super(color, filled);
		this.width  = width;
		this.length = length;
	}
	public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
		super(color, filled);
		this.topLeft = topLeft;
		this.width  = width;
		this.length = length;
	}
	public double getWidth() {
		return width;
	}
	public void setWidth(double width) {
		this.width = width;
	}
	public double getLength() {
		return length;
	}
	public void setLength(double length) {
		this.length = length;
	}
	
	public Point getTopLeft() {
		return topLeft;
	}
	public void setTopLeft(Point topLeft) {
		this.topLeft = topLeft;
	}
	public boolean equals(Object obj){
		if(obj instanceof Rectangle){
			Rectangle r = (Rectangle)obj;
			Point p1 = r.getTopLeft();
			Point p = this.getTopLeft();
			double x = p.getX();
			double y = p.getY();
			double x1 = p1.getX();
			double y1 = p1.getY();
			
if(Math.abs(this.length - r.length) >= 0 &&  Math.abs(this.length - r.length) <= 0.001 
			&& Math.abs(this.width - r.width) >= 0 &&  Math.abs(this.width - r.width) <= 0.001
			&& Math.abs(x - x1) >= 0 &&  Math.abs(x - x1) <= 0.001
			&& Math.abs(y - y1) >= 0 &&  Math.abs(y - y1) <= 0.001  ) 
				return true;
			return false;
		
}
		return false;
	}
	@Override
	public double getArea() {
		return this.getWidth()*this.getLength();
	}
	@Override
	public double getPerimeter() {
		return (this.getWidth() + this.getLength())*2;
	}
	@Override
	public String toString(){
		Point tl = this.getTopLeft();
		double x = tl.getX();
		double y = tl.getY();
		y = Math.round( (y*10)/10 );
		x = Math.round( (x*10)/10 );
		return "Rectangle[topLeft=("+ x + "," + y + ")" +  ",width=" + this.getWidth() + ",length=" 
		+ this.getLength() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
	}
	
}

public class Square extends Rectangle {
	public Square(){
	}
	public Square(double side){
		super(side, side);
	}
	public Square(double side, String color, boolean filled){
		super(side, side, color, filled);
	}
	public Square(Point topLeft, double side, String color, boolean filled){
		super(side, side, color, filled);
		this.topLeft = topLeft;
	}
	public boolean equals(Object obj){
		if(obj instanceof Square){
			Square sq = (Square) obj;
			Point p = this.getTopLeft();
			Point p1 = sq.getTopLeft();
			double x = p.getX();
			double y = p.getY();
			double x1 = p1.getX();
			double y1 = p1.getY();
			double s = this.getSide();
			double s1 = sq.getSide();
			if( Math.abs(x - x1) >= 0 &&  Math.abs(x - x1) <= 0.001
				&& Math.abs(y - y1) >= 0 &&  Math.abs(y - y1) <= 0.001
				&& Math.abs(s - s1) >= 0 &&  Math.abs(s - s1) <= 0.001
				) return true;
			return false;
			
		}
		return false;
	
}
	
	@Override
	public void setWidth(double width) {
		super.setWidth(width);
	}
	@Override
	public void setLength(double length) {
		super.setLength(length);
	}
	public void setSide(double side) {
		this.setLength(side);
		this.setWidth(side);
	}
	public double getSide(){
		return this.getWidth();
	}
	public String toString(){
		Point tl = this.getTopLeft();
		double x = tl.getX();
		double y = tl.getY();
		y = Math.round( (y*10)/10 );
		x = Math.round( (x*10)/10 );
	return "Square[topLeft=("+ x + "," + y + ")" + ",side=" + this.getWidth() + ",color="  
	+ this.getColor() + ",filled="  +this.isFilled() + "]";
	}
	
}
