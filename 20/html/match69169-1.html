class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point position, double radius, String color, boolean filled){
        super(color, filled);
        center = new Point(position.getX(), position.getY());
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea(){
        return Math.PI * radius * radius ;
    }

    public double getPerimeter(){
        return 2 * Math.PI * radius;
    }

    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle other = (Circle)obj;
            if(
Math.abs(this.radius - other.radius) <= 0.001){
                return true;
            }
        }
        return false;
    }

    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + this.radius + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }



}

public class Layer {
    private List<Shape> shapes;
    Layer(){
        shapes = new ArrayList();
    }
    public void addShape(Shape obj){
            shapes.add((Shape) obj);
    }

    public String getInfo(){
        String result = "Layer of crazy shapes: ";
        for(int i = 0; i < shapes.size(); i++){
            result += '\n';
            result += shapes.get(i).toString();
            //if (i != shapes.size() - 1) result += '\n';
        }
        return result;
    }
    public void removeCircles(){
        for(int i = 0; i < shapes.size(); ++i){
            if( shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            
}
        }
    }
    public void removeDuplicates(){
        for (int i = 0; i < shapes.size(); ++i){
            for(int j = i + 1; j < shapes.size(); j++){
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle) {
                    if (shapes.get(i).equals(shapes.get(j))
                && ((Circle) shapes.get(i)).getCenter().getX() == ((Circle) shapes.get(j)).getCenter().getX()
                && ((Circle) shapes.get(i)).getCenter().getY() == ((Circle) shapes.get(j)).getCenter().getY())
                    {
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle) {
                    if (shapes.get(i).equals(shapes.get(j))
     //                && shapes.get(i).getColor() == shapes.get(j).getColor()
                     && ((Rectangle) shapes.get(i)).getTopLeft().getX() == ((Rectangle) shapes.get(j)).topLeft.getX()
                     && ((Rectangle) shapes.get(i)).getTopLeft().getY() == ((Rectangle) shapes.get(j)).topLeft.getY())
                    {
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Square) {
                    if (shapes.get(i).equals(shapes.get(j))
                    && ((Square) shapes.get(i)).getTopLeft().getX() == ((Square) shapes.get(j)).topLeft.getX()
                    && ((Square) shapes.get(i)).getTopLeft().getY() == ((Square) shapes.get(j)).topLeft.getY())
                    {
                        shapes.remove(j);
                    }
                }
            }
        }
    }
}

import static java.lang.StrictMath.pow;

/**
 * Created by CCNE on 21/10/2019.
 */
public class Point {
    private double x;
    private double y;
    public Point(){
        this.x = 0;
        this.y = 0;
    }
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point other){
        double 
result = Math.sqrt(pow(this.x - other.x, 2) + pow(this.y - other.y, 2));
        return result;
    }
    public String toString(){
        return "(" + this.x + "," + this.y + ")";
    }

    public boolean equals(Object obj){
        if(obj instanceof Point){
            Point other = (Point) obj;
            if(this.x == other.x && this.y == other.y) {
                return true;
            }
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}

public class Rectangle extends Shape {
    protected Point topLeft;

    }
    public Rectangle(double width, double length){
        this.length = length;
        this.width = width;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point position, double width, double length, String color, boolean filled){
        super(color, filled);
        topLeft = new Point(position.getX(), position.getY());
        this.width = width;
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    
}
    public void setLength(double length) {
        this.length = length;
    }

    public double getArea(){
        return (double)Math.round(width * length * 100) / 100;
    }

    public double getPerimeter(){
        return (double)Math.round(2 * (width + length) * 100) / 100;
    }

    public boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle other = (Rectangle)obj;
            if(this.width - other.width < 0.001 && this.length - other.length < 0.001){
                
return true;
            }
        }
        return false;
    }


    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + this.width +",length=" + this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

    @Override
    public int hashCode() {
        return super.hashCode();

/**
 * Created by CCNE on 21/10/2019.
 */
public abstract class Shape {
    protected String color;
    protected boolean filled;
    public Shape(){

    }
    public Shape(String color, boolean filled){
        this.color = color;
        this.filled = filled;
    }

    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public boolean isFilled() {
        return filled;
    }

    public void setFilled(boolean filled) {
        this.filled = filled;
    }

    abstract public double getArea();
    abstract public double getPerimeter();

    
@Override
    public int hashCode() {
        return super.hashCode();
    }

    public String toString(){
        return "Shape[color=" + this.color + ",filled=" + this.filled + "]";
    }
}
