import java.awt.*;
import java.util.Objects;

public class Rectangle extends Shape{
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){}
    public Rectangle(double wod,double len){
        width=wod;
        length=len;
    }
    public Rectangle(double wod,double len,String col,boolean fil){
        super(col,fil);
        width=wod;
        length=len;
    }
    public Rectangle(Point topl,double wod,double len,String col,boolean fil){
        super(col,fil);
        topLeft=topl;
        width=wod;
        length=len;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }
    
@Override
    public double getArea(){
        return width*length*100.0/100.0;
    }
    @Override
    public double getPerimeter(){
        return 2*(width+length);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Rectangle rectangle = (Rectangle) o;
        return Double.compare(rectangle.width, width) == 0 &&
                Double.compare(rectangle.length, length) == 0 &&
                topLeft.equals(rectangle.topLeft);
    }

    @Override
    public int hashCode() {
        return Objects.hash(topLeft, width, length);
    }

    @Override
    public String toString() {
        return "Rectangle[" +
                "topLeft=" + topLeft +
                ",width=" + width +
                ",length=" + length +
                ",color=" +color+
                ",filled=" +filled+
                "]";
    }

    public static void main(String[] args) {
        Point point= new Point(10.5,5);
        Rectangle rectangle= new Rectangle(point,8.0,6.0, "BLUE", true);
        System.out.println(rectangle.toString());

    }
}

import java.awt.*;
import java.util.Objects;

public class Square extends Rectangle {
    public Square(){}
    public Square(double sid){
        super(sid,sid);
    }
    public Square(double sid,String col, boolean fil){
        super(sid,sid,col,fil);
    }
    public Square(Point topl,double sid,String col, boolean fil){
        super(topl,sid,sid,col,fil);
    }

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
    }
    public double getSide(){
        return width;
    }
    public void setSide(double side){
        width=side;
        length=side;
    }
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        
if (o == null || getClass() != o.getClass()) return false;
        Square square = (Square) o;
        return Double.compare(square.getSide(), getSide()) == 0 &&
                topLeft.equals(square.topLeft);
    }

    @Override
    public int hashCode() {
        return Objects.hash(topLeft,getSide());
    }

    @Override
    public String toString(){
        return "Square[topLeft="+topLeft+
                ",side="+width+
                ",color="+color+
                ",filled="+filled+
                "]";
    }

    public static void main(String[] args) {
        Point x=new Point(10,5);
        Shape shape7=new Square(1,"red",false);
        Shape shape8=new Square(x,1,"Blue",false);
        Shape shape9=new Square(2,"red",false);
        System.out.println(shape7.hashCode());
        System.out.println(shape8.hashCode());
        System.out.println(shape9.hashCode());
    }

}
