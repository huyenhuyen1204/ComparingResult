public class Circle extends Shape {
    protected Point center;
    protected double radius;

    //Constructor
    public Circle() {
};

    public Circle(double radius) {
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public  Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    //Function

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return radius * radius * Math.PI;
    }

    @Override
    public double getPerimeter() {
        return radius * 2.0 * Math.PI;
    }

    @Override
    public boolean equals(Object obj) {
        if(obj instanceof Circle) {
            Circle temp = (Circle) obj;
            return this.getCenter().equals(temp.getCenter()) && this.getRadius() == temp.getRadius();
        }
        return false;
    }

    @Override
    public int hashCode() {
        return this.hashCode();
    }

    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + color + ",filled=" + filled + "]";
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape other) {
        shapes.add(other);
    }

    public void removeCircles() {
        for (int i = 0; i < shapes.size(); i++) {
            
if (shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo() {
        String res = "Layer of crazy shapes: ";
        for(int i = 0; i < shapes.size(); i++) {
            res += "\n" + shapes.get(i);
        
}
        return res;
    }

    public void removeDuplicates() {
        for (int i = 0; i < shapes.size()-1; i++) {
            for (int j = i + 1; j < shapes.size(); j++) {
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

import java.util.*;

public class Point {
    private double x;
    private double y;

    //Constructor
    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    //Function
    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other) {
        return Math.sqrt((
other.getX() - this.getX())*(other.getX() - this.getX()) + (other.getY() - this.getY()) * (other.getY() - this.getY()));
    }

    @Override
    public 
boolean equals(Object obj) {
        if (obj instanceof Point) {
            Point temp = (Point) obj;
            return Math.round(this.x * 1000) == Math.round(temp.x * 1000)
                    && Math.round(this.y * 1000) == Math.round(temp.y * 1000);
        }
        return false;
    }

    @Override
    public int hashCode() {
        return this.hashCode();
    }

    public String toString() {
        return "(" + String.format("%.1f", this.getX()) + "," + String.format("%.1f",this.getY())+ ")";
    }
}

public class Square extends Rectangle {
    //Constructor
    public Square() {};

    public Square(double side) {
        this.width = side;
        this.length = side;
    }

    public Square(double side, String color, boolean filled) {
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled) {
        super(side, side, color, filled);
        this.topLeft = topLeft;
    }

    // Function
    public double getSide() {
        return this.length;
    }

    public void setSide(double side) {
        this.width = side;
        this.length = side;
    }

    @Override
    public void setWidth(double side) {
        this.width = side;
        this.length = side;
    }

    @Override
    public void setLength(double side) {
        this.length = side;
        this.width = side;
    }

    @Override
    public 
boolean equals(Object obj) {
        if(obj instanceof Square) {
            Square temp = (Square) obj;
            return this.getTopLeft().equals(temp.getTopLeft())  && this.getWidth() == temp.getWidth() && 
this.getLength() == temp.getLength();
        }
        return false;
    }

    @Override
    public int hashCode() {
        return this.hashCode();
    }

    public String toString() {
        return "Square[topLeft=" + this.topLeft.toString() + ",side=" + this.length + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
}
