public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center,double radius, String color, boolean filled)
    {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Circle) {
            Circle other = (Circle) obj;
            if(other.center.equals(this.center))
            {
               if(Math.abs(this.radius - other.radius) <= 0.001)
               {
                   return true;
               }
            }
        }
        return false;
    }

    public int hashCode(){
         return this.center.hashCode() ^ (int) (this.radius *37);
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + color + ",filled=" + filled + "]";
    }
    @Override
    public double getArea() {
        double area = radius * radius * PI;
        return area;
    }

    @Override
    public double getPerimeter() {
        double perimeter = 2 * radius * PI;
        return perimeter;
    }
}

import java.util.ArrayList;
import java.util.List;


public class Layer {
    private List<Shape> shapes;
    public Layer(){
        shapes = new ArrayList<Shape>();
    }
    public void addShape(Shape shape){
        shapes.add(shape);
    }
    public void removeCircles(){
        for(int i = 0; i < shapes.size() ; i++){
            if(shapes.get(i) instanceof Circle){
                //System.out.println("11");
                shapes.remove(i);
                i --;
            }
        }
    }
    public String getInfo(){
        String info = "Layer of crazy shapes: \n";
        for(int i = 0; i < shapes.size(); i ++){
            info += shapes.get(i).toString();
            info += "\n";
        }
        return info;
    }
    public void removeDuplicates(){
        for(int i=0; i < shapes.size(); i++) {
            for(int j = i+1; j < shapes.size(); j++) {
                if(shapes.get(i).equals(shapes.get(j)) ) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
        }
}

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){

    }
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.length = length;
        this.width = width;
    }
    public Rectangle(Point topLeft,double width, double length, String color, boolean filled){
        super(color, filled);
        this.length = length;
        this.width = width;
        this.topLeft = topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    @Override
    public double getArea() {
        double area = width * length;
        return area;
    }

    @Override
    public double getPerimeter() {
        double Perimeter = 2 * (width + length);
        return Perimeter;
    }
    public String toString(){
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" +color + ",filled=" + filled + "]";
    }

    public int hashCode() {
       return (int) this.length * this.topLeft.hashCode() ^ (int) (this.width *37) ;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle) {
            Rectangle other = (Rectangle) obj;
            if(other.topLeft.equals(this.topLeft))
            {
                if(Math.abs(this.width - other.width) > 0.001) return false;
                if(Math.abs(this.length - other.length) > 0.001) return false;
                return true;
            }
        }
        return false;
    }
}
