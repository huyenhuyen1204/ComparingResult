Week6-1/Circle.java


public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center,double radius, String color, boolean filled){
        super(color,filled);
        this.radius = radius;
        this.center = center;
    }
    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    public double getArea(){
        return (Math.PI*Math.pow(this.radius,2));
    }
    public double getPerimeter(){
        return(Math.PI*2*this.radius);
    }
    public String toString(){
        radius = Math.round(radius*10.0)/10.0f;
        return "Circle[center=" + center.toString() + ",radius=" + radius  +",color=" + color + ",filled=" +  filled + "]";
    }
    public boolean equals(Object obj){
        if(getClass() != obj.getClass()) return false;
        Circle circle = (Circle) obj;
        return Math.abs(center.getX() - circle.getCenter().getX()) <= 0.001 &&
                Math.abs(center.getY() - circle.getCenter().getY()) <= 0.001 &&
                Math.abs(radius - circle.getRadius()) <= 0.001;
    }
    public int hashCode(){
        return (int)radius*10 + 3;
    }

}

Week6-1/Point.java

public class Point {
    private  double x;
    private  double y;
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p){
        return Math.sqrt(Math.pow(this.x - p.x,2) + Math.pow(this.y - p.y,2));
    }
    public boolean equals(Object obj){
        if(obj.getClass() != getClass()) return false;
        Point p = (Point) obj;
        if(this.x == p.getX() && this.y == p.getY()) return true;
        return false;
    }
    public String toString(){
        x = Math.round(x*10.0)/10.0f;
        y = Math.round(y*10.0)/10.0f;
        return "(" + x + "," + y + ")";
    }

}

Week6-1/Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){

    }
    public Rectangle(double width, double length){
        this.length = length;
        this.width = width;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }public Rectangle(Point TopLeft,double width, double length, String color, boolean filled){
        this.topLeft = TopLeft;
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public double getArea(){
        return (this.length*this.width);
    }
    public double getPerimeter(){
        return (this.length + this.width)*2;
    }
    public Point getTopLeft(){
        return topLeft;
    }
    public void setTopLeft(Point p){
        this.topLeft = p;
    }
    public boolean equals(Object obj){
        if(getClass() != obj.getClass()) return false;
        Rectangle rectangle = (Rectangle) obj;
        return Math.abs(topLeft.getX() - rectangle.topLeft.getX()) <= 0.001 &&
                Math.abs(topLeft.getY() - rectangle.topLeft.getY()) <= 0.001 &&
                Math.abs(getLength() - rectangle.getLength()) <= 0.001 &&
                Math.abs(getWidth() - rectangle.getWidth()) <= 0.001;
    }
    public String toString(){
        width = Math.round(width*10.0f) / 10.0f;
        length = Math.round(length*10.0f) / 10.0f;
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + color +",filled=" + filled +"]";
    }
    public int hashCode(){
        return (int)(width + length) + (int)(width*length);
    }
}
