Circle.java

public class Circle extends Shape
{
    protected double radius;
    protected Point center;

    public Circle(){}

    public Circle(double radius)
    {
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled)
    {
        super(color, filled);
        this.radius = radius;
    }
    
    public Circle(Point e, double radius, String color, boolean filled)
    {
        super(color, filled);
        this.center = e;
        this.radius = radius;
    }

    public Point getCenter(){return this.center;}
    public void setCenter(Point e)
    {
        this.center = e;
    }

    public double getRadius()
    {
        return this.radius;
    }
    public void setRadius(double radius)
    {
        this.radius = radius;
    }

    @Override
    public double getArea()
    {
        return (Math.PI * radius * radius);
    }

    @Override 
    public double getPerimeter()
    {
        return (Math.PI * 2 * radius);
    }

    public String toString()
    {
        radius = Math.round(radius * 10) / 10.0;
        String s = "Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + color + ",filled=" + filled + "]";
        return s;
    }

    public boolean equals(Object b)
    {
        if(b instanceof Circle)
        {
            Circle e = (Circle) b;
            if(e.radius == this.radius && e.center.equals(this.center))
                return true;
        }
        return false;
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer
{
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape e)
    {
        shapes.add(e);
    }

    public String getInfo()
    {
        String s = "Layer of crazy shapes: \n";
        for(int i = 0; i < shapes.size(); i++)
        {
            s += shapes.get(i).toString() + "\n";
        }
        return s;
    }

    public void removeCircles()
    {
        for(int i = 0; i < shapes.size(); i++)
        {
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i = -1;
            }
        }
    }

    public void removeDuplicates()
    {
        for(int i = 0; i < shapes.size() - 1; i++)
            for(int j = i + 1; j < shapes.size(); j++)
            {
                if(shapes.get(i).equals(shapes.get(j)))
                {
                    shapes.remove(j);
                }
            }
    }
    
    public static void main(String[] args)
    {
        Point t = new Point(5, 8);
        Shape e = new Circle(t, 5, "RED", false);
        t = new Point(10, 9);
        Point i = new Point(2, 2);
        Shape e1 = new Rectangle(t, 10, 5, "BLUE", true);
        Shape e2 = new Rectangle(t, 12, 6, "BLUE", true);
        Shape e3 = new Rectangle(t, 14, 7, "BLUE", true);
        Shape e4 = new Rectangle(t, 16, 8, "BLUE", true);
        Shape e5 = new Square(t, 7, "YELLOW", false);
        Layer l = new Layer();
        l.shapes.add(e);
        l.shapes.add(e);
        l.shapes.add(e1);
        l.shapes.add(e2);
        l.shapes.add(e3);
        l.shapes.add(e4);
        l.shapes.add(e);
        l.shapes.add(e5);
        l.shapes.add(e5);
        l.removeDuplicates();
        System.out.println(l.getInfo());
        Shape e6 = new Square(i, 7, "RED", true);
        if(e6.equals(e5)) System.out.println("YES");
    }
}

import java.math.*;

public class Point{
    private double x;
    private double y;

    public Point(double x, double y)
    {
        this.x = x;
        this.y = y;
    }

    public double getX()
    {
        x = Math.round(x * 10) / 10.0;
        return this.x;
    }
    public void setX(double x)
    {
        this.x = x;
    }

    public double getY()
    {
        y = Math.round(y * 10) / 10.0;
        return this.y;
    }
    public void setY(double y)
    {
        this.y = y;
    }

    public double distance(Point e)
    {
        return Math.round((Math.sqrt(x * x + y * y)) * 10) / 10.0;
    }

    public boolean equals(Object b)
    {
        if(b instanceof Point)
        {
            Point e = (Point) b;
            if(e.x == this.x && e.y == this.y)
                return true;
        }
        return false;
    }

    public int hashCode(){return super.hashCode();}
    public String toString()
    {
        String s = "(" + x + "," + y + ")";
        return s;
    }
}

Rectangle.java

public class Rectangle extends Shape
{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){}
    public Rectangle(double width, double length)
    {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled)
    {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point e,double width, double length, String color, boolean filled)
    {
        super(color, filled);
        this.topLeft = e;
        this.width = width;
        this.length = length;
    }

    public double getWidth()
    {
        width = Math.round(width * 10) / 10.0;
        return this.width;
    }
    public void setWidth(double width)
    {
        this.width = width;
    }
    public double getLength()
    {
        length = Math.round(length * 10)/ 10.0;
        return this.length;
    }
    public void setLength(double length)
    {
        this.length = length;
    }
    public Point getTopLeft(){return this.topLeft;}
    public void setTopLeft(Point e)
    {
        this.topLeft = e;
    }

    @Override
    public double getArea()
    {
        return (length * width);
    }
    @Override
    public double getPerimeter()
    {
        return ((length + width) * 2);
    }

    public String toString()
    {
        width = Math.round(width * 10) / 10.0;
        length = Math.round(length * 10)/ 10.0;
        String s = "Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + color + ",filled=" + filled + "]";
        return s;
    }

    public boolean equals(Object b)
    {
        if(b instanceof Rectangle)
        {
            Rectangle e = (Rectangle) b;
            if(e.width == this.width && e.length == this.length && e.topLeft.equals(this.topLeft))
                return true;
        }
        return false;
    }
}

Square.java

public class Square extends Rectangle
{
    public Square(){}
    public Square(double side)
    {
        super(side, side);
    }
    public Square(double side, String color, boolean filled)
    {
        super(side, side, color, filled);
    }
    public Square(Point e, double side, String color, boolean filled)
    {
        super(e, side, side, color, filled);
    }

    public double getSide()
    {
        return super.getWidth();
    }
    public void setSide(double side)
    {
        super.setLength(side);
        super.setWidth(side);
    }
    
    public void setWidth(double side)
    {
        super.setWidth(side);
        super.setLength(side);
    }

    public void setLength(double side)
    {
        super.setLength(side);
        super.setWidth(side);
    }

    public String toString()
    {
        double x = Math.round(super.getLength() * 10) / 10.0;
        String s = "Square[topLeft=" + super.getTopLeft().toString() + ",side=" + x + ",color=" + super.color + ",filled=" + super.filled + "]";
        return s;
    }

    public boolean equals(Object b)
    {
        if(b instanceof Square)
        {
            Square e = (Square) b;
            if(e.getSide() == this.getSide() && e.topLeft.equals(this.topLeft))
                return true;
        }
        return false;
    }
}
