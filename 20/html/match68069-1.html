import java.util.Objects;

public 
class Circle extends Shape{
    protected Point center ;

    protected double radius;
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point point, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = point;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return this.radius* Math.PI * this.radius;
    }

    @Override
    public double getPerimeter() {
        return this.radius * 2 * Math.PI;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public boolean equals(Object obj) {
        if(obj instanceof Circle){
            Circle other = (Circle) obj;
            if(this.getCenter().getX() == other.getCenter().getX() && this.getCenter().getY() == other.getCenter().getY() && this.getRadius() == other.getRadius())
                return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "Circle[center=(" + this.getCenter().getX() + "," + this.getCenter().getY() + "),radius=" + this.getRadius() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }

}

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape other){
        this.shapes.add(other);
    }
    public void removeCircles(){
        for(int i=0; i<shapes.size()-1;i++){
            
if (shapes.get(i) instanceof Circle)
            {
                this.shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        String result = "Layer of crazy shapes: \n";
        for(Shape other : this.shapes){
            result += other.toString() + "\n";
        
}
        return result;
    }
    public void removeDuplicates(){
        for(int i = 0; i < shapes.size() - 1; i++){
            for(int j = i + 1; j < shapes.size(); j++)
                if(shapes.get(i).equals(shapes.get(j))) {
                    this.shapes.remove(j);
                    j--;
                }

        }

    }
}
