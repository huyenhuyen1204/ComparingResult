Circle.java

public class Circle extends Shape
{
    protected Point center;
    protected double radius;
    public final double PI = Math.PI;

    public Circle(){};
    public Circle(double radius)
    {
        this.setRadius(radius);
    }
    public Circle(double radius, String color, boolean filled)
    {
        super(color, filled);
        this.setRadius(radius);
    }
    public Circle(Point p, double radius, String color, boolean filled)
    {
        super(color, filled);
        this.setRadius(radius);
        this.setCenter(p);
    }
    public Point getCenter()
    {
        return this.center;
    }
    public void setCenter(Point p)
    {
        this.center = p;
    }
    public double getRadius()
    {
        return this.radius;
    }
    public void setRadius(double radius)
    {
        this.radius = radius;
    }
    public double getArea()
    {
        return this.getRadius()*this.getRadius()*PI;
    }

    public double getPerimeter()
    {
        return 2*PI*this.getRadius();
    }
    public boolean equals(Object o)
    {
        if (o instanceof Circle)
        {
            Circle 
c = (Circle) o;
            return  Math.abs(this.radius-c.radius) <=0.001
                    && Math.abs(this.getCenter().getX()-c.getCenter().getX()) <= 0.001
                    && Math.abs(this.getCenter().getY()-c.getCenter().getY()) <= 0.001 ;
        }
        return false;
    }
    public int hashCode()
    {
        return super.hashCode();
    }
    public String toString()
    {
        return "Circle[center=(" + Math.ceil(this.getCenter().getX()*10)/10 + "," + Math.ceil(this.getCenter().getY()*10)/10 + "),radius=" + Math.ceil(this.getRadius()*10)/10 + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
}

import java.util.*;
public class Layer
{
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape a)
    {
        shapes.add(a);
    }
    public void removeCircles()
    {
        for (int i=0; i<shapes.size(); i++)
        {

            
if (shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo()
    {
        String result = "Layer of crazy shapes: \n";
        for (Shape s : shapes)
        {
            result = result + 
s.toString() + "\n";
        }
        return result;


    }
    public void removeDuplicates()
    {
        for (int i=0; i<shapes.size(); i++)
        {
            for (int j=i+1; j<shapes.size(); j++)
            {
                if ( shapes.get(i).equals(shapes.get(j)) )
                {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

Rectangle.java

public class Rectangle extends Shape
{
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){};
    public Rectangle(double width, double length)
    {
        this.setWidth(width);
        this.setLength(length);
    }
    public Rectangle(double width, double length, String color, boolean filled)
    {
        super(color, filled);
        this.setWidth(width);
        this.setLength(length);
    }
    public Rectangle(Point p, double width, double length, String color, boolean filled)
    {
        super(color, filled);
        this.setWidth(width);
        this.setLength(length);
        this.setTopLeft(p);
    }
    public double getWidth()
    {
        return this.width;
    }
    public void setWidth(double width)
    {
        this.width = width;
    }
    public double getLength()
    {
        return this.length;
    }
    public void setLength(double length)
    {
        this.length = length;
    }
    public double getArea()
    {
        return this.getWidth() * this.getLength();
    }
    public double getPerimeter()
    {
        return (this.getWidth() + this.getLength()) * 2;
    }
    public Point getTopLeft()
    {
        return this.topLeft;
    }
    public void setTopLeft(Point p)
    {
        this.topLeft = p;
    }
    public boolean equals(Object o)
    {
        if (o instanceof Rectangle)
        {
            Rectangle 
r = (Rectangle)o;
            return  Math.abs(this.length-r.length) <=0.001
                    && Math.abs(this.width-r.width) <=0.001
                    && Math.abs(this.getTopLeft().getX()-r.getTopLeft().getX()) <= 0.001
                    && Math.abs(this.getTopLeft().getY()-r.getTopLeft().getY())<= 0.001 ;
        }
        return false;
    }
    public int hashCode()
    {
        return super.hashCode();
    }
    public String toString()
    {
        return "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + Math.ceil(this.getWidth()*10)/10 + ",length=" + Math.ceil(this.getLength()*10)/10 + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }
}
