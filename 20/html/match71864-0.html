Circle.java

import java.awt.*;

public class Circle extends Shape {
    private double vX = 0.1;
    private double vY = 0.1;
    protected Point center;
    protected double radius;
    public Circle(){}
    public Circle(double radius ){
        this.radius = radius;
    }
    public Circle(double radius,String color,boolean filled ){
        super(color,filled);
        this.radius = radius;
    }

    public Circle(Point center,double radius,String color,boolean filled ){
        super(color,filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return (double)Math.PI*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return (double)Math.PI*2*radius;
    }

    @Override
    public boolean equals(Object obj) {
        if( obj instanceof Circle){
            if( (double)Math.round(radius*1000)/1000 != (double)Math.round( ((Circle) obj).radius*1000)/1000)
                return false;
            else if( (double)Math.round(center.getX()*1000)/1000 != (double)Math.round( ((Circle)obj).center.getX()*1000)/1000)
                return false;
            else if( (double)Math.round(center.getY()*1000)/1000 != (double)Math.round( ((Circle)obj).center.getY()*1000)/1000)
                return false;
            else
                return true;
        }
        else
            return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "Circle[center="+center.toString()+",radius="+(double)Math.round(radius*10)/10+super.toString();
    }
    public void move(){
        if( this.center.getX()-radius <= 0 || this.center.getX() >= 300-radius) vX = -vX;
        if( this.center.getY()-radius <= 0 || this.center.getY() >= 300-radius) vY = -vY;
        this.center.setX(this.center.getX()+vX);
        this.center.setY(this.center.getY()+vY);
    }
    public void draw(Graphics graphics){
        graphics.setColor(Color.RED);
        graphics.drawOval((int) (center.getX()-radius),(int)(center.getY()-radius),(int)radius,(int)radius);
    }

}

import javax.swing.*;
import java.awt.*;
import java.util.ArrayList;
import java.util.List;

public class Layer extends JPanel {
    private List<Shape> shapes = new ArrayList<>();
    public void addShape(Shape shape){
        if (shape != null){
        this.shapes.add(shape);}
    }
    public void removeCircles(){
            for( int i = 0; i < shapes.size(); i++) {
                if (shapes.get(i) instanceof Circle) {
                    this.shapes.remove(i);
                }
            }
    }
    public String getInfo() {
        StringBuilder info = new StringBuilder("Layer of crazy shapes: ");
        for( Shape shape : shapes ){
            info.append("\n").append(shape.toString());
        }
        return info.toString();
    }
    public void removeDuplicates(){
        if( shapes.size() > 0) {
            for( int i = 0; i < shapes.size(); i++){
                for( int j = i; j < shapes.size(); j++){
                    if (shapes.get(j).equals(shapes.get(i)) && j != i) {
                        shapes.remove(j);
                    }
                }
            }
        }
    }

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        for( Shape a : shapes){
            a.draw(g);
        }
    }
}
