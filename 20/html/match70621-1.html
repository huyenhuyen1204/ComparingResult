class Circle extends Shape {
    protected  double radius;
    protected  Point center;

    public Circle(){

    }

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        this.center = center;
        this.radius = radius;
        this.color = color;
        this.filled = filled;
    }

    public double getRadius(){
        return radius;
    }

    public void setRadius(double radius){ this.radius = radius; }

    public Point getCenter(){return center;}

    public void setCenter(Point center){this.center = center;}

    @Override
    public double getArea() {
        double area = Math.PI * getRadius() * getRadius();
        return area;
    }

    @Override
    public double getPerimeter() {
        double perimeter = 2 * Math.PI * getRadius();
        return perimeter;
    }

    public boolean equals(Object obj){
        if (obj instanceof Circle){
            Circle compareCircle = (Circle)obj;
            if(Math.abs(compareCircle.getRadius() - this.getRadius()) <= 0.001) return true;
        }
        return false;
    }

    public String toString(){
        String result = "";
        result = "Circle[center=(" + (double)Math.round(center.getX() * 10) / 10 + "," +  (double)Math.round(center.getY() * 10) / 10
                +"),radius=" + (double)Math.round(getRadius() * 10) / 10 +",color=" + getColor() + ",filled=" + isFilled() + "]";
        return result;
    }

}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape newShape){
        shapes.add(newShape);
    }

    public 
String getInfo(){
        String s = "Layer of crazy shapes: \n";
        for(int i = 0; i < shapes.size(); i++){
            s = s + shapes.get(i).toString() + "\n";
        }
        return s;
    }

    public void removeCircles(){
        for(int i = 0; i< shapes.size(); i++){
            if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);// index or object are oke
                i--;
            }
        }
    }

    public void removeDuplicates(){
        for(int i = 0; i< shapes.size(); i++){
            for(int j = i + 1; j<shapes.size(); j++){
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle){
                    
if(((Circle) shapes.get(i)).getCenter().getY() == ((Circle) shapes.get(j)).getCenter().getY()
                    && ((Circle) shapes.get(i)).getCenter().getX() == ((Circle) shapes.get(j)).getCenter().getX()
                    && ((Circle) shapes.get(i)).getRadius() == ((Circle) shapes.get(j)).getRadius()) {
                        shapes.remove(j);
                        j--;
                    }
                }
                if(shapes.get(i) instanceof Rectangle && 
shapes.get(j) instanceof Rectangle){
                    if(((Rectangle) shapes.get(i)).getTopLeft().getY() == ((Rectangle) shapes.get(j)).getTopLeft().getY()
                            && ((Rectangle) shapes.get(i)).getTopLeft().getX() == ((Rectangle) 
shapes.get(j)).getTopLeft().getX()
                            && ((Rectangle) shapes.get(i)).getWidth() == ((Rectangle) shapes.get(j)).getWidth()
                            && ((Rectangle) shapes.get(i)).getLength() == ((Rectangle) shapes.get(j)).getLength()) {
                        shapes.remove(j);
                        j--;
                    }
                }
                if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Square){
                    
if(((Square) shapes.get(i)).getTopLeft().getY() == ((Square) shapes.get(j)).getTopLeft().getY()
                            && ((Square) shapes.get(i)).getTopLeft().getX() == ((Square) shapes.get(j)).getTopLeft().getX()
                            && ((Square) 
shapes.get(i)).getSide() == ((Square) shapes.get(j)).getSide()){
                        shapes.remove(j);
                        j--;
                    }
//                    if(shapes.get(i).equals(shapes.get(j))){
//                        shapes.remove(j);
//                        j--;
                }
            }
        }
    }
}

public class Square extends Rectangle{
    public Square(){

    }

    public Square(double side){
        super(side,side);
    }

    public Square(double side, String color, boolean filled){
        super(side,side);
        this.color = color;
        this.filled = filled;
    }

    public Square(Point center, double side, String color, boolean filled){
        super(center,side,side,color,filled);
    }

    public double getSide(){
        return this.length;
    
}

    public void setSide(double side){
        this.length = side;
        this.width = side;
    }

    public void setWidth(double side){
        this.width = side;
        this.length = side;
    }

    public void setLength(double side){
        this.length = side;
        this.width = side;
    }

    public boolean equals(Object obj){
        if (obj instanceof Square){
            Square compareSquare = (Square) obj;
            if(Math.abs(compareSquare.getSide() - this.getSide()) <= 0.001) return true;
        }
        return false;
    }


    public String toString(){
        String result = "";
        result = "Square[topLeft=(" + (double)Math.round(topLeft.getX() * 10) / 10 + "," +  (double)Math.round(topLeft.getY() * 10) / 10
                +"),side=" + (double)Math.round(getSide() * 10) / 10 +",color=" + getColor() + ",filled=" + isFilled() + "]";
        return result;
    }
