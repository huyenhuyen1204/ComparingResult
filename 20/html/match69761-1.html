import java.util.List;
import java.util.ArrayList;

 * Created by CCNE on 08/10/2019.
 */
public class Layer {
    private List <Shape> shapes = new ArrayList<Shape>();
    public Layer(){
    }
    public void addShape(Shape _shape){
        shapes.add(_shape);
    }
    public void removeCircles(){
        for (int i =0 ; i < shapes.size();i++){
            
if(shapes.get(i) instanceof Circle ){
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        String t = "Layer of crazy shapes: \n";
        for (int i =0 ; i < shapes.size();i++){
            t += shapes.get(i).toString() + "\n";
        
}
        return  t;
    }
    public void removeDuplicates(){
        for (int i = 0; i < shapes.size();i++) {
            for (int j = 0; j < shapes.size();j++){
                    if (shapes.get(i).equals(shapes.get(j)) && i!=j) {
                        shapes.remove(j);
                }
            }
        }
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){}
    public Rectangle(double w, double l){
        width = w;
        length = l;
    }
    public Rectangle(double w, double l, String _color, boolean _fill){
        width  = w;
        length = l;
        color  = _color;
        filled = _fill;
    }
    public Rectangle(Point p,double w, double l, String _color, boolean _fill){
        topLeft = p;
        width = w;
        length = l;
        color  = _color;
        filled = _fill;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }
    public double getLength(){
        return  length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public 
void setWidth(double width) {
        this.width = width;
    }
    public double getArea(){
        return this.width*this.length;
    }
    public double getPerimeter(){
        return (this.width+this.length)*2.0;
    }
    public String toString(){
        return "Rectangle[topLeft="+ this.topLeft +",width=" + this.width+ ",length=" + this.length + ",color=" + this.color +",filled=" + this.filled +"]";
    }
    public 
boolean equals(Object other){
        if (other instanceof Rectangle){
            Rectangle a = (Rectangle) other;
            if (this.topLeft.equals(a.getTopLeft()) && this.width == a.getWidth() && this.length == a.getLength())
                return  true;
        }
        return  false;
    }
}
