Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){
    }

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return PI*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return 2*PI*radius;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + (double)Math.round(this.radius * 10)/10 + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }

    public boolean equals(Object obj) {
        if (obj instanceof Circle) {
            Circle other = (Circle) obj;
            if (other.center.equals(this.center) && Math.abs(other.getRadius() - this.radius) <= 0.001)
                return true;
            return false;
        }
        return false;
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public Layer(){}

    public void addShape(Shape shape) {
        this.shapes.add(shape);
    }
    public void removeCircles() {
        for (int i = 0; i < shapes.size(); i++) {
            if (shapes.get(i) instanceof Circle) {
                this.shapes.remove(i);
            }
        }
    }
    public String getInfo() {
        String result = "";
        result += "Layer of crazy shapes: \n";
        for (Shape A : this.shapes) {
            result += A.toString() + "\n";
        }
        return result;
    }

    public void removeDuplicates() {
        for (int i = 0; i < this.shapes.size(); i++) {
            for (int j = this.shapes.size()-1; j > i; j--) {
                if (this.shapes.get(i).equals(this.shapes.get(j))) {
                    this.shapes.remove(j);
                }
            }
        }
    }

    public static void main(String...args) {
        Point pc = new Point(10, 5);
        Point pr = new Point(10, 5);
        Point pr1 = new Point(10.5, 5);
        Point ps = new Point(10, 5);
        Point p4 = new Point(2, 5);
        Circle c = new Circle(pc, 8, "RED", true);
        Rectangle r = new Rectangle(pr, 8, 6, "RED", true);
        Rectangle r1 = new Rectangle(pr1, 8, 6, "BLUE", true);
        Square s = new Square(ps, 8, "RED", false);
        Circle c1 = new Circle(pc, 8, "BLACK", false);
        Layer shapes = new Layer();
        shapes.addShape(c);
        shapes.addShape(r);
        shapes.addShape(r1);
        shapes.addShape(s);
        shapes.addShape(c1);
        //shapes.removeCircles();
        shapes.removeDuplicates();
        System.out.print(shapes.getInfo());
    }

}

public class Point {
    private double x;
    private double y;
    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return this.x;
    }
    public void setX(double x) {
        this.x = x;
    }
    public double getY() {
        return this.y;
    }
    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point otherPoint) {
        double a = Math.pow(otherPoint.getX() - this.x, 2) + Math.pow(otherPoint.getY() - this.y, 2);
        return (Math.sqrt(a));
    }
    public boolean equals(Object obj) {
        if (obj instanceof Point) {
            Point other = (Point) obj;
            if (other.distance(this) <= 0.001) return true;
            return false;
        }
        return false;
    }

    public String toString() {
        return ("(" + (double)Math.round(this.x * 10)/10 + "," + (double)Math.round(this.y * 10)/10 + ")");
    }
}

public class Square extends Rectangle {
    public Square(){
    }

    public Square(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    public Square(double side, String color, boolean filled){
        super.setLength(side);
        super.setWidth(side);
        super.setColor(color);
        super.setFilled(filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled){
        super.setTopLeft(topLeft);
        super.setLength(side);
        super.setWidth(side);
        super.setColor(color);
        super.setFilled(filled);
    }

    public double getSide(){
        return super.getWidth();
    }

    public void setSide(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public void setWidth(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public void setLength(double side) {
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public String toString() {
        return "Square[topLeft=" + super.topLeft.toString() + ",side=" + (double)Math.round(this.getSide() * 10)/10 + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }

    public boolean equals(Object obj) {
        if (obj instanceof Square) {
            Square other = (Square) obj;
            if (other.topLeft.equals(this.topLeft) && Math.abs(other.getSide() - this.getSide()) <= 0.001) return true;
            return false;
        }
        return false;
    }
}
