public 
class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle () {

    }
    public Circle (double radius) {
        this.radius = radius;
    }
    public Circle (double radius,String color,boolean filled) {
        super(color,filled);
        this.radius = radius;
    }
    public Circle (Point center,double radius,String color,boolean filled) {
        super(color,filled);
        this.radius = radius;
        this.center = center;
    }
    public Point getCenter() {
        return this.center;
    }
    public void setCenter(Point center) {
        this.center = center;
    }
    public double getRadius () {
        return Math.round((radius*10)/10);
    }
    public void setRadius(double radius) {
        this.radius=radius;
    }
    
@Override
    public double getArea() {
        return Math.PI*radius*radius;
    }
    @Override
    public double getPerimeter() {
        return 2*Math.PI*radius;
    }
    public boolean equals(Object a) {
        if (a instanceof Circle) {
            Circle b = (Circle) a;
            if (this.radius == b.getRadius() && this.center.equals(b.getCenter())) return true;
            else return false;
        }
        else return false;
    }

    public String toString() {
        return "Circle[center=(" + center.getX() + "," + center.getY() + ")" +",radius=" + radius + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }

}

import java.util.ArrayList;
import java.util.List;
import java.util.StringJoiner;

public class Layer {
    private List <Shape> shapes = new ArrayList<Shape>();
    public void addShape (Shape a) {
        shapes.add(a);
    }
    public void removeCircles() {
        int i=0;
        while ( i < shapes.size())
        {
            if (shapes.get(i) instanceof Circle) shapes.remove(i);
            else i++;
        }
    }
    public String getInfo() {
        String s = "";
        
s+="Layer of crazy shapes: \n";
        for (int i=0;i<shapes.size();i++) {
            s+=shapes.get(i).toString() + "\n";
        }
        return s;
    }
    public void removeDuplicates(){
       List<Shape> a = new ArrayList<>();
        
for (int i = 0; i < shapes.size(); i++) {
            if (!a.contains(shapes.get(i))) {
                a.add(shapes.get(i));
            }
        }
        shapes.clear();
        shapes.addAll(a);
    }
}

        return this.x;
    }

        return this.y;

        this.x = x;
    }
    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point a) {
        double kc = Math.sqrt((this.x - a.getX())*(this.x - a.getX()) + (this.y - a.getY())*(this.y - a.getY()));
        return Math.round((kc*10)/10) ;

    }
    public String toString() {
        return "(" + x + "," + y + ")";
    }

        if ( a instanceof Point) {
            Point b = (Point) a;
            if (this.x == b.getX() && this.y == b.getY()) return true;
            else  return false;
        }
        else return false;
    }

}

public class Rectangle extends Shape {

        this.width = width;
        this.length = length;
    }
    public Rectangle (double width,double length,String color,boolean filled) {
        super(color,filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle (Point topleft,double width,double length,String color,boolean filled) {
        super(color,filled);
        this.width = width;
        this.length = length;
        this.topLeft = topleft;
    }
    public Point getTopLeft() {
        return this.topLeft;
    }
    public void setTopLeft( Point topleft) {
        this.topLeft = topleft;
    }
    public double getWidth() {
        return this.width;
    }
    public double getLength() {
        return  this.length;
    }
    public void setWidth (double width) {
        this.width = width;
    
}
    public void setLength (double length) {
        this.length = length;
    }
    @Override
    public double getArea() {
        return width*length;
    }
    @Override
    public double getPerimeter() {
        return width+length+width+length;
    }
    public boolean equals(Object a) {
        if (a instanceof Rectangle) {
            Rectangle b = (Rectangle) a;
            if (this.width == b.getWidth()&& this.length == b.getLength() && this.topLeft.equals(b.getTopLeft())) return true;
            else return false;
        }
        else return false;
    }
    public String toString() {
        return "Rectangle[topLeft=(" + topLeft.getX() + "," + topLeft.getY() + ")" + ",width=" + getWidth() + ",length=" + getLength() + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
}

public abstract 
class Shape {
    protected String color;
    protected boolean filled;
    public Shape () {

    }
    public Shape (String color,boolean filled) {
        this.color = color;
        this.filled = filled;
    }
    public String getColor() {
        return color;
    }
    public boolean isFilled() {
        return filled;
    }
    public void setColor (String color) {
        this.color = color;
    }
    public void setFilled(boolean filled) {
        this.filled = filled;
    }
    abstract public double getArea() ;
    abstract public double getPerimeter();
    public String toString() {
        return "Shape[color=" + color + ",filled=" + filled + "]";
    }

}
