import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes =new ArrayList<Shape>();

    public void addShape(Shape n){
        shapes.add(n);
    }

    public void removeCircles(){
        for (int i = 0; i< shapes.size(); i++)
            if (shapes.get(i) instanceof Circle)
                shapes.remove(shapes.get(i));
    }

    public String getInfo(){
        String s = "";
        s += "Layer of crazy shapes: " + "\n";
        for (int i = 0; i < shapes.size(); i++) {
            //System.out.print( (i + 1) + ": ");
            s += shapes.get(i).toString().trim() + "\n";
        }
        return s;
    }

    public void removeDuplicates(){
        for (int i = 0; i < shapes.size()-1; i++) {
            for (int j = i + 1; j < shapes.size(); j++) {
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance( Point other){
        double t= Math.sqrt((x-other.getX())*(x-other.getX())+(y-other.getY())*(y-other.getY()));
        //t=Math.round(t*10)/10;
        return t;
    }

    public boolean equals(Object obj){
        if (obj instanceof Point) {
            Point s = (Point) obj;
            if (this.distance(s) == 0 ) return true;
        }
        return false;
    }
    public String toString (){
        return "("+this.x+","+this.y+")";
    }
}

        this.length=side;
        this.color=color;

    }
