Circle.java

public class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){}
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return Math.PI*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return Math.PI*radius*2;
    }
    public boolean equals(Object a) {
        if (a instanceof Circle) {
            Circle b = (Circle) a;
            if (this.radius - 0.001 <= b.radius && b.radius <= this.radius + 0.001) {
                return true;
            }
            else return false;
        } else return false;
    }

    public String toString(){
        return "Circle[center="+this.center+",radius="+this.radius+",color="+this.color+",filled="+this.filled+"]";
        }



}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape a){
        shapes.add(a);
    }
    public void removeCircles(){
        for (int i=0; i<shapes.size(); i++){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
            }
        }

    }
    public String getInfo(){
        String s = "Layer of crazy shapes: \n";
        for (int i=0; i<shapes.size(); i++){
            s+=shapes.get(i).toString()+"\n";
        }
        return s;
    }
    public void removeDuplicates(){
        for(int i=0; i<shapes.size(); i++){
            for (int j=0; j<shapes.size(); j++){
                if(shapes.get(i).equals(shapes.get(j)) && i!=j){
                    shapes.remove(j);
                }
            }
        }
    }
}

public class Square extends Rectangle {
    public Square(){}
    public Square(double Side){
        super(Side,Side);
    }
    public Square(double Side, String color, boolean filled){
        super(Side, Side);
        this.color = color;
        this.filled =filled;
    }
    public Square(Point point, double Side, String color, boolean filled){
        super(Side, Side);
        this.color = color;
        this.filled = filled;
        this.setTopLeft(point);
    }
    public double getSide(){
        return getWidth();
    }
    public void setSide(double Side){
        this.setWidth(Side);
        this.setLength(Side);
    }
    public void setWidth(double Side){
        super.setWidth(Side);
    }
    public void setLength(double Side){
        super.setLength(Side);
    }
    public boolean equals(Object obj) {
        if(obj instanceof Square) {
            Square b = (Square) obj;
            if (this.getSide() - 0.001 <= b.getSide() && b.getSide() <= this.getSide() + 0.001){
                return true;
            }
            else return false;
        }
        else return false;

    }

    @Override
    public String toString() {
        return "Square[topLeft="+this.getTopLeft()+",side="+this.getWidth()+",color="+this.color+",filled="+this.filled+"]";
    }

}
