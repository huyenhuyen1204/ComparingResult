Circle.java

import java.util.Objects;

public 
class Circle extends Shape{
    protected Point center;
    protected double radius;
    public Circle(){

    }
    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point point, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = point;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return this.radius* Math.PI * this.radius;
    }

    @Override
    public double getPerimeter() {
        return this.radius * 2 * Math.PI;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public boolean equals(Object o) {
        if(o instanceof Circle){
            Circle other = (Circle) o;
            if(this.getCenter().getX() == other.getCenter().getX() && this.getCenter().getY() == other.getCenter().getY() && this.getRadius() == other.getRadius())
                return true;
        }
        return false;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center + ",radius=" + this.getRadius() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }

}

import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape other){
        shapes.add(other);
    }
    public void removeCircles(){
        for(int i = 0; i < this.shapes.size(); i++){
            if(
this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        String result = "Layer of crazy shapes: \n";
        for(Shape other : this.shapes){
            result += 
other.toString() + "\n";
        }
        return result;
    }
    public void removeDuplicates(){
        for(int i = 0; i < shapes.size() - 1; i++){
            for(int j = i + 1; j < shapes.size(); j++)
                if(shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
        }
    }
}

public 
class Square extends Rectangle{

    public Square(){}
    public Square(double number){
        this.width = number;
        this.length = number;
    }
    public Square(double number, String color, boolean filled){
        this.color = color;
        this.filled = filled;
        this.width = number;
        this.length = number;
    }
    public Square(Point other, double number, String color, boolean filled){
        this.topLeft = other;
        this.width = number;
        this.length = number;
        this.color = color;
        this.filled = filled;
    }
    public double getSide(){
        return this.length;
    }
    public void setSide(double number){
        this.width = number;
        this.length = number;
    }
    public void setWidth(double number){
        this.width = number;
        this.length = number;
    }
    public void setLength(double number){
        this.length = number;
        this.width = number;
    }
    public String toString(){
        return "Square[topLeft=" + topLeft + ",side=" + this.getSide() + ",color=" + 
this.getColor() + ",filled=" + this.isFilled() + "]";
    }
    public boolean equals(Object o){
        if(o instanceof Square){
            Square other = (Square) o;
            if(this.getSide() == other.getSide() && this.getTopLeft().getX() == other.getTopLeft().getX() && this.getTopLeft().getY() == other.getTopLeft().getY())
                return true;
        }
        return false;
    }
}
