public class Circle extends Shape {
    protected Point center;
    protected double radius;
    protected final double pi = Math.PI;
    public Circle(){};
    public Circle (double radius) {
        this.radius = radius;
    }
    public Circle (double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }
    public Circle (Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center;
        this.radius = radius;
    }
    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public double getArea() {
        return pi * radius * radius;
    }
    @Override
    public double getPerimeter() {
        return radius * 2 * pi ;
    }
    @Override
    public String toString() {
        return "Circle[center=(" + center.getX() + "," + center.getY()
                + "),radius=" + radius + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }
    public Point getCenter() {
        return center;
    }
    public void setCenter(Point p) {
        center.setX(p.getX());
        center.setY(p.getY());
    }
    public int hashCode() {
        return 0;
    }
    public boolean equals(Object o) {
        if (o instanceof Circle) {
            if (Math.abs(((Circle) o).center.getY() - center.getY()) <= 0.001
                    && Math.abs(((Circle) o).center.getX() - center.getX()) <= 0.001
                    && Math.abs(((Circle) o).getRadius() - radius) <= 0.001) {
                return true;
            }
        }
        return false;
    }
    public static void main(String[] args) {
        Point center = new Point(2,2.4);
        Circle c = new Circle (center,11, "red", true);
        Point newCenter = new Point(1, 2.9);
        c.setCenter(newCenter);
        System.out.println(c.toString());
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return x;
    }
    public void setX (double x) {
        this.x = x;
    }
    public double getY() {
        return y;
    }
    public void setY (double y) {
        this.y = y;
    }
    public double distance (Point p) {
        return Math.sqrt((x - p.getX() * (x - p.getX())
                + (y - p.getY()) * (y - p.getY())));
    }
    public boolean equals(Object o) {
        if (o instanceof Point) {
            if (Math.abs(((Point) o).getX() - x) <= 0.001 &&
                    Math.abs(((Point) o).getY() - y) <= 0.001) {
                return true;
            }
        }
        return false;
    }
    public String toString(){
        return "(" + getX() + "," + getY() + ")";
    }
}

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){};
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point p, double width, double length, String color, boolean filled) {
        super(color, filled);
        topLeft = p;
        this.width = width;
        this.length = length;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    @Override
    public double getArea() {
        return width * length;
    }
    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }
    @Override
    public String toString() {
        return "Rectangle[topLeft=(" + topLeft.getX() + "," + topLeft.getY()
                + "),width=" + width + ",length=" + length
                + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }
    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point p) {
        topLeft.setX(p.getX());
        topLeft.setY(p.getY());
    }
    public boolean equals (Object o) {
        if (o instanceof Rectangle) {
            if (Math.abs(((Rectangle) o).getWidth() - width) <= 0.001
                    && Math.abs(((Rectangle) o).getLength() - length) <= 0.001
                    && Math.abs(((Rectangle) o).topLeft.getX() - topLeft.getX()) <= 0.001
                    && Math.abs(((Rectangle) o).topLeft.getY() - topLeft.getY()) <= 0.001){
                return true;
            }
        }
        return false;
    }
    public int hashCode() {
        return 1;
    }
    public static void main(String[] args) {
        Point topLeft = new Point(1, 2.2);
        Point top = new Point(2, 3);
        Rectangle c1 = new Rectangle (topLeft,3.5, 2.5,"red", true);
        Rectangle c2 = new Rectangle (topLeft,3.5, 2.5,"red", true);
        System.out.println(c1.equals(c2));
    }
}

public class Square extends Rectangle {
    public Square(){};
    public Square (double side) {
        setWidth(side);
        setLength(side);
    }
    public Square (double side, String color, boolean filled) {
        super(side, side, color, filled);
    }
    public Square (Point topLeft, double side, String color, boolean filled) {
        super (topLeft, side, side, color, filled);
    }
    public double getSide() {
        return getLength();
    }
    public void setSide(double side) {
        setWidth(side);
        setLength(side);
    }
    @Override
    public void setWidth(double side) {
        super.setWidth(side);
        super.setLength(side);
    }
    @Override
    public void setLength(double side) {
        super.setWidth(side);
        super.setLength(side);
    }
    @Override
    public String toString() {
        return "Square[topLeft=(" + getTopLeft().getX() + "," + getTopLeft().getY()
                + "),side=" + getLength() + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }
    public boolean equals(Object o) {
        if (o instanceof Square) {
            return (Math.abs(((Square) o).getSide() - this.getSide()) <= 0.001
                    && Math.abs(((Square) o).topLeft.getX() - topLeft.getX()) <= 0.001
                    && Math.abs(((Square) o).topLeft.getY() - topLeft.getY()) <= 0.001);
        }
        return false;
    }
    public int hashCode() {
        return 2;
    }
    public static void main(String[] args) {
        Point top = new Point(1,1.5);
        Square c = new Square (top,3.3,"red", true);
        Point newTop = new Point(5, 10);
        Square c2 = new Square (top,3.3,"blue", true);
        System.out.println(c.hashCode() + " " + c2.hashCode());
        c.setTopLeft(newTop);
        System.out.println(c.toString() + "\n" + c2.toString());
    }
}
