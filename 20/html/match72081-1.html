Circle.java

public class Circle extends Shape {

    protected double radius;
    protected Point center;

    public Circle()
    {
        super();
    }
    public Circle(double radius)
    {
        super();
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color, filled);
        this.radius=radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius=radius;
        this.center=center;
    }

    public Point getCenter(){
        return center;
    }
    public void setCenter(Point center){
        this.center=center;
    }

    public double getRadius()
    {
        return radius;
    }
    public void setRadius(double radius)
    {
        this.radius=radius;
    }

    
@Override
    public double getArea(){
        return radius*radius*PI;
    }

    @Override
    public double getPerimeter() {
        return 2*radius*PI;
    }

    public boolean equals(Object object){

        if(object instanceof Circle)
        {
            Circle obj = (Circle) object;
            if(this.getCenter().equals(obj.getCenter()))
                if(this.getRadius()==obj.getRadius())
                    return true;
        
}
        return false;
    }
    @Override
    public String toString() {
        return "Circle[center="+this.getCenter().toString()+",radius="+radius+",color="+color+",filled="+filled+"]";
    }

    public static void main(String[] args)
    {
        Circle circle=new Circle(9.8, "blue",true);
        System.out.println(circle.getPerimeter());
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public Layer(){

    }

    public void addShape(Shape obj)
    {
        shapes.add(obj);
    }
    public void removeCircles(){
        int i=0;

        while (i<shapes.size()){
            if(shapes.get(i) instanceof Circle)
                shapes.remove(i);
            else
                i++;
        }
    
}
    public String getInfo()
    {
        String res ="Layer of crazy shapes: \n";
        for(int i=0; i<shapes.size(); i++)
            res+=shapes.get(i).toString()+"\n";

        return res;
    }
    public void removeDuplicates(){
        int i=0;

        while (i<shapes.size()-1){
            int j=i+1;
            while (j<shapes.size()) {
                if (shapes.get(i).equals(shapes.get(j)))
                    shapes.remove(j);
                else j++;
            }
            i++;
        }
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){
        super();
    }
    public Rectangle(double width,double length){
        super();
        this.width=width;
        this.length=length;
    }
    public Rectangle(double width, double length, String color, boolean filled)
    {
        super(color, filled);
        this.width=width;
        this.length=length;
    }

    public Rectangle(Point topLeft,double width, double length, String color, boolean filled){
        super(color, filled);
        this.width=width;
        this.length=length;
        this.topLeft=topLeft;
    }

    public Rectangle(String color, boolean filled)
    {
        super(color, filled);
    }
    public double getWidth()
    {
        return  width;
    }
    public void setWidth(double width)
    {
        this.width=width;
    }
    public double getLength()
    {
        return length;
    }
    public void setLength(double length)
    {
        this.length=length;
    }

    
@Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return 2*(width+length);
    }

    public Point getTopLeft(){
        return  topLeft;
    }
    public void setTopLeft(Point topLeft)
    {
        this.topLeft=topLeft;
    }
    public boolean equals(Object other){
        if(other instanceof Rectangle)
            if(other instanceof Square)
            {
                return  false;
            }
            else
            {
                Rectangle obj= (Rectangle) other;
                boolean check=this.topLeft.equals(obj.getTopLeft());
                if(check)
                {
                    if(this.width==obj.getWidth()&&this.length==obj.getLength())
                        return  true;
                }
            }

         return  false;
    }

    public String toString(){
        return "Rectangle[topLeft="+topLeft.toString()+",width="+width+",length="+length+",color="+color+",filled="+filled+"]";
    }
}
