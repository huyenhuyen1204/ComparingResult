public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){
    }
    public Circle( double _radius){
        this.radius=_radius;
    }
    public Circle(double _radius, String _color, boolean _filled){
        super(_color, _filled);
        this.radius=_radius;

    }
    public Circle(Point _center,double _radius, String _color, boolean _filler){
        super(_color,_filler);
        this.center=_center;
        this.radius=_radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }


    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea(){
        return this.getRadius()*this.getRadius()*Math.PI;
    }
    public double getPerimeter(){
        return 2*this.getRadius()*Math.PI;
    }
    public double round(double a){
        a=Math.round(a*10);
        return a/10;
    }
    public String toString(){
        return "Circle[center=("+this.round(this.getCenter().getX())+","+this.round(this.getCenter().getY())+"),radius="+this.round(getRadius())+",color="+super.getColor()+ ",filled="+super.isFilled()+"]";
    }

    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle a= (Circle) obj;
            if(Math.abs(a.getRadius()-this.getRadius())<0.001 && a.getCenter().equals(this.getCenter())&&a.isFilled()==this.isFilled()){
                return true;
            }
        }
        
return false;
    }
    public boolean Duplicates(Object obj){
        if(obj instanceof Circle){
            Circle a= (Circle) obj;
            if(Math.abs(a.getRadius()-this.getRadius())<0.001 && a.getCenter().equals(this.getCenter())){
                return true;
            }
        }
        return false;
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes= new ArrayList<Shape>();
    public void addShape(Shape _shape){
        this.shapes.add(_shape);
    }
    public void removeCircles(){
        for(int i=0;i<shapes.size();++i){
            if(shapes.get(i) instanceof Circle){
                this.shapes.remove(shapes.get(i));
                i--;
            }
        }
    }
    public String getInfo(){
        if(this.shapes.isEmpty()){
            return "";
        }
        else {
            String s="";
            s="Layer of crazy shapes: \n";
            for(Shape i: shapes){
                s=s+i.toString()+"\n";
            }
            return s;
        }
    
}
    public void removeDuplicates(){
        for(int i=0;i< shapes.size();++i){
            for(int j=i+1;j<shapes.size();++j){
                if(i!=j&&(shapes.get(i) instanceof Circle&& shapes.get(j) instanceof  Circle)){
                    Circle c1= (Circle) shapes.get(i);
                    Circle c2= (Circle) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
                
if(i!=j&&(shapes.get(i) instanceof Rectangle&& shapes.get(j) instanceof  Rectangle)){
                    Rectangle c1= (Rectangle) shapes.get(i);
                    Rectangle c2= (Rectangle) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
                
if(i!=j&&(shapes.get(i) instanceof Square&& shapes.get(j) instanceof  Square)){
                    Square c1= (Square) shapes.get(i);
                    Square c2= (Square) shapes.get(j);
                    if(c1.Duplicates(c2)){
                        this.shapes.remove(j);
                    }
                }
            }
        }
    }

    public static void main(String[] args) {
        Point a1= new Point(10.0, 5.0);

        Shape a= new Circle(a1,8.0,"RED",true);
        Shape c= new Circle(a1,8.0,"RED",true);

src/Point.java

public class Point {
    private double x;
    private double y;
    public Point(double _x, double _y){
        x=_x;
        y=_y;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p){
        return Math.sqrt((p.x-this.x)*(p.x-this.x)+(p.y-this.y)*(p.y-this.y));
    }
    public 
String toString(){
        return "("+this.x+","+this.y+")";
    }
    public boolean equals(Object obj) {
        if(obj instanceof Point){
            Point a= (Point) obj;
            if(Math.abs(a.getX()-this.getX())<0.001 && Math.abs(a.getY()-this.getY())<0.001){
                return true;
            }
        }
        return false;
    }

}

    public
