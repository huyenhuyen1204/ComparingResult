Circle.java

public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle(){}

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius,String color,boolean filled){
        this.color = color;
        this.filled = filled;
        this.radius = radius;
    }

    public Circle(Point p,double radius,String color,boolean filled){
        this.radius = radius;
        this.color = color;
        this.filled = filled;
        this.center = p;
    }

    public double getRadius(){return this.radius;}

    public void setRadius(double radius){
        this.radius = radius;
    }

    public Point getCenter(){return this.center;}

    public void setCenter(Point p){
        this.center = p;
    }

    public boolean equals(Object obj){
        if (obj instanceof Circle){
            double t = this.radius-((Circle) obj).radius;
            if (t==0) return true;
            else return false;
        } else if (obj == null) return false;
        else return false;
    }

    @Override
    public double getArea(){
        double area = Math.pow(this.radius,2.)*Math.PI;
        return area;
    }

    @Override
    public double getPerimeter() {
        double perimeter = 2*Math.PI*this.radius;
        return perimeter;
    }

    public String toString(){
        String s = new String();
        s += ("Circle[center=("+this.center.getX()+","+this.center.getY()+"),radius="+this.radius+",color="+this.color+",filled="+this.filled+"]");
        return s;
    }

    public int hashCode(){
        return (int)(this.center.hashCode()*this.radius+1);
    }
}

import java.util.ArrayList;
import java.util.Iterator;
import java.util.LinkedHashSet;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape){
        this.shapes.add(shape);
    }

    public void removeCircles(){
        for (int i=0;i<shapes.size();i++){
            if (shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }

    public String getInfo(){
        String s = new String();
        s += ("Layer of crazy shapes:" +"\n");
        for (Shape sp : shapes){
            if (sp instanceof Circle) {
                sp = (Circle) sp;
                s += (sp.toString() + "\n");
            }
            else if (sp instanceof Rectangle){
                sp = (Rectangle)sp;
                s += (sp.toString()+"\n");
            }
            else if (sp instanceof Square){
                sp = (Square)sp;
                s += (sp.toString()+"\n");
            }
        }
        return s;
    }

    public void removeDuplicates(){
        for (int i =0;i<shapes.size();i++){
            for (int j=i+1;j<shapes.size();j++){
                if (shapes.get(i).equals(shapes.get(j)))
                    if (shapes.get(i).hashCode() == shapes.get(j).hashCode())
                        shapes.remove(j);
            }
        }
    }

    public static void main(String[] args){
        Layer list = new Layer();
        Point p = new Point(1,2);
        Point p1 = new Point(2,3);
        Point p2 = new Point(3,4);
        Circle c1 = new Circle(p,5.,"red",true);
        Circle c3 = new Circle(p,5.,"green",true);
        Circle c2 = new Circle(p1,5,"red",false);
        Square s1 = new Square(p1,5.0,"red",true);
        Square s2 = new Square(p1,5.0,"red",true);
        Rectangle r1 = new Rectangle(p2,3.0,4.0,"red",true);
        list.addShape(c1);
        list.addShape(c3);
        list.addShape(c2);
        list.addShape(s1);
        list.addShape(r1);
        list.addShape(s2);
        String s = new String();
        list.removeDuplicates();
        s = list.getInfo();
        System.out.print(s);
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(){}

    public Rectangle(double width,double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width,double length,String color,boolean filled){
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }

    public Rectangle(Point p,double width,double length,String color,boolean filled){
        this.topLeft = p;
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }

    public double getWidth(){return this.width;}

    public void setWidth(double width){
        this.width = width;
    }

    public double getLength(){return this.length;}

    public void setLength(double length){
        this.length = length;
    }

    public Point getTopLeft(){return this.topLeft;}

    public void setTopLeft(Point p){
        this.topLeft = p;
    }

    @Override
    public double getPerimeter() {
        double perimeter = (this.length+this.width)*2;
        return perimeter;
    }

    @Override
    public double getArea() {
        double area = this.width*this.length;
        return area;
    }

    public boolean equals(Object obj){
        if (obj instanceof Rectangle){
            if ((this.width == ((Rectangle) obj).width) && (this.length == ((Rectangle) obj).length)) return true;
            else return false;
        }
        else return false;
    }

    public String toString(){
        String s = new String();
        s += ("Rectangle[topLeft=("+this.topLeft.getX()+","+this.topLeft.getY()+"),width="+this.width+",length="+this.length+",color="+this.color+",filled="+this.filled+"]");
        return s;
    }

    public int hashCode(){
        return (int)(this.topLeft.hashCode()*this.width*this.length+1);
    }
}

public class Square extends Rectangle {
    public Square(){}

    public Square(double side){
        this.length = side;
        this.width = side;
    }

    public Square(double side,String color,boolean filled){
        this.width = side;
        this.length = side;
        this.color = color;
        this.filled = filled;
    }

    public Square(Point p,double side,String color,boolean filled){
        this.topLeft = p;
        this.width = side;
        this.length = side;
        this.color = color;
        this.filled = filled;
    }

    public double getSide(){return this.length;}

    public void setSide(double side){
        this.length = side;
        this.width = side;
    }

    public void setWidth(double side){
        this.width = side;
        this.length = side;
    }

    public void setLength(double side){
        this.length = side;
        this.width = side;
    }

    public String toString(){
        String s = new String();
        s += ("Square[topLeft=("+this.topLeft.getX()+","+this.topLeft.getY()+"),side="+this.length+",color="+this.color+",filled="+this.filled+"]");
        return s;
    }

    public boolean equals(Object obj){
        if (obj instanceof Square){
            if (this.getSide() == ((Square) obj).getSide()) return true;
            else return false;
        }
        else return false;
    }

    public int hashCode(){
        return (int)(this.topLeft.hashCode()*this.getSide()+1);
    }

}
