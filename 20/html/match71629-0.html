Circle.java


public class Circle extends Shape{
	protected double radius;
	protected Point center;
	final private double PI = Math.PI; 
	
	public Circle() {this.radius = 0.0;}
	public Circle(double r) {this.radius = r;}
	public Circle(double r, String c, boolean f) {super(c, f); this.radius = r;}
	public Circle(Point p, double r, String c, boolean f) {super(c, f); this.radius = r; this.center = p;}
	
	public void setRadius(double x) {this.radius = x;}
	public double getRadius() {return this.radius;}
	
	public void setCenter(Point x) {this.center = x;}
	public Point getCenter() {return this.center;}
	
	public boolean equals(Object o){
		if (o instanceof Circle){
			Circle c = (Circle) o;
			return (Math.abs(this.radius - c.radius) < 0.001 && this.center.equals(c.center));
		} else return false;
	}
	
	@Override 
	public double getArea(){
		return this.radius * this.radius * PI;
	}
	@Override
	public double getPerimeter(){
		return this.radius*2.0*PI;
	}
	
	public String toString(){
		return ("Circle[center=" + this.center.toString() + ",radius=" + String.format("%.1f",this.radius) +  ",color=" + this.color + ",filled=" + this.filled + "]");
	}
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
	private List <Shape> shapes = new ArrayList <Shape>();
	
	public void addShape(Shape s) {shapes.add(s);}
	
	public void removeCircles(){
		for (int i = 0; i < shapes.size(); ++i){
			if (shapes.get(i) instanceof Circle) {
				shapes.remove(i);
				--i;
			}
		}
	}
	
	public String getInfo(){
		String res = "Layer of crazy shapes: \n";
		for (int i = 0; i < shapes.size(); ++i) {
			res += shapes.get(i).toString() + "\n";
		}
		return res;
	}
	
	public void removeDuplicates(){
		for (int i = 0; i < shapes.size(); ++i){
			Shape s = shapes.get(i);
			for (int j = i+1; j < shapes.size(); ++j) {
				if (shapes.get(j).equals(s)) {
					shapes.remove(j);
					--j;
				}
			}
		}
	}
}

Point.java

public class Point {
	private double x;
	private double y;
	
	public Point(){
		this.x = 0.0;
		this.y = 0.0;
	}
	public Point(double x, double y){
		this.x = x;
		this.y = y;
	}
	
	public void setX(double x) {this.x = x;}
	public double getX() {return this.x;}
	public void setY(double x) {this.y = x;}
	public double getY() {return this.y;}
	
	public double distance(Point p) {return Math.sqrt((this.x - p.getX())*(this.x - p.getX()) 		
													+ (this.y - p.getY())*(this.y - p.getY()));}
	public boolean equals(Object ob){
		if (ob instanceof Point) {
			Point object = (Point) ob;
			return ((Math.abs(this.x - object.getX()) < 0.001) && (Math.abs(this.y - object.getY()) < 0.001));
		} else return false;
	}
	
	public String toString() {return ("(" + String.format("%.1f",this.x) + "," + String.format("%.1f",this.y) + ")");}
}

Rectangle.java

public class Rectangle extends Shape{
	protected double width;
	protected double length;
	protected Point topLeft;
	
	public Rectangle() {
		this.width = 0.0;
		this.length = 0.0;
		this.topLeft.setY(0.0);
		this.topLeft.setX(0.0);
	}
	public Rectangle(double w, double l){
		this.width = w;
		this.length = l;
	}
	public Rectangle(double w, double l, String c, boolean f){
		super(c, f);
		this.width = w;
		this.length = l;
	}
	public Rectangle (Point p, double w, double l, String c, boolean f){
		super(c, f);
		this.width = w;
		this.length = l;
		this.topLeft = p;
	}
	
	public double getWidth() {return this.width;}
	public void setWidth(double x) {this.width = x;}
	
	public double getLength() {return this.length;}
	public void setLength(double x) {this.length = x;}
	
	public Point getTopLeft() {return this.topLeft;}
	public void setTopLeft(Point p) {this.topLeft = p;}
	
	public boolean equals(Object o){
		if (o instanceof Rectangle) {
			Rectangle rec = (Rectangle) o;
			return ((Math.abs(this.getLength() - rec.getLength()) < 0.001) 
					&& (Math.abs(this.getWidth() - rec.getWidth()) < 0.001) && (this.topLeft.equals(rec.topLeft)));
		} else return false;
	}
	
	@Override
	public double getArea() {return (this.width * this.length);}
	@Override
	public double getPerimeter() {return (this.width + this.length) * 2.0;}
	
	public String toString(){
		return ("Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + String.format("%.1f",this.width) + 
				",length=" + String.format("%.1f",this.length)  + ",color=" + this.color + ",filled=" + this.filled + "]");
	}

}
