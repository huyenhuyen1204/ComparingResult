public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){};
    public Circle(double radius){
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled){
        super(color,filled);
        this.radius=radius;
    }
    public Circle(Point center,double radius,String color,boolean filled){
        super(color,filled);
        this.radius=radius;
        this.center=center;
    }
    public Point getCenter(){
        return center;
    }
    public void setCenter(Point center){
        this.center=center;
    }
    public double getRadius(){
        return radius;
    }
    public void setRadius(double radius){
        this.radius=radius;
    }
    public double getArea() {
        return Math.PI*radius*radius;
    }
    public double getPerimeter(){
        return 2*Math.PI*radius;
    }
    public boolean equals(Object cris) {
        if (cris instanceof Circle) {
            Circle cri = (Circle) cris;
            return this.center.equals(cri.center) && Math.abs(cri.getRadius() - this.getRadius()) <= 0.001;
        }

        return false;
    }

    public String toString(){
        return "Circle[center=" + center.toString()+",radius="+this.radius+",color="+this.color+",filled="+this.filled+"]";
    }

}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes=new ArrayList<>();
    public void addShape(Shape shape1){
        shapes.add(shape1);
    }
    public void removeCircles(){
        for(int i=0;i<shapes.size();i++){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            }
        }
    };
    public String getInfo(){
        String s="Layer of crazy shapes: \n";
        for(Shape shape: shapes){
            s=s+ shape.toString()+"\n";
        }
        return s;
    };
    public void removeDuplicates() {
        for (int i = 0; i < shapes.size()-1; i++) {
            for (int j = i+1; j < shapes.size(); j++) {
                if(shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }

}

public class Square extends Rectangle{
    public Square(){super();};
    public Square(double side){
        super(side,side);
    }
    public Square(double side,String color,boolean filled){
        super(side,side,color,filled);

    }
    public Square(Point topLeft,double side,String color,boolean filled){
        super(topLeft,side,side,color,filled);
    }
    public double getSide(){
        return this.getWidth();
    };
    public void setSide(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    public boolean equals(Object squar) {
        if (squar instanceof Square) {
            Square square = (Square) squar;
            return this.topLeft.equals(square.topLeft) && Math.abs(square.getSide() - this.getSide()) <= 0.001;
        }
        return false;
    }
    @Override
    public void setWidth(double side){
        this.setSide(side);
    }
    @Override
    public void setLength(double side){
        this.setSide(side);
    }

    public String toString(){
        return "Square[topLeft=" +topLeft.toString()+",side="+this.getSide()+",color="+this.color+",filled="+this.filled+"]";
    }
}
