import java.util.*;
public class Layer {
    private List<Shape> shapes= new ArrayList<>();
    
    public void addShape(Shape other) {
        shapes.add(other);
    }
    public void removeCircles() {
        for(int i=0;i<shapes.size();i++) {
            
if(shapes.get(i) instanceof Circle) 
                shapes.remove(i);
             
        }
    }
    public void removeDuplicates() {
       
        for(int i=0;i<shapes.size()-1;i++) {
            Object a=shapes.get(i);
            for(int j=shapes.size()-1;j>i;j--) {
                Object b=shapes.get(j);
                if(a.equals(b)) {
                    shapes.remove(j);
                }
            }
        }
    }
    public 
String getInfo() {
        String str="Layer of crazy shapes: \n";
        for (Object shape : shapes) 
            str += shape.toString() + "\n";
        
        return str;
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft=new Point();
    public Rectangle(){}
    public Rectangle(double w,double l){
        width=w;
        length=l;
    }
    public Rectangle(double w,double l,String color,boolean a){
        super(color,a);
        width=w;
        length=l;
    }
    public Rectangle(Point p,double w,double l,String color,boolean a){
        super(color,a);
        setTopLeft(p);
        width=w;
        length=l;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double a) {
        width=a;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double a) {
        length=a;
    }
    @Override
    public double getArea() {
        return width*length;
        //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
    }

    @Override
    public double getPerimeter() {
        return 2*(width+length);
        //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
    }
    public Point getTopLeft(){
        return topLeft;
    }
    public void setTopLeft(Point other) {
        topLeft.setX(other.getX());
        topLeft.setY(other.getY());
    
}
    public boolean equals(Object oth) {
        if(oth instanceof Rectangle) {
            Rectangle o= (Rectangle) oth;
            if(o.getWidth()==this.getWidth() && o.getLength()==getLength()
                    &&(o.getTopLeft().equals(getTopLeft()))&& o.hashCode()==hashCode()) 
                return true;
        }
        return false;
    }
    public int hashCode() {
        String str="Rectangle[topLeft="+getTopLeft().toString()+",width="+getWidth()+
                ",length="+getLength()+"]";
        return str.hashCode();
    }
    public String toString() {
        return "Rectangle[topLeft="+getTopLeft().toString()+",width="+getWidth()+
                ",length="+getLength()+",color="+getColor()+",filled="+isFilled()+"]";
    }
}
