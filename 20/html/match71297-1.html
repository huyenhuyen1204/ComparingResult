public class Circle extends Shape {
    protected double radius;
    protected Point center;
    public Circle(){};
    public Circle(double radius)
    {
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius=radius;
    }
    public Circle(Point x,double radius,String color,boolean filled)
    {
        super(color,filled);
        this.radius=radius;
        this.center=x;
    }
    public double getRadius() {
        return radius;
    }
    
    public Point getCenter(){
        return this.center;
    }
    void setCenter(Point x){
        this.center=x;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea() {
        return radius*radius*Math.PI;
    }
    public double getPerimeter(){
        return Math.PI*radius*2;
    }
    public boolean equals(Object c){
        if (c instanceof Circle){
            Circle cir=(Circle) c;
            if (this.center==cir.center&&this.radius==cir.radius)
                return true;
        }
        return false;
    }
    public 
int hashCode(){
        return this.hashCode();
    }
    @Override
    public String toString() {
        return "Circle[center=("+this.center.x+","+this.center.y+")"+",radius="+getRadius()+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }
}

import java.util.*;

public class Layer {
    List<Shape> shapes=new LinkedList<Shape>();
    public void addShape(Shape x){
        shapes.add(x);
    }
    public void removeCircles(){
        for (int i=0;i<shapes.size();i++){
            
if (shapes.get(i) instanceof Circle ){
                shapes.remove(i);
            }
        }
    }
    public String getInfo(){
        String s="Layer of crazy shapes: \n";
        for (Shape shape: shapes){
            
s=s+shape.toString()+'\n';
        }
        return s;
    }
    public void removeDuplicates(){
        for (int i=0;i<shapes.size();i++){
            for (int j=0;j<shapes.size();j++){
                if (i!=j &&shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                }
            }
        }
    }
}
