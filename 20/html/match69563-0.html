Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Circle(double radius, String color, boolean filled) {
        this(null, radius, color, filled);
    }

    public Circle(double radius) {
        this(null, radius, null, false);
    }

    public Circle() {
        this(1);
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    @Override
    public double getArea() {
        return radius * radius * Math.PI;
    }

    @Override
    public double getPerimeter() {
        return 2 * Math.PI * radius;
    }

    @Override
    public String toString() {
        return "Circle[center=" + center + ",radius=" + Math.round(radius * 10) / 10.0 + ",color=" + color + ",filled=" + filled + "]";
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Circle) {
            return Math.abs(this.center.getX() - ((Circle) obj).center.getX()) <= 0.001 &&
                    Math.abs(this.center.getY() - ((Circle) obj).center.getY()) <= 0.001 &&
                    Math.abs(this.radius - ((Circle) obj).radius) <= 0.001;
        }
        return false;
    }

    public static void main(String[] args) {
        Point p1 = new Point(1, 2);
        Shape c1 = new Circle(p1, 3, "red", false);
        System.out.println(c1.getArea());
        System.out.println(c1.getPerimeter());
        System.out.println(c1);
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape newShape) {
        shapes.add(newShape);
    }

    public String getInfo() {
        StringBuilder info = new StringBuilder("Layer of crazy shapes: \n");
        for (Shape shape : shapes) {
            info.append(shape.toString()).append("\n");
            // CAREFUL here
        }
//        info.append("------------");
        return info.toString();
    }

    public void removeCircles() {
        for (int i = shapes.size() - 1; i >= 0; i--) {
            if (shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    }

    public void removeDuplicates() {
        for (int i = 0; i < shapes.size(); i++) {
            for (int j = shapes.size() - 1; j > i; j--) {
                if (shapes.get(i).equals(shapes.get(j)) &&
                        shapes.get(i).getClass().equals(shapes.get(j).getClass())) {
                    shapes.remove(j);
                }
            }
        }
    }

    public static void main(String[] args) {
        Shape c1 = new Circle(new Point(3, 4), 3, "red", false);
        Shape r1 = new Rectangle(new Point(2.1424, 9), 2, 5, "blue", false);

        Shape s2 = new Square(new Point(2.1424, 9), 2, "blue", false);
        Shape r3 = new Square(new Point(2.1424, 9), 2, "blue", false);

        layer.addShape(c1);
        layer.addShape(c1);
        layer.addShape(r1);
        layer.addShape(s1);
        layer.addShape(s1);
        layer.addShape(r1);
        layer.addShape(r2);

        System.out.println(layer.getInfo());
//        layer.removeCircles();
        layer.removeDuplicates();
        System.out.println(layer.getInfo());
    }

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        this(null, width, length, color, filled);
    }

    public Rectangle(double width, double length) {
        this(null, width, length, null, false);
    }

    public Rectangle() {
        this(1, 1);
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getWidth() {
        return width;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getLength() {
        return length;
    }

    @Override
    public double getArea() {
        return width * length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }

    @Override
    // TODO: round
    public String toString() {
//        double roundedX = Math.round(this.topLeft.getX() * 10) / 10.0;
//        double roundedY = Math.round(this.topLeft.getY() * 10) / 10.0;
//        Point roundedTopLeft = new Point(roundedX, roundedY);
        return "Rectangle[topLeft=" + topLeft + ",width=" + width + ",length=" + length + ",color=" + color + ",filled=" + filled + "]";
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle) {
            return Math.abs(this.topLeft.getX() - ((Rectangle) obj).topLeft.getX()) <= 0.001 &&
                    Math.abs(this.topLeft.getY() - ((Rectangle) obj).topLeft.getY()) <= 0.001 &&
                    Math.abs(this.width - ((Rectangle) obj).width) <= 0.001 &&
                    Math.abs(this.length - ((Rectangle) obj).length) <= 0.001;
        }
        return false;
    }

    public static void main(String[] args) {
        Shape r1 = new Rectangle(2, 3);
        System.out.println(r1.getArea());
        System.out.println(r1.getPerimeter());
        System.out.println(r1);

        Rectangle r2 = (Rectangle) r1;
        System.out.println(r2.getArea());
        r2.setWidth(100);
        System.out.println(r1.getArea());
    }
}
