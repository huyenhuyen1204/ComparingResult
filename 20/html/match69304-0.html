import java.awt.*;

public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle(){

    }

    public Circle(double radius){
        this.radius= radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius= radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius= radius;
        this.center= center;
    }

    public Point getCenter(){
        return center;
    }

    public void setCenter(Point center){
        this.center= center;
    }

    public double getRadius(){
        return radius;
    }

    public void setRadius(double radius){
        this.radius= radius;
    }


    public double getArea(){
        Circle circle= new Circle();
        return getRadius()*getRadius()*Math.PI;
    }

    public double getPerimeter(){
        Circle circle= new Circle();
        return getRadius()*2*Math.PI;
    }

    public boolean equals(Object obj) {
        if(obj instanceof Circle){
            Circle circle = (Circle) obj;
            if(circle.radius == this.radius && circle.center == this.center)
                return true;
            else
                return false;
        }
        else
            return false;
    }

    @Override
    public String toString() {
        return "Circle[center=("+getCenter().getX()+","+getCenter().getY()+"),radius="+getRadius()+
                ",color="+getColor()+",filled="+isFilled()+"]";
    }
}

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public class Layer {
    protected List<Shape> shapes= new ArrayList<Shape>();

    public void addShape(Shape shape){
        shapes.add(shape);
    }

    public void removeCircles(){
        int i, j;
        for (i = 0; i< shapes.size(); i++){
            if (shapes.get(i) instanceof Circle){
                shapes.remove(shapes.get(i));
            }
        }
    }

    public String getInfo(){
        int i;
        String s;
        s = "Layer of crazy shapes: "+"\n";
        for(i = 0; i< shapes.size(); i++){
            s = s + shapes.get(i).toString()+"\n";
        }
        return s;
    }

    public void removeDuplicates(){
        int i, j;
        for(i = 0; i< shapes.size() -1; i++){
            for(j = i + 1; j< shapes.size(); j++){
                if(shapes.get(i).equals(shapes.get(j)) ){
                    shapes.remove(shapes.get(j));
                }

            }
        }
    }
}

src/Point.java

public class Point {
    protected double x;
    protected double y;

    public Point(double x, double y){
        this.x= x;
        this.y= y;
    }

    public double getX(){
        return x;
    }

    public void setX(double x){
        this.x= x;
    }

    public double getY(){
        return y;
    }

    public void setY(double y){
        this.y= y;
    }

    public double distance(Point obj){
        return Math.sqrt((this.x- obj.getX())*(this.x- obj.getX())+(this.y-obj.getY())*(this.y-obj.getY()));
    }

    public boolean equals(Object obj){
        if(obj instanceof Point){
            Point point= (Point) obj;
            if(point.x== getX() && point.y== getY())
                return true;
            else
                return false;
        } else
            return false;
    }

    public String toString(){
        return "("+getX()+","+getY()+")";
    }

}

public class Rectangle extends  Shape{
    protected double width;
    protected double length;
    protected Point topLeft;


    public Rectangle(){

    }

    public Rectangle(double width, double length){
        this.width= width;
        this.length= length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color,filled);
        this.width= width;
        this.length= length;

    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.width= width;
        this.length= length;
        this.topLeft= topLeft;
    }

    public double getWidth(){
        return width;
    }

    public void setWidth(double width){
        this.width= width;
    }

    public double getLength(){
        return length;
    }

    public void setLength(double length){
        this.length= length;
    }

    public double getArea(){
        Rectangle rectangle= new Rectangle();
        return getWidth()*getLength();
    }

    public double getPerimeter(){
        Rectangle rectangle= new Rectangle();
        return 2*(getLength()+getWidth());
    }

    public Point getTopLeft() {
        return topLeft;
    };

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public boolean equals(Object object){
        if (object instanceof  Rectangle){
            Rectangle rect = (Rectangle) object;
            if(rect.width == this.width && rect.length == this.length && rect.topLeft == this.topLeft)
                return true;
            else
                return false;
        }
        else
            return false;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=("+getTopLeft().getX()+","+getTopLeft().getY()+"),width="+getWidth()+",length="+getLength()+
                ",color="+getColor()+",filled="+isFilled()+"]";
    }
}

public class Square extends Rectangle {

    public Square(){

    }
    public Square(double side){
        super(side, side);
    }

    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled){
        super(topLeft,side, side, color, filled);
    }

    public double getSide(){
        return getWidth();
    }

    public void setSide(double side){
        setWidth(side);
        setLength(side);
    }

    @Override
    public void setLength(double length) {
        setLength(length);
        setWidth(length);
    }

    @Override
    public void setWidth(double width) {
        setWidth(width);
        setLength(width);
    }

    @Override
    public String toString() {
        return "Square[topLeft=("+topLeft.getX()+","+topLeft.getY()+"),side="+getSide()+
                ",color="+getColor()+",filled="+isFilled()+"]";
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Square){
            Square other = (Square) obj;
            if (this.width==other.width && this.length==other.length&& this.topLeft.equals(other.topLeft)) return true;
            else return false;
        }
        else return false;
    }
}
