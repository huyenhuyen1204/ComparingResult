return false;

import java.util.*;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();
    public void addShape(Shape shape){
       this.shapes.add(shape);
    }
    public void removeCircles(){
        for(int i = 0 ; i < shapes.size() ; i++){
            if(
shapes.get(i).getClass() == Circle.class){
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        String s = "Layer of crazy shapes: \n";
        for(Shape i : shapes){
            s = s + i.toString() + "\n";
        }
        return s;
    }
    public void removeDuplicates(){
            List<Shape> newList = new ArrayList<Shape>();
            for (Shape element : shapes) {
                if (!newList.contains(element)) {
                    newList.add(element);
                }
            }

            shapes = newList;
        }
    }

Week6-1/Rectangle.java

public class Rectangle extends Shape{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){

    }
    public Rectangle(double width, double length){
        this.length = length;
        this.width = width;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }public Rectangle(Point TopLeft,double width, double length, String color, boolean filled){
        this.topLeft = TopLeft;
        this.width = width;
        this.length = length;
        this.color = color;
        this.filled = filled;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public double getArea(){
        return (this.length*this.width);
    }
    public double getPerimeter(){
        return (this.length + this.width)*2;
    }
    public Point getTopLeft(){
        return topLeft;
    }
    public void setTopLeft(Point p){
        this.topLeft = p;
    }
    public boolean equals(Object obj){
        if(getClass() != obj.getClass()) 
return false;
        Rectangle rectangle = (Rectangle) obj;
        return Math.abs(topLeft.getX() - rectangle.topLeft.getX()) <= 0.001 &&
                Math.abs(topLeft.getY() - rectangle.topLeft.getY()) <= 0.001 &&
                Math.abs(getLength() - rectangle.getLength()) <= 0.001 &&
                Math.abs(getWidth() - rectangle.getWidth()) <= 0.001;
    }
    public String toString(){
        width = Math.round(width*10.0f) / 10.0f;
        length = Math.round(length*10.0f) / 10.0f;
        return "Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + color +",filled=" + filled +"]";
    }
    public int hashCode(){
        return (int)(width + length) + (int)(width*length);
    }
}
