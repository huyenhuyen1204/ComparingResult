Circle.java

public class Circle extends Shape{
    protected double radius;
    protected Point center;
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getRadius() {
        return radius;
    }

    public Circle(Point P,double radius ,String color,boolean filled)
    {
        super(color,filled);
        this.radius = radius;
        this.center = P;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public Point getCenter() {
        return center;
    }

    @Override
    public double getArea() {
        return 3.14159*radius*radius;
    }
    public double getPerimeter(){
        return 3.14159*radius*2;
    }

    @Override
    public boolean equals(Object obj){
        if (obj instanceof Circle){
            Circle other = (Circle)obj;
            {
                if (Math.abs(this.radius-other.radius)<=0.001 && Math.abs(this.center.getX()-other.center.getX())<=0.001 && Math.abs(this.center.getY()-other.center.getY())<=0.001){
                    return true;
                }
            }
        }
        return false;
    }
    @Override
    public int hashCode(){
        return super.hashCode();
    }
    @Override
    public String toString(){
        radius = (double)Math.round(radius*10)/10;
        return "Circle[center="+getCenter().toString()+",radius="+radius+",color="+color+",filled="+filled+"]";
    }

}

import java.util.*;

public class Layer
{
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape sh) {
        shapes.add(sh);

    }

    public void removeCircles() {

        for (int i=0;i<shapes.size();i++){
            {

                if  (shapes.get(i) instanceof Circle) {
                    shapes.remove(shapes.get(i));
                    i--;
                }
            }
        }

    }
    public String getInfo(){
        String list="Layer of crazy shapes: "+"\n";
        for (int i=0;i<shapes.size();i++){
            list+=shapes.get(i).toString()+"\n";
        }
        return list;
    }

    public void removeDuplicates() {
        List<Shape> list = new ArrayList<Shape>();
        for (int i = 0; i < shapes.size(); i++) {
            if (list.contains(shapes.get(i)) == false) {
                list.add(shapes.get(i));
            }
        }
        while (shapes.size() != 0) {
            shapes.remove(0);
        }
        for (int i = 0; i < list.size(); i++) {
            shapes.add(list.get(i));
        }
    }


    public static void main(String []args)
    {
        Layer a = new Layer();
        Point point = new Point(10,5);
        Shape circle = new Circle(point,8.0,"RED",true);
        a.addShape(circle);
        Shape rectangle = new Rectangle(point,8.0,6.0,"RED",true);
        a.addShape(rectangle);
        Shape circle1 = new Circle(point,8.0,"RED",true);
        a.addShape(circle1);
        Shape square = new Square(point, 8.0, "RED", true);
        a.addShape(square);
        Shape circle2 = new Circle(point,8.0,"RED",true);
        a.addShape(circle2);
        Shape circle3 = new Circle(point,8.0,"RED",true);
        a.addShape(circle3);
        a.removeDuplicates();
        System.out.println(a.getInfo());

    }
}

Point.java

public class Point
{
    private double x;
    private double y;
    public Point(double x,double y)
    {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point a)
    {
        return Math.sqrt( Math.pow(getX() - a.getX() ,2) +Math.pow(getY() - a.getY() ,2));
    }
    public  boolean equals(Object obj)
    {
        if(obj instanceof Point)
        {
            Point other = (Point)obj;
            if(Math.abs(this.x -other.x) <= 0.001 && Math.abs(this.y -other.y) <= 0.001)
            {
                return  true;
            }
        }
        return false;
    }

    @Override
    public int hashCode(){
        return super.hashCode();
    }

    public String toString()
    {
        x = (double)Math.round(x*10)/10;
        y = (double)Math.round(y*10)/10;
        return "("+x+","+y+")";
    }

}

public class Rectangle extends Shape{

    public  Rectangle(double width, double length ,String color ,boolean filled)

        super(color,filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle() {

    }

    public  Rectangle(Point P,double width, double length ,String color ,boolean filled)
    {
        super(color,filled);

        this.topLeft = P;
    }

    public void setWidth(double width)

        this.topLeft = P;

    public Point getTopLeft() {
        return topLeft;
    }

    @Override
    public double getArea() {
        return width*length;

    public double getPerimeter() {
        return 2*width+2*length;

    {
        if(obj instanceof Rectangle)

            Rectangle other = (Rectangle)obj;
            if(Math.abs(this.width - other.width)<=0.001 && Math.abs(this.length - other.length)<=0.001
                    && Math.abs(this.topLeft.getX()- other.topLeft.getX() )<=0.001 && Math.abs(this.topLeft.getY()- other.topLeft.getY() )<=0.00)

        width = (double)Math.round(width*10)/10;
