public class Circle extends Shape {
    protected  Point center;
    protected double radius;
    public Circle(){}
    public Circle(double radius){this.radius = radius;}
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }
    public Point getCenter(){return this.center;}
    public void setCenter(Point center){this.center = center;}
    public double getRadius(){return this.radius;}
    public void setRadius(double radius){this.radius = radius;}
    public double getArea(){return Math.PI*radius*radius;}
    public double getPerimeter(){return 2.0*Math.PI*radius;}
    public 
boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle circle = (Circle) obj;
            if(this.center.equals(circle.center)&&Math.abs(this.radius - circle.radius)<=0.001) return true;
        }
        
return false;
    }
    public String toString(){
        return "Circle[center=" + this.center.toString() +",radius=" + this.radius + ",color=" + getColor() + ",filled=" + isFilled() + "]";
    }
}

import java.util.Objects;

/**
 * Created by CCNE on 10/10/2019.
 */
public class Point {
    private double x;
    private double y;
    public 
Point(double x, double y){
        this.x = x;
        this.y = y;
    }
    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }
    public void setX(double x){this.x = x;}
    public void setY(double y){this.y = y;}
    public double distance(Point other){
        double d=Math.sqrt(Math.pow(this.x-other.x, 2.0)+Math.pow(this.y-other.y, 2.0));
        d = (double)Math.round(d*10)/10;
        return d;
    }
    public boolean equals(Object obj){
        if(obj instanceof Point){
            Point other = (Point) obj;
            if(other.x==this.x&&other.y==this.y) return true;
        }
        return false;
    }
    public String toString(){
        return "(" + this.x + "," + this.y + ")";
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){}
    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    
}
    public double getWidth(){return this.width;}
    public double getLength(){return this.length;}
    public void setWidth(double width){this.width = width;}
    public void setLength(double length){this.length = length;}
    public double getArea(){return this.width*this.length;}
    public double getPerimeter(){return 2.0*(this.width+this.length);}
    public Point getTopLeft(){return this.topLeft;}
    public void setTopLeft(Point topLeft){this.topLeft = topLeft;}
    public boolean equals(Object obj){
        if(obj instanceof Rectangle){
            Rectangle rect = (Rectangle)obj;
            if(this.topLeft.equals(rect.topLeft)&&Math.abs(this.width-rect.width)<=0.001&&Math.abs(this.length-rect.length)<=0.001) return true;
        
}
        return false;
    }
    public String toString(){
        return "Rectangle[topLeft=" + this.topLeft.toString() +",width=" + this.width + ",length=" + this.length + ",color=" + getColor() + ",filled=" +isFilled() +"]";
    }
}

public class Square extends Rectangle {
    public Square(){}
    public Square(double side){
        super(side, side);
    }
    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }
    public Square(Point topLeft, double side, String color, boolean filled){
        super(topLeft, side, side, color, filled);
    }
    public double getSide(){
        return super.getLength();
    }
    public void setSide(double side){
        setLength(side);
        setWidth(side);
    }
    @Override
    public void setWidth(double side){super.setWidth(side);
        super.setLength(side);
    }
    public void setLength(double side){setWidth(side);}
    public 
boolean equals(Object obj){
        if(obj instanceof Square){
            Square sq = (Square)obj;
            if(this.topLeft.equals(sq.topLeft)&&Math.abs(this.width-sq.width)<=0.001) return true;
        
}
        return false;
    }
    public String toString(){
        return "Square[topLeft="+ super.getTopLeft().toString()+ ",side=" + getSide() + ",color=" + getColor() + ",filled=" + isFilled() +"]";
    }
}
