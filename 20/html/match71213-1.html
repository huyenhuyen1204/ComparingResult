import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape s) {
        if(s != null) this.shapes.add(s);
    }

    public void removeCircles() {
        for(int i = 0; i < shapes.size(); i++) {
            
if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }
        }
    }

    public String getInfo() {
        String ans = "Layer of crazy shapes: \n" ;
        for(int i = 0; i < shapes.size(); i++) {
            ans += shapes.get(i).toString() + "\n";
        
}
        return ans;
    }

    public void removeDuplicates() {
        for(int i = 0; i < shapes.size() - 1; i++) {
            for (int j = i + 1; j < shapes.size(); j++) {
                if(shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    public static void main(String[] args) {
        Point p = new Point(1.2, 2.4);
        Point p1 = new Point(2.3345, 4.2);
        Point p2 = new Point(1.2345, 2.3456);
        Circle c1 = new Circle(p, 2.5551, "red", false);
        Circle c2 = new Circle(p, 2.555, "blue", true);
        Rectangle r1 = new Rectangle(p1, 2.4, 4.2, "yellow", true);
        Rectangle r2 = new Rectangle(p1, 2.4, 4.2, "red", false);
        Square s1 = new Square(p2, 2.555, "red", true);
        Square s2 = new Square(p2, 2.5551, "blue", true);
        //System.out.println(c1.equals(c2));
        //System.out.println(c1.toString());
        Layer shapes = new Layer();
        shapes.addShape(c1);
        
shapes.addShape(c2);
        shapes.addShape(r1);
        shapes.addShape(r2);
        shapes.addShape(s1);
        shapes.addShape(c2);
        shapes.addShape(s2);
        System.out.println(s1.equals(s2));
        //System.out.print(s);
    }
}
