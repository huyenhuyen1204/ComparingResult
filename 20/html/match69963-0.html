Layer.java

import java.util.*;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape shape){
        shapes.add(shape);
    }

    public void removeCircles(){
        for (int i = 0; i < shapes.size(); i++)
            if(shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
                i--;
            }
    }

    public void removeDuplicates(){
        for(int i = 0; i < shapes.size(); i++)
            for(int j = i + 1; j < shapes.size(); j++)
            {
                if(shapes.get(j).equals(shapes.get(i)))
                {
                    shapes.remove(j);
                    j--;
                }
            }
    }

    public String getInfo(){
        String s = "Layer of crazy shapes: " + "\n" ;
        for(int i = 0; i < shapes.size(); i++)
            s = s + shapes.get(i).toString();
        return s;
    }

}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x,  double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point a){
        double distance = Math.sqrt(Math.pow((this.x - a.x), 2) + Math.pow(this.y - a.y, 2));
        return distance;
    }

    @Override
    public boolean equals(Object obj) {
        if(this == obj)
                return true;
        if (obj == null || this.getClass() != obj.getClass())
            return false;
        Point p = (Point) obj;
        if(Math.abs(x - p.x) <= 0.001 && Math.abs(y - p.y) <= 0.001)
            return true;
        else return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "(" + x + "," + y + ")";
    }
}
