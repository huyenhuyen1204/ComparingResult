import java.util.Objects;

public class Circle extends Shape {
    protected double radius;
    protected final double PI =3.14;
    protected Point center;
    public Circle(){}
    public Circle(double radius)
    {
        this.radius=radius;
    }
    public Circle(double radius,String color,boolean filled)
    {
        super(color, filled);
        this.radius=radius;
    }
    public Circle (Point center,double radius,String color,boolean filled)
    {
        super(color, filled);
        this.center=center;
        this.radius=radius;
    }
    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius)
    {
        this.radius=radius;
    }
    public Point getCenter()
    {
        return center;
    }
    public void setCenter(Point center) {
        this.center = center;
    }
    @Override
    public double getArea() {
        return radius*radius*PI;
    }
    @Override
    public double getPerimeter() {
        return radius*2*PI;
    }
    @Override
    public boolean equals(Object obj)
    {
        if(obj instanceof Circle) {
            Circle a = (Circle) obj;
            if (Math.abs(a.getRadius() - this.getRadius()) <= 0.001 && a.center.equals(this.center) == true)
                return true;
            else return false;
        }
        return false;
    }
    @Override
    public int hashCode()
    {
        return super.hashCode();
    }
    @Override
    public String toString()
    {
        return "Circle[center=("+this.center.getX()+","+this.center.getY()+"),radius="+this.getRadius()+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }
    public static void main(String[] args)
    {
        Point a=new Point(4.88888,5.00002);
        Circle c=new Circle(a,3.0001,"blue",false);
        Point b=new Point(4.88888,5.00002);
        Circle d= new Circle(b,3.0001,"blue",false);
        System.out.println(a.toString());
        System.out.println(c.toString());
        System.out.println(c.getArea());
        System.out.println(d.getPerimeter());
        System.out.println(d.equals(c));

    }
}

import java.util.List;
import java.util.ArrayList;
import java.util.Scanner;

public class Layer {
     private List<Shape> shapes=new ArrayList<>();
     public void addShape(Shape a)
         {
             shapes.add(a);
         }
         public void removeCircles()
         {
          for(int i=0;i<shapes.size();i++)
          {
              if(shapes.get(i) instanceof Circle){ shapes.remove(i);
              i--;}
          }
     }
     public String getInfo()
     {
         String s="";
         s+="Layer of crazy shapes: "+"\n";
         for(int i=0;i<shapes.size();i++)
         {
             s+= shapes.get(i).toString()+"\n";
         }
         return s;
     }
     public void removeDuplicates()
     {
         for(int i=0;i<shapes.size()-1;i++)
         {
             for(int j=i+1;j<shapes.size();j++)
             {
                 if(shapes.get(i).equals(shapes.get(j)))
                 {
                     shapes.remove(j);
                     j--;
                 }
             }
         }
     }
}

import java.text.DecimalFormat;

public class Rectangle extends Shape {
    protected double width;
        protected double length;
        protected Point topLeft;
        DecimalFormat f=new DecimalFormat("#.0");

    public Rectangle(){}
    public Rectangle(double width,double length)
            {
                this.width=width;
                this.length=length;
            }
    public Rectangle(double width,double length,String color, boolean filled)
            {
                super(color, filled);
                this.width=width;
                this.length=length;
            }
    public Rectangle(Point topLeft,double width,double length,String color,boolean filled)
            {
                super(color, filled);
                this.topLeft=topLeft;
                this.width = width;
        this.length=length;
    }
    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }
    @Override
    public double getArea() {
        return width*length;
    }
    @Override
    public double getPerimeter() {
        return 2*(width+length);
    }
    @Override
    public boolean equals(Object obj)
    {
        if(obj instanceof Rectangle) {
            Rectangle a = (Rectangle) obj;
            if (Math.abs(a.width - this.width) <= 0.001 && Math.abs(a.length - this.length) <= 0.001 && a.topLeft.equals(this.topLeft))
                return true;
        }
        return false;
    }
    @Override
    public int hashCode()
    {
        return super.hashCode();
    }
    public String toString()
    {
        return "Rectangle[topLeft="+topLeft.toString()+",width="+ f.format(width)+",length="+f.format(length)+",color="+color+",filled="+filled+"]";
    }
    public static void main(String[] args)
    {
        Point x=new Point(2.999993,3.862738);
        Point y=new Point(2.9998,3.86275);
        Point z=new Point(4,11);
        Rectangle a= new Rectangle(x,7.55554,11.5134,"red",true);
        Rectangle b=new Rectangle(y,7.5555,11.5135,"blue",true);
        Rectangle c= new Rectangle(z,8,12,"blue",false);
        System.out.println(x.toString());
        System.out.println(a.toString());
        System.out.println(a.equals(b));
        System.out.println(a.equals(c));

    }

}

import java.text.DecimalFormat;

public class Square extends Rectangle {
    DecimalFormat f=new DecimalFormat("#.0");
    public Square(){}
    public Square(double side)
    {
        super(side,side);
    }
    public Square(double side,String color,boolean filled)
    {
        super(side,side,color,filled);
    }
    public Square(Point topLeft,double side,String color,boolean filled)
    {
        super(topLeft,side,side,color,filled);
    }
    public double getSide() {
        return width;
    }
    public void setSide(double side) {
        this.width = side;
        this.length=side;
    }
    public void setWidth(double side) {
        this.width=side;
        this.length=side;
    }
    public void setLength(double side) {
        this.width=side;
        this.length=side;
    }
    public boolean equals(Object obj)
    {
        if(obj instanceof Square)
        {
            Square a=(Square)obj;
            if(Math.abs(a.getSide()-this.getSide())<=0.001&&a.topLeft.equals(this.topLeft)) return true;
            else return false;
        }
        return false;
    }
    public String toString()
    {
        return "Square[topLeft="+topLeft.toString()+",side="+f.format(getSide())+",color="+color+",filled="+filled+"]";
    }
    public static void main(String[] args)
    {
        Point a=new Point(3,4);
        Point b=new Point(2,3);
        Square c=new Square(a,3,"black",true);
        Square d=new Square(a,3,"black",true);
        System.out.println(a.toString());
        System.out.println(b.equals(a));
        System.out.println(c.equals(d));
        System.out.println(d.toString());
    }
