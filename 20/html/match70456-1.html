public class Circle extends Shape
{
    protected double radius;
    protected Point center = new Point();
    public 
Circle()
    {
    }
    public Circle(double radius)
    {
        this.radius = radius;
    }
    public Circle(double radius,String color, boolean filled)
    {
        this.radius = radius;
        setColor(color);
        setFilled(filled);
    }
    public Circle(Point p ,double radius,String color,boolean filled)
    {
        this.center = p;
        this.radius = radius;
        setColor(color);
        setFilled(filled);
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public double getArea()
    {
        return this.radius*this.radius*Math.PI;
    }
    public double getPerimeter()
    {
        return 2.0*this.radius*Math.PI;
    }
    public String toString() {
        String s = "Circle[center=(" + Math.round(center.getX()*10.0)/10.0 + "," + Math.round(center.getY()*10.0)/10.0 + "),radius=" + Math.round(this.radius*10.0)/10.0 + ",color=" + this.color + ",filled=" + this.filled + "]";
        return s;
    }
    public boolean equals(Object o)
    {
        Circle c = (Circle) o;
        if (Math.round(c.getRadius()*1000.0)/1000.0 == Math.round(this.radius * 1000.0)/1000.0)
        {
            return true;
        }
        return false;
    }
    public int hashCode()
    {
        return this.hashCode();
    }}

import java.util.List;
import java.util.ArrayList;

public class Layer {
    private List<Shape> shapes = new ArrayList<>() ;
    public void addShape(Shape s)
    {
        this.shapes.add(s);
    }
    public void removeCircles()
    {
        for (int i=0;i<shapes.size();i++)
        {
            
if (shapes.get(i) instanceof Circle)
            {
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo()
    {
        String s = "Layer of crazy shapes: " +"\n";
        int size = shapes.size();
        for (int i = 0;i < size - 1;i++)
        {
            s = s + shapes.get(i).toString() + "\n";
        }
        s = s + shapes.get(shapes.size() - 1).toString() ;
        
return s;
    }
    public void removeDuplicates()
    {
        for (int i = 0;i<shapes.size();i++)
        {
            for (int j = i+1;j<shapes.size();j++)
            {
                if (shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle)
                {
                    Circle c1 = (Circle) shapes.get(i);
                    Circle c2 = (Circle) shapes.get(j);
                    if (c1.getCenter().equals(c2.getCenter()) && c1.getRadius() == c2.getRadius())
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
                else 
if (shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle)
                {
                    Rectangle r1 = (Rectangle) shapes.get(i);
                    Rectangle r2 = (Rectangle) shapes.get(j);
                    if (r1.getTopLeft().equals(r2.getTopLeft()) && r1.getWidth() == r2.getWidth() && r1.getLength() == r2.getLength())
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
                else 
if (shapes.get(i) instanceof Square && shapes.get(j) instanceof Square)
                {
                    Square s1 = (Square) shapes.get(i);
                    Square s2 = (Square) shapes.get(j);
                    if (s1.getTopLeft().equals(s2.getTopLeft()) && s1.getSide() == s2.getSide())
                    {
                        shapes.remove(j);
                        j--;
                    }
                }
            }
        }
    }
}

class Rectangle extends Shape
{
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle()
    {
    }
    public Rectangle(double width,double length)
    {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width,double length,String color,boolean filled)
    {
        this.width = width;
        this.length = length;
        setColor(color);
        setFilled(filled);
    }
    public Rectangle(Point point,double width,double length,String color,boolean filled)
    {
        this.topLeft = point;
        this.width = width;
        this.length = length;
        setColor(color);
        setFilled(filled);
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }
    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }
    public int hashCode()
    {
        return this.hashCode();
    }
    public boolean equals(Object o)
    {
        Rectangle 
r = (Rectangle) o;
        if (Math.round(this.width*1000.0)/1000.0 == Math.round(r.width*1000.0)/1000.0 && Math.round(this.length*1000.0)/1000.0 == Math.round(r.length*1000.0)/1000.0)
        {
            return true;
        }
        return false;
    }
    @Override
    public double getArea()
    {
        return this.width*this.length;
    }
    public double getPerimeter()
    {
        return (this.width + this.length)*2.0;
    }
    public String toString()
    {
        String s = "Rectangle[topLeft=(" + Math.round(topLeft.getX()*10.0)/10.0 + "," + Math.round(topLeft.getY()*10.0)/10.0 + "),width=" + Math.round(this.width*10.0)/10.0 + ",length=" + Math.round(this.length*10.0)/10.0 + ",color=" + this.color + ",filled=" + this.filled +"]";
        return s;
    }
}
