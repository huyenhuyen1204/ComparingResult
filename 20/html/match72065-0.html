import java.awt.*;
import java.util.Objects;

public class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){}
    public Circle(double red){
        radius=red;
    }
    public Circle(double rad,String col,boolean fil){
        radius=rad;
        color=col;
        filled=fil;
    }
    public Circle(Point cent,double rad,String col,boolean fil){
        center=cent;
        radius=rad;
        color=col;
        filled=fil;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public double getArea(){
        return 3.14*radius*radius*100.0/100.0;
    }
    @Override
    public double getPerimeter(){
        return 2*3.14*radius*100.0/100.0;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Circle circle = (Circle) o;
        return Double.compare(circle.radius, radius) == 0 &&
                center.equals(circle.center);
    }

    @Override
    public int hashCode() {
        return Objects.hash(center, radius);
    }

    @Override
    public String toString(){
        return "Circle[center="+center+
                ",radius="+radius+
                ",color="+color+
                ",filled="+filled+"]";
    }

    public static void main(String[] args) {
        Point x=new Point(10.0,5.0);
        Circle circle= new Circle(x,1,"red",false);
        Circle circle1= new Circle(x,2,"blue",true);
        System.out.println(circle.toString());
        System.out.println(circle1);
    }
}

        String a="Layer of crazy shapes: "+"\n";
        for (int i=0;i<s.size();i++){
            a+=s.get(i).toString()+"\n";
        }
        return a;
    }
    public void removeDuplicates(){
        List<Shape> s=this.shapes;
        for(int i=0;i<s.size();i++)
            {
                for (int j=i+1;j<s.size();j++){
                    if((s.get(i) instanceof Circle && s.get(j) instanceof Square) || (s.get(i) instanceof Square && s.get(j) instanceof Circle)) continue;

                        s.remove(j);
                        j--;
                    }
                }
            }

import java.util.Objects;

public class Rectangle extends Shape{
    protected Point topLeft;
    protected double width;
    protected double length;
    public Rectangle(){ }
    public Rectangle(double wod,double len){
        width=wod;
        length=len;
    }
    public Rectangle(double wid,double len,String col,boolean fil){
        super(col,fil);
        color=col;
        filled=fil;
        width=wid;
        length=len;
    }
    public Rectangle(Point topl,double wod,double len,String col,boolean fil){
        color=col;
        filled=fil;
        topLeft=topl;
        width=wod;
        length=len;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }
    @Override
    public double getArea(){
        return width*length*100.0/100.0;
    }
    @Override
    public double getPerimeter(){
        return 2*(width+length);
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Rectangle rectangle = (Rectangle) o;
        return Double.compare(rectangle.width, width) == 0 &&
                Double.compare(rectangle.length, length) == 0 &&
                topLeft.equals(rectangle.topLeft);
    }

    @Override
    public int hashCode() {
        return Objects.hash(topLeft, width, length);
    }

    @Override
    public String toString() {
        return "Rectangle[" +
                "topLeft=" + topLeft +
                ",width=" + width +
                ",length=" + length +
                ",color=" +color+
                ",filled=" +filled+
                "]";
    }

    public static void main(String[] args) {
        Point y= new Point(10.5,5);
        Point x=new Point(10,6);
        Shape shape3= new Rectangle(y,1,1,"blue",true);
        Shape shape4= new Rectangle(y,2,1,"blue",true);
        Shape shape5= new Rectangle(y,1,1,"blue",true);
       // Shape shape=rectangle;

        System.out.println(shape3.hashCode());
        System.out.println(shape4.hashCode());
        System.out.println(shape5.hashCode());

    }
}

import java.awt.*;
import java.util.Objects;

public class Square extends Rectangle {
    public Square(){}
    public Square(double sid){
        super(sid,sid);
    }
    public Square(double sid,String col, boolean fil){
        super(sid,sid,col,fil);
    }
    public Square(Point topl,double sid,String col, boolean fil){
        super(topl,sid,sid,col,fil);
    }

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
        super.setLength(width);
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
        super.setWidth(length);
    }
    public double getSide(){
        return width;
    }
    public void setSide(double side){
        width=side;
        length=side;
    }
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Square square = (Square) o;
        return Double.compare(square.getSide(), getSide()) == 0 &&
                topLeft.equals(square.topLeft);
    }

    @Override
    public int hashCode() {
        return Objects.hash(topLeft,getSide());
    }

    @Override
    public String toString(){
        return "Square[topLeft="+topLeft+
                ",side="+width+
                ",color="+color+
                ",filled="+filled+
                "]";
    }

    public static void main(String[] args) {
        Point x=new Point(10,5);
        Shape shape7=new Square(1,"red",false);
        Shape shape8=new Square(x,1,"Blue",false);
        Shape shape9=new Square(2,"red",false);
        System.out.println(shape7.hashCode());
        System.out.println(shape8.hashCode());
        System.out.println(shape9.hashCode());
    }

}
