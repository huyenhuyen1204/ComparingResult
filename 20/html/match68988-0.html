import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes=new ArrayList<>();
    public void addShape(Shape newShape){
        shapes.add(newShape);
    }
    public void removeCircles(){
        for (int i=0;i<shapes.size();i++){
            if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i--;
            }
        }
    }
    public void removeDuplicates(){
        for (int i=0;i<shapes.size();i++){
            for (int j=i+1;j<shapes.size();j++) {
                if (shapes.get(i) instanceof Circle){
                    if (shapes.get(i).equals(shapes.get(j)) && ((Circle) shapes.get(i)).getCenter().equals(((Circle) shapes.get(j)).getCenter())){
                        shapes.remove(j);
                    }
                }
                else if (shapes.get(i) instanceof Rectangle){
                    if (shapes.get(i).equals(shapes.get(j)) && ((Rectangle) shapes.get(i)).getTopLeft().equals(((Rectangle) shapes.get(j)).getTopLeft())){
                        shapes.remove(j);
                    }
                }
                else {
                    if (shapes.get(i).equals(shapes.get(j)) && ((Square) shapes.get(i)).getTopLeft().equals(((Square) shapes.get(j)).getTopLeft())){
                        shapes.remove(j);
                    }
                }
            }
        }
    }
    public String getInfo(){
        String temp = "Layer of crazy shapes: ";
        for (Shape shape : shapes){
            temp = temp +"\n"+ shape.toString();
        }
        return temp;
    }
    }

Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){
    };
    public Rectangle(double width,double length){
        this.width=width;
        this.length=length;
    }
    public Rectangle(double width,double length,String color,boolean filled){
        super(color, filled);
        this.length=length;
        this.width=width;
    }
    public Rectangle(Point topLeft,double width,double length,String color,boolean filled){
        super(color, filled);
        this.length=length;
        this.width=width;
        this.topLeft=topLeft;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return 2*(width+length);
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft="+getTopLeft().toString()+",width="+this.getWidth()+",length="+this.getLength()+",color="+super.color+",filled="+super.filled+"]";
    }

    @Override
    public boolean equals(Object obj) {
        if (obj instanceof Rectangle == false) return false;
        else {
            Rectangle other=(Rectangle) obj;
            return (Math.abs(this.getWidth() - other.getWidth()) <= 0.001
                    && Math.abs(this.getLength() - other.getLength()) <= 0.001
                    && Math.abs(this.getTopLeft().getX() - other.getTopLeft().getX()) <= 0.001
                    && Math.abs(this.getTopLeft().getY() - other.getTopLeft().getY()) <= 0.001);
        }
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}

    }

    }

    public void setLength(double side) {

    }

    @Override

        return "Square[topLeft="+getTopLeft().toString()+",side="+this.getSide()+",color="+this.getColor()+",filled="+this.isFilled()+"]";
    }

    public boolean equals(Object obj) {

                    &&  Math.abs(getTopLeft().getX() - other.getTopLeft().getX()) <= 0.001
