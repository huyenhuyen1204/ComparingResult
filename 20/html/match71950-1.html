import java.util.*;
public class Layer {
    private List<Shape> shapes = new LinkedList<>() ;
    public Layer(){};
    public Layer(List<Shape> shapes){
        this.shapes = shapes;
    }
    public void addShape(Shape a){
        shapes.add(a);
    }
    public void removeCircles(){
        for(int i =0; i< shapes.size(); i++)
        {
            
if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }


        }
    }
    public String getInfo(){
        String info= "Layer of crazy shapes:\n";
        for(Shape i : shapes){
            info += i.toString() + "\n" ;

        }
        return info;
    }
    public void removeDuplicates(){
        for(int i= 0; i < shapes.size() ; i++){
            for(int j = i+1 ; j < shapes.size(); j++)
            {
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle && 
shapes.get(i).equals(shapes.get(j))
                        ||shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle  &&shapes.get(i).equals( shapes.get(j))
                        ||shapes.get(i) instanceof Square && shapes.get(j) instanceof Square && shapes.get(i).equals(shapes.get(j)) )
                {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    public static void main(String[] args){
        Layer 
l = new Layer() ;
        Point p = new Point(2,3);
        Shape a = new Circle(p,3.222221, "do", true);
        Shape c = new Circle(p,3.222222, "do", true);
        Shape b = new Square( p , 3.2, "sds" ,true);
        Shape d = new Rectangle(p, 2, 3.2, "sds" ,true);
        Shape e = new Square( p , 3.3, "sds" ,true);
        Shape f = new Square( p , 3.3, "sds" ,true);

        l.addShape(a);
        l.addShape(b);
        l.addShape(c);
        l.addShape(f);
        l.addShape(e);

        l.addShape(d);

        l.removeDuplicates();
        System.out.print(l.getInfo());
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;
    public Rectangle(){}
    public Rectangle(double width, double length){
        this.width= width;
        this.length= length;
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point p ,double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = p;
    }

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }


    public double getArea() {
        return getLength() * getWidth();
    }


    public double getPerimeter() {
        return 2.0 *( getWidth() + getLength() );
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }


    public boolean equals(Object obj) {

        return  (this.width == ((Rectangle) obj).width && this.length == ((Rectangle) obj).length && topLeft.equals(((Rectangle)obj).topLeft)) ;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    public String toString() {
        String 
res = "Rectangle[topLeft=" + topLeft.toString() +",width=" + String.valueOf(getWidth()) + ",length=" + String.valueOf(getLength()) + ",color=" + getColor() + ",filled=" + String.valueOf( isFilled()) + "]";
        return res;
    }

}
