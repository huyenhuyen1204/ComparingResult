src/Circle.java

public class Circle extends  Shape{
    protected double radius;
    protected  Point center;
    // constructer
    public Circle (){}
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius , String color , boolean filled){
        super(color , filled);
        this.radius = radius;
    }
    public Circle(Point point , double radius , String color , boolean filled){
        super(color, filled);
        this.center = point;
        this.radius = radius;
    }
    //method

    public void setCenter(Point center) {
        this.center = center;
    }
    public Point getCenter() {
        return center;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }
    public double getRadius() {
        return radius;
    }
    @Override
    public double getArea() {
        return this.radius * this.radius * Math.PI;
    }
    @Override
    public double getPerimeter() {
        return this.radius * 2.0 * Math.PI;
    }
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle a = (Circle) obj;
            return this.radius == a.radius && this.center.equals(a.center);
        }
        return false;
    }
    public 
int hashCode(){
        return this.hashCode();
    }
    @Override
    public String toString() {
        return "Circle[center="+this.center.toString()+",radius="+this.radius+",color="+this.getColor()+",filled="+this.isFilled() + "]";
    }
}

import java.util.LinkedList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new LinkedList<>();

     // method
    public void addShape(Shape shape){
        if(shape != null){
            shapes.add(shape);
        }
    }
    public void removeCircles (){
        for(int i = 0 ; i < shapes.size() ; i++){
            
if(shapes.get(i) instanceof Circle ){
                shapes.remove(i);
                i--;
            }

        }
    }
    public String getInfo() {
       String r = ("Layer of crazy shapes: ");
         r = r + '\n';
        for (int i = 0; i < shapes.size(); i++) {
            r= r + shapes.get(i).toString();
            r = r + '\n';
        
}
        return r;
    }

    public void removeDuplicates(){
        for(int i = 0 ; i < shapes.size() - 1  ; i++){
            for(int j = i + 1 ; j < shapes.size() ; j++){
                if(shapes.get(i).equals(shapes.get(j))){
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }

}

src/Point.java

public class Point {
    private double x;
    private  double y;

 // constructer
    public Point (double x , double y ){
        this.x = x;
        this.y = y;
    }
// method
    public void setX(double x) {
        this.x = x;
    }
    public double getX(){
        return this.x;
    }
    public void setY(double y ){
        this.y = y;
    }
    public double getY(){
        return this.y;
    }
    public double distance (Point a){
        double k = (this.x - a.x) * (this.x - a.x)  + (this.y - a.y) * (this.y - a.y);
        return Math.sqrt(k);
    
}
    public boolean equals(Object obj){
        if( obj instanceof Point ){
            Point a = (Point) obj;
            return this.x == a.x && this.y == a.y;
        }
        return false;
    }
    public int hashCode(){
            return this.hashCode();
    }
    public String toString(){
        return "(" + String.format("%.1f", this.x) + "," + String.format("%.1f", this.y) + ")";
    }
}

public class Square extends Rectangle{
    //constructer
    public Square(){}
    public Square(double side){
        this.width = side;
        this.length = side;
    }
    public Square(double side , String color , boolean filled){
        super(side , side , color , filled);
    }
    public Square(Point point , double side , String color , boolean filled){
        super(point , side , side , color , filled);
    }
    //method
    public void  setSide(double side){
        this.width = side;
        this.length= side;
    }
    public double getSide(){
        return this.length;
    
}
    @Override
    public void setWidth(double side) {
        this.width = side;
        this.length = side;
    }
    @Override
    public void setLength(double side) {
        this.length = side;
        this.width = side;
    }

    @Override
    public boolean equals(Object obj) {
      if(obj instanceof  Square){
         Square a = (Square) obj;
          return this.width == a.width && this.length == a.length && this.topLeft.equals(a.topLeft);
      }
      return false;
    }

    @Override
    public String toString() {
        return "Square[topLeft=" + this.topLeft + ",side=" + this.width + ",color=" + this.color +",filled=" + this.filled + "]";
    }
}
