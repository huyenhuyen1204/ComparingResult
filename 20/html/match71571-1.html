import java.util.concurrent.Callable;

public class Circle extends Shape {
    protected final double pi=Math.PI;
    protected Point center;
    protected double radius;
    public Circle(){super();
}
    public Circle(double radius){
        this.radius=radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color,filled);
        this.radius=radius;
    }
    public Circle(Point a, double radius, String color, boolean filled){
        super(color,filled);
        this.radius=radius;
        this.center=a;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    public double getArea(){
        return pi*this.radius*this.radius;
    }

    public double getPerimeter() {
        return pi*2*this.radius;
    }
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle other = (Circle) obj;
            return other.getRadius()==this.getRadius() && other.getCenter()==this.getCenter();
        }
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    public String toString() {
        return "Circle[center=" + this.center.toString() + "," +
                "radius=" + String.valueOf(this.radius) + ",color=" + getColor() + ",filled=" + String.valueOf(filled)+ "]";
    }
}

import java.util.*;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape a){
        shapes.add(a);
    }
    public void removeCircles(){
        for(int i= 0; i<shapes.size(); i++){
            
if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }
        }
    }
    public String getInfo(){
        String s="Layer of crazy shapes: ";
        s+= '\n';
        for(int i=0; i<shapes.size(); i++){
           s += shapes.get(i).toString();
            s+= '\n';
        }
        return s;
    }
    public void removeDuplicates(){
        for(int i=0; i<shapes.size()-1; i++){
            for(int j=i+1; j<shapes.size(); j++){
                if(shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    public static void main(String [] args){
        Point 
point = new Point(1.0, 2.0);
        Circle circle= new Circle(point, 2.0, "RED", true);
        Layer layer=new Layer();
        layer.addShape(circle);
        String a= layer.getInfo();
        System.out.print(a);
    }
}

public class Square extends Rectangle{

    public Square(){
        super();
    }
    public Square(double side){
        setWidth(side);
        setLength(side);
    }
    public Square(double side, String color, boolean filled){
        setWidth(side);
        setLength(side);
        this.color=color;
        this.filled=filled;
    }
    public Square(Point a,double side, String color, boolean filled){
        super(a, side, side, color, filled);
    }
    public double getSide() {
        return super.getLength();
    }
    public void setSide(double side) {
        setWidth(side);
        setLength(side);
    
}
    public void setWidth(double side){
        this.width=side;
        this.length=side;

    }
    public void setLength(double side){
        this.length=side;
        this.width=side;
    }
    public boolean equals(Object obj){
        return super.equals(obj);
    }
    public String toString(){
        return "Square[topLeft=(" + 
String.valueOf(this.topLeft.getX()) + "," + String.valueOf(this.topLeft.getY())+ ")," +
                "side=" + String.valueOf(this.getSide()) +  ",color=" + getColor() + ",filled=" + String.valueOf(filled) +"]";
    }
}
