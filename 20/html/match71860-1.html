import java.util.Objects;

public class Circle extends Shape{
    private static final double PI = Math.PI;
    protected double radius;
    protected Point center;

    public Circle() {
        super();
    };

    public Circle(double radius) {
        super();
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled) {
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled) {
        super(color, filled);
        this.center = center; this.radius = radius;
    }

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getArea() {
        return PI * radius * radius;
    }

    public double getPerimeter() {
        return PI * 2 * radius;
    }

    public String toString() {
        return new String("Circle[center=" + center.toString() + ",radius=" + radius + ",color=" + color + ",filled=" + filled + "]");
    }

    @Override
    public boolean equals(Object o) {
        if (o instanceof Circle) {
            Circle otherCircle = (Circle)o;
            
if (this.center.equals(otherCircle.center) && Math.abs(this.radius - otherCircle.radius) <= EP) return true;
            else return false;
        }
        else return false;
    }


}

import java.util.ArrayList;

public class Layer {
    private List<Shape> shapes;

    public Layer() {
        shapes = new ArrayList<Shape>();
    }
    public void addShape(Shape newShape) {
        shapes.add(newShape);
    }
    public void removeCircles() {
        shapes.removeIf(s -> s instanceof Circle);
    }


    public String getInfo() {
        StringBuffer info = new StringBuffer();
        info.append("Layer of crazy shapes: \n");
        for (Shape s: shapes) {
            info.append(s.toString() + "\n");
        }
        return new String(info);
    
}

    /*public void removeDuplicates() {

    public void removeDuplicates() {
        for (int i = 0; i < shapes.size() - 1; i++)
            for (int j = i + 1; j < shapes.size(); j++)
                if (shapes.get(i).equals(shapes.get(j))) {
                    shapes.remove(j); j--;
                }
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return this.y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point p) {
        return Math.sqrt((this.x - p.x) * (this.x - p.x) + (this.y - p.y) * (this.y - p.y));
    }

    @Override
    public boolean equals(Object p) {
        if (p instanceof Point) {
            Point newPoint  = (Point)p;
            
if (newPoint.x == this.x && newPoint.y == this.y) return true;
            else return false;
        }
        else return false;
    }

    public String toString() {
        return new String("(" + x + "," + y + ")");
    }
}

import java.util.Objects;

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {
        super();
    }

    public Rectangle(double width, double length) {
        super();
        this.width = width; this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width; this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft; this.width = width; this.length = length;
    }

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getArea() {
        return width * length;
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public double getPerimeter() {
        return (width + length) * 2;
    }

    public String toString() {
        return new String("Rectangle[topLeft=" + topLeft.toString() + ",width=" + width + ",length=" + length + ",color=" + color + ",filled=" + filled + "]");
    }

    @Override
    public boolean equals(Object o) {
        if (o instanceof Rectangle) {
            Rectangle otherRectangle = (Rectangle)o;
            if (this.topLeft.equals(otherRectangle.topLeft) &&
                Math.abs(this.width - otherRectangle.width) <= EP &&
                
Math.abs(this.length - otherRectangle.length) <= EP) return true;
            else return false;
        }
        else return false;
    }



}
