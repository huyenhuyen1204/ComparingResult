class Circle extends Shape{
    protected double radius;
    protected Point center;

    public Circle(){}

    public Circle(double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }

    public Circle(Point center, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = center;
    }

    public Point getCenter(){
        return this.center;
    }

    public void setCenter(Point center){
        this.center = center;
    }

    public double getRadius(){
        return this.radius;
    }

    public void setRadius(double radius){
        this.radius = radius;
    }

    public double getArea(){
        return (double) Math.PI * this.getRadius() * this.getRadius();
    }

    public double getPerimeter(){
        return (double) Math.PI * 2 * this.getRadius();
    }

    public boolean equals(Object other){
        if (other instanceof Circle){
            if (
this.getCenter().equals(((Circle) other).getCenter()) &&
                Math.abs(this.getRadius() - ((Circle) other).getRadius()) <= 0.001)
                    return true;
        }
        return false;
    }

    public String toString(){
        return "Circle[center=" + this.getCenter().toString() + ",radius=" + this.getRadius() + ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
}

public class Layer {

    }

    public void removeCircles(){
        for (int i = 0; i < shapes.size(); i++)
            
if (shapes.get(i) instanceof Circle)
                shapes.remove(shapes.get(i));
    }

    public String getInfo(){
        String res = "Layer of crazy shapes: \n";
        for (int i = 0; i < shapes.size() - 1; i++)
            res = res + shapes.get(i).toString() + "\n";
        res = res + shapes.get(shapes.size() - 1).toString();
        
return res;
    }

    public void removeDuplicates(){
        List<Shape> shapes2 = new ArrayList<Shape>();
        for (int i = 0; i < shapes.size(); i++) {
            boolean ok = false;
            for (int j = 0; j < shapes2.size(); j++)
                if (shapes.get(i).equals(shapes2.get(j))) {
                    ok = true;
                    break;
                }
            if (ok == false) shapes2.add(shapes.get(i));
        }
        shapes.clear();
        for (int i = 0; i < shapes2.size(); i++)
            shapes.add(shapes2.get(i));
    }
}

src/Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX(){
        return this.x;
    }

    public void setX(double x){
        this.x = x;
    }

    public double getY(){
        return this.y;
    }

    public void setY(double y){
        this.y = y;
    }

    public double distance(Point other){
        return Math.sqrt((this.getX() - other.getX())*(
this.getX() - other.getX()) +
                         (this.getY() - other.getY())*(this.getY() - other.getY()));
    }

    public boolean equals(Object other){
        if (other instanceof Point){
            if (
Math.abs(this.getX() - ((Point) other).getX()) <= 0.001 &&
                Math.abs(this.getY() - ((Point) other).getY()) <= 0.001)
                    return true;
        }
        return false;
    }

    public String toString(){
        return "(" + this.getX() + "," + this.getY() + ")";
    }
}

public class Rectangle extends Shape{
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle(){}

    public Rectangle(double width, double length){
        this.width = width;
        this.length = length;
    }

    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }

    public double getWidth(){
        return this.width;
    }

    public void setWidth(double width){
        this.width = width;
    }

    public double getLength(){
        return this.length;
    }

    public void setLength(double length){
        this.length = length;
    }

    public double getArea(){
        return this.getLength() * this.getWidth();
    }

    public double getPerimeter(){
        return (this.getWidth() + this.getLength() ) * 2;
    }

    public Point getTopLeft(){
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft){
        this.topLeft = topLeft;
    }

    public boolean equals(Object other){
        if (other instanceof Rectangle){
            if (
this.getTopLeft().equals(((Rectangle) other).getTopLeft()) &&
                Math.abs(this.getWidth() - ((Rectangle) other).getWidth()) <= 0.001 &&
                Math.abs(this.getLength() - ((Rectangle) other).getLength()) <= 0.001)
                    return true;
        }
        return false;
    }

    public String toString(){
        return "Rectangle[topLeft=" + this.getTopLeft().toString() +
                ",width=" + this.getWidth() + ",length=" + this.getLength() +
                ",color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
    }
}
