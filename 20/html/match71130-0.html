import javax.swing.*;
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Layer {
    private List<Shape> shapes=new ArrayList<>();
    public void addShape(Shape shape1){
    shapes.add(shape1);
    }
    public void removeCircles(){
     for(int i=shapes.size()-1;i>=0;i--){
         if(shapes.get(i) instanceof Circle){
             shapes.remove(i);
         }
     }
    }
    public String getInfo(){
        String a="";
        a+="Layer of crazy shapes: "+"\n";
        for (int i = 0; i < shapes.size(); i++)
        {
            a=a+shapes.get(i).toString()+"\n";

        }
    return a;
    }
    public void removeDuplicates(){
        for(int i=0;i<shapes.size()-1;i++){
            for(int j=i+1;j<shapes.size();j++){
                if(shapes.get(i) instanceof Circle&&shapes.get(j) instanceof Circle){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Rectangle&&shapes.get(j) instanceof Rectangle){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
                if(shapes.get(i) instanceof Square&&shapes.get(j) instanceof Square){
                    if(shapes.get(i).equals(shapes.get(j))){
                        shapes.remove(j);
                    }
                }
            }

        }
    }

}

import javax.annotation.processing.SupportedAnnotationTypes;
import java.awt.*;
import java.util.Objects;

public class Square extends Rectangle {

    public Square(){
    }
    public Square(double side){
        this.length=this.width=side;
    }
    public Square(double side, String color, boolean filled){
        this.color=color;
        this.filled=filled;
        this.length=this.width=side;
    }
    public Square(Point topLeft,double side,String color,boolean filled){
        this.topLeft=topLeft;
        this.color=color;
        this.filled=filled;
        this.length=this.width=side;

    }
    public double getSide(){
        return this.length;
    }
    public void setSide(double side){
        this.length=this.width=side;
    }
    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Square square = (Square) o;
        return square.width==square.length&&square.topLeft==this.topLeft
                &&square.color==this.color&&square.filled==this.filled;}

    @Override
    public void setWidth(double side) {
        this.length=side;
        this.width=side;
    }

    @Override
    public void setLength(double side) {
        this.length=side;
        this.width=side;
    }
    public String toString(){
         return "Square[topLeft="+topLeft.toString()+",side="+Shape.Round(getSide())+",color="+getColor()+",filled="+isFilled()+"]";
    }
