Circle.java

import java.lang.Math;
public class Circle extends Shape {

	protected double radius;
	protected Point center;
	public Circle(){
		super();
	}

	public Circle(double radius) {
		super();
		this.radius = radius;
	}

	public Circle(double radius, String color,boolean filled) {
		super(color,filled);
		this.radius = radius;
	}

	public Circle(Point center,double radius) {
		super();
		this.radius = radius;	
		this.center = center;
	}

	public Circle(Point center,double radius, String color,boolean filled) {
		super(color,filled);
		this.radius = radius;
		this.center = center;
	}

	public double getRadius() {
		return radius;
	}

	public void setRadius(double radius) {
		this.radius = radius;
	} 

	public double getArea() {
		return radius * radius * Math.PI;
	}

	public 
double getPerimeter() {
		return radius * 2 * Math.PI;
	}

	public Point getCenter() { return center; }
	public void setCenter(Point newCenter) { this.center = newCenter; }

	public String toString() {
		return new String("Circle[center="+center+",radius="+radius+",color="+color+",filled="+filled+"]");
	
}

	public boolean equals(Object that) {
		if (that instanceof Circle) {
			Circle t = (Circle)that;
			if(this.center.equals(t.center) && this.radius == t.radius) return true;
		}
		return false;
	}
}

import java.util.*;

public class Layer {
	private List<Shape> shapes = new ArrayList<>();
	public void addShape(Shape newShape) {
		shapes.add(newShape);
	}

	public void removeCircles() {
		for(int i=shapes.size()-1;i>=0;i--) 
			if(shapes.get(i) instanceof Circle) shapes.remove(i);
	}

	public String getInfo() {
		String r = "Layer of crazy shapes: \n";
		for(int i=0;i<shapes.size();i++) r+=
shapes.get(i).toString()+"\n";
			return r;
	}

	public void removeDuplicates() {
		for(int i=0;i<shapes.size();i++) for(int j=shapes.size()-1;j>i;j--) if(shapes.get(i).equals(shapes.get(j))) shapes.remove(j);
	} 	
}
