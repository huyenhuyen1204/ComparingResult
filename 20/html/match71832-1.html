public class Circle extends Shape {
    public final double PI = 3.14159;
    protected Point center = new Point(0.0,0.0);
    protected double 
radius;

    public Circle(){}

    public Circle(double radius){
        this.radius=radius;
    }

    public Circle(double radius,String color,boolean filled){
        super(color, filled);
        this.radius =radius;
    }
    public Circle(Point center,double radius,String color,boolean filled ){
        super(color, filled);
        this.center = center;
        this.radius =radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    
@Override
    public double getArea() {
        return PI*radius*radius ;
    }

    @Override
    public double getPerimeter() {
        return 2*radius*PI;
    }

    @Override
    public String toString() {
        return "Circle[center="+center.toString()+",radius="+ Math.round(radius*10.0)/10.0+",color="+super.getColor()+",filled="+super.isFilled()+"]";
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj)
            return true;
        
if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        Circle other = (Circle) obj;
        if ( !center.equals(other.getCenter()))
            return false;
        else if (radius!=other.getRadius()) {
            return  false;
        }
        return true;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape){
        shapes.add(shape);
    }
    private boolean isCircle(Shape sp){

        if(sp.getClass() == Circle.class) return true;
        return  false;
    }
    public void removeCircles(){
        for (int i = 0; i < shapes.size() ; i++) {
            
if (isCircle(shapes.get(i))) {
                shapes.remove(i);
                i--;
            }
        }

    }
    public String getInfo(){
        int s = shapes.size();
        String q = "Layer of crazy shapes: \n";
        for (int i = 0; i < s; i++) {
            q = q + shapes.get(i).toString()+"\n";
        }
        return q;
    }
    public void removeDuplicates(){
        int size=shapes.size();
        for (int i = 0; i < size-1 ; i++) {
            for (int j = size-1; j > i ; j--) {
                if (shapes.get(i).equals(shapes.get(j))) shapes.remove(j);
            }
            size=shapes.size();
        }
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft = new Point(0.0,0.0);
    public Rectangle(){}
    public Rectangle(double width,double length ){
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width,double length,String color,boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft,double width,double length,String color,boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public double getWidth() {
        return width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    @Override
    public double getArea() {
        return width*length;
    }

    @Override
    public double getPerimeter() {
        return (width+length)*2;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft="+topLeft.toString()+",width="+width+",length="+length+",color="+super.getColor()+",filled="+ super.isFilled()+"]";
    }

    @Override
    public boolean equals(Object obj) {
        if (this == obj)
            return true;
        
if (obj == null)
            return false;
        if (getClass() != obj.getClass())
            return false;
        Rectangle other = (Rectangle) obj;
        if ( !topLeft.equals(other.getTopLeft()))
            return false;
        else {
            if (width!=other.getWidth()) return  false;
            else{
                if (length!=other.getLength()) return  false;
            }

        }
        return true;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}
