src/Circle.java

public class Circle extends Shape {
	protected double radius = 1.0;
	protected Point center;

	public Circle() {
	}

	public Circle(double r) {
		radius = r;
	}

	public Circle(double r, String c, boolean f) {
		radius = r;
		color = c;
		filled = f;
	}

	public Circle(Point cen, double r, String c, boolean f) {
		center = cen;
		radius = r;
		color = c;
		filled = f;
	}

	public double getRadius() {
		return radius;
	}

	public void setRadius(double r) {
		radius = r;
	}

	public Point getCenter() {
		return center;
	}

	public void setCenter(Point c) {
		center = c;
	}

	@Override
	public double getArea() {
		return PI * radius * radius;
	}

	@Override
	public double getPerimeter() {
		return PI * 2 * radius;
	}

	public boolean equals(Object o) {
		if (o instanceof Circle) {
			Circle c = (Circle)o;
			return center.equals(c.getCenter()) && Math.abs(radius - c.getRadius()) <= 0.001;
		}
		return false;
	}

	public String toString() {
		return String.format("Circle[center=%s,radius=%.1f,color=%s,filled=%s]", 
			center.toString(), radius, color, String.valueOf(filled));
	}
}

import java.util.List;
import java.util.ArrayList;

public class Layer {
	private List<Shape> shapes = new ArrayList<Shape>();

	public void addShape(Shape s) {
		shapes.add(s);
	} 

	public void removeCircles() {
		ArrayList<Shape> not_circle_shapes = new ArrayList<Shape>();

		for (Shape shape: shapes) {
			if (!(shape instanceof Circle)) {
				not_circle_shapes.add(shape);
			}
		}

		shapes = not_circle_shapes;
	}

	public void removeDuplicates() {
		ArrayList<Shape> unique_shapes = new ArrayList<Shape>();

		for (Shape shape: shapes) {
			boolean is_in_unique = false;
			for (Shape unique_shape: unique_shapes)
				if (shape.equals(unique_shape)) {
					is_in_unique = true;
					break;
				}

			if (!is_in_unique)
				unique_shapes.add(shape);
		}

		shapes = unique_shapes;
	}

	public String getInfo() {
		String res = "Layer of crazy shapes: \n";
		for (Shape shape: shapes) {
			res += shape.toString() + "\n";
		}
		return res;
	}
}

src/Point.java

import java.lang.Math;

public class Point {
	private double x;
	private double y;

	public Point(double x, double y) {
		this.x = x;
		this.y = y;
	}

	public double getX() {
		return x;
	}

	public void setX(double x) {
		this.x = x;
	}

	public double getY() {
		return y;
	}

	public void setY(double y) {
		this.y = y;
	}

	public double distance(Point p) {
		return Math.sqrt(Math.pow(p.getX() - this.x, 2) + Math.pow(p.getY() - this.y, 2));
	} 

	public boolean equals(Object o) {
		if (o instanceof Point) {
			Point p = (Point)o;
			return Math.abs(this.x - p.getX()) <= 0.001 && Math.abs(this.y - p.getY()) <= 0.001;
		}
		return false;
	}

	public String toString() {
		return String.format("(%.1f,%.1f)", this.x, this.y);
	}
}

public class Rectangle extends Shape {
	protected double width;
	protected double length;
	protected Point topLeft;

	public Rectangle() {
	}

	public Rectangle(double w, double l) {
		width = w;
		length = l;
	}

	public Rectangle(double w, double l, String c, boolean f) {
		width = w;
		length = l;
		color = c;
		filled = f;
	}

	public Rectangle(Point top, double w, double l, String c, boolean f) {
		topLeft = top;
		width = w;
		length = l;
		color = c;
		filled = f;
	}

	public double getWidth() {
		return width;
	}

	public void setWidth(double w) {
		width = w;
	}

	public double getLength() {
		return length;
	}

	public void setLength(double l) {
		length = l;
	}

	public Point getTopLeft() {
		return topLeft;
	}

	public void setTopLeft(Point top) {
		topLeft = top;
	}

	@Override
	public double getArea() {
		return width * length;
	}

	@Override
	public double getPerimeter() {
		return 2 * (width + length);
	}

	public boolean equals(Object o) {
		if (o instanceof Rectangle) {
			Rectangle r = (Rectangle)o;
			return topLeft.equals(r.getTopLeft()) && Math.abs(width - r.getWidth()) <= 0.001 && Math.abs(length - r.getLength()) <= 0.001;
		}
		return false;
	}

	public String toString() {
		return String.format("Rectangle[topLeft=%s,width=%.1f,length=%.1f,color=%s,filled=%s]",
			topLeft.toString(), width, length, color, String.valueOf(filled));
	}
}
