Circle.java

import java.util.*;
public class Circle extends Shape {

    protected double radius;
    protected Point center;

    public Point getCenter() {
        return this.center;
    }

    public void setCenter(Point center) {
        this.center = center;
    };


    @Override
    public boolean equals(Object o) {
        if (o == this)
            return true;
        if (!(o instanceof Circle)) {
            return false;
        }
        Circle circle = (Circle) o;
        return Math.abs(center.getX() - circle.getCenter().getX()) <= 0.001 &&
                Math.abs(center.getY() - circle.getCenter().getY()) <= 0.001 &&
                Math.abs(radius - circle.getRadius()) <= 0.001;
    }

    public Circle() {

    }

    public Circle(double radius) {
        this.radius = radius;
    }
    
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }  
    public Circle(Point point, double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = point;
    }  
    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    @Override
    public double getArea(){
        return Math.PI*radius*radius;
    }

    @Override
    public double getPerimeter(){
        return Math.PI*2*radius;
    }

    public static double round(double a){
        return (double)Math.round(a*10) / 10;
    }

    @Override
    public String toString(){
        return "Circle[center=" + this.center.toString() + ",radius=" + round(this.radius) + ",color=" + this.color + ",filled=" + this.filled + "]";
    }
}

Layer.java


import java.util.*;

public class Layer {

    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape shape){
        this.shapes.add(shape);
    }

    public void removeCircles() {
        shapes.removeIf(shape -> shape instanceof Circle);
    }

    public void removeDuplicates(){
        for (int x = 0; x < shapes.size()-1; ++x){
            for (int y = x + 1; y < shapes.size(); ++y)
                if (shapes.get(x).equals(shapes.get(y))) shapes.remove(y);
        }
    }

    public String getInfo(){
        String init = "Layer of crazy shapes: \n";
        String main = "";
        for (Shape shape : shapes){
            main += shape.toString();
            main += "\n";
        }
        return init + main;
    }
}

Point.java


public class Point {

    private double x;
    private double y;

    public Point() {
    }

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return this.y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public Point x(double x) {
        this.x = x;
        return this;
    }

    public Point y(double y) {
        this.y = y;
        return this;
    }

    @Override
    public boolean equals(Object o) {
        if (o == this)
            return true;
        if (!(o instanceof Point)) {
            return false;
        }
        Point point = (Point) o;
        return x == point.x && y == point.y;
    }


    @Override
    public String toString() {
        return "(" + this.getX() + "," + this.getY() + ")";
    }


    public double distance(Point other){
        return Math.sqrt(Math.pow(this.x - other.x, 2) + Math.pow(this.y - other.y, 2));
    }


}

Rectangle.java


public class Rectangle extends Shape{

    protected double width;
    protected double length;
    protected Point topLeft;
    

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return this.length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return this.topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    public Rectangle width(double width) {
        this.width = width;
        return this;
    }

    public Rectangle length(double length) {
        this.length = length;
        return this;
    }

    public Rectangle topLeft(Point topLeft) {
        this.topLeft = topLeft;
        return this;
    }

    @Override
    public boolean equals(Object o) {
        if (o == this)
            return true;
        if (!(o instanceof Rectangle)) {
            return false;
        }
        Rectangle rectangle = (Rectangle) o;
        return Math.abs(topLeft.getX() - rectangle.topLeft.getX()) <= 0.001 &&
                Math.abs(topLeft.getY() - rectangle.topLeft.getY()) <= 0.001 &&
                Math.abs(getLength() - rectangle.getLength()) <= 0.001 &&
                Math.abs(getWidth() - rectangle.getWidth()) <= 0.001;
    }


    public Rectangle() {
    }

    public Rectangle(double width, double length){
        
    }
    public Rectangle(double width, double length, String color, boolean filled){
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled){
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length= length;
    }

    @Override
    public double getArea(){
        return width*length;
    };

    @Override
    public double getPerimeter(){
        return 2*(width + length);
    };

    public static double round(double a){
        return (double) Math.round(a*10) / 10;
    }

    @Override
    public String toString(){
        return "Rectangle[topLeft=" + this.topLeft.toString() + ",width=" + round(this.width) + ",length=" + round(this.length) + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

}

public class Square extends Rectangle {

    public Square(){

    }

    public Square(double side){
        super(side, side);
    }
    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }
    public Square(Point point, double side, String color, boolean filled){
        super(point, side, side, color, filled);
        
    }
    public double getSide(){
        return this.width;
    }
    public void setSide(double side){
        this.length = side;
        this.width = side;
    }

    @Override
    public double getLength(){
        return this.length;
    }
    @Override
    public void setLength(double length) {
    	this.length = length;
    }
    @Override
    public double getWidth() {
    	return this.width;
    }
    @Override
    public void setWidth(double width) {
    	this.width = width;
    }

    public boolean equals(Object o) {
        if (!(o instanceof Square)) return false;
        if (this == o) return true;
        Square _o = (Square) o;
        return  Math.abs(topLeft.getX() - _o.topLeft.getX()) <= 0.001 &&
                Math.abs(topLeft.getY() - _o.topLeft.getY()) <= 0.001 &&
                Math.abs(getSide() - _o.getSide()) <= 0.001;
    }

    @Override
    public String toString() {
        return "Square" + '[' +
                "topLeft=" + topLeft.toString() +
                ",side=" + String.format("%.1f", length) +
                ",color=" + color +
                ",filled=" + filled +
                ']';
    }
