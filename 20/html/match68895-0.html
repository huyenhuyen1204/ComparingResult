Circle.java


public class Circle extends Shape {
    protected Point center = new Point(0,0);
    protected double radius;
    public Circle(){ super();}
    public Circle(double radius){this.radius = radius;}
    public Circle(double radius, String color, boolean filled){
        super(color,filled);
        this.radius = radius;
    }
    public Circle(Point point, double radius, String color, boolean filled){
        super(color,filled);
        this.radius = radius;
        this.center = point;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center.setX(center.getX());
        this.center.setY(center.getY());
    }

    public double getRadius() {
        return this.radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }
    @Override
    public boolean equals(Object object){
        if(object instanceof Circle){
            Circle temp = (Circle) object;
            if(Math.abs(this.getCenter().getX() - temp.getCenter().getX()) <= 0.001  && Math.abs(this.getCenter().getY() - temp.getCenter().getY()) <= 0.001 && Math.abs(this.getRadius() - temp.getRadius()) <= 0.001) return true;
        }
        return false;
    }
    @Override
    public int hashCode(){
        return this.hashCode();
    }

    @Override
    public double getArea(){
        return Math.PI*radius*radius;
    }

    @Override
    public double getPerimeter() {
        return 2*Math.PI*radius;
    }

    @Override
    public String toString() {
        String temp = "Circle[center=(" + this.getCenter().getX()+","+ this.getCenter().getY()+"),radius=" + this.getRadius() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
        return temp;
    }
}

import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public Layer() {
    }

    public void addShape(Shape shape) {
        shapes.add(shape);
    }

    public void removeCircles() {
        int sizeOfList = shapes.size();
        for (int i = 0; i < sizeOfList ; i++) {
            if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
            }
        }
    }

    public void removeDuplicates() {
        int sizeOfList = shapes.size();
        for (int i = 0; i < sizeOfList - 1; i++) {
            for (int j = i + 1; j < sizeOfList; j++) {
                if (shapes.get(j).equals(shapes.get(i))) shapes.remove(j);
            }
        }
    }

    public String getInfo() {
        String info = "Layer of crazy shape:\n";
        int sizeOfList = shapes.size();
        for (int i = 0; i < sizeOfList; i--) {
            info += shapes.get(i).toString() + "\n";
        }
        return info;
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point other) {
        double temp = Math.pow((this.x - other.x), 2) + Math.pow((this.y - other.y), 2);
        temp = Math.sqrt(temp);
        return temp;
    }

    public boolean equals(Object object) {
        if(object instanceof Point){
            Point temp = (Point) object;
            if(temp.getY() == this.getY() && temp.getX() == this.getX()) return  true;
        }
        return false;
    }

    public int hashCode() {
        return this.hashCode();
    }

    public String toString() {
        String temp ="(" + this.getX() +"," + this.getY() + ")";
        return  temp;
    }
}

Rectangle.java

public class Rectangle extends Shape {
    protected Point topLeft = new Point(0,0);
    protected double length;
    protected double width;

    public Rectangle() {
        super();
    }

    public Rectangle(double width, double length) {
        this.length = length;
        this.width = width;
    }

    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }

    public Rectangle(Point point, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = point;
        this.width = width;
        this.length = length;
    }

    public double getWidth() {
        return this.width;
    }

    public void setWidth(double width) {
        this.width = width;
    }

    public double getLength() {
        return length;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return this.width * this.length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (this.length + this.width);
    }
    @Override
    public boolean equals(Object object){
        if( object instanceof Rectangle){
            Rectangle temp = (Rectangle) object;
            if(Math.abs(this.getLength() - temp.getLength()) <= 0.001 && Math.abs(this.getWidth() - temp.getWidth()) <= 0.001){
                return true;
            }
        }
        return(false);
    }

    @Override
    public int hashCode() {
        return this.hashCode();
    }

    @Override
    public String toString() {
        String temp = "Rectangle[topLeft=(" +this.getTopLeft().getX() + "," + this.getTopLeft().getY()+ "),width=" + this.getWidth() + ",length=" + this.getLength() + ",";
        temp += "color=" + super.getColor() + ",filled=" + super.isFilled() + "]";
        return temp;
    }
}
