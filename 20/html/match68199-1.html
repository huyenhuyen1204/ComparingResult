import java.util.*;
public class Circle extends Shape {
    protected double radius;
    protected Point center;

    public Circle(){}
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point p,double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
        this.center = p;
    }
    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        center = center;
    }

    public double getArea() {
        double S = Math.PI * getRadius() * getRadius();
        return S;
    }


    public double getPerimeter() {
        double perimeter = Math.PI * 2.0 * getRadius();
        return perimeter;
    }


    public boolean equals(Object obj) {
       // Circle a = (Circle) obj;
        return (this.getRadius() == 
((Circle) obj).getRadius() && this.getArea() == ((Circle) obj).getArea());
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    public String toString() {
        String str = "";
        str = "Circle[center=" + center.toString() + ",radius=" + String.valueOf(getRadius()) + ",color=" + getColor() + ",filled=" + isFilled() +"]";
        return str;
    }

}

import java.util.*;
public class Layer {
    private List<Shape> shapes = new LinkedList<>() ;
    public Layer(){};
    public Layer(List<Shape> shapes){
        this.shapes = shapes;
    }
    public void addShape(Shape a){
        shapes.add(a);
    }
    public void removeCircles(){
        for(int i =0; i< shapes.size(); i++)
        {
            
if(shapes.get(i) instanceof Circle) {
                shapes.remove(i);
                i--;
            }


        }
    }
    public String getInfo(){
        String info= "Layer of crazy shapes:\n";
        for(Shape i : shapes){
            info += 
i.toString() + "\n" ;

        }
        return info;
    }
    public void removeDuplicates(){
        for(int i= 0; i < shapes.size() ; i++){
            for(int j = i+1 ; j < shapes.size(); j++)
            {
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle && shapes.get(i).equals(shapes.get(j))
                        ||shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle  &&shapes.get(i).equals( shapes.get(j))
                        ||shapes.get(i) instanceof Square && shapes.get(j) instanceof Square && shapes.get(i).equals(shapes.get(j)) )
                {
                    shapes.remove(j);
                    j--;
                }
            }
        }
    }
    public static void main(String[] args){
        Layer l = new Layer() ;
        Point p = new Point(2,3);
        Shape a = new Circle(p,3.222221, "do", true);
        Shape c = new Circle(p,3.222222, "do", true);
        Shape b = new Square( p , 3.2, "sds" ,true);
        Shape d = new Rectangle(p, 2, 3.2, "sds" ,true);
        Shape e = new Square( p , 3.3, "sds" ,true);
        Shape f = new Square( p , 3.3, "sds" ,true);

        
l.addShape(a);
        l.addShape(b);
        l.addShape(c);
        l.addShape(f);
        l.addShape(e);

        l.addShape(d);

        l.removeDuplicates();
        System.out.print(l.getInfo());
    }
}

Point.java

public class Point {
    private double x;
    private double y;
    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point a){
        double 
d = Math.sqrt( (a.x -x )*(a.x -x) + (a.y -y)*(a.y - y) );
        return d;
    }

    @Override
    public boolean equals(Object obj) {
        Point a = (Point) obj;
        if(this.x == a.x && this.y == a.y) return true;
        else  return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }


    public String toString() {
        String res = "(" + String.valueOf(x) + "," + String.valueOf(y) + ")";
        return res;
    }
}

public class Square extends Rectangle{

    public Square(){}
    public Square(double size){
        this.length = size;
        this.width = size;
    }
    public Square(double size, String color, boolean filled){
        this.color = color;
        this.filled = filled;
        this.length = size;
        this.width = size;
    }
    public Square(Point p ,double size, String color, boolean filled){
        this.color = color;
        this.filled = filled;
        this.length = size;
        this.width = size;
        this.topLeft = p;
    }
    public double getSide() {
        return this.width;
    }

    public void setSide(double size) {

        this.length = size;
        this.width = size;
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
    }

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
    }

    public boolean equals(Object obj) {
        return super.equals(obj);
    
}

    @Override
    public String toString() {
        String res= "Square[topLeft=" + topLeft.toString() +",side=" + String.valueOf(getSide()) + ",color=" + getColor() + ",filled=" + String.valueOf(isFilled()) + "]";
        return res;
    }

}
