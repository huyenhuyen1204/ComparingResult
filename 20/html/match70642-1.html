import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<>();
    
    public void addShape(Shape shape){
        shapes.add(shape);
    }
    
    public void removeCircles(){
        for (int i = 0; i < this.shapes.size(); i++) {
            if(this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);
            }
        }
    }
    
    public String getInfo(){
        String infro = "Layer of crazy shapes:\n";
        for (Shape shape : shapes) {
            if(shape instanceof Circle){
                Circle sp = (Circle) shape;
                infro += sp.toString() + "\n";
            }else if(shape instanceof Rectangle){
                Rectangle sp = (Rectangle) shape;
                infro += sp.toString() + "\n";
            }else if(shape instanceof Square){
                Square sp = (Square) shape;
                infro += sp.toString() + "\n";
            }
        }
        return infro;
    
}
    
    public void removeDuplicates(){
        for (int i = 0; i < this.shapes.size() - 1; i++) {
            for (int j = i; j < this.shapes.size(); j++) {
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle){
                    Circle sp1 = (Circle) shapes.get(i);
                    Circle sp2 = (Circle) shapes.get(j);
                    if(sp1.getRadius() == sp2.getRadius() && sp1.getCenter().equals(sp2)){
                        shapes.remove(j);
                    }
                }else 
if(shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle){
                    Rectangle sp1 = (Rectangle) shapes.get(i);
                    Rectangle sp2 = (Rectangle) shapes.get(j);
                    if(sp1.getWidth()== sp2.getWidth() && sp1.getLength()== sp2.getLength() && sp1.getTopLeft().equals(sp2)){
                        shapes.remove(j);
                    }
                }else if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Square){
                    Square sp1 = (Square) shapes.get(i);
                    Square sp2 = (Square) shapes.get(j);
                    if(sp1.getWidth()== sp2.getWidth() && sp1.getLength()== sp2.getLength() && sp1.getTopLeft().equals(sp2)){
                        shapes.remove(j);
                    }
                }else if(shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Square){
                    Rectangle sp1 = (Rectangle) shapes.get(i);
                    Square sp2 = (Square) shapes.get(j);
                    if(sp1.getWidth()== sp2.getWidth() && sp1.getLength()== sp2.getLength() && sp1.getTopLeft().equals(sp2)){
                        shapes.remove(j);
                    }
                }else 
if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Rectangle){
                    Square sp1 = (Square) shapes.get(i);
                    Rectangle sp2 = (Rectangle) shapes.get(j);
                    if(sp1.getWidth()== sp2.getWidth() && sp1.getLength()== sp2.getLength() && sp1.getTopLeft().equals(sp2)){
                        shapes.remove(j);
                    }
                }
            }
        }
    }
    
}

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }
    
    public double distance(Point p){
        double a = Math.abs(this.x - p.getX());
        double b = Math.abs(this.y - p.getY());
        
        return Math.pow(a*a + b*b, 1/2);
    
}
    
    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle sp = (Circle) obj;
            if(this.getX() == sp.getCenter().getX() && this.getY() == sp.getCenter().getY()){
                return true;
            }
        }else 
if(obj instanceof Rectangle){
            Rectangle sp = (Rectangle) obj;
            if(this.getX() == sp.getTopLeft().getX() && this.getY() == sp.getTopLeft().getY()){
                return true;
            }
        }else if(obj instanceof Square){
           Square sp = (Square) obj;
            if(
this.getX() == sp.getTopLeft().getX() && this.getY() == sp.getTopLeft().getY()){
                return true;
            } 
        }
        
        return false;
    }
    
    public int hashCode(){
        return -1;
    }
    
    public String toString(){        
        return "(" + this.x + "," + this.y + ")";
    }
}
