    }

    public double getArea() {
        return Math.PI * radius * radius;
    }

        return 2 * Math.PI * radius;
    }

    public boolean equals(Object o) {
        if (o instanceof Circle) {
            if ( (this.center.equals(((Circle) o).center)) && (this.radius - ((Circle) o).radius <= 0.001) )
                
return true;

                return false;
        }
        else return false;
    }

        double r= (double) Math.round(radius*10)/10;

import java.util.*;
public class Layer {
    private List<Shape> shapes = new ArrayList<>();

    public void addShape(Shape shape) {
        shapes.add(shape);
    }
    public String getInfo() {
        String str = "Layer of crazy shapes: ";
        for ( Shape i : shapes) {
            str +="\n"+ 
i.toString() ;
        }
        return str;
    }
    public void removeCircles() {
        for ( int i = 0; i < shapes.size(); i++) {
            if ( shapes.get(i) instanceof Circle)
                shapes.remove(i);
        }
    }
    public void removeDuplicates() {
        Set<Shape> remove_shape = new LinkedHashSet<>(shapes);
        shapes = new ArrayList<Shape>(remove_shape);
    }
}

Point.java

public class Point {
    private double x;
    private double y;

    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point point) {
        return Math.sqrt((this.x - point.x)*(this.x - point.x) + (this.y - point.y)*(this.y - point.y));
    }

    @Override
    public boolean equals(Object obj) {
        if ( obj instanceof Point ) {
            
if ( (this.x - ((Point)obj).x <= 0.001) && (this.y - ((Point)obj).y <= 0.001))
                return true;
            else
                return false;
        }
        else
            return false;
    }

    public String toString() {
        return "(" + (double) Math.round(x*10)/10 + "," + (double) Math.round(y*10)/10 + ")";
    }

    @Override
    public int hashCode() {
        return (int)(x+y);
    }
}

public class Rectangle extends Shape {
    protected Point topLeft;
    protected double width;
    protected double length;

    public Rectangle(){}
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, Point topLeft) {
        this.width = width;
        this.length = length;
        this.topLeft = topLeft;
    }
    public Rectangle(Point topLeft, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = topLeft;
        this.width = width;
        this.length = length;
    }


    public double getWidth() {
        return width;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return length;
    }
    public void setLength(double length) {
        this.length = length;
    }
    public Point getTopLeft() {
        return topLeft;
    }
    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    
@Override
    public double getArea() {
        return width * length;
    }

    @Override
    public double getPerimeter() {
        return 2 * (width + length);
    }

    public boolean equals(Object o) {
        if (o instanceof Rectangle) {
            if ( (
this.topLeft.equals(((Rectangle) o).topLeft)) && (this.width - ((Rectangle) o).width <= 0.001) && (this.length - ((Rectangle) o).length <= 0.001) )
                return true;
            else
                return false;
        }
        else return false;
    }

    @Override
    public String toString() {
        return "Rectangle[topLeft=" + topLeft.toString() + ",width="+ width + ",length=" + length + ",color=" + color +",filled=" + filled + "]";
    }

    @Override
    public int hashCode() {
        return (int)(width + length);
    }
}
