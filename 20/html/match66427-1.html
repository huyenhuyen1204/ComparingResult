import java.lang.Math;
public class Point {
    private   double x;
    private double y;


    public Point(double x, double y) {
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public void setX(double x) {
        this.x = x;
    }
    public double getY() {
        return y;
    }
    public void setY(double y) {
        this.y = y;
    }
    public double distance(Point p) {
        double dis = 0;
        double x1 = this.x - p.x;
        double y1 = this.y - p.y;
        dis = Math.sqrt(Math.pow(x1, 2) + Math.pow(y1, 2));
        return dis;
    }

    @Override
    public 
boolean equals(Object obj) {
        if(obj instanceof Point) {
            Point p = (Point) obj;
            if(!(Math.round(this.x*1000)/1000 == Math.round(p.x*1000)/1000)){
                return false;
            }else if(!(Math.round(this.y*1000)/1000 == Math.round(p.y*1000)/1000)){
                return false;
            }else {
                return true;
            }
        }else {return false;}
    }
    public int hashCode() {
        return (int)x;
    }

    @Override
    public String toString() {
        return "("+this.x + "," + this.y + ")";
    }
}

public class Rectangle extends Shape {
    protected double width;
    protected double length;
    protected Point topLeft;

    public Rectangle() {

    }
    public Rectangle(double width, double length) {
        this.width = width;
        this.length = length;
    }
    public Rectangle(double width, double length, String color, boolean filled) {
        super(color, filled);
        this.width = width;
        this.length = length;
    }
    public Rectangle(Point point, double width, double length, String color, boolean filled) {
        super(color, filled);
        this.topLeft = point;
        this.length = length;
        this.width = width;

    }
    public double getWidth() {
        return Math.round(this.width*10)/10;
    }
    public void setWidth(double width) {
        this.width = width;
    }
    public double getLength() {
        return Math.round(this.length*10)/10;
    }

    public void setLength(double length) {
        this.length = length;
    }

    public Point getTopLeft() {
        return topLeft;
    }

    public void setTopLeft(Point topLeft) {
        this.topLeft = topLeft;
    }

    @Override
    public double getArea() {
        return Math.round((this.length*this.width)*10)/10;
    }

    @Override
    public double getPerimeter() {
        return Math.round((2*this.width+2*this.length)*10)/10;
    
}

    public boolean equals(Object obj) {
        if(obj instanceof Rectangle) {
            Rectangle re = (Rectangle) obj;
            if(!(Math.round((this.length)*1000)/1000 != Math.round((re.length)*1000)/1000)){
                return false;
            }else if(!(Math.round((this.width)*1000)/1000 != Math.round((re.width)*1000)/1000)) {
                return false;
            }else if(!this.topLeft.equals(re)) {
                return false;
            }else {
                return true;
            }
        }else {return false;}
    }
    public int hashCode() {
        return (int)width;
    }

    public String toString() {
        return "Rectangle[topLeft="+topLeft.toString()+",width=" + this.getWidth() + ",length=" + this.getLength() + ",color=" + this.getColor() +",filled=" + this.isFilled() + "]";
    }
}
