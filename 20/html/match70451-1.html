import java.util.ArrayList;
import java.util.List;

public class Layer {
    private List<Shape> shapes = new ArrayList<Shape>();

    public void addShape(Shape shape){
        shapes.add(shape);
    }

    public void removeCircles(){
        for(int i =0; i<shapes.size(); i++){
            
if(shapes.get(i) instanceof Circle){
                shapes.remove(i);
            }
        }
    }

    public String getInfo(){
        String s = "Layer of crazy shapes: "+"\n";
        for(Shape i: shapes){
            s += i.toString()+"\n";
        }
        return s;
    }

    public void removeDuplicates(){
        for(int i=0; i<shapes.size(); i++){
            for(int j=i+1; j<shapes.size(); j++){
                if(shapes.get(i) instanceof Circle && shapes.get(j) instanceof Circle){
                    Circle obj1 = (Circle) shapes.get(i);
                    Circle obj2 = (Circle) shapes.get(j);
                    if(obj1.getRadius() == obj2.getRadius() && obj1.getCenter().equals(obj2.getCenter())) shapes.remove(j);
                }
                if(shapes.get(i) instanceof Rectangle && shapes.get(j) instanceof Rectangle){
                    Rectangle obj1 = (Rectangle) shapes.get(i);
                    Rectangle obj2 = (Rectangle) shapes.get(j);
                    
if(obj1.getLength() == obj2.getLength() && obj1.getWidth() == obj2.getWidth() && obj1.getTopLeft().equals(obj2.getTopLeft())) shapes.remove(j);
                }
                if(shapes.get(i) instanceof Square && shapes.get(j) instanceof Square){
                    Square obj1 = (Square) shapes.get(i);
                    Square obj2 = (Square) shapes.get(j);
                    
if(obj1.getLength() == obj2.getLength() && obj1.getWidth() == obj2.getWidth() && obj1.getTopLeft().equals(obj2.getTopLeft())) shapes.remove(j);
                }
            }
        }
    }
}

import java.lang.Math.*;

public class Point {
    private double x;
    private   double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return x;
    }

    public double getY() {
        return y;
    }

    public void setX(double x) {
        this.x = x;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point point){
        double a = this.getX()-point.getX();
        double b = this.getY()-point.getY();
        return Math.sqrt((a*a)+(b*b));
    }

    public 
boolean equals(Object object){
        if(object instanceof Point){
            Point other = (Point) object;
            if(this.getX() == other.getX() && this.getY() == other.getY()){
                return  true;
            }
            return false;
        
}
        return false;
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }

    @Override
    public String toString() {
        return "("+getX()+","+getY()+")";
    }
}

public class Square extends Rectangle {
    public Square(){}

    public Square(double length){
        setLength(length);
        setWidth(length);
    }

    public Square(double length, String color, boolean filled){
        setLength(length);
        setWidth(length);
        setColor(color);
        setFilled(filled);
    }

    public Square(Point side,double length, String color, boolean filled){
        setColor(color);
        setFilled(filled);
        setLength(length);
        setWidth(length);
        setTopLeft(side);
    }

    public void setSide(double side){
        setLength(side);
        setWidth(side);
    }

    public double getSide(){
        return getLength();
    
}

    @Override
    public void setWidth(double width) {
        super.setWidth(width);
        super.setLength(width);
    }

    @Override
    public void setLength(double length) {
        super.setLength(length);
        super.setWidth(length);
    }

    public boolean equals(Object object){
        if(object instanceof Square){
            Square other = (Square) object;
            if(this.width == other.width && this.length == other.length){
                return true;
            }
            return false;
        
}
        return false;
    }

    @Override
    public String toString() {
        return "Square[topLeft="+getTopLeft()+",side="+getLength()+",color="+getColor()+",filled="+isFilled()+"]";
    }
}
