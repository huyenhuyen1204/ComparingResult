import java.util.*;

public class Layer {
    private List<Shape> shapes;

    public Layer(){
        this.shapes =  new ArrayList<Shape>();
    }

    public void addShape(Shape S){
        this.shapes.add(S);
    }

    public String getInfo(){
        String Out = "Layer of crazy shapes: \n" ;
        for (int i=0; i<this.shapes.size(); i++){
            if (this.shapes.get(i) instanceof Circle){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }

            else if (this.shapes.get(i) instanceof Rectangle){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }

            else if (this.shapes.get(i) instanceof Square){
                Out = Out + this.shapes.get(i).toString() + "\n";
            }
        }
        return Out;
    }

    public void removeCircles(){
        for(int i=0; i<this.shapes.size(); i++){
            if (this.shapes.get(i) instanceof Circle){
                this.shapes.remove(i);

                if (i==this.shapes.size()-1)
                    this.shapes.remove(i);
            }
        }
    }

    public void removeDuplicates() {
        for (int i = 0; i < this.shapes.size() - 1; i++)
            for (int j = i + 1; j < this.shapes.size(); j++) {
                if (this.shapes.get(i).equals(this.shapes.get(j)))
                        this.shapes.remove(j);
            }
    }

        Layer L = new Layer();

public class Point {
    private double x;
    private double y;

    public Point(double x, double y){
        this.x = x;
        this.y = y;
    }

    public double getX() {
        return this.x;
    }

    public void setX(double x) {
        this.x = x;
    }

    public double getY() {
        return this.y;
    }

    public void setY(double y) {
        this.y = y;
    }

    public double distance(Point B){
        return Math.sqrt((this.x - B.x)*(this.x - B.x) + (this.y - B.y)*(this.y - B.y));
    }

    public boolean equals(Object O){
        if (O instanceof Point){
            if (this.x==((Point) O).x && this.y==((Point) O).x) return true;
            else return false;
        }
        else return false;
    }

    //public int hashCode()

    @Override
    public String toString() {
        return "(" + this.x + "," + this.y + ")";
    }
}

public class Square extends Rectangle {
    public Square(){
        super(1.0, 1.0);
    }

    public Square(double side){
        super(side, side);
    }

    public Square(double side, String color, boolean filled){
        super(side, side, color, filled);
    }

    public Square(Point topLeft, double side, String color, boolean filled){
        super(topLeft, side, side, color, filled);
    }

    public double getSide(){
        return super.getLength();
    }
    public void setSide(double side){
        super.setLength(side);
        super.setWidth(side);
    }

    @Override
    public void setWidth(double side){
        super.setWidth(side);
        super.setLength(side);
    }

    @Override
    public void setLength(double side){
        super.setWidth(side);
        super.setLength(side);
    }

    public boolean equals(Object O){
        if (O instanceof Square){
            if (this.topLeft == ((Square) O).topLeft && this.width == ((Square) O).width && this.length == ((Square) O).length)
                return true;
            else return false;
        }
        else return false;
    }

    @Override
    public String toString(){
        return "Square[topLeft=(" + super.getTopLeft().getX() + "," + super.getTopLeft().getY() + "),side="
                + this.getSide() + ",color=" + this.getColor() + ",filled=" + this.isFilled() + "]";
    }

}
