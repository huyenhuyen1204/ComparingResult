Circle.java

public class Circle extends Shape {
    protected Point center;
    protected double radius;
    public Circle(){
    }
    public Circle(double radius){
        this.radius = radius;
    }
    public Circle(double radius, String color, boolean filled){
        super(color, filled);
        this.radius = radius;
    }
    public Circle(Point position, double radius, String color, boolean filled){
        super(color, filled);
        center = new Point(position.getX(), position.getY());
        this.radius = radius;
    }

    public Point getCenter() {
        return center;
    }

    public void setCenter(Point center) {
        this.center = center;
    }

    public double getRadius() {
        return radius;
    }
    public void setRadius(double radius) {
        this.radius = radius;
    }

    public double getArea(){
        return Math.PI * radius * radius ;
    }

    public double getPerimeter(){
        return 2 * Math.PI * radius;
    }

    public boolean equals(Object obj){
        if(obj instanceof Circle){
            Circle other = (Circle)obj;
            if(Math.abs(this.radius - other.radius) <= 0.001 && this.center.equals(other.center)){
                return true;
            }
        }
        return false;
    }

    public String toString() {
        return "Circle[center=" + center.toString() + ",radius=" + this.radius + ",color=" + this.color + ",filled=" + this.filled + "]";
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }



}

public class Layer {
    private List<Shape> shapes;
    Layer(){
        shapes = new ArrayList();
    }
    public void addShape(Shape obj){
        shapes.add(obj);
    }

    public String getInfo(){
        String result = "Layer of crazy shapes: ";
        for(int i = 0; i < shapes.size(); i++){
            result += '\n';
            result += shapes.get(i).toString();
            //if (i != shapes.size() - 1) result += '\n';
        }
        return result;
    }
    public void removeCircles(){
        for(int i = 0; i < shapes.size(); ++i){
            if( shapes.get(i) instanceof Circle){
                shapes.remove(i);
                i = 0;
            }
        }
    }
    public void removeDuplicates(){
        for (int i = 0; i < shapes.size(); ++i){
            for(int j = i + 1; j < shapes.size(); ++j){

                if(shapes.get(i).equals(shapes.get(j)) && i != j)
                {
                    shapes.remove(j);
