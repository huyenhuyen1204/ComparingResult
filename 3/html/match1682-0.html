public class Student{

    private String name;
    private String id;
    private String group;
    private String email;

    public String getName (){
        return name;
    }

    public void setName (String name){
        this.name = name;
    }

    public String getId (){
        return id;
    }

    public void setId (String id){
        this.id = id;
    }

    public String getGroup(){
        return group;
    }

    public void setGroup (String group){
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    Student (){
        setName("Student");
        setGroup("K62CB");
        setId("000");
        setEmail("uet@vnu.edu.vn");
    }

    Student (String name, String id, String email){
        setName(name);
        setId(id);
        setEmail(email);
        setGroup("K62CB");
    }

    Student (Student s){
        name = s.getName();
        id = s.getId();
        email = s.getEmail();
        group = s.getGroup();
    }

    String getInfo(){
        String Info = name + " - " + id + " - " + group + " - " + email;
        return Info;
    }

}

public class StudentManagement {

    private static Student students[] = new Student[100];
    private static int num_of_student = 0;

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    public static void addStudent(Student newStudent){
        students[num_of_student] = newStudent;
        ++num_of_student;
        return;
    }

    public static String studentsByGroup() {
        String list = "";
        int check[] = new int[num_of_student];
        for (int i = 0; i < num_of_student; ++i) check[i] = 1;

        for (int i = 0; i < num_of_student; ++i)
        {
            if (check[i] == 1)
            {
                String group = students[i].getGroup();
                list += group + "\n" + students[i].getInfo() + "\n";
                check[i] = 0;

                for (int j = i + 1; j < num_of_student; ++j)
                    if (students[j].getGroup().equals(group))
                    {
                        list += students[j].getInfo() + "\n";
                        check[j] = 0;
                    }
            }
        }
        return list;
    }

    public static void removeStudent(String id) {
        int i = 0;
        while (i < num_of_student)
        {
            if (students[i].getId().equals(id))
            {
                for (int j = i ; j < num_of_student - 1; ++j)
                {
                    students[j] = students[j + 1];
                }
                --num_of_student;
            }
            else ++i;
        }

        i = 0;

        while (i < num_of_student - 1)
        {
            id = students[i].getId();
            int t = i + 1;
            while (t < num_of_student)
            {
                if (students[t].getId().equals(id))
                {
                    for (int j = t ; j < num_of_student - 1; ++j)
                    {
                        students[j] = students[j + 1];
                    }
                    --num_of_student;
                }
                else ++t;
            }
            ++i;
        }

        return;
    }

    public static void main(String[] args)
    {
        Student s1 = new Student("Nguyen Van An","17020001", "17020001@vnu.edu.vn");
        Student s2 = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        Student s3 = new Student("Nguyen Van C", "17020003", "17000003@vnu.edu.vn");
        Student s4 = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");

        s1.setGroup("K62CC");
        s2.setGroup("K62CC");

        addStudent(s1);
        addStudent(s2);
        addStudent(s3);
        addStudent(s4);

        System.out.println(studentsByGroup());
    }
}
