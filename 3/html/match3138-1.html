public class Student {

	private String id;
	private String group;

String getName() {

	}
	
	public String getId() {
		return this.id;
	}
	
	public String getGroup() {
		return this.group;
	}
	
	public String getEmail() {
		return this.email;
	}
	
	public String getInfo() {
		String info = this.name + " - " + this.id + " - " + this.group + " - " + this.email;
		return info;
	
}
	
	// set info
	public void setName(String n) {
		this.name = n;
	}
	
	public void setId(String n) {
		this.id = n;
	}
	
	public void setGroup(String n) {
		this.group = n;
	}
	
	public void setEmail(String n) {
		this.email = n;
	}
	
	//

public class StudentManagement {

	public static boolean sameGroup(Student s1, Student s2) {
		boolean kq;
		if (s1.getGroup().contentEquals(s2.getGroup())) kq = true;
		else kq = false;
		return kq;
	}
	
	Student[] students = new Student[100];
	
	String[] groupList = new String[100];
	
	int[] numberOfStudentInGroup = new int[100];
	
	private int index = 0;
	private int indexGroup = 0;
	
	public void addStudent(Student newStudent) {
		if (index < 100) {
			students[index] = newStudent;
			boolean newGroup = true;
			for (int i = 0; i < indexGroup; i++) {
				if (newStudent.getGroup().contentEquals(groupList[i])) {
					newGroup = false;
					numberOfStudentInGroup[i]++;
					break;
				}
			}
			if (newGroup) {
				groupList[indexGroup] = newStudent.getGroup();
				numberOfStudentInGroup[indexGroup] = 1;
				indexGroup++;
			}
			index++;
		}
	
}
	
	public void removeStudent(String id) {
		for (int i = 0; i < index; i++) {
			if (students[i].getId().contentEquals(id)) {
				// remove group if necessary
				boolean removeGroup = false;
				for (int j = 0; j < indexGroup; ++j) {
					if (students[i].getGroup().contentEquals(groupList[j])) {
						if (numberOfStudentInGroup[j] == 1) {
							for (int k = j; k < indexGroup - 1; ++k) {
								groupList[k] = groupList[k+1];
							}
							removeGroup = true;
							break;
						}
					}
				}
				if (removeGroup) indexGroup--;
				
				// remove student
				
for (int j = i; j < index - 1; ++j) {
					students[j] = students[j+1];
				}
				index--;
				
				break;
			}
		}
	}
	
	public String studentsByGroup() {
		String kq = "";
		for (int i = 0; i < indexGroup; ++i) {
			
kq += groupList[i] + "\n";
			for (int j = 0; j < index; ++j) {
				if (students[j].getGroup().contentEquals(groupList[i])) {
					kq += students[j].getInfo() + "\n";
				}
			}
		}
		return kq;
	}
