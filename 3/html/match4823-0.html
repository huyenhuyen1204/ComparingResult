import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class StudentManagement {

    Student[] students = new Student[100];
    public int count = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student newStudent) {
        students[count] = newStudent;
        count++;
    }

    public String studentsByGroup() {
        String res = "";
        List<String> groups = new ArrayList<String>();
        for (int i = 0; i < count; i++) {
            boolean exist = false;
            for (int j = 0; j < groups.size(); j++) {
                if (groups.get(j).equals(students[i].getGroup())) {
                    exist = true;
                    break;
                }
            }
            if (!exist) groups.add(students[i].getGroup());
        }
        for (int i = 0; i < groups.size(); i++) {
            res += (groups.get(i) + '\n');
            for (int j = 0; j < count; j++) {
                if (students[j].getGroup().equals(groups.get(i)))
                    res += (students[j].getInfo() + '\n');
            }
        }
        return res;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < count; i++) {
            if (id.equals(students[i].getId())) {
                for (int j = i; j < count; j++)
                    students[j] = students[j + 1];
                count--;
                break;
            }
        }
    }

    public static void main(String[] args) {
        StudentManagement SM = new StudentManagement();
        Student Bang = new Student();
        Student Huong = new Student("Pham Thanh Huong", "061", "pthuong9a1@gmail.com");
        SM.addStudent(Bang);
        SM.addStudent(Huong);
        Student Phanh = new Student(Huong);
        Phanh.setName("Nguyen Phuong Anh");
        SM.addStudent(Phanh);
        Huong.setGroup("Anh 12");
        SM.removeStudent("000");
        System.out.println(SM.count);
        System.out.println(SM.studentsByGroup());
    }
}
