public class StudentManagement {
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    private Student[] students = new Student[100];
    private Boolean[] gr = new Boolean[100];
    private int count = 0;
    private String s = "";

    public void addStudent(Student newStudent) {
        students[count] = new Student(newStudent);
        gr[count] = false;
        count++;
    }

    public String studentsByGroup() {
        String print = "";
        for (int i = 0; i < count; i++) {
            if (gr[i] == false) {
                print = print + students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
                gr[i] = true;
                for (int j = i + 1; j < count; j++) {
                    if (students[j].getGroup().equals(students[i].getGroup()) && gr[j] == false) {
                        print = print + students[j].getInfo() + "\n";
                        gr[j] = true;
                    }
                }
            }
        }
        return print;
    }


    public void removeStudent(String id) {

        for (int i = 0; i < count; i++) {
            if (students[i].getId().compareTo(id) == 0) {
                for (int j = i; j < count - 1; j++)
                    students[j] = students[j + 1];
                students[count - 1] = null;
                count--;
            }
        }
    }
}
