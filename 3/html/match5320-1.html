public class StudentManagement {
    Student[] students = new Student[100];
    int dem=0;

    public void addStudent(Student newStudent) {
        students[dem]=newStudent;
        dem++;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public String studentsByGroup() {
        String list_sv ="";
        String[] list_group = new String[dem];
        list_group[0] = students[0].getGroup();
        int 
n=1;
        for(int i=1; i<dem; i++) {
            for(int j=0; j<n; j++) {
                if(students[i].getGroup().equals(list_group[j]))
                    break;
                if(( !(students[i].getGroup().equals(list_group[j]))) && (j == n-1)) {
                    list_group[n]=students[i].getGroup();
                    n++;
                    break;
                }
            }
        }
        for(int i=0; i<n; i++) {
            String str=list_group[i] +"\n";
            for(int j=0; j<dem; j++) {
                if(students[j].getGroup().equals(list_group[i])) {
                    str += students[j].getInfo() + "\n";
                }
            }
            list_sv += str;
        
}
        return list_sv;
    }
    public void removeStudent(String id) {
        for(int i=0; i<dem; i++) {
            if(students[i].getId().equals(id)) {
                for(int j=i+1; j<dem; j++)
                    students[j-1]=students[j];
            }
        }
        dem--;
        System.out.println(studentsByGroup());
    }

}
