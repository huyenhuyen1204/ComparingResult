public class Student {
    private String name;
    private String id;
    private String group;
    private String email;

    Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    Student(String initName, String initID, String initEmail) {
        this.name = initName;
        this.id = initID;
        this.group = "K62CB";
        this.email = initEmail;
    }

    Student(Student s) {
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

    Student(String initName, String initID, String initGroup, String initEmail) {
        this.name = initName;
        this.id = initID;
        this.group = initGroup;
        this.email = initEmail;
    }

    public String getName() {
        return this.name;
    }

    public String getGroup() {
        return this.group;
    }

    public String getId() {
        return this.id;
    }

    public String getEmail() {
        return this.email;
    }

    public void setName(String nameValue) {
        this.name = nameValue;
    }

    public void setId(String id) {
        this.id = id;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public String getInfo() {
        return this.name + " - " + this.id + " - " + this.group + " - " + this.email;
    }
}

/**

public class StudentManagement {
    private Student[] students = new Student[100];
    private int student_count = 0;

    StudentManagement() {
        this.students = new Student[100];
        student_count = 0;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        if (this.student_count < 100) {
            students[student_count] = new Student(newStudent.getName(), newStudent.getId(), newStudent.getGroup(), newStudent.getEmail());
            student_count++;
        }
    }

    public String studentsByGroup() {
        String result = "";
        Boolean[] checked = new Boolean[student_count];
        for (int i = 0; i < student_count; i++)
            checked[i] = false;
        for (int i = 0; i < student_count; i++) {
            if (!checked[i]) {
                checked[i] = true;
                result += this.students[i].getGroup() + "\n";
                result += this.students[i].getInfo() + "\n";
                for (int j = i + 1; j < student_count; j++)
                    if (!checked[j] && this.sameGroup(students[i], students[j])) {
                        result += students[j].getInfo() + "\n";
                        checked[j] = true;
                    }
            }
        }

        return result;
    }

    public void removeStudent(String findid) {
        for (int i = 0; i < student_count; i++)
            if (this.students[i].getId().equals(findid)) {
                for (int k = i; k < student_count; k++)
                    students[k] = students[k + 1];
                student_count--;
            }
    }

    public static void main(String[] args) {
        Student studentAn = new Student("Nguyen Van An", "17020001", "K62CC", "17020001@vnu.edu.vn");
        Student studentB = new Student(studentAn);
        studentB.setId("17020002");
        studentB.setEmail("17020002@vnu.edu.vn");
        studentB.setName("Nguyen Van B");
        Student studentC = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        Student studentD = new Student();
        studentD.setName("Nguyen Van D"); studentD.setGroup("K62CB"); studentD.setId("17020004"); studentD.setEmail("17020004@vnu.edu.vn");
        StudentManagement arr = new StudentManagement();
        arr.addStudent(studentAn); arr.addStudent(studentB); arr.addStudent(studentC); arr.addStudent(studentD);
        System.out.println(arr.studentsByGroup());
        arr.removeStudent("17020001");
        arr.removeStudent("17020002");
        arr.removeStudent("17020003");
        arr.removeStudent("");
        System.out.println(arr.studentsByGroup());
    }
}
