public class StudentManagement {

    static Student[] students = new Student[100];
    static int countStudents = 0;
    String[] groups = new String[100];
    
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup()))
            return true;
        return false;
    }

    public static void addStudent(Student newStudent) {
        for(int i=0;i<countStudents;i++){
            if (students[i].getId().equals(newStudent.getId()))
                return;
        }
        students[countStudents] = newStudent;
        countStudents++;
    }

    public static String studentsByGroup() {
        String str = students[0].getGroup();
        int countGroup = 0;
        int NextGroup = countGroup;
        while (true) {
            for (int i = countGroup; students[i] != null; i++) {

                if (sameGroup(students[countGroup], students[i])) {
                    str = str + "\n" + students[i].getInfo();
                } else if (NextGroup == countGroup && !str.contains(students[i].getGroup())) {
                    NextGroup = i;
                }
            }
            if (countGroup == NextGroup) {
                break;
            } else {
                countGroup = NextGroup;
                str = str + "\n" + students[countGroup].getGroup();
            }
        }
        return str;
    }

    public static void removeStudent(String id) {
        Student[] temp = new Student[100];
        int counttemp = 0;
        for (int i = 0; i < countStudents; i++) {
            if (!students[i].getId().equals(id)) {
                temp[counttemp] = students[i];
                counttemp++;
            }
        }
        students = temp.clone();
        countStudents--;
    }



    public static void main(String[] args) {
    }
}
