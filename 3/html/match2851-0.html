public class StudentManagement {
    Student[] students = new Student[100];
    int count_Student = 0;
    public static boolean sameGroup(Student s1, Student s2)
    {
        if(s1.getGroup().equals(s2.getGroup())) return true;
        else    return false;
    }
    public void addStudent(Student newStudent)
    {
        students[count_Student] = new Student();
        students[count_Student] = newStudent ;
        count_Student++;
    }
    public String studentsByGroup()
    {
        String[] Group = new String[100];
        boolean test;
        int count_group =0;
        for(int i=0;i<count_Student;i++)
        {
            if(i==0)    Group[count_group]=students[i].getGroup();
            else {
                test = true;
                for (int j = 0; j < i; j++) {
                    if (sameGroup(students[i], students[j])) {
                        test = false;
                        break;
                    }
                }
                if (test) {
                    count_group++;
                    Group[count_group] = students[i].getGroup();
                }
            }
        }
        String Students = "";
        for(int i=0;i<=count_group;i++)
        {
            Students += Group[i]+ "\n";
            for(int j=0;j<count_Student;j++)
            {
                if(Group[i].equals(students[j].getGroup()))    Students += students[j].getInfo() + "\n";
            }
        }
        return Students;
    }
    public void removeStudent(String id)
    {
        for(int i=0;i<count_Student;i++)
        {
            if(students[i].getId().equals(id))
            {
                for(int j=i;j<count_Student-1;j++)
                {
                    students[j]=new Student(students[j+1]);
                }
                count_Student--;
            }
        }
    }

    public static void main(String[] args) {
        StudentManagement s = new StudentManagement();
        Student s1 = new Student();
        Student s2 = new Student("cuong","002","asdhfbh");
        s.addStudent(s1);
        s.addStudent(s2);

        System.out.println(s.studentsByGroup());
        s.removeStudent("002");
        System.out.println(s.studentsByGroup());
    }
}
