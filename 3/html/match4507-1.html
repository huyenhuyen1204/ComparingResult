public class StudentManagement {
    public Student[] students = new Student[100];
    public int length =0;
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup()))return true;
        else return false;
    }

    public void addStudent(Student newStudent){
        students[length] = newStudent;
        length++;
    }

    public String studentsByGroup() {
        String[] arrtemp = new String[100];
        int i, j;
        int dem = 1;
        arrtemp[0] = this.students[0].getGroup();
        int tam = 2;
        for (i = 1; i < length; i++) {
            for (j = 0; j < i; j++) {
                if (this.students[j].getGroup().equals(this.students[i].getGroup())) {
                    tam = 0;
                    break;
                } else
                    tam = 1;
            }
            if (tam == 1) {
                arrtemp[dem] = this.students[j].getGroup();
                dem++;
            }
        }
        String print = "";
        for (int e = 0; e < dem; e++) {
            print += arrtemp[e] + "\n";
            for (int f = 0; f < length; f++) {
                if (arrtemp[e].equals(this.students[f].getGroup())) {
                    print += this.students[f].getInfo() + "\n";
                }
            }
        }
        return print;
    }

    public void removeStudent(String id) {
        int i;
        for (
i = 0; i < length; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < length-1; j++)
                    students[j] = students[j + 1];
                length--;
            }
        }
    }

    {
