/**
 *
 * @author Le Long
 */
public class StudentManagement {
    private Student[] students = new Student[100];
    private int numbOfStudents = 0;
    
    public static boolean sameGroup(Student s1, Student s2) {
        // TODO:
        if (s1.getGroup().equals(s2.getGroup())) {
            return true;
        }
        return false;
    }

    public void addStudent(Student newStudent){
	 // TODO:
         students[numbOfStudents] = new Student(newStudent);
         numbOfStudents++;
    }

    public String studentsByGroup() {
        // TODO:
        int[][] groups = new int[100][100];
        String list = new String();
        int i, j, numbOfGroups = 0;
        int numbInAGroup[] = new int[100];
        boolean flag;
        for (i = 0; i < numbOfStudents; i++) {
            flag = true;
            for (j = 0; j < numbOfGroups; j++) {
                if (students[i].getGroup().equals(students[groups[j][0]].getGroup())) {
                    groups[j][numbInAGroup[j]] = i;
                    numbInAGroup[j]++;
                    flag = false;
                    break;
                } 
            } 
            if (flag) {
                groups[numbOfGroups][0] = i;
                numbInAGroup[numbOfGroups] = 1;
                numbOfGroups++;
            }
        } 
        for (i = 0; i < numbOfGroups; i++) {
            list = list + students[groups[i][0]].getGroup() + "\n";
            for (j = 0; j < numbInAGroup[i]; j++) {
                list = list + students[groups[i][j]].getInfo() + "\n";
            }
        }
	return list;
    }

    public void removeStudent(String id) {
        // TODO:
        int i, j;
        for (i = 0; i < numbOfStudents; i++) {
            if (students[i].getId().equals(id)) {
                break;
            }
        }
        for (j = i; j < numbOfStudents - 1; j++) {
            students[j] = new Student(students[j+1]);
        }
        numbOfStudents--;
    }

}
