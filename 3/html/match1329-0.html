    public static boolean sameGroup(Student s1, Student s2) {
        if((s1.getGroup()).equals(s2.getGroup()))

    public static void addStudent(Student newStudent){
        if(stt==100)

            System.out.println("ERROR");
        }
        else {
            students[stt]=newStudent;

        }
    }

    public static String studentsByGroup() {
        String s = "";
            for(int i=0;i<stt;i++)
            {
                if(s.indexOf(students[i].getGroup())==-1)
                {
                    s+=students[i].getGroup()+"\n"+students[i].getInfo()+"\n";
                    for (int j=i+1;j<stt;j++)
                    {
                        if(sameGroup(students[i],students[j])==true)
                        {
                            s+=students[j].getInfo()+"\n";
                        }

                    }
                }

            }
            if(s.equals("")==true)
            {
                return null;
            }
            else return s;
    }
