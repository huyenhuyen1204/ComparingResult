public class StudentManagement {
    public int length = 0;
    private Student [] 
students = new Student[100];
    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }
    public void addStudent(Student newStudent){
        if(length < 100){
            students[length] = newStudent;
            length++;
        }
    }
    public void removeStudent(String id){
        if(length > 0){
            
for(int i  = 0 ; i < length ; i++){
                if(id.equals(students[i].getId())){
                    for(int j = i   ; j < length -1 ; j++)
                        students[j] = students[j+1];
                }}
            length--;
            if(id.equals(students[length - 1].getId()))
                length--;
        }}

    public String studentsByGroup(){
        String str ="";
        String newGroup = "";
        for(int i = 0 ; i < length ; i++){
            if(!newGroup.contains(students[i].getGroup())){
                newGroup += students[i].getGroup();
                str += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
            }
            if( i < length - 1 ){
                for(int j = i + 1 ; j < length ; j++){
                    if(sameGroup(students[i],students[j]))
                        
str += students[j].getInfo() + "\n";
                }
            }
        }
        return str;
    }


}
