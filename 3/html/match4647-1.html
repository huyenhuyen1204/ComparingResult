public class StudentManagement{
    private int n;
    private Student[] students;

    public StudentManagement(){
        n = 0;
        students = new Student[100];
    
}
    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().compareTo(s2.getGroup()) == 0;
    }
    public void addStudent(Student newStudent){
        students[n++] = newStudent;
    }
    public String studentsByGroup(){
        String ret = "";
        boolean[] outed = new boolean[n + 1];
        for(int i = 0; i < n; i++){
            if(!outed[i]){
                outed[i] = true;
                ret += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
                for(int j = i + 1; j < n; j++){
                    if(sameGroup(students[i], students[j])){
                        ret += students[j].getInfo() + "\n";
                        outed[j] = true;
                    }
                }
            }
        }
        return ret;
    }

    public void removeStudent(String id) {
        // TODO:
        Student[] tmp = new Student[100];
        int count = 0;
        for(int i = 0; i < n; i++){
            if(students[i].getId().compareTo(id) == 0){
                count ++;
            } else {
                tmp[i - count] = students[i];
            }
        }
        n -= count;
        students = tmp;
    }

    public static void main(String[] args){
        StudentManagement sm = new StudentManagement();
        for(int i = 0; i < 10; i++){
            Student s = new Student("aa",Integer.toString(i % 4), "a");
            s.setGroup(Integer.toString(i % 3));
            sm.addStudent(s);
        }
        sm.removeStudent("3");
        System.out.println(sm.studentsByGroup());
    }
}
