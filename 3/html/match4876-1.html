import java.util.ArrayList;
import java.util.Iterator;

public class StudentManagement {
    private Student[] 
students = new Student[100];
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup()) )
            return true;
        else return false;
    }
    public void addStudent(Student newStudent){

        for (int i=0; i<100; i++){
            if (students[i] == null){
                students[i] = new Student(newStudent);
                break;
            }
        }
    }
    public String studentsByGroup() {
        String fullStudents = "";
        for (int i=0; this.students[i] != null; i++){
            if (!(fullStudents.contains(students[i].getGroup()))) {
                fullStudents = fullStudents + students[i].getGroup() + "\n";
                for (int j = 0; students[j] != null; j++) {
                    if (
sameGroup(students[i],students[j]))
                        fullStudents = fullStudents + students[j].getInfo() + "\n";
                }
            }
        }
        return fullStudents;
    }
    public void removeStudent(String id){
        for (int i=0; this.students[i] != null; i++){
            if (students[i].getId().equals(id)){
                for (int j = i; students[j] != null; j++){
                    students[j] = students[j+1];
                }
                break;
            }
        }

    }
}
