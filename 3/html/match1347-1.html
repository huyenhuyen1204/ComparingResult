        this.name = name;
    }

        return id;

    {
        this.id = id;
    }

    // getter,setter cho thuộc tính group
    public String getGroup() {
        return group;
    }
    public void setGroup(String group) {
        this.group = group;
    }

    // getter,setter cho thuộc tính email
    public String getEmail() {
        return email;
    }
    public void setEmail(String email) {
        this.email=email;
    }

    //Phương thức nhận thông tin
    public 
String getInfo()
    {
        return (name+" - "+ id + " - " + group + " - " + email);
    }

    //Phương thức khởi tạo không tham số
    public Student()
    {
        setName("Student");
        setId("000");
        setGroup("K62CB");
        setEmail("uet@vnu.edu.vn");
    }
    //Phương thức khởi tạo có tham số
    public Student(String name, String id, String email)
    {
        setName(name);
        setId(id);
        setGroup("K62CB");
        setEmail(email);
    }
    //Phương thức khởi tạo kiểu sao chép
    public Student(Student s)
    {
        setName(s.name);
        setId(s.id);
        setGroup(s.group);
        setEmail(s.email);
    }

}

class StudentManagement
{
    //Khởi tạo thuộc tính mảng Student;
    Student[] students = new Student[100];
    private int count=0;

    public static boolean sameGroup(Student s1, Student s2)
    {
        if ((s1.getGroup()).equals(s2.getGroup()))
            return true;
        else
            return false;
    }

    public void addStudent(Student newStudent)
    {
        students[count]= new Student();
        students[count]= newStudent;
        count++;
    }

    public String studentsByGroup()
    {
        String[] group_name = new String[100];
        int k=0;
        boolean ex;
        group_name[k]=students[0].getGroup();
        for (int i=1 ; i<count ; i++)
        {
            ex=true;
            for (int j=0 ; j<i ; j++)
            {
                if (sameGroup(students[i], students[j]))
                {
                    ex = false;
                    break;
                }
            }
            if (ex)
            {
                k++;
                group_name[k]=students[i].getGroup();
            }
        }
        String GroupStudent="";

        for (int i=0 ; i<=k ; i++)
        {
            GroupStudent += group_name[i] + "\n";
            for (int j = 0; j < count ; j++)
            {
                if (group_name[i]==students[j].getGroup())
                    GroupStudent += students[j].getInfo() + "\n";
            }
        }

        return GroupStudent;
    }

    public void removeStudent(String id)
    {
        for (int i=0 ; i<count ; i++)
        {
            if ((students[i].getId()).equals(id))
            {
                for (int j=i ; j<(count-1) ; j++ )
                    students[j]=new Student(students[j+1]);
                count--;
            }
        }
    }

    public static void main(String[] args)

    {
        Student s1 = new Student("Nguyen Van An","17020001", "17020001@vnu.edu.vn");
        System.out.println(s1.getInfo());
    }
}
