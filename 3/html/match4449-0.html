public class StudentManagement
{
    int pos=0;
    Student[] studentarr = new Student[100];
    // kiem tra 2 sinh vien co cung lop hay khong
    public static  boolean sameGroup(Student s1, Student s2)
    {
       if(s1.getGroup().equals(s2.getGroup())) return true;
       else return false;
    }

    public void addStudent (Student newStudent)
    {
      studentarr[pos]= newStudent;
      pos++;
    }


    public String studentsByGroup()
    {
       String  s="";
       int [] a= new int[pos];
       for(int i=0;i<pos;i++) a[i]=0;
       for (int i=0;i<pos;i++)
       {
           if (a[i]==0)
           {
               String str=studentarr[i].getGroup();
               s+=str+"\n";
               a[i]=1;
               s+=studentarr[i].getInfo()+"\n";
               for (int j=i+1;j<pos;j++)
               {
                   if (str.equals(studentarr[j].getGroup())&&a[j]==0)
                   {
                       s+=studentarr[j].getInfo()+"\n";
                       a[j]=1;
                   }
               }
           }
       }
       return s;

    }
    public void removeStudent(String Id)
    {
        for (int i=0;i<pos;i++)
        {
            if (studentarr[i].getId().equals(Id))
            {
                for (int j=i;j<pos-1;j++)
                    studentarr[j]=studentarr[j+1];
                pos--;
            }
        }
    }


}
