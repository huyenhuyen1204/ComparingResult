public class StudentManagement {
    static int k = 0;
    static Student students[] = new Student[100];
    public static boolean sameGroup( Student s1, Student s2) {
        if ( s1.getGroup().equals(s2.getGroup()) ) return true;
        else return false;
    }
    public void addStudent ( Student newStudent){
        students[k] = newStudent;
        k++;
    }
    public String studentsByGroup(){
        String string = "";
        int dem[] = new int[k];
        for ( int i = 0; i < k; i++) {
            dem[i] = i;
        }
        for ( int i = 0; i < k; i++){
            if ( dem[i] != - 1){
                string += students[i].getGroup() + "\n";
                for ( int j = i; j < k; j++) {
                    if ( students[i].getGroup().equals(students[j].getGroup()) && dem[j] != -1){
                        dem[j] = -1;
                        string += students[j].getInfo() + "\n";
                    }
                }
                dem[i] = -1;
            }
        }
        //System.out.print(string);
        return string;

    }

    public void removeStudent(String id) {
        //String id_remove;
        for ( int i = 0; i < k; i++){
            if ( students[i].getId().equals(id) ){
                for ( int j = i; j < k - 1; j++){
                    students[j].setName(students[j+ 1].getName());
                    students[j].setId(students[j+ 1].getId());
                    students[j].setGroup(students[j+ 1].getGroup());
                    students[j].setEmail(students[j+ 1].getEmail());
                }
            }
        }
        k= k - 1;
        //System.out.print(k);
    }
    public static void main(String[] args) {
        StudentManagement studentManagement = new StudentManagement();
    }
}
