public class Student {

    private String name;
    private String id;
    private String group;
    private String email;

    public String getName(){
        return name;
    }

    public void setName(String name){
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    /**
     * Constructor 1
     */
    Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    /**
     * Constructor 2
     * @param name
     * @param id
     * @param email
     */
    Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    /**
     * Constructor 3
     * @param s
     */
    Student(Student s) {
       this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

    
Student(String name, String id, String group, String email) {
        this.name = name;
        this.id = id;
        this.group = group;
        this.email = email;
    }

    String getInfo() {
        return name + " - " + id + " - " + group + " - " + email;
    }
}

class StudentManagement {

    public Student[] students = new Student[100];

    public Student[] getStudents() {
        return students;
    }

    public int getSize(){
        int size = 0;
        for(int i = 0;i<100;i++){
            if(getStudents()[i]!=null) {
                size++;
            } else {
                break;
            }
        }
        return size;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup()))
            return true;
        return false;
    }

    public void addStudent(Student newStudent){
    	for(int i = 0;i<100;i++){
                if(getStudents()[i]==null) {
                    getStudents()[i] = new Student(newStudent);
                    break;
                }
            }
    }

    public String studentsByGroup() {
        String str = "";
        boolean[] check = new boolean[100];
        for(int i = 0;i<
getSize();i++)
            check[i] = false;
        for(int i = 0;i<getSize();i++){
            if(check[i] == false){
                str += getStudents()[i].getGroup() + "\n";
                str += getStudents()[i].getInfo() + "\n";
                check[i] = true;
                for(int j=i+1;j<getSize();j++){
                    if(getStudents()[j].getGroup().equals(getStudents()[i].getGroup())){
                        str += getStudents()[j].getInfo() + "\n";
                        check[j] = true;
                    }
                }
            }
        }
        return str;
    }

    public void removeStudent(String id) {
        int tmp = 
getSize();
        for(int i = 0;i<tmp;i++){
            if(getStudents()[i].getId().equals(id)){
                for(int j = i;j<tmp-1;j++)
                    getStudents()[j] = new Student(getStudents()[j+1]);
                getStudents()[tmp-1] = null;
                break;
            }
        }
    }

    public static void main(String[] args) {
        Student s1 = new Student();
        System.out.println(s1.getInfo());
        Student s4 = new Student(s1);
        System.out.println(s4.getName());
        Student s2 = new Student("Quyet", "17020993", "17020993@vnu.edu.vn");
        System.out.println(s2.getId());
        Student s3 = new Student(s2);
        System.out.println(s3.getEmail());

        //same group s1, s4
        if(sameGroup(s1,s4))
            System.out.println("Same\n\n");

        //students by group
        StudentManagement sm = new StudentManagement();
        Student an = new Student("Nguyen Van An", "17020001", "K62CC", "17020001@vnu.edu.vn");
        Student b = new Student("Nguyen Van B", "17020002", "K62CC", "17020002@vnu.edu.vn");
        Student c = new Student("Nguyen Van C", "17020003", "K62CB", "17020003@vnu.edu.vn");
        Student d = new Student("Nguyen Van D", "17020004", "K62CB", "17020004@vnu.edu.vn");
        sm.addStudent(an);
        sm.addStudent(b);
        sm.addStudent(c);
        sm.addStudent(d);
        System.out.println(sm.studentsByGroup() + "\n\n");

        //remove id="17020003"
        sm.removeStudent("17020003");
        
for(int i = 0;i<sm.getSize();i++){
            System.out.println(sm.getStudents()[i].getInfo());
        }
    }
