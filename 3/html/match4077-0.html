public class StudentManagement
{
    //Khởi tạo thuộc tính mảng Student;
    Student[] students = new Student[100];
    private int count=0;

    public static boolean sameGroup(Student s1, Student s2)
    {
        if ((s1.getGroup()).equals(s2.getGroup()))
            return true;
        else
            return false;
    }

    public void addStudent(Student newStudent)
    {
        students[count]= new Student();
        students[count]= newStudent;
        count++;
    }

    public String studentsByGroup()
    {
        String[] group_name = new String[100];
        int k=0;
        boolean ex;
        group_name[k]=students[0].getGroup();
        for (int i=1 ; i<count ; i++)
        {
            ex=true;
            for (int j=0 ; j<i ; j++)
            {
                if (sameGroup(students[i], students[j]))
                {
                    ex = false;
                    break;
                }
            }
            if (ex)
            {
                k++;
                group_name[k]=students[i].getGroup();
            }
        }
        String GroupStudent="";

        for (int i=0 ; i<=k ; i++)
        {
            GroupStudent += group_name[i] + "\n";
            for (int j = 0; j < count ; j++)
            {
                if (group_name[i]==students[j].getGroup())
                    GroupStudent += students[j].getInfo() + "\n";
            }
        }

        return GroupStudent;
    }

    public void removeStudent(String id)
    {
        for (int i=0 ; i<count ; i++)
        {
            if ((students[i].getId()).equals(id))
            {
                for (int j=i ; j<(count-1) ; j++ )
                    students[j]=new Student(students[j+1]);
                count--;
            }
        }
    }

    public static void main(String[] args)

    {
        Student s1 = new Student("Nguyen Van An","17020001", "17020001@vnu.edu.vn");
        System.out.println(s1.getInfo());
    }
}
