import java.util.*;

public class StudentManagement {
    public static boolean sameGroup(Student s1, Student s2) {
        String a1 = s1.getGroup();
        String a2 = s2.getGroup();
        if (a1.equals(a2)) return true;
        else return false;
    }

    private Student[] students = new Student[100];
    private int n = 0;

    public void addStudent(Student newStudent) {
        students[n] = newStudent;
        n = n + 1;
    }

    public String studentsByGroup() {
        StringBuilder res = new StringBuilder();
        Vector<String> group = new Vector<String>();

        for (int i = 0; i < n; i++) {
            if (!group.contains(students[i].getGroup())) {
                group.add(students[i].getGroup());
            }
        }

        for (int i = 0; i < group.size(); ++i) {
            res.append(group.get(i)).append("\n");
            for (int j = 0; j < n; j++) {
                if (students[j].getGroup().equals(group.get(i))) {
                    res.append(students[j].getInfo()).append("\n");
                }
            }
        }
        return res.toString();
    }


    public void removeStudent(String id) {
        for (int i = 0; i < n; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < n - 1; j++) {
                    students[j] = students[j + 1];
                }
                students[n - 1] = null;
                n--;
                break;
            }
        }
    }

    public static void main(String[] args) {
        Student student = new Student();
        StudentManagement studentManagement = new StudentManagement();
        studentManagement.addStudent(student);
        student.setGroup("A");
        System.out.print(studentManagement.studentsByGroup());
    }

}
