import java.util.ArrayList;
import java.util.List;

public class StudentManagement {

    public Student[] students;
    public int n;
    public StudentManagement() {
        this.students = new Student[100];
        this.n = 0;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        if (n < 100) students[n++] = newStudent;
    }

    public String studentsByGroup() {
        StringBuilder my_list = new StringBuilder("");
        List<String> my_groups = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (!my_groups.contains(students[i].getGroup())) {
                my_groups.add(students[i].getGroup());
                my_list.append(students[i].getGroup());
                my_list.append("\n");
                my_list.append(students[i].getInfo());
                my_list.append("\n");
                if (i < n - 1)
                    for (int j = i + 1; j < n; j++) {
                        if (sameGroup(students[i], students[j])) {
                            my_list.append(students[j].getInfo());
                            my_list.append("\n");
                        }
                    }
            }
        }
        return my_list.toString();
    }

    public void removeStudent(String id) {
        for (int i = 0; i < n; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < n - 1; j++) {
                    students[j] = students[j+1];
                }
                n--;
                break;
            }
        }
    }
}
