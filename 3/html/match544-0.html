public class Student {

    private String name;
    private String id;
    private String group;
    private String email;

    public String getGroup() { return group; }

    public String getName() { return name; }

    public String getId() { return id; }

    public String getEmail() { return email; }

    public void setName(String name)
    {
        this.name = name;
    }

    public void setGroup(String group)
    {
        this.group = group;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public void setId(String id) {
        this.id = id;
    }

    public Student()
    {
        name = "Student";
        id = "000";
        group = "K62CB";
        email = "uet@vnu.edu.vn";
    }

    public Student(String name, String id, String email)
    {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    public Student(Student s)
    {
        name = s.getName();
        id = s.getId();
        group = s.getGroup();
        email = s.getEmail();
    }

    public String getInfo() 
    {
        String info = name + " - " + id + " - " + group + " - " + email;
        return info;
    }
}

public class StudentManagement {

    Student[] students = new Student[100];
    int index = 0;

    public StudentManagement() {}

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        if (index < 100)
            students[index++] = newStudent;
    }

    public String studentsByGroup() {
        String st = "";
        boolean[] checked = new boolean[index];
        for (int i = 0; i < index; i++)
            checked[i] = false;
        for (int i = 0; i < index; i++)
        {
            if (!checked[i]) {
                String gr = students[i].getGroup();
                st = st + gr + "\n";
                for (int j = 0; j < index; j++)
                {
                    if (students[j].getGroup().equals(gr))
                    {
                        st = st + students[j].getInfo() + "\n";
                        checked[j] = true;
                    }
                }
            }
        }
	    return st;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < index; ++i)
        {
            if (students[i].getId().equals(id))
            {
                for (int j = i; j < index; j++) {
                    students[j] = students[j + 1];
                }
                students[--index] = null;
                i = -1;
            }
        }
    }

    public static void main(String[] args) {


        Student s2 = new Student("Nguyen Van E", "17020002", "17020001@vnu.edu.vn");
        s2.setGroup("K64CA");

        Student s4 = new Student("Nguyen Van C", "17020004", "17020001@vnu.edu.vn");
        Student s3 = new Student("Nguyen Van B", "17020003", "17020001@vnu.edu.vn");
        s3.setGroup("K63CB");
        Student s5 = new Student("Nguyen Van B", "17020005", "17020001@vnu.edu.vn");
        s5.setGroup("K63CB");

        StudentManagement sm = new StudentManagement();

        sm.addStudent(s1);
        sm.addStudent(s1);
        sm.addStudent(s2);
        sm.addStudent(s1);
        sm.addStudent(s1);
        sm.addStudent(s2);
        sm.addStudent(s1);

        System.out.println(sm.studentsByGroup());
        System.out.println("------------");

        String test1 = new String("17020001");

        sm.addStudent(s3);

    }
}
