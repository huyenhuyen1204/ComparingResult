class StudentManagement {
    private Student[] students = new Student[100];
    private int top=-1;

    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup()))
            return true;
        return false;
    }
    public void addStudent(Student newStudent){
         top++;
         students[top] = newStudent;
    }
    public String studentsByGroup(){
         String[] gr = new String[100];
         int l=0,kt=0;
         gr[0]=students[0].getGroup();
         for(int i=1;i<=top;i++){
             kt=0;
             for(int j=0;j<=l;j++){
                 if(students[i].getGroup().equals(gr[j]))
                     kt=1;
             }
             if(kt==0) {
                 l++;
                 gr[l]=students[i].getGroup();
             }
         }
         String print="";
         
for(int i=0;i<=l;i++){
             print+=gr[i]+"\n";
             for(int j=0;j<=top;j++){
                 if(students[j].getGroup().equals(gr[i]))
                     print+=students[j].getInfo()+"\n";
             }
         }
         return print;
    }
    public void removeStudent(String id){
        if(students[top].getId().equals(id)){
            top--;
        }
         
for (int i=0;i<=top;i++)
             if(students[i].getId().equals(id)){
                 for(int j=i;j<=top;j++) {
                     students[j] = students[j + 1];
                 }
                 top--;
             }
    }
}
