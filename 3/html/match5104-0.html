public class Student {
    private String name, id, group, email;

    /**
     * Getter cho name, id, group, email
     */

    public String getName() {
        return name;
    }

    public String getId() {
        return id;
    }

    public String getGroup() {
        return group;
    }

    public String getEmail() {
        return email;
    }

    /**
     * Setter cho name, id, group, email
     */
    public void setName(String name) {
        this.name = name;
    }

    public void setId(String id) {
        this.id = id;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    /**
     * Phuong thuc khoi tao khong co tham so
     */
    public Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    /**
     * Phuong thuc khoi tao co tham so
     * @param name
     * @param id
     * @param email
     */
    public Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.group = "K62CB";
        this.email = email;
    }
    /**
     * Phuong thuc khoi tao co tham so
     * @param name
     * @param group
     * @param id
     * @param email
     */
    public Student(String name, String id, String group, String email) {
        this.name = name;
        this.id = id;
        this.group = group;
        this.email = email;
    }
    /**
     * Phuong thuc khoi tao sao chep
     * @param s
     */
    public Student(Student s) {
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

    /**
     * Getter cho info
     */
    public String getInfo(){
        return getName() + " - " + getId() + " - " + getGroup() + " - " + getEmail();
    }
}

/**

 * @author Nguyen Quang Huy
 * @since 7/7/2019
 */
public class StudentManagement {
    Student[] students = new Student[100];
    int length = 0;
    /**
     * Ham kiem tra hai sinh vien cung lop
     * @param s1
     * @param s2
     * @return true hoac false
     */
    public static boolean sameGroup(Student s1, Student s2){
       return s1.getGroup().equalsIgnoreCase(s2.getGroup());
    }

    /**
     * Them hoc sinh vao danh sach hoc sinh
     * @param newStudent
     */
    public void addStudent(Student newStudent){
        students[length++] = newStudent;
    }

    /**
     * Sap xep hoc sinh theo lop va thu tu
     * @return danh sach hoc sinh
     */
    public String studentsByGroup(){
        String temp = new String();
        boolean visited[] = new boolean[length];
        for (int i = 0; i < length; i++){
            visited[i] = false;
        }

        for (int i = 0; i < length; i++){
            if(!visited[i])
                temp += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
            for (int j = i + 1; j < length; j++){
                if(sameGroup(students[i], students[j])) {
                    visited[j] = true;
                    temp += students[j].getInfo() + "\n";
                }
            }
        }

        return temp;
    }

    /**
     * Xoa hoc sinh ra khoi danh sach
     * @param id
     */
    public void removeStudent(String id){
        for (int i = 0; i < length; i++){
            if(students[i].getId().equals(id)) {
                length = length - 1;
                for (int j = i; j < length; j++) {
                    i++;
                    students[j] = students[i];
                }
                break;
            }
        }
    }

    public static void main(String[] args) {

    }
}
