import java.util.ArrayList; // import the ArrayList class
public class StudentManagement {
    public Student[] students;
    private int count = 0;
    StudentManagement(){
        students = new Student[100];

    }
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup()) ) return true;
        else return false;
    }
    public void addStudent(Student newStudent){
        students[count] = newStudent;
        count++;
    }

    public String studentsByGroup(){
        String result = "";
        int gr = 0;
         ArrayList<String> group = new ArrayList<String>(100);
        for(int i = 0; i < count; i++){
            gr++;
            group.add(students[i].getGroup()) ;
        }
        for(int i = 0 ; i < gr; i++){

            for (int j = 0; j < gr;j++){
                if(i != j && group.get(i).equals(group.get(j))){
                    gr--;
                    if(i > j) group.remove(i);
                    else group.remove(j);

                }
            }
        }
        result += group.get(0);
        for(int i = 0 ; i < gr; i++){
            if(i != 0 ) result =  result + "\n" +  group.get(i);
            //result = result + "\n" + students[0].getInfo();
            for(int j = 0; j < count; j++){

                if(students[j].getGroup().equals(group.get(i)) )result = result + "\n" + students[j].getInfo();
            }
        }
        return result;
    }

    public void removeStudent(String id){
        for(int i = 0; i < count; i++){
            if(students[i].getId().equals(id)) {
              for(int j = i; j < count;j++){
                 students[j] = students[j + 1];
              }
              count--;
            };
        }
    }
    public static void main(String[] args){
        StudentManagement sm = new StudentManagement();
        Student s = new Student("Nguyen Van An","17020001","17020001@vnu.edu.vn");
        Student v = new Student("Nguyen Van B","17020002","17020002@vnu.edu.vn");
        Student a = new Student("Nguyen Van C","17020003","17020003@vnu.edu.vn");
        Student b = new Student("Nguyen Van D","17020004","17020004@vnu.edu.vn");
        //Student a = new Student("a", "0" ,"fa");
        sm.addStudent(s);
        sm.addStudent(v);
        a.setGroup("K63");
        sm.addStudent(a);
        sm.addStudent(b);

                System.out.println(sm.studentsByGroup());
        sm.removeStudent(v.getId());
        System.out.println(sm.studentsByGroup());
    }
}
