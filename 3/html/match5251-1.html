/**

 * @author Pham Quyen
 */
public class StudentManagement {

    static Student[] students = new Student[100];
    static int num = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        boolean rs = false;
        if (s1.getGroup().equals(s2.getGroup())) {
            rs = true;
        }
        return rs;
    }

    public static void addStudent(Student newStudent) {
        boolean exist = false;
        for (int i = 0; i < num; i++) {
            if (students[i].getInfo().equals(newStudent.getInfo())) {
                exist = true;
                break;
            }
        }
        if (!exist) {
            students[num] = newStudent;
            num++;
        }
    
}

    public static String studentsByGroup() {
        String rs = "";
        String[] groups = new String[10];
        groups[0] = students[0].getGroup();
        int countGroups = 1;
        for (int i = 1; i < num; i++) {
            boolean exist = false;
            for (int j = 0; j < countGroups; j++) {
                if (students[i].getGroup().equals(groups[j])) {
                    exist = true;
                
}
            }
            if (!exist) {
                groups[countGroups] = students[i].getGroup();
                countGroups++;
            }
        }
        for (int m = 0; m < countGroups; m++) {
            
rs = rs + groups[m] + "\n";
            for (int n = 0; n < num; n++) {
                if (students[n].getGroup().equals(groups[m])) {
                    rs = rs + students[n].getInfo() + "\n";
                }
            }
        
}
        return rs;
    }

    public static void removeStudent(String id) {
        for (int i = 0; i < num; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < num - 1; j++) {
                    students[j] = students[j + 1];
                }
                num--;
                students[num] = null;
                break;
            }
        }
    }

    public static void main(String[] args) {
        Student a = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        a.setGroup("K62CC");
        Student b = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        b.setGroup("K62CC");
        Student c = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        Student d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");

        addStudent(a);
        addStudent(b);
        addStudent(c);
        addStudent(d);

        System.out.println(studentsByGroup());
    }
}
