public class StudentManagement {
    Student[] students = new Student[100];
    int student_size = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().compareTo(s2.getGroup()) == 0;
    }
    public void addStudent(Student newStudent) {
        students[student_size] = newStudent;
        student_size++;
    }
    public void removeStudent(String id) {
        for (int i = 0; i < student_size; i++) {
            if (students[i].getId().compareTo(id) == 0) {
                student_size--;
                for (int j = i; j < student_size; j++) {
                    students[j] = students[j + 1];
                }
            }
        }
    }
    public String studentsByGroup() {
        boolean[] printed_student = new boolean[student_size];
        String current_class;
        String output = "";
        for (int i = 0; i < printed_student.length; i++) {
            printed_student[i] = false;
        }
        for (int i = 0; i < student_size; i++) {
            if (printed_student[i] == false) {
                current_class = students[i].getGroup();
                output += current_class + "\n";
                for (int j = i; j < student_size; j++) {
                    if (sameGroup(students[i], students[j])) {
                        output += students[j].getInfo() + "\n";
                        printed_student[j] = true;
                    }
                }
            }
        }
    return output;
    }
}
