public class StudentManagement {
		public static boolean sameGroup(Student s1, Student s2){
			return s1.getGroup().equals(s2.getGroup());
		}

		public Student students[] = new Student[100];
		
		public int sizeArray() {
			int i = 0;
			while (students[i] != null) i++;
			return i;
		}
		
		public void addStudent(Student newStudent){
			int StudentSize = sizeArray();
			students[StudentSize] = new Student(newStudent);
		}

		public String studentsByGroup(){
			int n = sizeArray();
			int m = n;
			String[] listGroup = new String[n];
			
			for(int i=0;i<n;i++){
				listGroup[i] = 
students[i].getGroup();
			}

			for(int i=0;i < n - 1;i++)
					for(int j=i + 1;j < n;j++)
						if(listGroup[i].equals(listGroup[j])) {
							for (int k = j; k < n -1 ; k++)
								listGroup[k] = listGroup[k+1]; 
							n--;
						}
			
			String list = "";
			
for(int i=0;i<n;i++) {
				list += listGroup[i] + "\n";
				for(int j=0;j<m;j++)
					if(listGroup[i].equals(students[j].getGroup()))
						list += students[j].getInfo() + '\n';
			}
			
			return list;
		}

		public void removeStudent(String id){
			int stusize = sizeArray();
			
for(int i=0;i<stusize - 1;i++)
				if(students[i].getId().equals(id))
					for(int j=i; j < stusize; j++)
						students[j] = students[j+1];
		}
}
