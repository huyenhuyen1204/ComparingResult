class StudentManagement {
    public static Student[] students = new Student[100];
    public static int studentCount = 0;
    
    public static boolean sameGroup(Student s1, Student s2){
        return s2.getGroup().equals(s1.getGroup());
    }
    
    public static void addStudent(Student newStudent){
        students[studentCount] = newStudent;
        studentCount ++;
    }
    
    public static String studentsByGroup(){
        String studentList = "";
        boolean studentCheck[] = new boolean[studentCount];
        
        for(int i=0 ; i<studentCount ; i++) {
            for(int j = i+1 ; j<studentCount ; j++) {
                if(students[j].getId().equals(
students[i].getId())) {
                    for(int k = j ; k<studentCount ; k++) {
                        students[k] = students[k+1];
                        studentCount --;
                    }
                }
            }
        }
        
        for(int i=0 ; i<studentCount ; i++){
            if(studentCheck[i] == false){
                studentList += students[i].getGroup() + "\n";
                studentList += students[i].getInfo()+ "\n";
                studentCheck[i] = true;
                for(int j=i+1 ; j<studentCount ; j++){
                    if(students[j].getGroup().equals(students[i].getGroup())) {
                        if(students[j].getId().equals(students[i].getId())== false){
                            studentList += students[j].getInfo()+ "\n";
                        }
                        studentCheck[j] = true;
                    }
                }
                
            }
        }
        return studentList;
    }
    
    public static void removeStudent(String id) {
        
for(int i=0; i<studentCount ; i++){
            if(students[i].getId().equals(id)) {
                for(int j=i ; j<studentCount ; j++) {
                    students[j] = students[j+1];
                }
                studentCount --;
            }
        }
    }
    
    public static void main(String[] args) 
    {
    	Student s1 = new Student("Nguyen Van An", "1701", "1701");
    	Student s2 = new Student("Nguyen Van B", "1702", "1702");
    	Student s3 = new Student("Nguyen Van C", "1703", "1703");
    	Student s4 = new Student("Nuyen Van D", "1704", "1704");
    	
    	s1.setGroup("K62CC");
    	s2.setGroup("K62CC");
    	

    	addStudent(s1);
    	addStudent(s2);
    	addStudent(s3);
    	addStudent(s2);
    	
    	addStudent(s4);
    
    	
        removeStudent("1702");
//        for(int i = 0; i < studentCount ; i++){
//            System.out.println(students[i].getInfo());
//        }
   	
    	System.out.println(studentsByGroup());
    }
}
