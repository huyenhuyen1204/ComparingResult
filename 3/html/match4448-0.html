public class StudentManagement {
    //them thuoc tinh mang students
    public Student students[] = new Student[100];
    //tao 1 bien dem
    private byte count = 0;

    //ham khoi tao lop
    public StudentManagement(){
    }

    //kiem tra 2 student cung lop
    public static boolean sameGroup(Student s1, Student s2){
        if (s1.getGroup().equals(s2.getGroup()))
            return true;
        else
            return false;
    }

    //them 1 student vao mang
    public void addStudent(Student newStudent){
        students[count] = new Student(newStudent);
        count++;
    }

    public String studentsByGroup() {
        byte check[] = new byte[count];
        //check = 1 -> da duyet, check = 0 -> chua duyet
        for (int i = 0; i < count; i++){
            check[i] = 0;
        }

        if (count == 0)
            return "";
        else {
            String result = "";
            byte currentPosition = 0;
            for (byte i = 0; i < count; i++)
            {
                if (check[i] == 0)
                {
                    currentPosition = i;
                    if (currentPosition == 0){
                        result += students[i].getGroup();
                    } else {
                        result += "\n" + students[i].getGroup();
                    }
                    for (byte j = i; j < count; j++)
                    {
                        if (sameGroup(students[j], students[currentPosition])){
                            result += "\n" + students[j].getInfo();
                            check[j] = 1;
                        }
                    }
                }
            }
            return result;
        }
    }

    public void removeStudent(String ID){
        for (byte i = 0; i < count; i++)
        {
            if (students[i].getId().equals(ID))
            {
                for (byte j = i; j < count; j++)
                    students[j] = students[j+1];
                count--;
                i--;
            }
        }
    }
