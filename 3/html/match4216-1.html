class StudentManagement {
    private Student[] students;
    private int numStudents;

    public StudentManagement() {
        students = new Student[100];
        numStudents = 0;
    }

    public void addStudent(Student newStudent) {
        students[numStudents++] = newStudent;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < numStudents; ++i)
            if (students[i].getId().compareTo(id) == 0) {
                for (int j = i; j < numStudents - 1; ++j) {
                    students[j] = students[j + 1];
                }
                students[--numStudents] = null;
                break;
            }
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().compareTo(s2.getGroup()) == 0;
    }

    public String studentsByGroup() {
        StringBuilder studentList = new StringBuilder();
        String[] groups = new String[100];
        int numGroups = 0;
        for (int i = 0; i < numStudents; ++i) {
            boolean check = false;
            for (int j = 0; j < numGroups; ++j)
                if (students[i].getGroup().compareTo(groups[j]) == 0) {
                    check = true;
                    break;
                }
            if (!check) {
                
groups[numGroups++] = students[i].getGroup();
                studentList.append(groups[numGroups - 1]);
                studentList.append('\n');
                for (int j = i; j < numStudents; ++j)
                    if (students[j].getGroup().compareTo(groups[numGroups - 1]) == 0) {
                        studentList.append(students[j].getInfo());
                        studentList.append('\n');
                    }
            }
        }
        return studentList.toString();
    }

    public static void main(String[] args) {
        Student student17020001 = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        student17020001.setGroup("K62CC");

        Student student17020002 = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        student17020002.setGroup("K62CC");

        Student student17020003 = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        student17020003.setGroup("K62CB");

        Student student17020004 = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
        student17020004.setGroup("K62CB");

        Student dumpStudent = new Student();

        StudentManagement studentManagement = new StudentManagement();
        studentManagement.addStudent(student17020001);
        studentManagement.addStudent(student17020002);
        studentManagement.addStudent(dumpStudent);
        studentManagement.addStudent(student17020003);
        studentManagement.addStudent(student17020004);
        studentManagement.removeStudent(dumpStudent.getId());
        System.out.println(studentManagement.studentsByGroup());
    }
}
