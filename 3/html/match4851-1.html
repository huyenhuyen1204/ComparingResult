public class StudentManagement {
    Student[] students = new Student[100];
    int numsofstudent;

    StudentManagement() {
        numsofstudent = 0;
    }
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public void addStudent(Student newStudent){
        students[numsofstudent] = newStudent;
        numsofstudent++;
    
}

    public String studentsByGroup() {
        boolean[] groupStudent = new boolean[100];
        for (int i = 0; i < numsofstudent; i++) {
            groupStudent[i] = false;
        }

        String list = "";

        for (int i = 0; i < numsofstudent; i++) {
            if (groupStudent[i] == false) {
                list += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
                groupStudent[i] = true;

                for (int j = i + 1; j < numsofstudent; j++) {
                    if (sameGroup(students[i], students[j]) == true) {
                        list += students[j].getInfo() + "\n";
                        groupStudent[j] = true;

                    }
                }
            }
        }
        return list;
    }

    public void removeStudent(String id) {
        
for (int i = 0; i < numsofstudent; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < numsofstudent - 1; j++) {
                    students[j] = students[j + 1];
                }
                numsofstudent--;
            }
        }
    }
}
