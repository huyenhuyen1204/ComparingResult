public class StudentManagement {

    // TODO: khai bao thuoc tinh students la array chua cac doi tuong thuoc lop Student (max. 100)
    private Student[] students = new Student[100];
    private Integer currentGroup = 0;
    private int top;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        Student temp = new Student(newStudent);
        boolean belongToNewGroup = false;
        for(Integer i = 0; i < top; ++i)
            if (students[i].getAlive() && sameGroup(students[i], newStudent)) {
                temp.setGroupNumber(students[i].getGroupNumber());
                belongToNewGroup = true;
            }
        if (!belongToNewGroup){
            temp.setGroupNumber(currentGroup);
            currentGroup += 1;
        }
        students[top] = temp;
        top += 1;
    }

    public String studentsByGroup() {
        String ret = "";
        for(Integer i = 0; i < currentGroup; ++i){
            boolean printed = false;
            for(int j = 0; j < top; ++j)
                if (students[j].getAlive() && students[j].getGroupNumber().equals(i)){
                    if (!printed){
                        ret += students[j].getGroup() + "\n";
                        printed = true;
                    }
                    ret += students[j].getInfo() + "\n";
                }
        }
        return ret;
    }

    public void removeStudent(String id) {
        for(int i = 0; i < top; ++i)
            if (students[i].getId().equals(id))
                students[i].setAlive(false);
    }

    public static void main(String[] args){
        StudentManagement tmp = new StudentManagement();

        Student a = new Student();
        a.setGroup("K63C-CLC");
        tmp.addStudent(a);
        tmp.addStudent(new Student());
        tmp.addStudent(new Student());
        tmp.addStudent(new Student());
        Student b = new Student();
        b.setGroup("adsaf");
        tmp.addStudent(b);
        tmp.removeStudent("001");

        System.out.print(tmp.studentsByGroup());

    }

}
