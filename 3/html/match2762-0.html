public class StudentManagement{
	Student[] students=new Student[100];
	public static boolean sameGroup(Student s1, Student s2){
		return s1.getGroup().equals(s2.getGroup()); 
	}
	int size=0;
	public void addStudent(Student newStudent){
		students[size]= new Student(newStudent);
		size++;
    }

    public String studentsByGroup() {
    	String s="";
    	String g=students[0].getGroup();
    	s=s+g+"\n"+students[0].getInfo()+"\n";
    	students[0].setGroup("");
    	for(int i=1;i<size;i++){
        	if(students[i].getGroup().equals("")){
        		continue;
        	}
        	if(students[i].getGroup()!=g){
        		g=students[i].getGroup();
        		s=s+g+"\n";
    		}
    		for(int j=i;j<size;j++){
       			if(students[j].getGroup().equals(g)){
        			s=s+students[j].getInfo()+"\n";
        			students[j].setGroup("");
        		}
        	}
        }
		return s;
    }
