public class StudentManagement {
    int doDai = 0;
    Student[] students = new Student[100];

    public static 
boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent(Student newStudent) {
        students[doDai++] = newStudent;
    }

    public String studentsByGroup() {
        String ketqua = "";
        int[] a = new int[doDai];
        for (int i = 0; i < doDai; i++) a[i] = 1;
        for (int i = 0; i < doDai; i++)
            for (int j = i + 1; j < doDai; j++) if (students[j].getGroup().equals(students[i].getGroup())) a[j] = 0;
        for (int i = 0; i < doDai; i++) {
            if (a[i] == 1) {
                ketqua += students[i].getGroup() + "\n";
                for (int j = 0; j < doDai; j++)
                    if (students[j].getGroup().equals(students[i].getGroup()))
                        ketqua += students[j].getInfo() + "\n"
;
            }
        }
        return ketqua;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < doDai; i++)
            if (students[i].getId().equals(id)) {
                for (int j = i; j < doDai; j++) students[j] = students[j + 1];
                doDai--;
            }
    }

    public static void main(String[] args) {
        Student a = new Student("nguyenvanan", "100", "nvagmailcom");
        Student b = new Student("nguyenvanbn", "101", "nvbgmailcom");
        Student c = new Student("nguyenvancn", "103", "nvcgmailcom");
        Student d = new Student("nguyenvandn", "104", "nvdgmailcom");
        Student e = new Student();
        Student f = new Student();
        StudentManagement moi = new StudentManagement();
        moi.addStudent(a);
        moi.addStudent(e);
        moi.addStudent(c);
        moi.addStudent(d);
        moi.addStudent(b);
        moi.addStudent(f);
        System.out.print(moi.studentsByGroup());
        // moi.removeStudent("101");
        //for(int i=0;i<moi.doDai;i++)
        //System.out.println(moi.students[i].getInfo());

    }
}
