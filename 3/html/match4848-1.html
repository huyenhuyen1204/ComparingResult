class StudentManagement {
    static Student[] students = new Student[100];
    int sosv;

    public static boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent(Student newStudent){
        if (sosv < 100)
            this.students[sosv] = new Student(newStudent);
        sosv++;
    }

    public String studentsByGroup() {
        boolean[] value = new boolean[sosv];
            for (int i=0; i<sosv; i++){
                value[i] = false;
            }
        String in = "";
        for(int i=0;i<sosv;i++)
        {
            if(value[i] == false)
            {
                String lop = students[i].getGroup();
                in = in + lop + "\n";
                for (int j = i; j < sosv; j++) {
                        if (students[j].getGroup().equals(lop)) {
                            in = in + students[j].getInfo() + "\n";
                            value[j] = true;
                        }
                }
            }
        }
        return in;
    }

    public void removeStudent(String id) {

        int pos = 0;
        for (int i = 0; i < sosv; i++) {
            if (students[i].getId().equals((id))) {
                pos = i;
                break;
            }
        }

        for (int i = pos; i < sosv; i++) {
            students[i] = students[i + 1];
        }
        sosv--;
    }

    }
    
}
