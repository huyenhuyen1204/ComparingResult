public class StudentManagement {
    int present_quantity;
    Student[] students= new Student[100];
    StudentManagement(){
        
present_quantity = 0;
    }

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }
    public void addStudent(Student newStudent){
        students[present_quantity] = newStudent;
        present_quantity++;
    }
    public String studentsByGroup(){
        int[] check_group = new int[100];
        int count_group = 0;
        String s="";
        for(int i = 0; i < present_quantity; i++){
            count_group = 0;
            for(int j = i; j < present_quantity; j++){
                if(check_group[j] != 1){
                    if(sameGroup(students[i], students[j]) == true){
                        check_group[j]=1;
                        if(count_group < 1){
                            count_group++;
                            s = s + students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
                        } else s = s + students[j].getInfo() + "\n";
                    }

                }
            }
        
}
        return s;
    }
    public void removeStudent(String id){
        for(int i = 0; i < present_quantity; i++){
            if(students[i].getId().equals(id)){
                for(int j = i; j<present_quantity-1; j++){
                    students[j] = students[j+1];
                }present_quantity--;
                break;
            }
        }
    }
}
