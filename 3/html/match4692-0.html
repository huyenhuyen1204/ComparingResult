public class  StudentManagement {
    public static Student[] students = new Student[100];
    static int soSV = 0;

    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().compareTo(s2.getGroup()) == 0){
            return true;
        }
        else return false;
    }

    public void addStudent(Student newStudent){
        soSV++;
        students[soSV-1] = new Student(newStudent);
    }

    public static String studentsByGroup(){
        String s = "";
        int d = 0;
        String[] sGroup = new String[100];
        for(int i = 0; i < soSV; i++){
            sGroup[d] = students[i].getGroup();
            d++;
        }

        for (int i = 1; i < d; i++) {
            for (int j = 0; j < i; j++) {
                if (sGroup[i].equals(sGroup[j])) {
                    for (int t = i; t < d; t++) {
                        sGroup[t] = sGroup[t + 1];
                        d--;
                        i--;
                    }
                }
            }
        }

        for(int i = 0; i < d; i++){
            s +=  sGroup[i] + "\n";
            for(int j = 0; j < soSV; j++){
                if(students[j].getGroup().equals(sGroup[i])){
                    s += students[j].getInfo() + "\n";
                }
            }

        }
        return s;
    }

    public void removeStudent(String id){
        for(int i = 0; i < soSV; i++){
            if(students[i].getId().compareTo(id) == 0) {
                for(int j = i; j < soSV; j++){
                    students[i] = students[i+1];
                    if(j == soSV - 1) soSV--;
                }
                break;
            }
        }
    }

}
