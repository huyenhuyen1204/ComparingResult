public class Student {
    private  String name;
    private  String id;
    private  String group;
    private  String email;
    public void setName(String name){
        this.name=name;
    }
    public String getName(){
        return name;
    }

    public void setId(String id){
        this.id=id;
    }
    public String getId(){
        return id;
    }

    public void setGroup(String group){
        this.group=group;
    }
    public String getGroup(){
        return group;
    }

    public void setEmail(String email){
        this.email=email;
    }
    public String getEmail(){
        return email;
    }
    Student() {
        setName("Student");
        setId("000");
        setGroup("K62CB");
        setEmail("uet@vnu.edu.vn");
    }


    Student(String name, String id, String email) {
        setName(name);
        setId(id);
        setGroup("K62CB");
        setEmail(email);
    }


    Student(Student s) {
        name = s.getName();
        id = s.getId();
        group = s.getGroup();
        email = s.getEmail();
    }

    String getInfo()
    {
        String result = name + " - " + id + " - " + group + " - " + email;
        return result;

    }

}

public class StudentManagement {

    private static Student students[] = new Student[100];

    private static int count = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public static void addStudent(Student newStudent) {
        students[count] = newStudent;
        count++;
    }

    public static String studentsByGroup() {
        int[] check = new int[count];
        String sTB = "";
        for (int i = 0; i < count; i++) {
            check[i] = 1;
        }
        for (int i = 0; i < count; i++) {
            if (check[i] == 1) {
                String group = students[i].getGroup();
                sTB += group + "\n" + students[i].getInfo() + "\n";

                check[i] = 0;

                for (int j = i + 1; j < count; j++) {
                    if (students[j].getGroup().equals(group)) {
                        sTB += students[j].getInfo() + "\n";
                        check[j] = 0;
                    }
                }
            }
        }
        return sTB;
    }

    public static void removeStudent(String id) {
        int m = 0;
        while (m < count) {
            if (id.equals(students[m])) {
                for (int i = m; i < count; i++) {
                    students[i] = students[i + 1];
                }
                count--;
            }
            m++;
        }
        for (int i = 0; i < count; i++) {
            String id1 = students[i].getId();
            for (int j = i + 1; j < count; j++) {
                if (id.equals(students[j].getId())) {
                    for (int k = j; k < count; k++) {
                        students[k] = students[k + 1];
                    }
                    count--;
                }
            }
        }
        count--;
    }
