class StudentManagement {
    public static boolean sameGroup(Student s1, Student s2)
    {
        if(s2.getGroup().equals(s1.getGroup()))return true;
        else return false;
    }
     Student[] students= new Student[100];
     int number=0;
    public void addStudent(Student newStudent)
    {
        students[number]=newStudent;
        number++;
    }
    public String studentsByGroup()
    {
        int[] test=new int[number];
        for(int i=0;i<number;i++) test[i]=0;
        String newlist="";
        for(int i=0;i<number;i++)
        {
            if(test[i]==0)
            {
                String g=students[i].getGroup();
                newlist+=g+"\n";
                newlist+=students[i].getInfo()+"\n";
                test[i]=1;
                for(int j=i+1;j<number;j++)
                {
                    if(students[j].getGroup().equals(g)&&test[j]==0)
                    {
                        newlist+=students[j].getInfo();
                        newlist+="\n";
                        test[j]=1;
                    }
                }

            }
        }
        return newlist;

    }
    public void removeStudent(String id)
    {
        
for(int i=0;i<number;i++)
        {
            if(students[i].getId().equals(id))
            {
                for(int j=i;j<number-1;j++)
                students[j]=students[j+1];
                number--;
            }
        }
    }
}
