public 
class StudentManagement {

    private static final int MAX = 100;
    private int size = 0;
    private Student[] students = new Student[MAX];

    public static boolean sameGroup(Student s1, Student s2) {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student newStudent) {
        if (size + 1 < MAX) {
            students[size] = newStudent;
            ++size;
        }
    }

    public String studentsByGroup() {
        if (size == 0) return "";
        StringBuilder result = new StringBuilder();
        boolean[] 
isOut = new boolean[size];
        String curGroup;
        for (int cur = 0; cur < size; cur++) {
            if (!isOut[cur]) {
                curGroup = students[cur].getGroup();
                result.append(curGroup).append("\n");
                for (int i = cur; i < size; ++i) {
                    if (!isOut[i] && curGroup.equals(students[i].getGroup())) {
                        result.append(students[i].getInfo()).append("\n");
                        isOut[i] = true;
                    }
                }
            }
        }
        return result.toString();
    }

    public void removeStudent(String id) {
        for (int i = 0; i < size; i++) {
            if (id.equals(students[i].getId())) {
                --size;
                if (size - i >= 0) System.arraycopy(students, i + 1, students, i, size - i);
            }
        }
    }
}
