public class StudentManagement {

    static Student[] students = new Student[100];
    static int countStudents = 0;
    String[] groups = new String[100];
    
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup()))
            return true;
        return false;
    }

    public static void addStudent(Student newStudent) {
        for(int i=0;i<countStudents;i++){
            if (students[i].getId().equals(newStudent.getId()))
                return;
        }
        students[countStudents] = newStudent;
        countStudents++;
    }

    public static String studentsByGroup() {
        String str = students[0].getGroup();
        int countGroup = 0;
        int NextGroup = countGroup;
        while (true) {
            for (int i = countGroup; students[i] != null; i++) {

                if (sameGroup(students[countGroup], students[i])) {
                    str = str + "\n" + students[i].getInfo();
                } else if (NextGroup == countGroup && !str.contains(students[i].getGroup())) {
                    NextGroup = i;
                }
            }
            if (countGroup == NextGroup) {
                break;
            } else {
                countGroup = NextGroup;
                str = str + "\n" + students[countGroup].getGroup();
            }
        }
        return str;
    }

    public static void removeStudent(String id) {
        Student[] temp = new Student[100];
        int counttemp = 0;
        for (int i = 0; i < countStudents; i++) {
            if (!students[i].getId().equals(id)) {
                temp[counttemp] = students[i];
                counttemp++;
            }
        }
        students = temp.clone();
        countStudents--;
    }



    public static void main(String[] args) {
            Student a = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
           Student b = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
            Student c = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
            Student d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");

            a.setGroup("K63CC");
            b.setGroup("K63CC");

           addStudent(a);
            addStudent(b);
            addStudent(c);
            addStudent(d);

            for(int i=0; i<countStudents; i++){
                System.out.println(students[i].getInfo());
            }
            System.out.println("\n");
            addStudent(a);
            removeStudent("17020003");
            for(int i=0; i<countStudents; i++){
                System.out.println(students[i].getInfo());
            }
    }
}
