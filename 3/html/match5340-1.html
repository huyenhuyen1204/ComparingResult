public class StudentManagement {

    Student[] students = new Student[100];
    int index = 0;

    public 
StudentManagement() {}

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup()); // xoa dong nay sau khi cai dat
    }

    public void addStudent(Student newStudent){
        if (index < 100)
            students[index++] = newStudent;
    }

    public String studentsByGroup() {
        String s = "";
        boolean[] checked = new boolean[index];
        for (int i = 0; i < index; i++)
            checked[i] = false;
        for (int i = 0; i < index; i++)
        {
            if (!checked[i]) {
                String 
g = students[i].getGroup();
                s = s + g + "\n";
                for (int j = 0; j < index; j++)
                {
                    if (students[j].getGroup().equals(g))
                    {
                        s = s + students[j].getInfo() + "\n";
                        checked[j] = true;
                    }
                }
            }
        }
	    return s;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < index; ++i)
        {
            if (students[i].getId().equals(id))
            {
                int j = i;
                for (; j < index; j++) {
                    
students[j] = students[j + 1];
                }
                students[--index] = null;
                i = -1;
            }
        }
    }

    public static void main(String[] args) {


        Student s7 = new Student("Nguyen Van E", "17020002", "17020001@vnu.edu.vn");
        s7.setGroup("K64CA");
        Student s1 = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        // Student s2 = new Student();
        // Student s3 = new Student();
        Student s4 = new Student("Nguyen Van C", "17020001", "17020001@vnu.edu.vn");
        Student s5 = new Student("Nguyen Van D", "17020001", "17020001@vnu.edu.vn");
        Student s6 = new Student("Nguyen Van B", "17020003", "17020001@vnu.edu.vn");
        s6.setGroup("K63CB");
        Student s8 = new Student("Nguyen Van B", "17020005", "17020001@vnu.edu.vn");
        s6.setGroup("K63CB");

        StudentManagement sm = new StudentManagement();

        sm.addStudent(s1);
        sm.addStudent(s5);
        sm.addStudent(s4);
        // sm.addStudent(s3);
        sm.addStudent(s7);
        sm.addStudent(s6);
        // sm.addStudent(s2); 
        sm.addStudent(s8);

        System.out.println(sm.studentsByGroup());

        System.out.println(sm.studentsByGroup());
    }
}
