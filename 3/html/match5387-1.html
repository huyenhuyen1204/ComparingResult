import java.util.ArrayList;
public class StudentManagement {
    private int n;
    Student[] students = new Student[100];
    
StudentManagement(){
        n=0;
    }
    public static boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent(Student newStudent){
        students[n]=newStudent;
        n++;
    }

    public String studentsByGroup() {
        String s="";
        int count=0;
        int i,j;
        ArrayList<String> str = new ArrayList<String>();
        for(i=0;i<n;++i){
            
if(students[i]!=null){
                if(str.contains(students[i].getGroup())==false){
                    str.add(students[i].getGroup());
                    count++;
                }
            }
        }
        for(i=0;i<count;++i){
            s=s+str.get(i) + '\n';
            for(j=0;j<n;++j){
                if(students[j]!=null){
                    if(
str.get(i).equals(students[j].getGroup())){
                        s=s+ students[j].getName()+" - "+students[j].getId()+" - "+students[j].getGroup()+" - "+students[j].getEmail()+'\n';
                    }
                }
            }
        }
        return s;
    }

    public void removeStudent(String id) {
        int i,j;
        for(i=0;i<n;++i){
            if(id.equals(students[i].getId())){
                for(j=i;j<n-1;++j) students[j]=students[j+1];
            }
        }
        n--;
    }

    public static void main(String[] args) {
        Student An = new Student("Nguyen Van An", "17020001","17020001@vnu.edu.vn");
        An.setEmail("K62CC");
        Student B = new Student("Nguyen Van B", "17020002","17020002@vnu.edu.vn");
        B.setEmail("K62CC");
        Student C = new Student("Nguyen Van C", "17020003","17020003@vnu.edu.vn");
        C.setEmail("K62CB");
        Student D = new Student("Nguyen Van D", "17020004","17020004@vnu.edu.vn");
        D.setEmail("K62CB");
        StudentManagement students = new StudentManagement();
        students.addStudent(C);
        students.addStudent(B);
        students.addStudent(An);
        students.addStudent(D);

        System.out.println(students.studentsByGroup());
        students.removeStudent("17020003");
        System.out.println(students.studentsByGroup());
    }
}
