public class Student {

    private String name;
    private String id;
    private String group;
    private String email;

    public String getName(){
        return name;
    }
    public void setName(String n){
        this.name = n;
    }
    public String getId(){
        return id;
    }
    public void setId(String i){
        this.id = i;
    }
    public String getGroup(){
        return group;
    }
    public void setGroup(String g){
        this.group = g;
    }
    public String getEmail(){
        return email;
    }
    public void setEmail(String e){
        this.email = e;
    }

    /**
     * Constructor 1
     */
    Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    /**
     * Constructor 2
     * @param n
     * @param sid
     * @param em
     */
    Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    /**
     * Constructor 3
     * @param s
     */
    Student(Student s) {
        this.name = s.getName();
        this.id = s.getId();
        this.group = s.getGroup();
        this.email = s.getEmail();
    }

    String getInfo() {
        return this.name+" - "+this.id+" - "+this.group+" - "+this.email;
    }
}

import java.util.Comparator;

public class StudentManagement {

    Student[] students = new Student[100];

    public static boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public void addStudent(Student newStudent){
        for(int i=0; i<100; i++){
            if(students[i] == null){
                students[i] = newStudent;
                break;
            }
        }
    }

    public String studentsByGroup() {

        int c1 = 0;
        for (int i = 0; i < students.length; i++) {
            if (students[i] != null) {
                c1++;
            }
        }
        String S = "";
        for (int i = 0; i < c1; i++) {
            int kt = 1;
            for (int j = 0; j < i; j++) {
                if (students[i].getGroup().equals(students[j].getGroup())) {
                    kt = 0;
                    break;
                }
            }
            if (kt == 1) {
                S = S + "\n" + students[i].getGroup() + "\n" + students[i].getInfo();
                for (int k = i + 1; k < c1; k++) {
                    if (students[i].getGroup().equals(students[k].getGroup())) {
                        S = S + "\n" + students[k].getInfo();
                    }
                }
            }
        }

        return S;
    }
