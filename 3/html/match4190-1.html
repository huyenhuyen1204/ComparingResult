public class StudentManagement {
     Student [] students = new Student [100] ;
    private  int n=0;

    public static boolean sameGroup(Student s1, Student s2)
    {
        if ((s1.getGroup()).equals(s2.getGroup()))
        {
            return true;
        }
        else
        {
            return false;
        }
    }

    public  void addStudent(Student newStudent)
    {
        students[n]= new Student();
        students[n]=newStudent;
        n++;
    }
    public  String studentsByGroup()
    {
        String[] StudentSameGroup = new String[100];
        int count=0,temp;
        StudentSameGroup[count]=students[0].getGroup();
        
for(int i=1;i<n;i++)
        {
            temp=1;
            for(int j=0;j<i;j++)
            {
                if (sameGroup(students[j],students[i]))
                {
                    temp=0;
                    break;
                }
            }
            if (temp==1)
            {
                count++;
                StudentSameGroup[count]=students[i].getGroup();
            }
        }

        String Students ="";

        
for(int i=0;i<=count;i++)
        {
            Students += StudentSameGroup[i] + "\n";
            for(int j=0;j<n;j++)
            {
                if((students[j].getGroup()).equals(StudentSameGroup[i]))
                {
                    Students += students[j].getInfo() + "\n";

                }
            }
        }
        return Students;

    }

    public void removeStudent(String id)
    {
        for(int i=0;i<n;i++)
        {
            if(students[i].getId().equals(id))
            {
                for(int j=i;j<n-1;j++)
                {
                    students[j]=students[j+1];
                }
                n--;

            }

        }
    }

    public static void main(String[] args)

    {
        StudentManagement s=new StudentManagement();

        Student s1 = new Student("Nguyen Van A","18020252","17020001@vnu.edu.vn");
        s1.setGroup("K62CC");
        Student s2 = new Student("Nguyen Van B","18020244","17020002@vnu.ude.vn");
        s2.setGroup("K62CC");
        Student s3 = new Student("Nguyen Van C","18020254","17020003@vnu.edu.vn");
        Student s4 = new Student("Nguyen Van D","18020242","17020004@vnu.edu.vn");


        s.addStudent(s1);
        s.addStudent(s2);
        s.addStudent(s3);
        s.addStudent(s4);

        System.out.println(s.studentsByGroup());

        s.removeStudent("18020252" );
        System.out.println(s.studentsByGroup());



    }




}
