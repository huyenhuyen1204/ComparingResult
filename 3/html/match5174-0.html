class Student {
    private String name, id, group, email;

    public Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    public Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    Student(Student s) {
        this.name = s.getName();
        this.id = s.getId();
        this.group = s.getGroup();
        this.email = s.getEmail();
    }

    public String getName() {
        return name;
    }
    public void setName(String name) {
        this.name = name;
    }

    public String getId() {
        return id;
    }
    public void setId(String id) {
        this.id = id;
    }

    public String getEmail() {
        return email;
    }
    public void setEmail(String email) {
        this.email = email;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    String getInfo() {
        return name + " - " + id + " - " + group + " - " + email;
    }
}
public class StudentManagement {
    Student[] students = new Student[100];
    int numsofstudent;

    StudentManagement() {
        numsofstudent = 0;
    }
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public void addStudent(Student newStudent){
        students[numsofstudent] = newStudent;
        numsofstudent++;
    }

    public String studentsByGroup() {
        boolean[] groupStudent = new boolean[100];
        for (int i = 0; i < numsofstudent; i++) {
            groupStudent[i] = false;
        }

        String list = "";

        for (int i = 0; i < numsofstudent; i++) {
            if (groupStudent[i] == false) {
                list += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
                groupStudent[i] = true;
                
                for (int j = i + 1; j < numsofstudent; j++) {
                    if (sameGroup(students[i], students[j]) == true) {
                        list += students[j].getInfo() + "\n";
                        groupStudent[j] = true;

                    }
                }
            }
        }
        return list;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < numsofstudent; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < numsofstudent - 1; j++) {
                    students[j] = students[j + 1];
                }
                numsofstudent--;
            }
        }
    }
}
