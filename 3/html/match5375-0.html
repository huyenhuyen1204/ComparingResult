import java.util.HashSet;
import java.util.Set;

public class StudentManagement {
    static Student [] arr_student=new Student[100];
    int soSV=0;
    public static boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup())) {
            return true;
        } else return false;
    }
    public void addStudent(Student newStudent){
        for(int a = 0; a < arr_student.length ; a++) {
            if (arr_student[a] == null) {
                arr_student[a] = newStudent;
                soSV++;
                break;
            }
        }
    }
    public String  studentsByGroup() {
        String mangsv="";
        String []group=new String[100];
        int dem=0;
        group[dem]=arr_student[0].getGroup();
        for(int a=1;a<soSV;a++)
        {
            for(int b=0;b<=dem;b++)
            {
               if(b==dem&&!group[b].equals(arr_student[a].getGroup()))
               {
                   dem++;
                   group[dem]=arr_student[a].getGroup();
                   break;
               }
               else if(group[b].equals(arr_student[a].getGroup())){
                   break;
               }
            }
        }
        for(int b=0;b<=dem;b++){
             mangsv += group[b];
            for(int a=0;a<soSV;a++)
            {
                if(arr_student[a].getGroup().equals(group[b])){
                    mangsv+="\n";
                    mangsv+=arr_student[a].getInfo();
                }
            }
            mangsv+="\n";
        }
        return mangsv;
    }
    public void removeStudent(String id){
        for(int a=0;a<soSV;a++) {
                if (arr_student[a].getId().equals(id)) {
                    for (int b = a+1; b < soSV; b++) {
                        arr_student[b-1] = arr_student[b];
                    }
                    soSV--;
                    break;
                }
            }
        }

    }



}
