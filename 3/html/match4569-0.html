public class Student {

    private String name;
    private String id;
    private String group;
    private String email;

    public Student() 
    {
        name = "Student";
        id = "000";
        group = "K62CB";
        email = "uet@vnu.edu.vn";
    }

    public Student(String name, String id, String email)
    {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    public Student(Student s)
    {
        name = s.name;
        id = s.id;
        group = s.group;
        email = s.email;
    }

    public String getInfo() 
    {
        String info = name + " - " + id + " - " + group + " - " + email;
        return info;
    }

    public String getGroup()
    {
        return group;
    }

    public String getName()
    {
        return name;
    }

    public String getId()
    {
        return id;
    }

    public String getEmail()
    {
        return email;
    }

    public void setName(String n)
    {
        name = n;
    }

    public void setGroup(String n)
    {
        group = n;
    }

    public void setId(String n)
    {
        id = n;
    }

    public void setEmail(String n)
    {
        email = n;
    }
}

public class StudentManagement {

    int index = 0;

    Student[] s = new Student[100];

    public StudentManagement() {}

    public static boolean sameGroup(Student s1, Student s2)
    {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent)
    {
        if (index < 100)
            s[index++] = newStudent;
    }

    public String studentsByGroup() {
        String str = "";
        boolean[] checkedGr = new boolean[index];
        for (int i = 0; i < index; i++)
            checkedGr[i] = false;
        for (int i = 0; i < index; i++)
        {
            if (!checkedGr[i]) 
            {
                String g = s[i].getGroup();
                str = str + g + "\n";
                for (int j = 0; j < index; j++)
                {
                    if (s[j].getGroup().equals(g))
                    {
                        str = str + s[j].getInfo() + "\n";
                        checkedGr[j] = true;
                    }
                }
            }
        }
	    return str;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < index; ++i)
        {
            if (s[i].getId().equals(id))
            {
                for (int j = i; j < index; j++) {
                    s[j] = s[j + 1];
                }
                s[--index] = null;
                i = -1;
            }
        }
    }

    public static void main(String[] args) {


        Student s1 = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        s1.setGroup("K64CA");
        Student s2 = new Student("Nguyen Van E", "17020002", "17020001@vnu.edu.vn");
        Student s3 = new Student("Nguyen Van B", "17020003", "17020001@vnu.edu.vn");
        s3.setGroup("K63CB");
        Student s4 = new Student("Nguyen Van C", "17020004", "17020001@vnu.edu.vn");
        Student s5 = new Student("Nguyen Van B", "17020005", "17020001@vnu.edu.vn");
        s5.setGroup("K63CB");

        StudentManagement StudManage = new StudentManagement();

        StudManage.addStudent(s1);
        StudManage.addStudent(s1);
        StudManage.addStudent(s2);
        StudManage.addStudent(s3);
        StudManage.addStudent(s4);
        StudManage.addStudent(s2);
        StudManage.addStudent(s5);

        System.out.println(StudManage.studentsByGroup());
        System.out.println("------------");

        String test1 = new String("17020001");

        StudManage.removeStudent(test1);

        StudManage.addStudent(s3);
        StudManage.removeStudent(new String("17020002"));

        System.out.println(StudManage.studentsByGroup());
    }
}
