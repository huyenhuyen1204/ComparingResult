    }

     Student(String name, String id, String email){
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    Student(Student s){
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

    public String getName(){
        return name;
    }
    public void setName(String name){
        this.name= name;
    }

    public String getId(){ return id;}
    public void setId(String id){
        this.id= id;
    }

    public String getGroup(){
        return group;
    }
    public void setGroup(String group){
        this.group= group;
    }

    public String getEmail(){
        return email;
    }
    public void setEmail(String email){
        this.email= email;
    }

    public String getInfo(){
        String Info = this.getName() +" - " + this.getId() + " - "+ this.getGroup() + " - " + this.getEmail();
        return Info;
    }
}

public class StudentManagement{
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public Student [] students = new Student[100];

    public int count(){
        int i = 0;
        while (students[i] != null){
            i++;
        }
        return i;
    }

    public void addStudent (Student newStudent) {
        int n = count();
        students[n] = newStudent;
    }

    public String studentsByGroup(){
        String [] s = new String [100];
        int n = count();
        int k = 1;
        if(n!=0){
            s[0] = students[0].getGroup();
            for(int i =1; i< n; i++) {
                int check = 0;
                for (int j = 0; j < k; j++) {
                    if (students[i].getGroup().equals(s[j])) {
                        check = 1;
                        break;
                    }
                }
                if (check == 0) {
                    s[k] = students[i].getGroup();
                    k++;
                }
            }
        }
        String res = "";
        for(int i = 0; i < k; i++){
            res += s[i];
            res += "\n";
            for(int j=0;j < n; j++){
                if(s[i].equals(students[j].getGroup())){
                    res += students[j].getInfo();
                    res +="\n";
                }
            }
        }
        return res;
    }
    public void removeStudent(String id){
        int n = count();
        int vtrixoa= 0;
        for(int i = 0; i < n; i++){
            if((students[i].getId()).equals(id))
                vtrixoa = i;
        }
        for(int k = vtrixoa; k < n-1; k++){
            students[k] = students[k+1];
        }
        students[n-1] = null;
    }
}
