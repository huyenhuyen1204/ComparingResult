import java.util.*;

public class StudentManagement {
    public int size;
    private Student[] students = new Student[100];

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        if (size < 99) {
            students[size] = newStudent;
            size++;
        }
    }

    public void removeStudent(String id) {
        if (size > 0) {
            for (int i = 0; i < size; i++)
                if (students[i].getId().equals(id)) {
                    for (int j = i; j < size - 1; j++) {
                        students[j] = students[j + 1];
                    }
                    size--;
                }
            if (students[size - 1].getId().equals(id))
                size--;
        }
    }

    public String studentsByGroup() {
        String res = "";
        Set<String> sortedGroup = new HashSet<>();
        for (int i = 0; i < size; i++)
            if (!sortedGroup.contains(students[i].getGroup())) {
                sortedGroup.add(students[i].getGroup());
                res += students[i].getGroup() + '\n' + students[i].getInfo() + '\n';
                if (i < size - 1)
                    for (int j = i + 1; j < size; j++)
                        if (sameGroup(students[i], students[j]))
                            res += students[j].getInfo() + '\n';
            }
        return res;
    }
}
