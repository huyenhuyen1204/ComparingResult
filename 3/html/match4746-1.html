class Student {
    private String name;
    private String id;
    private String group;
    private String email;

    public String getName() {
        return name;
    }

    public String getId() {
        return id;
    }

    public String getGroup() {
        return group;
    }

    public String getEmail() {
        return email;
    }

    public void setName(String n) {
        this.name = n;
    }

    public void setId(String n) {
        this.id = n;
    }

    public void setGroup(String n) {
        this.group = n;
    }

    public void setEmail(String n) {
        this.email = n;
    }

    public String getInfo() {
        String result = name + " - " + id + " - " + group + " - " + email;
        return result;
    }

    Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }

    Student(Student s) {
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

}

public 
class StudentManagement {
    private Student[] students = new Student[100];
    int C;

    StudentManagement() {
        C = 0;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        students[C] = newStudent;
        C++;
    }

    public String studentsByGroup() {
        String sh = "";
        String[] group = new String[100];
        int count_G = 0;
        for (int i = 0; i < C; i++) {
            if (count_G == 0) {
                group[count_G] = students[i].getGroup();
                count_G++;
            } else {
                int p = 0;
                for (int k = 0; k < count_G; k++) {
                    if (group[k].equals(students[i].getGroup())) {
                        p = 1;
                        break;
                    }
                }
                if (p == 0) {
                    group[count_G] = students[i].getGroup();
                    count_G++;
                }
            }
        }
        for (int i = 0; i < count_G; i++) {
            sh=sh + group[i] + "\n";
            for (int k = 0; k < C; k++) {
                if (group[i].equals(students[k].getGroup())) {
                    sh = sh  + students[k].getInfo() + "\n";
                }
            }
        }
        return sh;
    }

    public void removeStudent(String id) {
        
for (int i = 0; i < C; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < C; j++) {
                    students[j] = students[j + 1];
                }
                C--;
                break;
            }
        }

    }

    public static void main(String[] args) {
        StudentManagement test = new StudentManagement();
        Student a = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        a.setGroup("K62CC");
        Student b = new Student("Nguyen Van B", "17020003", "17020003@vnu.edu.vn");
        b.setGroup("K62CC");
        Student c = new Student("Nguyen Van C", "17020002", "17020002@vnu.edu.vn");
        Student d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
        test.addStudent(b);
        test.addStudent(c);
        test.addStudent(d);
        test.addStudent(a);


        test.removeStudent("17020004");
        System.out.print(test.studentsByGroup());
    }
