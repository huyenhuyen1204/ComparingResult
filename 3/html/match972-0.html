public class StudentManagement {

    Student[] students = new Student[100];
    int numS = 0;
    int numG = 0;

    private static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());

    }

    public void addStudent(Student newStudent) {
        students[numS] = newStudent;
        numS++;
    }

    public  String studentsByGroup() {
        String tempG = "";
        String[] nameGroup = new String[100];
        String printStudent = "";
        for (int i = 0; i < numS; i++){
            if(students[i].getGroup().equals(tempG) == false) {
                tempG = students[i].getGroup();
                nameGroup[numG] = students[i].getGroup();
                numG++;
            }
        }

        for(int i = 0; i < numG; i++){
            printStudent += nameGroup[i] + "\n";
            for(int j = 0; j < numS; j++){
                if(students[j].getGroup().equals(nameGroup[i]) == true) {
                    printStudent += students[j].getInfo() + "\n";
                }
            }
        }

        return printStudent;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < numS; i++) {
            if (students[i].getId().equals(id) == true) {
                for (int j = i; j < numS; j++) {
                    students[j] = students[j + 1];
                }
                numS--;
            }
        }
    }
