public class StudentManagement
{
    private Student [] students = new Student[100];
    private int count = 0;

    public int getCount()
    {
        return count;
    }

    public Student[] getStudents() {
        return students;
    }

    public static boolean sameGroup(Student s1, Student s2)
    {
        return s1.getGroup().compareTo(s2.getGroup()) == 0;
    }

    public void addStudent(Student newStudent)
    {
        if (count < 100)
        {
            students[count] = newStudent;
            count++;
        }
    }

    public String studentsByGroup()
    {
        String sortStudents = "";
        String listGroup = "";
        String [] groupArr = new String [count];
        int countGroup = 0;

        for (int i = 0; i < count; i++)
        {
            if (!listGroup.contains(students[i].getGroup()))
            {
                listGroup += students[i].getGroup();
                
groupArr[countGroup] = students[i].getGroup();
                countGroup++;
            }
        }

        for (int j = 0; j < countGroup; j ++)
        {
            sortStudents += "\n" + groupArr[j];
            for (int i = 0; i < count; i++)
            {
                if (students[i].getGroup().compareTo(groupArr[j]) == 0)
                {
                    sortStudents = sortStudents + "\n" + students[i].getInfo();
                }
            }
        }
        return sortStudents;
    }

    public void removeStudent(String id)
    {
        for (int i = 0; i < count; i++)
        {
            if (id.equals(students[i].getId()))
            {
                for (int j = i; j < count - 1; j++)
                {
                    students[j] = students[j + 1];
                }
                count--;
                i = count + 5;
            }
        }
    }

}
