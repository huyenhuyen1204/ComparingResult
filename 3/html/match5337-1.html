public class Student {

    private String name;
    private String id;
    private String group;
    private String email;

    //Getter & Setter
    public String getName() {
        return name;
    }

    public void setName(String n) {
        this.name = n;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    //Get info
    String getInfo() {
        String result = name + " - " + id + " - " + group + " - " + email;
        
return result;
    }

    //Constructor
    Student() {
        setName("Student");
        setId("000");
        setGroup("K62CB");
        setEmail("uet@vnu.edu.vn");
    }

    Student(String name, String id, String email) {
        setName(name);
        setId(id);
        setGroup("K62CB");
        setEmail(email);
    }

    Student(Student s) {
        name = s.getName();
        id = s.getId();
        group = s.getGroup();
        email = s.getEmail();
    }
}

public class StudentManagement {

    private static Student students[] = new Student[100];
    private static int cnt = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        String g1 = s1.getGroup();
        String g2 = s2.getGroup();
        return g1.equals(g2);
    }

    public static void addStudent(Student newStudent) {
        if (newStudent != null) {
            students[cnt] = newStudent;
            cnt++;
        }
        return;
    }

    public static String studentsByGroup() {
        String printList = "";
        int check[] = new int[cnt];
        for (int i = 0; i < cnt; ++i)
            check[i] = 1;

        for (int i = 0; i < cnt; ++i) {
            if (check[i] == 1) {
                String 
group = students[i].getGroup();
                printList += group + "\n" + students[i].getInfo() + "\n";
                check[i] = 0;

                for (int j = i + 1; j < cnt; ++j)
                    if (students[j].getGroup().equals(group)) {
                        printList += students[j].getInfo() + "\n";
                        check[j] = 0;
                    }
            }
        }
        return printList;
    }

    public static void removeStudent(String id) {
        int i = 0;
        while (i < cnt) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < cnt - 1; ++j) {
                    students[j] = students[j + 1];
                }
                cnt--;
            } else i++;
        }
        for (i = 0; i < cnt - 1; ++i) {
            for (int j = i + 1; j < cnt; ++j) {
                if (
students[i].getId().equals(students[j].getId())) {
                    for (int k = j; k < cnt - 1; ++k) {
                        students[k] = students[k + 1];
                    }
                    --cnt;
                }
            }
        }
        return;
    }
}
