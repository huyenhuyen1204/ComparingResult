import java.util.ArrayList;

public class StudentManagement
{
    Student[] students = new Student[100];
    private int num_student;

    StudentManagement()
    {
        num_student = 0;
    }

    public static boolean sameGroup(Student s1, Student s2)
    {
        if (s1.getGroup().equals(s2.getGroup()))
        {
            return true;
        }
        else
        {
            return false;
        }
    }

    public void addStudent(Student newStudent)
    {
        students[num_student] = newStudent;
        num_student += 1;
    }

    public String studentsByGroup()
    {
        ArrayList<String> group = new ArrayList<String>();
        String result = "";

        for (Student i : students)
        {
            if (i != null)
            {
                if (!group.contains(i.getGroup()))
                {
                    group.add(i.getGroup());
                }
            }
        }

        for (String i : group)
        {
            result += i + '\n';

            for (Student j : students)
            {
                if (j != null)
                {
                    if (i.equals(j.getGroup()))
                    {
                        result += j.getName() + " - " + j.getId() + " - " + j.getGroup() + " - " + j.getEmail() + '\n';
                    }
                }
            }
        }

        return result;
    }

    public void removeStudent(String id)
    {
        int index = 0;

        for(Student i : students)
        {
            if (i.getId().equals(id))
            {
                break;
            }
            index += 1;
        }

        for (int i = index; i < num_student - 1; i++)
        {
            students[i] = students[i + 1];
        }
        students[num_student - 1] = null;
    }
