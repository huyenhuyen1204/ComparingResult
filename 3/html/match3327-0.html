import java.util.*;

public class StudentManagement {
    private Student students[] = new Student[100];
    int numerbOfStudent = 0;
    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public  void addStudent(Student newStudent){
        Student a = new Student(newStudent);
        students[numerbOfStudent] = a;
        numerbOfStudent ++;
    }

    public String studentsByGroup() {
        Set<String> save = new LinkedHashSet<>();
        for(Student a : students){
            if(a != null && save.contains(a.getGroup()) == false){
                save.add(a.getGroup());
            }
        }
        String result = "";
        for(String s : save){
            String AGroup = s;
            for(Student a : students){
                if(a!= null && a.getGroup().equals(s)){
                    AGroup = AGroup + "\n" + a.getInfo();
                }
            }
            result = result + AGroup + "\n";
        }
        return result;
    }

    public  void removeStudent(String id) {
        for(int i = 0 ; i < numerbOfStudent ; i++){
            if(students[i].getId().equals(id)){
                for(int j = i ; j < numerbOfStudent ; j ++){
                    students[j] = students[j +1];
                }
                numerbOfStudent --;
                break;
            }
        }
    }
