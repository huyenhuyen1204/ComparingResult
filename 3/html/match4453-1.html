class StudentManagement {

    Student [] students = new Student[100];
    
    public static boolean sameGroup(Student s1, Student s2) {
        return ((s1.getGroup()).equals(s2.getGroup())) ;
    }   
    
    public void addStudent(Student newStudent) {
        int i=0;
        while(i<100) {
            if(students[i]==null) {
                students[i]= new Student(newStudent);
                return;
            }
            i++;
        }
    }
    
    public void removeStudent(String id) {
        int j=0,co=0;
        for(int i=0;i<100;i++) {
            if((students[i].getId()).equals(id)) {
                j=i;
                co=1;
                break;
            
}
        }
        if(co==0) return;
        int n=0;
        for(int i=0;i<100;i++) {
            if(students[i]!=null) n++;
            else break;
        }
        
        for(int i=j;i<n-1;i++) 
            students[i]= new Student(students[i+1]);
        //students[n-1] = new Student();
        students[n-1] = null;
    }
    
    public String studentsByGroup() {
        String result="",gr="";
        int []a=new int[100];
        int n=0;
        for(int i=0;i<100;i++) {
            if(students[i]==null) break;
            else n++
;
        }
        for(int i=0;i<n-1;i++) {
           // if(students[i]==null) continue;
            if(a[i]==0) {
                gr=students[i].getGroup();
                
                result+=gr+"\n";
                result+= students[i].getInfo() +"\n";
                a[i]=1;
            }
            
            for(int j=i+1;j<n;j++) {
                if(a[j]==0 && sameGroup(students[i], students[j])) {
                    result+= students[j].getInfo() +"\n";
                    a[j]=1;
                }
            }
        }
        return result;
    }
    
}
