public class StudentManagement {
    static Student[] students = new Student[100];
    int s;
    public static 
boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().compareTo(s2.getGroup()) == 0) return true;
        else return false;
    }

    public String studentsByGroup() {
        boolean[] value = new boolean[s];
            for (int i=0; i<s; i++){
                value[i] = false;
            }
        String in = "";
        for(int i=0;i<s;i++)
        {
            if(value[i] == false)
            {
                String lop = students[i].getGroup();
                in = in + lop + "\n";
                for (int j = i; j < s; j++) {
                        if (students[j].getGroup().equals(lop)) {
                            in = in + students[j].getInfo() + "\n";
                            value[j] = true;
                        }
                }
            }
        }
        return in;
    }

    public void addStudent(Student newStudent){
        if(s<100)
            students[s] = new Student(newStudent);
        s++;
    }
    
    public void removeStudent(String id) {
    	
for(int i = 0;i<s;i++)
        {
            if (students[i].getId().equals(id)){
                for (int j=i;j<s; j++){
                    students[j] = students[j+1];
                }
                s--;
            }
        }

    }

    }
    
}
