public class StudentManagement {

    private Student students[] = new Student[100];

    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }
    public int count()
    {
        int n = 0;
        for(int i=0;i<100;i++)
        {
            if(students[i] == null)
            {
                break;
            }
            n++;
        }
        return n;
    }
    public void addStudent(Student newStudent) {
        int n = count();
        students[n] = newStudent;
    }

    public String studentsByGroup() {
        String [] counstgroup = new String[100];
        int n = count();
        counstgroup[0] = students[0].getGroup();
        int q = 1;
        for(int i=1;i<n;i++)
        {
            int k = 0;
            for(int j=0;j<i;j++)
            {
                if(students[i].getGroup().equals( students[j].getGroup() ) )
                {
                    k++;
                    break;
                }
            }
            if(k == 0 )
            {
                counstgroup[q] = students[i].getGroup();
                q++;
            }
        }
        int m =0;
        String studentbygroup = "";
        for(int i=0; i<q; i++){
            studentbygroup += counstgroup[i];
            studentbygroup += "\n";
            for(int j=0; j<n; j++){
                if(counstgroup[i].equals(students[j].getGroup())){
                    studentbygroup += students[j].getInfo();
                    m++;
                    if(m <=n-1)studentbygroup += "\n";
                }
            }
        }
        return studentbygroup;
    }

    public void removeStudent(String id) {
        int n = count();
        for (int i = 0; i < n; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < n; j++) {
                    students[j] = students[j + 1];
                }
                i--;
                n--;
            }
        }
    }
}
