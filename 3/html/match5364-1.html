public class StudentManagement {
    StudentManagement(){
}
    public static boolean sameGroup(Student s1,Student s2){
        if (s1.getGroup().equals(s2.getGroup())==true) return true; else return false;
    }
    private int length=0;
    private Student[] students=new Student[100];
    public void addStudent(Student newStudent){
        students[length]= new Student();
        students[length].setName(newStudent.getName());
        students[length].setId(newStudent.getId());
        students[length].setGroup(newStudent.getGroup());
        students[length].setEmail(newStudent.getEmail());
        length++;
    }
    public String studentsByGroup(){
        String result="";
        int[] ds=new int[length];
        for (int i=0;i<length;i++) ds[i]=0;
        int i=0;
        while (i<length){
            if (ds[i]==1) i++; else{
                ds[i]=1;
                result=result+"\n"+students[i].getGroup()+"\n"+students[i].getInfo();
                for (int j=i+1;j<length;j++) if (sameGroup(students[i],students[j])==true){
                    ds[j]=1;
                    result=result+"\n"+students[j].getInfo();
                }
            }
        }
        return result;
    }
    public void removeStudent(String id){
        int i=0;
        while( (i<length) && (students[i].getId().equals(id)==false)) i++;
        if (i<length) {
            for (int j=i;j<length-1;j++){
                students[j].setName(students[j+1].getName());
                students[j].setId(students[j+1].getId());
                students[j].setGroup(students[j+1].getGroup());
                students[j].setEmail(students[j+1].getEmail());
            }
            length--;
        }
    }
}
