class StudentManagement {

	Student[] students = new Student[100];

	public static boolean sameGroup(Student s1, Student s2) {
		if (s1.getGroup().equals(s2.getGroup())) return true;
		return false;
	}
	
	public void addStudent(Student newStudent) {
		int realSize = 0;
		while (students[realSize] != null) realSize++;
		students[realSize] = new Student(newStudent);
	}
	
	public String studentsByGroup() {
		String list = "";
		int [] note = new int[100];
		for (int i = 0; students[i] != null; i++) note[i] = 0;
		
		for (int i = 0; students[i] != null; i++) {
			String groupNow = students[i].getGroup();
			boolean differentGroup = false;
			int indexDifferent = 0;
			list = list + groupNow + "\n" + students[i].getInfo() + "\n";
			
			for (int j = i+1; students[j] != null; j++) {
				if (students[j].getGroup() == groupNow) {
					list = list + students[j].getInfo() + "\n";
					note[j] = 1;
				}
				else if (note[j] == 0 && !differentGroup) {
					indexDifferent = j;
					differentGroup = true;
				}
			}
			if (differentGroup) i = indexDifferent - 1;
			else break;
		
}
		return list;
	}
	
	public void removeStudent(String id) {
		for (int i = 0; students[i] != null; i++) 
			if (students[i].getId().equals(id)) 
				for (int j = i; students[j] != null; j++) 
						students[j] = students[j+1];
	}
}
