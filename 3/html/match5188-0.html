import java.util.*;

public class StudentManagement
{
	Student students[] = new Student[100];
	private int studentCount = 0;

	public static boolean sameGroup(Student s1, Student s2)
	{
		return s1.getGroup().equals(s2.getGroup());
	}

	public void addStudent(Student newStudent)
	{
		students[studentCount++] = newStudent;
	}

	public void removeStudent(String id)
	{
		int pos = -1;
		for (int i = 0; i < studentCount; ++i)
		{
			if (students[i].getId().equals(id)) 
			{
				pos = i;
				break;
			}
		}

		if (pos == -1) return;
		for (int i = pos + 1; i < studentCount; ++i)
		{
			students[i - 1] = students[i]; 
		}
		--studentCount;
	}

	public String studentsByGroup()
	{
		String ans = "";
		List<String> seenGroups = new ArrayList<String>();
		for (int i = 0; i < studentCount; ++i)
		{
			if (seenGroups.contains(students[i].getGroup())) continue;
			ans += students[i].getGroup() + "\n";
			seenGroups.add(students[i].getGroup());
			for (int j = i; j < studentCount; ++j)
			{
				if (students[i].getGroup().equals(students[j].getGroup()))
				{
					ans += students[j].getInfo() + "\n";
				}
			}
		}
		return ans;
	}

	public static void main(String args[]) {
		Student an = new Student();
		System.out.println(an.getInfo());

		an.setName("Nguyen Van An");
		an.setId("17020001");
		an.setGroup("K62CC");
		an.setEmail("17020001@vnu.edu.vn");

		System.out.println(an.getName());
		System.out.println(an.getId());
		System.out.println(an.getGroup());
		System.out.println(an.getEmail());
		System.out.println(an.getInfo());

		Student b = new Student(an);
		System.out.println(b.getInfo());
		
		b = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
		System.out.println(b.getInfo());

        b.setGroup("K62CC");
		System.out.println(StudentManagement.sameGroup(b, an));
		
		StudentManagement sm = new StudentManagement();
		sm.addStudent(an);
		sm.addStudent(b);
		Student c = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
		Student d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
		sm.addStudent(c);
		sm.addStudent(d);
		System.out.println(sm.studentsByGroup());
		sm.removeStudent("17020001");
		System.out.println(sm.studentsByGroup());
    }
}
