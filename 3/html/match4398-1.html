public class StudentManagement {

    Student[] students = new Student[100];
    int listSize = 0;

    StudentManagement() {
        listSize = 0;
    }

    public static 
boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        students[listSize++] = newStudent;
    }

    public String studentsByGroup() {
        String ans = "";
        boolean[] ok = new boolean[100];
        for (int i = 0; i < listSize; ++i) {
            if (!ok[i]) {
                ok[i] = true;
                ans += students[i].getGroup() + "\n";
                ans += students[i].getInfo() + "\n";
                
for (int j = i + 1; j < listSize; ++j) {
                    if (sameGroup(students[i], students[j])) {
                        ok[j] = true;
                        ans += students[j].getInfo() + "\n";
                    }
                }
            }
        }
        if (ans.equals(""))
            return null;
        
return ans;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < listSize; ++i) {
            if (id.equals(students[i].getId())) {
                listSize--;
                for (int j = i; j < listSize; ++j) {
                    students[j] = students[j + 1];
                }
                break;
            }
        }
    }

}
