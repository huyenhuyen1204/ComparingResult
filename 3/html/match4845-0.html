public class Student {
    private String name;
    private String id;
    private String group;
    private String email;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.group = "K62CB";
        this.email = email;
    }

    public Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    public String getInfo(){
        return this.name + " - " + this.id + " - " + this.group + " - " + this.email;
    }

    public Student(Student s){
        this.name = s.getName();
        this.group = s.getGroup();
        this.email = s.getEmail();
        this.id = s.getId();
    }

}

public class StudentManagement {

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    Student[] students = new Student[100];
    int sz = 0;
    public void addStudent(Student newStudent){
        students[sz++] = newStudent;
    }

    public String studentsByGroup(){
        String res = "";
        int msz = 0;
        String[] nameGroup = new String[100];
        for(int i = 0; i < sz; i++){
            boolean ok = false;
            for(int j = 0; j < msz; j++){
                if(students[i].getGroup().equals(nameGroup[j])) {
                    ok = true;
                    break;
                }
            }
            if(ok == false){
                nameGroup[msz++] = students[i].getGroup();
                res = res +  students[i].getGroup() + "\n";
                for(int j = i; j < sz; j++){
                    if(sameGroup(students[i], students[j])){
                        res += students[j].getInfo() + "\n";
                    }

                }
            }
        }

        return res;

    }

    public void removeStudent(String id){
        for(int i = 0; i < sz; i++){
            if(students[i].getId().equals(id)){
                for(int j = i; j < sz - 1; j++){
                    students[j] = students[j + 1];
                }
                sz--;
            }
        }
    }
