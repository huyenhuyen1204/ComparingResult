//package student;
public class StudentManagement {

    Student[] students = new Student[100];
    int length=0;

    String[] GroupName = new String[100];
    int soGroup = 0;
    int count;

    public static 
boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
            students[length++] = newStudent;
    }

    public  String studentsByGroup() {
        String Result="" ;
        for (int i=0;i<=length-1;i++) {
            if (i==0)
            {
                GroupName[soGroup]=students[i].getGroup();
            }
            else
            {
            count = 0;                       
            for (int j=0;j<=soGroup;j++){
               if (students[i].getGroup().equals(GroupName[j])) 
               {
                   count++;
               }
            
            }
            if (count == 0 ) 
            { 
                soGroup++;
                GroupName[soGroup] = students[i].getGroup();
            }
            }
        }
        
for (int i=0;i<=soGroup;i++) {
            Result=Result+GroupName[i]+"\n";
            for (int j=0;j<=length-1;j++){
                if (GroupName[i].equals(students[j].getGroup()))  
                {
                    Result=Result+students[j].getInfo()+"\n";
                }
            }
        }
        return Result;
    }

    public void removeStudent(String id) {
        for (int i=0 ; i< length ; i++) {
                if (students[i].getId().equals(id)) {
                    length--;
                    for (int j=i;j< length;j++) 
                    {
                        i++;
                        students[j]=students[i];
                    }
                    
                }
        }
    }

    public static void main(String[] args) {
        StudentManagement list = new StudentManagement();
        Student sv1 = new Student();
        Student sv2 = new Student();
        Student sv3 = new Student();
        sv1.setName("Nguyen Van An"); sv2.setName("Nguyen Van B"); sv3.setName("Nguyen Van C");
        sv1.setId("17020001"); sv2.setId("17020002"); sv3.setId("17020003");
        sv1.setGroup("K62CC"); sv2.setGroup("K62CB"); sv3.setGroup("K62CC");
        sv1.setEmail("17020001@vnu.edu.vn"); sv2.setEmail("17020002@vnu.edu.vn"); sv3.setEmail("17020003@vnu.edu.vn");
        list.addStudent(sv1);
        list.addStudent(sv2);
        list.addStudent(sv3);
        System.out.println(list.studentsByGroup());
    }
}
