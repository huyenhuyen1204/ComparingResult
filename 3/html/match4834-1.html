public class StudentManagement {

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    Student[] students = new Student[100];
    int size = 0;
    public void addStudent(Student newStudent){
        students[size++] = newStudent;
    }

    public String studentsByGroup(){
        String ans = "";
        int msize = 0;
        String[] nameGroup = new String[100];
        for(int i = 0; i < size; i++){
            boolean check = false;
            for(int j = 0; j < msize; j++){
                if(students[i].getGroup().equals(nameGroup[j])) {
                    check = true;
                    break;
                }
            }
            if(check == false){
                
nameGroup[msize++] = students[i].getGroup();
                ans = ans +  students[i].getGroup() + "\n";
                for(int j = i; j < size; j++){
                    if(students[i].getGroup().equals( students[j].getGroup())){
                        ans += students[j].getInfo() + "\n";
                    }

                }
            }
        
}

        return ans;

    }

    public void removeStudent(String id){
        for(int i = 0; i < size; i++){
            if(students[i].getId().equals(id)){
                for(int j = i; j < size - 1; j++){
                    students[j] = students[j + 1];
                }
                size--;
            }
        }
    }
    public static void main(String[] x){
        StudentManagement list = new StudentManagement();

        Student nguyen_van_an = new Student("Nguyen Van An", "17020001",  "17020001@vnu.edu.vn");
        nguyen_van_an.setGroup("K62CC");
        list.addStudent(nguyen_van_an);

        Student nguyen_van_b = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        nguyen_van_b.setGroup("K62CC");
        list.addStudent(nguyen_van_b);

        Student nguyen_van_c = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");

        list.addStudent(nguyen_van_c);

        Student nguyen_van_d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");

        list.addStudent(nguyen_van_d);

        String s = list.studentsByGroup();

        System.out.println(s);


    }

}
