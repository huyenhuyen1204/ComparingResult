public class StudentManagement {
    Student[] students = new Student[100];
    int size = 0;
    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        students[size] = new Student(newStudent);
        size++;
    }

    public void removeStudent(String id){
        for(int i = 0; i < size - 1; i++){
            if(students[i].getId().equals(id)){
               for(int j = i; j < size - 1; j++){
                    students[j] = students[j+1];
               } 
               size--;
            }
        }
        if(students[size-1].getId().equals(id)){
            size--;
        }
    }

    public String studentsByGroup(){
        String res = "";
        String[] groups = new String[100];
        int sizeG = 0;

        for(int i = 0; i < size; i++){
            boolean check = true;
            for(int j = 0; j < sizeG; j++){
                if(students[i].getGroup().equals(groups[j])){
                    check = false;
                    break;
                }
            }
            if(check == true){
                groups[sizeG] = students[i].getGroup();
                sizeG++;
            }
        }
        
        for(int i = 0; i < sizeG; i++){
            res = res + groups[i] + "\n";
            for(int j = 0; j < size; j++){
                if(groups[i].equals(students[j].getGroup())){
                    res = res + students[j].getInfo() + "\n";
                }
            }
        }
        return res;
    }

    }
}
