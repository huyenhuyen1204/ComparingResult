class Student {
	private String name;
	private String id;
	private String group;
	private String email;

	public String getName() {
		return this.name;
	}
	public String getId() {
		return this.id;
	}
	public String getGroup() {
		return this.group;
	}
	public String getEmail() {
		return this.email;
	}
	public void setName(String n) {
		this.name = n;
	}
	public void setId(String n) {
		this.id = n;
	}
	public void setGroup(String n) {
		this.group = n;
	}
	public void setEmail(String n) {
		this.email = n;
	}

	public String getInfo() {
		String st = " - ";
		return this.name + st + this.id + st + this.group + st + this.email;
	}

	public Student() {
		this.name = "Student";
		this.id = "000";
		this.group = "K62CB";
		this.email = "uet@vnu.edu.vn";
	}

	public Student(String name , String id , String email) {
		this.group = "K62CB";
		this.name = name;
		this.id = id;
		this.email = email;
	}

	public Student(Student s) {
		this.name = s.name;
		this.id = s.id;
		this.group = s.group;
		this.email = s.email;
	}


}
