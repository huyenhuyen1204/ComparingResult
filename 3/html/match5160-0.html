import java.util.Arrays;
import java.util.Comparator;

public class StudentManagement {
    //1.attributes
    private Student[] students = new Student[100];
    private int count = 0;

    //2.methods
    public static boolean sameGroup(Student s1, Student s2) {
        if((s1!= null) && (s2!= null) && (s1.getGroup().equals(s2.getGroup())))
            return true;
        else
            return false;
    }
    public void addStudent(Student newStudent) {
        students[count] = new Student(newStudent);
        count++;
    }
    public String studentsByGroup() {
        Arrays.sort(students,0, count, new Comparator<Student>() {
            @Override
            public int compare(Student o2, Student o1) {
                return o1.getGroup().compareTo(o2.getGroup());
            }
        });
        String group_temp = students[0].getGroup();
        String result = group_temp;
        result += "\n";
        for (int i = 0; i < count; i++) {
            if(!students[i].getGroup().equals(group_temp)) {
                group_temp = students[i].getGroup();
                result += group_temp;
                result += "\n";
            }
            result += students[i].getInfo();
            result += "\n";
        }
        return result;
    }
    public void removeStudent(String id) {
        for (int i = 0; i < count; i++) {
            if(students[i].getId().equals(id)) {
                for (int j = i; j < count ; j++) {
                    students[j] = students[j+1];
                }
                students[count] = null;
                count--;
                break;
            }
        }
    }
}
