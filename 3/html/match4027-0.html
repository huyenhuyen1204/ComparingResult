import java.util.Scanner;

public class StudentManagement {
    private Student[] students;
    private int number;
    StudentManagement() {
        students = new Student[100];
        number = 0;
    }

    StudentManagement (Student[] students, int number) {
        this.students = students;
        this.number = number;
    }

    public static boolean sameGroup (Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent (Student newStudent) {
        students[number] = newStudent;
        number++;
    }

    public String studentsByGroup() {
        String[] groups = new String[number];

        int idx = 0;
        groups[idx] = students[0].getGroup();
        for (int i = 1; i < number; i++)
            if (!students[i].getGroup().equals(groups[idx])) {
                idx++;
                groups[idx] = students[i].getGroup();
            }

        StringBuilder print = new StringBuilder();
        for (int i = 0; i <= idx; i++) {
            if (groups[i] != "") {
                print.append(groups[i]).append("\n");
                for (int j = 0; j < number; j++)
                    if (students[j].getGroup().equals(groups[i]))
                        print.append(students[j].getInfo()).append("\n");
            }
        }
        return print.toString();
    }

    public void removeStudent (String id) {
        int idx = 0;
        for (int i = 0; i < number; i++)
            if (students[i].getId().equals(id)) {
                idx = i;
                break;
            }
        for (int i = idx; i < number-1; i++) {
            students[i] = students[i+1];
        }
        students[number-1].setName("");
        students[number-1].setId("");
        students[number-1].setGroup("");
        students[number-1].setEmail("");
        number--;
    }
}
