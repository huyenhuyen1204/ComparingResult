//package com.company;

public class StudentManagement {

    Student[] students = new Student[100];
    int n = 0;

    String[] allG = new String[100];
    int m = 0;
    int k;
    public static 
boolean sameGroup(Student s1, Student s2) {

        return (s1.getGroup().equals( s2.getGroup()));
    }

    public void addStudent(Student newStudent){

        students[n++] = newStudent;
    }

    public String studentsByGroup() {
        String kq = "";

        //allG[0] = students[0].getGroup();

        for (int i = 0; i<n; i++) {
            if (i==0) allG[m] = students[i].getGroup();
             k = 0;
            for (int j =0; j<=m; j++) {
                if (students[i].getGroup() == allG[j]) {
                    k++;
                }
            }
                if (k==0) {
                    m++;
                    allG[m] = students[i].getGroup();
                
}


        }

        for (int i = 0; i<=m; i++) {
            kq = kq + allG[i] + "\n";
            for (int j = 0; j<n; j++) {
                if (allG[i] == 
students[j].getGroup() ) {
                    kq = kq + students[j].getInfo() + "\n";
                }
            }
        }
        return kq;
    }

    public void removeStudent(String id) {

        for(int i = 0; i<n;i++) {
            if(students[i].getId().equals(id)) {
                n--;
                for (int j = i ; j<n; j++) {
                    i++;
                    students[j] = students[i];
                }
                break;
            }

        }


    }

    public static void main(String[] args) {

    }


}
