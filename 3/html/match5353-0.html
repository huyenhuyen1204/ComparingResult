public class StudentManagement {
    Student[] students = new Student[100];
    int cnt = 0;
    public static boolean sameGroup(Student s1, Student s2) {
         if(s1.getGroup().equals(s2.getGroup())){
             return true;
        }else return false;
    }
    public void addStudent(Student newStudent){
        students[cnt] = newStudent;
        cnt++;
    }
    public String studentsByGroup() {
        boolean[] mark = new boolean[cnt];
        for (int i = 0; i < cnt; i++) {
            mark[i] = false;
        }
        String abc = "";
        for (int i = 0; i < cnt; i++) {
            if (mark[i] == false) {
                abc += students[i].getGroup() + "\n";
                for (int j = i; j < cnt; j++) {
                    if (sameGroup(students[j], students[i])) {
                        abc +=  students[j].getInfo() + "\n";
                        mark[j] = true;
                    }

                }
            }
        }
        return abc;
    }
    public void removeStudent(String id){
        int n = 0;
        for (int i = 0; i < cnt; i++){
            if(students[i].getId().equals(id)) n = i;
        }
        for(int i = n; i < cnt-1; i++){
            students[i] = students[i+1];
        }
        cnt--;
    }
   public static void main(String[] args){
        Student a = new Student();
        a.setName("Nhung");
        a.setId("123");
        a.setGroup("K63CACLC3");
        a.setEmail("@gmail.com");
        Student b = new Student();
        b.setName("Linh");
        b.setId("456");
        b.setGroup("K63CACLC2");
        b.setEmail("@gmail.com");
        Student c = new Student();
        c.setName("Nguyen");
        c.setId("789");
        c.setGroup("K63CACLC3");
        c.setEmail("@gmail.com");
        StudentManagement ds = new StudentManagement();
        ds.addStudent(a);
        ds.addStudent(b);
        ds.addStudent(c);
        System.out.println(ds.studentsByGroup());
    }
}
