public class StudentManagement {
    private  static Student[] students = new Student[100];
    private  static int count = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        String a=s1.getGroup();
        String b=s2.getGroup();
        if ( a.equals(b) ) {
            return true;
        }
            return false;

    }

    public static void addStudent(Student newStudent) {
        students[count] = newStudent;
        count++;
    }

    public static String studentsByGroup() {
        String list = "";
        int[] c = new int[count];
        for (int i = 0; i < count; i++) {
            c[i] = 1;
        }
        for (int j = 0; j < count; j++) {
            if (c[j] == 1) {
                String gr = students[j].getGroup();
                list += gr + "\n" + students[j].getInfo() + "\n";
                c[j] = 0;

                for (int k = j + 1; k < count; k++)
                    if ( students[k].getGroup().equals(gr) ) {
                        list += students[k].getInfo() + "\n";
                        c[k] = 0;
                    }
            }
        }
        return list;
    }

    public static void removeStudent(String id){
        int i=0;
        while (i<count){
            //từ vị trí đầu tiên
            if(students[i].getId().equals(id)){
                //nếu có id=id đã cho thì từ vị đó đến cuối mảng students-1 thì
                for (int j=i; j<count-1; j++) {
                    students[j] = students[j + 1];
                }
                count--;
            }
            else i++;
        }

        //mỗi sinh viên chỉ có 1 id tương đương mỗi id = 1 sinh viên nên vòng này sẽ xóa những sinh viên trùng id và chỉ lấy sv đầu tiên.
        i = 0;
        while (i < count - 1) {
            id = students[i].getId();
            int t = i + 1;
            while (t < count) {
                if (students[t].getId().equals(id)) {
                    for (int j = t ; j < count - 1; j++) {
                        students[j] = students[j + 1];
                    }
                    count--;
                }
                else t++;
            }
            i++;
        }
    }

        {
