class StudentManagement {

    private static final int MAX = 100;
    private int size = 0;
    private Student[] students = new Student[MAX];

    public static boolean sameGroup(Student s1, Student s2) {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student newStudent) {
        if (size + 1 < MAX) {
            students[size] = newStudent;
            ++size;
        }
    }

    public String studentsByGroup() {
        if (size == 0) return "";
        String result = "";
        boolean[] isOut = new boolean[size];
        String curGroup;
        for (int cur = 0; cur < size; cur++) {
            if (!isOut[cur]) {
                curGroup = students[cur].getGroup();
                result = result + curGroup + "\n";
                for (int i = cur; i < size; ++i) {
                    if (!isOut[i] && curGroup.equals(students[i].getGroup())) {
                        result = result + students[i].getInfo() + "\n";
                        isOut[i] = true;
                    }
                }
            }
        }
        return result;
    }

    public void removeStudent(String id) {
        
for (int i = 0; i < size; i++) {
            if (id.equals(students[i].getId())) {
                --size;
                for (int j = i; j < size; ++j) {
                    students[j] = students[j + 1];
                }
            }
        }
    }

    public static void main(String[] args) {

    }
}
