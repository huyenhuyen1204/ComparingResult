public class StudentManagement {

    // TODO: khai bao thuoc tinh students la array chua cac doi tuong thuoc lop Student (max. 100)
    private int count = 0;
    Student students[] = new Student[100];

    public static boolean sameGroup(Student s1, Student s2) {
        // TODO:
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
     // TODO:
        students[count] = newStudent;      
        this.count++; 
    }

    public String studentsByGroup() {
        // TODO:
        String listStudent = "";
        for(int i = 0; i < count - 1; i++){
            for(int j = 0; j < count - 1; j++){
                if (students[j].getGroup().compareTo(students[j+1].getGroup()) < 0){
                    Student temp = students[j];
                    students[j] = students[j+1];
                    students[j+1] = temp;
                }
            }
        }
        for (int i = 0; i < count; i++) {            
            if (i == 0 || !sameGroup(students[i], students[i-1])){
                listStudent += students[i].getGroup() + "\n";
            }
            listStudent += students[i].getInfo() + "\n";
        }
	    return listStudent;
    }

    public void removeStudent(String id) {
        // TODO:
        for(int i = 0; i < count; i++){
            if (students[i].getId().equals(id)) {
                for(int j = i; j < count - 1; j++){
                    students[j] = students[j+1];
                }
                students[count-1] = null; 
                count--;
                return ;
            }
        }
    }


    public static void main(String[] args) {
        // TODO:
        StudentManagement st = new StudentManagement();
        
        
        st.addStudent(new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn"));
        st.addStudent(new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn"));       
        st.addStudent(new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn"));
        st.addStudent(new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn"));

        st.students[3].setGroup("K62CC");
        st.students[1].setGroup("K62CB");
        st.students[0].setGroup("K62CB");
        st.students[2].setGroup("K62CC");
        st.count = 4;
        System.out.println(st.studentsByGroup());
        st.removeStudent("17020002");
        System.out.println(st.studentsByGroup());
