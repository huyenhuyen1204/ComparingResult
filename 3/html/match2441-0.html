public class StudentManagement {
    private Student[] students = new Student[100];
    private int numS = 0;
    private Group[] listGroup = new Group[100];
    private int numGroup = 0;

    public class Group {
        private String name;
        private String result;

        public Group (String name) {
            this.name = name;
            this.result = name + "\n";
        }

        public void addStudent (Student s) {
            this.result += s.getInfo() + "\n";
        }

        public String print() {
            return result;
        }
    }

    public static boolean sameGroup (Student s1, Student s2) {
        return s1.getGroup().compareTo(s2.getGroup()) == 0;
    }

    private void addToGroup(Student s) {
        for (int i = 0 ; i < numGroup ; i ++)
            if (s.getGroup().compareTo(listGroup[i].name) == 0){
                listGroup[i].addStudent(s);
                return;
            }
        listGroup[numGroup] = new Group(s.getGroup());
        listGroup[numGroup].addStudent(s);
        numGroup++;
    }

    public void addStudent(Student newStudent) {
        students[numS] =  new Student(newStudent);
        numS ++;
    }

    public String studentsByGroup() {
        numGroup = 0;
        for (int i = 0 ; i < numS ; i ++) addToGroup(students[i]);

        String res = "";
        for (int i = 0 ; i < numGroup ; i ++) res += listGroup[i].print();
        return res;
    }

    public void removeStudent(String id) {

        for (int i = 0 ; i < numS ; i ++)
            if (students[i].getId().compareTo(id) == 0) {
                for (int j = i; j < numS - 1; j++)
                    students[j] = students[j + 1];
                students[numS - 1] = null;
                numS--;
            }
    }
}
