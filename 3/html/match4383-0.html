public class StudentManagement {
    private Student[] students = new Student[100];
    public int num = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
            students[num] = newStudent;
            num++;
    }

    public String studentsByGroup() {
        String print = "";
        String group = "";
        group = students[0].getGroup();
        print += group + "\n";

        for (int i = 0; i < num; i++){
            if (students[i].getGroup() == group)
                print += students[i].getInfo() + "\n";
        }

        for (int i = 0; i < num; i++){
            if (students[i].getGroup() == group){
                continue;
            } else {
                group = students[i].getGroup();
                print += group + "\n";
            }
        }

        for (int i = 0; i < num; i++){
            if (students[i].getGroup() == group)
                print += students[i].getInfo() + "\n";
        }
        
        return print;
    }

    public void removeStudent(String id) {
        for(int i = 0; i < num; i++) {
            if(students[i].getId().equals(id)) {
                for(int j = i; j < num-1; j++) {
                    students[j] = students[j+1];
                }
                students[num - 1] = null;
                num--;
            }
        }
    }

    public static void main(String[] args) {
        Student A = new Student("A", "001", "Aa");
        Student B = new Student("B", "002", "Bb");
        Student An = new Student("nguyen an", "003", "an");
        Student Bn = new Student("nguyen bn", "004", "bn");

        A.setGroup("K62CC");
        B.setGroup("K62CC");

//        System.out.println(sameGroup(A, B));

        StudentManagement st = new StudentManagement();
        st.addStudent(An); //cb
        st.addStudent(A);  //cc
        st.addStudent(B);  //cc
        st.addStudent(Bn); //cb

        //st.removeStudent("001");
        //st.removeStudent("001");
        //st.removeStudent("003");
        System.out.println(st.studentsByGroup());
    }
}
