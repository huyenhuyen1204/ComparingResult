class StudentManagement {

	public Student[] students;
	public int n;

	public StudentManagement() {
		this.students = new Student[100];
		this.n = 0;
	}
	
	public static boolean sameGroup(Student s1, Student s2) {
		return s1.getGroup().equals(s2.getGroup());
	}
	
	public void addStudent(Student newStudent) {
		if (n == 100)
			System.out.println("ERROR");
		else 
			students[n++] = newStudent;
	}
	
	public String studentsByGroup() {
		String result = "";
		for (int i = 0; i < n; i++) {
			boolean checkUniqueGroup = true;
			for (int j = 0; j < i; j++) {
				if (students[i].getGroup().equals(students[j].getGroup())) {
					checkUniqueGroup = false;
					break;
				}
			}
			
			if (checkUniqueGroup == true) {
				result += (students[i].getGroup() + "\n");
				for (int j = i; j < n; j++) {
					if (students[i].getGroup().equals(students[j].getGroup()))
						
result += (students[j].getInfo() + "\n");
				}
			}
		}
		return result;
	}
	
	public void removeStudent(String id) {
		for (int i = 0; i < n; i++) {
			if (students[i].getId().equals(id)) {
				for (int j = i; j < n - 1; j++) {
					students[j] = students[j + 1];
				}
				n--;
				break;
			}
		}
	}
