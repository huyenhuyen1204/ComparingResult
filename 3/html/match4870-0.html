    public Student getsudents(int i)
    {
        return  students[i];
    }
    public static boolean sameGroup(Student s1, Student s2)
    {
        if ( s1.getGroup().equals(s2.getGroup()) ) return true ;
        else  return  false ;
    }

    public  void  addStudent(Student newStudent)
    {
        students[size]  = newStudent ;
        size ++ ;
    }
    public String studentsByGroup()
    {
        String studentsBG = new String();

        for ( int i = 0 ; i < size ; i ++ )
        {

            boolean test = true ;
            for ( int j = 0 ; j < i ; j ++ )
            {
                if ( students[i].getGroup().equals(students[j].getGroup())) test = false;
            }
            if ( test)
            {
                studentsBG = studentsBG + students[i].getGroup() + "\n";
                studentsBG = studentsBG + students[i].getInfo() + "\n";
                for (int k = i +1; k < size ; k ++ )
                {
                    if ( students[k].getGroup().equals(students[i].getGroup()))
                    {
                        studentsBG = studentsBG + students[k].getInfo() + "\n";
                    }

                }



            }
        }
        return  studentsBG;
    }
    public void removeStudent(String id)
    {
        for ( int i = 0 ; i < size ; i ++)
        {
            if ( students[i].getId().equals(id) )
            {
                for ( int j = i ; j < 99 ; j ++ )
                {
                    students[j] = students[j+1] ;
                }
                size -- ;
            }
        }
    }
