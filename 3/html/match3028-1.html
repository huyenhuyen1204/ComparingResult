public class StudentManagement {

    static Student [] students = new Student[100];
    static int numbStudent=0;
    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public static void addStudent(Student newStudent){
        for(int i=0; i<numbStudent; i++){
            
if(students[i].getId().equals(newStudent.getId()))
                return;
        }
        students[numbStudent]=newStudent;
        numbStudent++;
    }


    public static String studentsByGroup() {
        String ByGroups=students[0].getGroup();
        int Group=0;
        int NextGroup=Group;
        while (true) {
            for (int i = Group; students[i] != null; i++) {

                if (sameGroup(students[Group], students[i])) {
                    ByGroups = ByGroups +"\n"+ students[i].getInfo();
                } else if (NextGroup == Group && !ByGroups.contains(students[i].getGroup())){
                    NextGroup = i;
                }
            }
            if(Group==NextGroup){
                break;
            }else {
                Group=NextGroup;
                ByGroups =ByGroups+"\n"+students[Group].getGroup();
            }
        }
        return ByGroups;
    }

    public static void removeStudent(String id) {
        int i=0;
        while (i<numbStudent){
            if(students[i].getId().equals(id)){
                System.arraycopy(students, i+1, students, i, students.length-1-i );
                numbStudent--;
                return;
            }
            i++;
        }
    }
