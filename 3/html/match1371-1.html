public class StudentManagement {
    public static 
boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup()))
            return(true);
        else
            return(false);
    }
    
    private Student[] students=new Student[100];
    public int num=0;
    
    public void addStudent(Student newStudent){
	 students[num]=newStudent;
         num++;
    }
    
    public String studentsByGroup() {
        String[] arr=new String[100];
        int cnt=0;
        for(int i=0; i<num; i++){
            int detec=0;
            for(int j=0; j<cnt; j++){
                if(students[i].getGroup().equals(arr[j])){
                    detec++;
                    
break;
                }
            }
            if(detec==0){
                arr[cnt]=students[i].getGroup();
                cnt++;
            }
        }
	String ans="";
        for(int i=0; i<cnt; i++){
            ans+=arr[i]+"\n";
            for(int j=0; j<num; j++){
                if(students[j].getGroup().equals(arr[i]))
                    
ans+=students[j].getInfo()+"\n";
            }
        }
        return(ans);
    }

    public void removeStudent(String id) {
        int i, j;
        for(i=0; i<num; i++){
            if(students[i].getId().equals(id)){
                 for(j=i; j<num-1; j++){
                        students[j]=students[j+1];
                 }
                num--;
            }
            
        }
    }
}
