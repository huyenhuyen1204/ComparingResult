class StudentManagement {
    public Student[] students = new Student[100];
    public int len;
    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public void addStudent(Student newStudent) {
        students[len] = newStudent;
        len++;
    }

    public String studentsByGroup() {
        String[] arr = new String[100];
        int count = 1;
        arr[0] = students[0].getGroup();
        int temp = 2,i,j;
        for (i = 1; i<len;i++) {
            for (j = 0; j<i; j++)
                if (students[j].getGroup().equals(students[i].getGroup())) {
                    temp = 0;
                    break;
                } else {
                    temp = 1;
                }
            if (temp == 1) {
                arr[count] = students[j].getGroup();
                count++;
            }
        }
        String result = "";
        boolean val = false;
        for (int u = 0; u < count; u++) {
            for (int v=0; v<len;v++) {
                if (students[v].getGroup().equals(arr[u]) && students[v].getId() != "000") {
                    if (val == false) {
                        result += arr[u] + "\n";
                        val = true;
                    }
                    result += students[v].getInfo() + "\n";
                }
            }
            val = false;
        }
        return result;
    }

    public void removeStudent (String id) {
        int index=0;
        for (int i = 0; i< len; i++) {
            if (students[i].getId().equals(id) && 
students[i].getId() != "000") index = i;
        }
        for (int j = index; j< len-1;j++) students[j] = students[j+1];
        len--;
    }

    public static void main(String[] args) {
        Student s1 = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        s1.setGroup("K62CC");
        Student s2 = new Student("Nguyen Van B", "17020002","17020002@vnu.edu.vn");
        s2.setGroup("K62CC");
        Student s3 = new  Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        Student s4 = new  Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
        StudentManagement play = new StudentManagement();
        play.addStudent(s1);
        play.addStudent(s2);
        play.addStudent(s3);
        play.addStudent(s4);
        System.out.println(play.studentsByGroup());
    }
}
