public class StudentManagement {
    private Student[] students = new Student[100];
    private int nStudent;

    private String[] classList = new String[100];

    public StudentManagement(){

    
}

    public static boolean sameGroup(Student s1, Student s2) {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student newStudent){
        students[++nStudent] = newStudent;
    }

    public String studentsByGroup() {
        int nClass = 0;
        for (int i = 1; i <= nStudent; i++) {
            boolean Exist = false;
            for (int j = 1; j <= nClass; j++) {
                if (students[i].getGroup().equals(classList[j])) {
                    Exist = true;
                    break;
                }
            }

            if (!Exist) {
                
classList[++nClass] = students[i].getGroup();
            }
        }

        StringBuilder result = new StringBuilder();

        for (int i = 1; i <= nClass; i++){
            
result.append(classList[i]).append('\n');
            for (int j = 1; j <= nStudent; j++)
                if (!students[j].getId().equals("-1") && students[j].getGroup().equals(classList[i]))
                    result.append(students[j].getInfo()).append('\n');
        }

        return result.toString();
    }

    public void removeStudent(String id) {
        for (int i = 1; i <= nStudent; i++)
            if (students[i].getId().equals(id))
                students[i].setId("-1");
    }

    public static void main(String[] args) {

    }
}
