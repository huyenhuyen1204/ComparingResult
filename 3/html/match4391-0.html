public class Student {
    private String name;
    private String id;
    private String group;
    private String email;

    public String getName() {
        return name;
    }

    public void setName(String inp) {
        name = inp;
    }

    public String getId() {
        return id;
    }

    public void setId(String inp) {
        id = inp;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String inp) {
        group = inp;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String inp) {
        email = inp;
    }

    Student(String _name, String _id, String _group, String _email) {
        name = _name;
        id = _id;
        group = _group;
        email = _email;
    }
    
    Student() {
        name = "Student";
        id = "000";
        group = "K62CB";
        email = "uet@vnu.edu.vn";
    }

    Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.group = "K62CB";
        this.email = email;
    }

    Student(Student s) {
        name = s.name;
        id = s.id;
        group = s.group;
        email = s.email;
    }

    String getInfo() {
        String ans = getName() + " - " + getId() + " - " + getGroup() + " - " + getEmail();
        return ans;
    }
}

public class StudentManagement {

    Student[] students = new Student[100];
    int listSize = 0;

    StudentManagement() {
        listSize = 0;
    }

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        students[listSize++] = newStudent;
    }

    public String studentsByGroup() {
        String ans = "";
        boolean[] ok = new boolean[100];
        for (int i = 0; i < listSize; ++i) {
            if (!ok[i]) {
                ok[i] = true;
                ans += students[i].getGroup() + "\n";
                ans += students[i].getInfo() + "\n";
                for (int j = i + 1; j < listSize; ++j) {
                    if (sameGroup(students[i], students[j])) {
                        ok[j] = true;
                        ans += students[j].getInfo() + "\n";
                    }
                }
            }
        }
        if (ans.equals(""))
            return null;
        return ans;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < listSize; ++i) {
            if (id.equals(students[i].getId())) {
                listSize--;
                for (int j = i; j < listSize; ++j) {
                    students[j] = students[j + 1];
                }
                break;
            }
        }
    }

}
