import java.util.Arrays;

public class StudentManagement {
    private Student[] students = new Student[100];
    int nuM = 0;
    int nUmGr = 0;
    public static boolean sameGroup( Student s1, Student s2 ) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent( Student newStudent ) {
        if (nuM < 100) {
            students[nuM] = new Student(newStudent);
            nuM++;
        }
    }

    public class Group
    {
        String name;
        String print;

        public void Group( String name )
        {
            this.name = name;
            this.print = name + "/n";
        }

        public String addToAGr( Student std ){
            this.print = std.getInfo() + "/n";
            return this.print;
        }
    }
        public String[] addGroups()
    {
        for( int i = 0; i < nuM; i++){
            int count = 0;
            String gR = students[i].getGroup();
            for( int j = 0; j < i + 1; j++ )
            {
                if( gR.equals(students[j].getGroup()))
                {
                    break;
                }
                else
                {
                    count++;
                }
            }
            if( count == i ) nUmGr++;
        }

        String[] groupS = new String[nUmGr];
        groupS[0] = students[0].getGroup();
        int count = 1;
        for( int i = 1; i < nuM; i++ )
        {
            int count1 = 0;
            for( int j = 0; j < count; j++ )
            {
                if( groupS[j].equals(students[i].getGroup())) break;
                else count1 ++;
            }
            if( count == count1 )
            {
                groupS[count] = students[i].getGroup();
                count++;
            }
        }
        return groupS;
    }
    public String studentsByGroup()
    {
        String[] groupS = addGroups();
        String res = "";
        for( int i = 0; i < nUmGr; i++ )
        {
            res += groupS[i] + "\n";
            for (int j = 0; j < nuM; j++ )
            {
                if( groupS[i].equals(students[j].getGroup())) res += students[j].getInfo() + "\n";
            }
        }
        return res;
    }
    public void removeStudent(String id)
    {
        for( int i = 0; i < nuM; i++ )
        {
            if( id.equals(students[i].getId()) )
            {
                for( int j = i; j < nuM - 1; j++ )
                {
                    students[j] = students[j + 1];
                }
                students[nuM - 1] = null;
                nuM--;
                i = i - 1;
            }

        }
    }
}
