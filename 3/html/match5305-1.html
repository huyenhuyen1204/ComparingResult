public class StudentManagement {
    public static 
boolean sameGroup(Student s1, Student s2){
        if (s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent(Student newStudent){
        students[numOfStudents] = newStudent;
        numOfStudents++;
    }

    public String studentsByGroup(){
        int i;
        String[] id_ = new String[numOfStudents];
        String sum = "";

        for (i=0; i<numOfStudents; i++)
            id_[i] = students[i].getId();

        for (i=0; i<numOfStudents; i++)
            if (!id_[i].equals("___"))
            {
                sum = sum + students[i].getGroup() + "\n";
                sum = sum + students[i].getInfo() + "\n";
                id_[i] = "___";

                for (int j=i+1; j<numOfStudents; j++)
                    if (sameGroup(students[i], students[j])){
                        sum = sum + students[j].getInfo() + "\n";
                        id_[j] = "___";
                    }
            }
        return sum;
    }

    public void removeStudent(String id_){
        
for (int i=0; i<numOfStudents; i++)
            if (students[i].getId().equals(id_))
            {
                for (int j=i; j<numOfStudents; j++)
                    students[j] = students[j+1];
                numOfStudents--;
            }
    }
