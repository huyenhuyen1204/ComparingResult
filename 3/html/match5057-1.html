public class StudentManagement {
    Student students[] = new Student[100];
    int count = 0;

    public StudentManagement() {

    }

    public static  
boolean sameGroup(Student s1, Student s2) {
        if ( (s1.getGroup()).equals(s2.getGroup()) ) return true;
        return false;
    }

    public void addStudent(Student newStudent) {
        students[count] = new Student(newStudent);
        count ++;
    }

    public String studentsByGroup() {
        int[] tick;
        tick = new int[students.length];
        for (int i = 0; i < count; i++) {
            tick[i] = 0;
        }

        String result = "";
        for (int i = 0; i < count; i++) {
            if (tick[i] == 0) {
                result += students[i].getGroup();
                result += "\n";
                result += students[i].getInfo();
                result += "\n";
                tick[i] = 1;
                for (int j = i+1; j < count; j++) {
                    if (sameGroup(students[i], students[j])) {
                        result += students[j].getInfo();
                        result += "\n";
                        tick[j] = 1;
                    }
                }
            }

        }
        return result;
    }

    public void removeStudent(String id) {
        
for (int i = 0; i < count; i++) {
            if ( (students[i].getId()).equals(id) ) {
                for (int j = i; j < count-1; j++) {
                    students[j] = new Student(students[j+1]);
                }
                count--;
                i--;
            }
        }
    }

    public static void main(String[] args) {
        StudentManagement studentManagement = new StudentManagement();
        Student a = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        Student b = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        Student c = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        Student d = new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
        a.setGroup("K62CC");
        b.setGroup("K62CC");

		studentManagement.addStudent(a);
        studentManagement.addStudent(d);
        studentManagement.addStudent(c);
        studentManagement.addStudent(b);

        //studentManagement.removeStudent("17020004");
        studentManagement.removeStudent("17020003");
        studentManagement.addStudent(c);
        //studentManagement.addStudent(d);

        String result = studentManagement.studentsByGroup();
        System.out.println(result);

    }
}
