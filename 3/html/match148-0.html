public class Student {
    private String name, id, group, email;

    public Student() {
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    public Student(String name, String id, String email) {
        setName(name);
        setId(id);
        setEmail(email);
        setGroup("K62CB");
    }

    public Student(Student s) {
        this.name =s.getName();
        this.group =s.getGroup();
        this.id =s.getId();
        this.email =s.getEmail();
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public String getInfo() {
        return (name + " - " + id + " - " + group + " - " + email);
    }
}

import java.util.Arrays;

public class StudentManagement {

    private Student[] students = new Student[100];
    private int size = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        students[size] =newStudent;
        size++;
    }

    public String studentsByGroup() {
        String[] groups = new String[100];
        int dem = 0;
        for (int i = 0; i < size; i++) {
            if (!Arrays.asList(groups).contains(students[i].getGroup())) {
                groups[dem] = students[i].getGroup();
                dem++;
            }
        }
        String ans = "";
        for (int i = 0; i < dem; i++) {
            ans = ans + groups[i] + "\n";
            for (int j = 0; j < size; j++)
                if (students[j].getGroup().equals(groups[i])) ans = ans + students[j].getInfo() + "\n";
        }
        return ans;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < size; i++) {
            if (id.equals(students[i].getId())) {
                for (int j = i; j < size - 1; j++) students[j] = students[j + 1];
            }
			
        }
       size--; 
    }
}
