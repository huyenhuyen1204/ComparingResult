public class StudentManagement {
    Student[] students = new Student[10001];
    int n  = 0; // so sinh vien;
    public static boolean sameGroup(Student s1, Student s2) {
       if(s1.getGroup().compareTo(s2.getGroup()) == 0) return true;
       return false;
    }
    public void addStudent(Student newStudent){
        n ++;
        students[n] = newStudent;
    }
    public String studentsByGroup() {
        int[] a = new int[10001];
        String result = "";
        for(int i = 1; i <= n ; i ++)
        {
            if(a[i] == 1) continue;
            a[i] = 1;
            result +=  students[i].getGroup();
            result += "\n";
            result += students[i].getInfo();
                    result += "\n";
            for(int j =  i + 1; j <= n ; j ++)
            {
                if(a[j] == 1) continue;
                if(students[i].getGroup().compareTo(students[j].getGroup()) == 0)
                {
                    a[j] = 1;
                    result += students[j].getInfo();
                    result += "\n";
                }
            }
        }
        return result; 
   }
    public void removeStudent(String id) {
         n --;
         int index = 0;
         for(int i = 1; i <= n ; i ++)
         {
             if(students[i].getId().compareTo(id) == 0)
             {
                 index = i;
                 break;
             }
         }
         for(int i = index; i <= n; i ++)
         {
             students[i] = students[i+1];
         }
    }
