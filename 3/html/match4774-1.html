public class StudentManagement {
    private Student[] students=new Student[100];
    private int soStudents=0;
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup()))
            return true;
        return false;
    }
    public void addStudent(Student newStudent){
        students[soStudents]=new Student(newStudent);
        soStudents++;
    }
    public String studentsByGroup(){
        Boolean[] groupStudent=new Boolean[100];
        for(int i=0;i<soStudents;i++)
            groupStudent[i]=false;
        String list="";
        for(int i=0;i<soStudents;i++){
            if(groupStudent[i]==false) {
                list += students[i].getGroup()+"\n"+students[i].getInfo()+"\n";
                groupStudent[i]=true;

                for (int j = i + 1; j < soStudents; j++) {
                    if (sameGroup(students[i], students[j]) == true) {
                        list += students[j].getInfo() + "\n";
                        groupStudent[j]=true;
                    }
                }
            }
        }
        return list;
    }
    public void removeStudent(String id){
        
for(int i=0;i<soStudents;i++){
            if(students[i].getId().equals(id)){
                for(int j=i;j<soStudents-1;j++){
                    students[j]=new Student(students[j+1]);
                }
                soStudents--;
            }
        }
    }
    public static void main(String[] args){

        Student s1=new Student();
        Student s2= new Student("Nguyen Van An","17020001","17020001@vnu.edu.vn");
        StudentManagement S= new StudentManagement();
        S.addStudent(s1);
        S.addStudent(s2);
        System.out.println(S.studentsByGroup());
        S.removeStudent("17020001");
        System.out.println(S.studentsByGroup());
    }
}
