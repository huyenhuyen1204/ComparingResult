class StudentManagement extends Student {
    static Student[] students;
    static int i;
    static {
        students = new Student[100];
        i = 0;
    }
    public static boolean sameGroup (Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) {
            return true;
        }
        return false;
    }
    static void addStudent (Student newStudent){
        students[i] = newStudent;
        i ++;
    }
    static boolean checkIfNotInGroup (String[] arr, String s, int p) {
        for (int j = 0; j < p; j ++) {
            if (arr [j].equals(s)) {
                return false;
            }
        }
        return true;
    }
    static String studentsByGroup() {
        String[] groups = new String[i];
        int p = 0;
        groups[p] = students[p].getGroup();
        p ++;
        for (int j = 1; j < i; j ++) {
            if (checkIfNotInGroup (groups, students[j].getGroup(), p) == true) {
                groups[p] = students[j].getGroup();
                p ++;
            }
        }
        String list = "";
        for (int k = 0; k < p; k ++) {
            list += groups[k] + "\n";
            for (int j = 0; j < i; j ++) {
                if (students[j].getGroup().equals(groups [k])) {
                    list += students[j].getInfo() + "\n";
                }
            }
        }
        return list;
    }
    static int getPositionOfStudent (String id) {
        for (int j = 0; j < i; j ++) {
            if (students[j].getId() == id) {
                return j;
            }
        }
        return 0;
    }
    static void removeStudent(String id) {
        int pos = getPositionOfStudent(id);
        Student[] tem = new Student [i - 1];
        int x = 0;
        for (int j = 0; j < i; j ++) {
            if (j != pos) {
                tem [x ++] = students [j];
            }
        }
        for (int j = 0; j < i - 1; j ++) {
            students [j] = tem [j];
        }
        i --;
    }
    public static void main (String[] args) {

        Student An = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");;
        An.setGroup("K62CC");

        Student B = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        B.setGroup("K62CC");

        Student C = new Student ("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        C.setGroup("K62CB");

        Student D = new Student ("Nguyen Van D", "17020004", "17020004@vnu.edu.vn");
        D.setGroup("K62CB");

        Student F = new Student ("Nguyen Van F", "17020004", "17020004@vnu.edu.vn");
        F.setGroup("K62CD");

        addStudent(An);
        addStudent(B);
        addStudent(C);
        addStudent(D);
        addStudent(F);


        System.out.println (studentsByGroup());
        removeStudent (C.getId());
        System.out.println (studentsByGroup());

    }
}
