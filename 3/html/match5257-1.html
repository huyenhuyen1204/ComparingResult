import java.util.*;
public class StudentManagement {
    // TODO: khai bao thuoc tinh students la array chua cac doi tuong thuoc lop Student (max. 100)
    private Student[] students = new Student[100] ;
    private int count = 0 ;
    public static boolean sameGroup(Student s1, Student s2) {
        // TODO:
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student newStudent){
        this.students[count] = newStudent;
        count++;
        // TODO:
    }
    public void removeStudent(String id) {
        for (int i = 0 ; i < count ; i++  ){
            if (this.students[i].getId().equals(id)){
                for (int j = i ; j < count ; j++){
                    if (j == count-1) break;
                    students[j] = students[j+1];
                }
                count--;
                break;

            }
        }
        // TODO:
    }
    public String studentsByGroup() {
        StringBuilder SBG = new StringBuilder("");
        boolean[] check = new boolean[100];
        Arrays.fill(check,false);
        for (int i = 0; i < count ; i++){
            if(!check[i]) {
                
SBG.append(students[i].getGroup() + "\n");
                for (int j = i; j < count; j++) {
                    if (sameGroup(students[j], students[i])) {
                        SBG.append(students[j].getInfo() + "\n");
                        check[j] = true;
                    }
                }
            }
        }

        return SBG.toString();
    }

    public static void main(String[] args) {
        StudentManagement s = new StudentManagement();
    }
}
