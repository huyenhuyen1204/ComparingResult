public 
class StudentManagement {

    private Student[] students = new Student[100];
    private int soSV = 0;


    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }


    public void addStudent(Student newStudent) {
        students[soSV++] = new Student(newStudent);
    
}
    public void removeStudent(String id){
        for(int i = 0; i < soSV; i++){
            if(students[i].getId().equals(id)){
                for(int j = i; j < soSV-1; j++){
                    students[j] = students[j+1];
                }
                soSV--;
                break;
            }
        }
    }
    public String findByGroup(String group) {
        String res = group;
        for(int i = 0; i < soSV; i++) {
            if(group.equals(students[i].getGroup())) {
                res = res + "\n" + students[i].getInfo();
            }
        }
        return res;
    }
    public String studentsByGroup() {
        String 
allGroup =students[0].getGroup();
        String res = findByGroup(allGroup);
        for(int i = 1; i < soSV; i++) {
            String aGroup = students[i].getGroup();
            if(!allGroup.contains(aGroup)) {
                allGroup = allGroup + aGroup;
                res = res + "\n" + findByGroup(aGroup);
            }
        }
        return res;
    }

    }
    
}
