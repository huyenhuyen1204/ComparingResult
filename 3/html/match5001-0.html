public class Student {
    private String name,id,group,email;

    public void setName(String st){
        name=st;
    }
    public void setId(String st){
        id=st;
    }
    public void setGroup(String st){
        group=st;
    }
    public void setEmail(String st){
        email=st;
    }
    public String getName(){
        return this.name;
    }
    public String getId(){
        return this.id;
    }
    public String getGroup(){
        return this.group;
    }
    public String getEmail(){
        return this.email;
    }

    public String getInfo(){
        return (name + " - " + id + " - " + group + " - " + email);
    }

    public Student(){
        this.setName("Student");
        this.setId("000");
        this.setGroup("K62CB");
        this.setEmail("uet@vnu.edu.vn");
    }
    public Student(String stN, String stId, String stEm){
        this.setName(stN);
        this.setId(stId);
        this.setGroup("K62CB");
        this.setEmail(stEm);
    }
    public Student(Student st){
        this.setName(st.getName());
        this.setId(st.getId());
        this.setGroup(st.getGroup());
        this.setEmail(st.getEmail());
    }
}

import java.util.*;

public class StudentManagement {
    private static final int N=100;
    int slS=0;
    int[] did=new int[N];
    Student[] students=new Student[N];

    public static void main(String[] args){
        Student an=new Student();
        an.setName("Nguyen Van An");
        an.setId("17020001");
        an.setGroup("K62CC");
        an.setEmail("17020001@vnu.edu.vn");

        Student B=new Student(an);
        B.setName("Nguyen Van B");
        B.setId("17020002");
        B.setEmail("17020002@vnu.edu.vn");

        Student C=new Student("Nguyen Van C","17020003","17020003@vnu.edu.vn");

        Student D=new Student("Nguyen Van D","17020004","17020004@vnu.edu.vn");

        StudentManagement test=new StudentManagement();
    }
    public static boolean sameGroup(Student s1, Student s2){
        return (s1.getGroup().compareTo(s2.getGroup()) == 0);
    }
    public void addStudent(Student newStudent){
        if (slS == 100) return;
        students[slS++]=new Student(newStudent);
    }
    public String studentsByGroup(){
        String rs="";
        Arrays.fill(did,0);
        for (int i=0; i < slS; i++)
            if (did[i] == 0){
                rs=rs + students[i].getGroup() + '\n';
                for (int j=i; j < slS; j++)
                    if (sameGroup(students[i],students[j])){
                        did[j]=1;
                        rs=rs + students[j].getInfo() + '\n';
                    }
            }
        return rs;
    }
    public void removeStudent(String stId){
        int i=0;
        while (i < slS){
            if (stId.compareTo(students[i].getId()) == 0){
                for (int j=i; j < slS-1; j++)
                    students[j]=new Student(students[j+1]);
                slS--;
            }
            else{
                i++;
            }
        }
    }
}
