class Student {
    private String name;
    private String id;
    private String group;
    private String email;

    Student(){
        name = "Student";
        id = "000";
        group = "K62CB";
        email = "uet@vnu.edu.vn";
    }

    public String getName(){
        return name;
    }
    public void setName(String name){
        this.name = name;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public String getGroup() {
        return group;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getEmail() {
        return email;
    }

    public void setEmail(String email) {
        this.email = email;
    }
    
String getInfo(){
        return(name + " - " + id + " - " + group + " - " + email);
    }
    Student(String name, String id, String email) {
        this.name = name;
        this.id = id;
        this.email = email;
        group = "K62CB";
    }
    Student(Student s){
        this.name = s.getName();
        this.id = s.getId();
        this.group = s.getGroup();
        this.email = s.getEmail();
    }

   
}

public 
class StudentManagement {
    StudentManagement() {
    }

    Student[] students = new Student[100];
    int StudentCount = 0;
    String[] group = new String[100];

    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        else return false;
    }

    public void addStudent(Student newStudent) {
        students[StudentCount] = newStudent;
        StudentCount++;
    }

    public String studentsByGroup() {
        int dem = 1;
        int kt = 0;
        group[0] = students[0].getGroup();
        for (int i = 0; i < StudentCount; i++) {
            for (int j = 0; j < dem; j++) {
                if (students[i].getGroup().equals(group[j])) {
                    kt = 1;
                    break;
                }
            }
            if (kt == 0) {
                group[dem] = students[i].getGroup();
                dem++;

            }
            kt = 0;
        }

        String result = "";

        for (int j = 0; j < dem; j++) {
            result += group[j] + "\n";
            for (int i = 0; i < StudentCount; i++) {
                if (group[j].equals(students[i].getGroup())) {
                    result += students[i].getInfo() + "\n";
                }
            }

        }

        return result;
    }

    public void removeStudent(String id) {
        int x = 0;
        for (int i = 0; i < StudentCount; i++) {
            if (students[i].getId().equals(id)) {
                x = i;
                break;
            }
        }
        for (int i = x; i < StudentCount - 1; i++) {
            students[i] = students[i + 1];
        }
        StudentCount--;
    }
}
