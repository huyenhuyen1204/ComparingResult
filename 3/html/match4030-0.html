import java.util.*;

public class StudentManagement {
    private Student[] students = new Student[100];
    private int countStudents =0;

    public static boolean sameGroup(Student s1, Student s2){
       return (s1.getGroup().equals(s2.getGroup()));

    }
    public void addStudent(Student newStudent){
        if (countStudents < 99 ) {
            students[countStudents] = newStudent;
            countStudents ++;
        }
    }
    public String studentsByGroup(){
        String a = "";
        Set<String> groups = new HashSet<>();
        for (int i = 0; i < countStudents;i++)
        {
            if (!groups.contains(students[i].getGroup()))
            {
                groups.add(students[i].getGroup());
                a += students[i].getGroup() + '\n' + students[i].getInfo() +'\n';
                if (i < countStudents - 1)
                {
                    for (int j = i+1;j<countStudents;j++)
                    {
                        if(sameGroup(students[i],students[j])) a += students[j].getInfo()+'\n';
                    }
                }
            }

        }
        return a;
    }
    public void removeStudent(String id){
        for (int i = 0;i < countStudents; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < countStudents - 1; j++) {
                    students[j] = students[j + 1];
                }
                countStudents--;
            }
        }
        if (students[countStudents -1].getId().equals(id)) countStudents--;
    }
}
