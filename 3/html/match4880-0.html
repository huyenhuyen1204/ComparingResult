public class StudentManagement {

    public static Student[] students = new Student[100];

        public static boolean sameGroup(Student s1, Student s2) {

            if(s1.getGroup().equals(s2.getGroup())) return true;
            return false;
        }
        public static  int i=0;

        public static void addStudent(Student newStudent){
            students[i]= newStudent;
            i++;
        }

        public static String studentsByGroup() {
            String print = "";
            int[] save = new int[100];
            print+= students[0].getGroup()+"\n"+ students[0].getInfo()+"\n";
            save[0]= 1;
            for(int j=0; j<i;j++)
            {
                for(int k =j+1; k<i;k++){
                    if(students[j].getInfo().equals(students[k].getInfo())){
                        save[k]=1;
                    }
                }

            }
            for(int j =0; j<i;j++)
            {
                if(save[j]==0) {
                    print += students[j].getGroup() + "\n" + students[j].getInfo() + "\n";
                    save[j] = 1;
                }
                    for (int k = 1; k < i; k++) {
                        if (sameGroup(students[j], students[k]) == true&& save[k]==0) {
                            print += students[k].getInfo() + "\n";
                            save[k] = 1;
                        }

                    }
                }


            return print;
        }


        public static void removeStudent(String id) {
            for(int j=0; j<i;j++)
            {
                if(students[j].getId().equals(id)){
                    for(int k =j;k<i-1;k++) {
                        students[k] = students[k + 1];
                    }
                    i--;
                   // break;
                }
            }

        }

    public static void main(String[] args) {
        
    }
}
