public class Student {
        private String name,id,email,group;

        public String getName() {
            return name;
        }

        public void setName(String name) {
            this.name = name;
        }

        public String getId() {
            return id;
        }

        public void setId(String id) {
            this.id = id;
        }

        public String getEmail() {
            return email;
        }

        public void setEmail(String email) {
            this.email = email;
        }

        public String getGroup() {
            return group;
        }

        public void setGroup(String group) {
            this.group = group;
        }
        public String getInfo()
        {
            
return name+" - "+id+" - "+group+" - "+email;
        }
        public Student()
        {
            setName("Student");
            setId("000");
            setGroup("K62CB");
            setEmail("uet@vnu.edu.vn");
        }
        public Student(String name,String id,String email)
        {
            setEmail(email);
            setName(name);
            setId(id);
            setGroup("K62CB");
        }
        public Student(Student s)
        {
            setName(s.getName());
            setId(s.getId());
            setGroup(s.getGroup());
            setEmail(s.getEmail());
        }
}

public 
class StudentManagement {
    Student[] students = new Student[100];
    int count_Student = 0;
    public static boolean sameGroup(Student s1, Student s2)
    {
        if(s1.getGroup().equals(s2.getGroup())) return true;
        else    return false;
    }
    public void addStudent(Student newStudent)
    {
        students[count_Student] = new Student();
        students[count_Student] = newStudent ;
        count_Student++;
    }
    public String studentsByGroup()
    {
        String[] Group = new String[100];
        boolean test;
        int count_group =0;
        for(int i=0;i<count_Student;i++)
        {
            if(i==0)    Group[count_group]=students[i].getGroup();
            else {
                test = true;
                for (int j = 0; j < i; j++) {
                    if (sameGroup(students[i], students[j])) {
                        test = false;
                        break;
                    }
                }
                if (test) {
                    count_group++;
                    Group[count_group] = students[i].getGroup();
                }
            }
        }
        String Students = "";
        for(int i=0;i<=count_group;i++)
        {
            Students += Group[i]+ "\n";
            
for(int j=0;j<count_Student;j++)
            {
                if(Group[i].equals(students[j].getGroup()))    Students += students[j].getInfo() + "\n";
            }
        }
        return Students;
    }
    public void removeStudent(String id)
    {
        for(int i=0;i<count_Student;i++)
        {
            if(students[i].getId().equals(id))
            {
                for(int j=i;j<count_Student-1;j++)
                {
                    students[j]=new Student(students[j+1]);
                }
                count_Student--;
            }
        }
    }

    public static void main(String[] args) {
        StudentManagement s = new StudentManagement();
        Student s1 = new Student();
        Student s2 = new Student("cuong","002","asdhfbh");
        s.addStudent(s1);
        s.addStudent(s2);

        System.out.println(s.studentsByGroup());
        s.removeStudent("002");
        System.out.println(s.studentsByGroup());
    }
}
