import java.util.ArrayList;
	
public class StudentManagement {
	private static final int MAX = 100;

	Student[] 
students = new Student[MAX];
	private int length = 0;

	public static boolean sameGroup(Student s1, Student s2) {
		return s1.getGroup().compareTo(s2.getGroup()) == 0;
	}

	public void addStudent(Student newStudent) {
		// if new length is smaller than maximum of array
		if (length + 1 <= MAX) {
			students[length] = newStudent;
			length += 1;
		} else {
			// System.out.println("Oh no!");
		}
	}

	private ArrayList<String> getGroups() {
		ArrayList<String> result = new ArrayList<String>();
		for (int i = 0; i < length; i++) {
			boolean is_existed = false;
			String this_group = students[i].getGroup();
			for (String grouped_str: result) {
				// System.out.println(this_group + ' ' + grouped_str);
				if (this_group.compareTo(grouped_str) == 0) {
					is_existed = true;
					break;
				}
			}
			if (!is_existed) {
				result.add(this_group);
			}
		}
		return result;
	}

	public String studentsByGroup() {
		ArrayList<String> all_groups = getGroups();

		String result = "";
		for (String group: all_groups) {
			
result += group + "\n";
			for (int i = 0; i < length; i++) {
				if (students[i].getGroup().compareTo(group) == 0) {
					result += students[i].getInfo() + "\n";
				}
			}
		}
		return result;
	}

	public void removeStudent(String id) {
		if (length > 0) {
			
for (int i = 0; i < length; i++)
				if (students[i].getId().compareTo(id) == 0) {
					// erase student i
					for (int j = i+1; j < length; j++)
						students[j-1] = students[j];

					length -= 1;
					break;
				}
		} else {
			// System.out.println("Oh no!");
		}
	}
