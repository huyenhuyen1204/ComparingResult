//package com.company;
import java.util.ArrayList;
public class StudentManagement {
    Student[] students;
    int count;
    StudentManagement(){
        students = new Student[100];
        count = 0;
    }
    StudentManagement(Student[] st , int count){
        this.students = st;
        this.count = count;
    }
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }//checked
    public void addStudent(Student newStudent){
        students[count] = newStudent;
        count++;

    }//checked
    //equal

    public String studentsByGroup(){
        ArrayList<String> al = new ArrayList<String>();
        String output="";
        for(int i=0; i<count; i++){
            if(!al.contains(students[i].getGroup())){
                al.add(students[i].getGroup());
            }

        }
        for(int i=0; i<al.size(); i++){
            output += al.get(i)+"\n";
            for(int j=0; j<count; j++){

                if(students[j].getGroup()==al.get(i)){
                    output+= students[j].getInfo()+"\n";
                }
            }
        }
        return output;

    }
    public int getAddressId(String id){
        for(int i=0; i<count; i++){
            if(students[i].getId().equals(id)) return i;
        }
        return -1;
    }//checked

    public void removeStudent(String id){
        int idAddress = getAddressId(id);
        for(int i=idAddress; i<count; i++)
            students[i]=students[i+1];
        //delete(student[idArress]);
        count--;
    }//checked
}
