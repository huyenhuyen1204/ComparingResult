import java.io.*;
import java.util.*;

public class StudentManagement {

        Student[] students = new Student[100];
        int count=0;


        public static boolean sameGroup(Student s1, Student s2) {
            if (s1.getGroup().equals(s2.getGroup())) {
                return true;
            }
            return false;
        
}

        public void addStudent(Student newStudent) {
            if(count<100) {
                students[count] = new Student(newStudent);
                count++;
            }
        }

        public String studentsByGroup() {
            boolean[] value = new boolean[count];
            for (int i=0; i<count; i++){
                value[i] = false;
            }
            String result = "";
            for (int i=0; i<count; i++) {
                if (value[i] == false) {
                    String temp = students[i].getGroup();
                    result = result + temp + "\n";
                    for (int j = i; j < count; j++) {
                        if (students[j].getGroup().equals(temp)) {
                            result = result + students[j].getInfo() + "\n";
                            value[j] = true;
                        }
                    }
                }
            }
            return result;
        }


        public void removeStudent(String id){
            
for (int i=0; i<count; i++){
                if (students[i].getId().equals(id)){
                    for (int j=i;j<count-1; j++){
                        students[j] = students[j+1];
                    }
                    count--;
                }
            }

        }

    public static void main(String[] args) {
        StudentManagement SM = new StudentManagement();
        Student s = new Student("18020122", "Vanh", "K63CC", "dgdg");
        Student t = new Student("18020123", "Va", "K63CH", "dgdg");
        Student n = new Student("18020124", "V", "K63CT", "dgdg");
        Student l = new Student("18020125", "ls", "K63CLC", "dgdg");
        Student i = new Student("afeefefe", "LONG", "K63CM", "Longngooo@");
        Student XD = new Student("LOL", "Loud", "K63CLC", " LaughOut@g");
        Student c = new Student("0000000", "NoName", "DoesntMatter", "NoNeed");
        SM.addStudent(c);
        SM.addStudent(t);
        SM.addStudent(n);
        SM.addStudent(l);
        SM.addStudent(i);
        SM.addStudent(XD);
        //SM.removeStudent("18020123");
        System.out.println(SM.studentsByGroup());

        System.out.println(SM.sameGroup(l, XD));



    }
}
