src/Student.java

public class Student {
	//getters
	public String getName() {
		return name;
	}
	public String getId() {
		return id;
	}
	public String getGroup() {
		return group;
	}
	public String getEmail() {
		return email;
	}
	
	//setters
	public void setName(String _name) {
		name = _name;
	}
	public void setId(String _id) {
		id = _id;
	}
	public void setGroup(String _group) {
		group = _group;
	}
	public void setEmail(String _email) {
		email = _email;
	}
	public String getInfo() {
		return name + " - " + id + " - " + group + " - " + email;
	}
	
	//constructors
	public Student(){
		name = "Student";
		id = "000";
		group = "K62CB";
		email = "uet@vnu.edu.vn";
	}
	public Student(String _name, String _id, String _email){
		name = _name;
		id = _id;
		group = "K62CB";
		email = _email;
	}
	public Student(Student s){
		name = s.name;
		id = s.id;
		group = s.group;
		email = s.email;
	}
	
	private String name;
	private String id;
	private String group;
	private String email;
}

src/StudentManagement.java

public class StudentManagement {
	private Student[] students = new Student[100];
	private int number = 0;
	
	//methods
	public static boolean sameGroup(Student s1, Student s2) {
		return s1.getGroup().equals(s2.getGroup());
	}
	
	public void addStudent(Student newStudent) {
		students[number] = newStudent;
		number++;
	}
	
	public String studentsByGroup() {
		String tempGroup = number > 0 ? students[0].getGroup() : "";
		String s = number > 0 ? tempGroup + '\n' : "";
		boolean[] check = new boolean[number];
		
		for(int i = 0; i < number; i++) {
			if(!students[i].getGroup().contains(tempGroup)) {
				tempGroup = students[i].getGroup();
				s += tempGroup + '\n';
			}
			if(!check[i]) {
				for(int j = i; j < number; j++) {
					if(students[j].getGroup().contains(tempGroup)) {
						s += students[j].getInfo() + '\n';
						check[j] = true;
					}
				}
			}
		}
		return s;
	}
	public void removeStudent(String id) {
		for(int i = 0; i < number; i++) {
			if(students[i].getId().equals(id)) {
				for(int j = i; j < number - 1; j++) {
					students[j] = students[j + 1];
				}
				students[number - 1] = null;
				number--;
				System.gc();
			}
		}
	}
	public static void main(String[] args) {
		Student s = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
		s.setGroup("K62CC");
		StudentManagement std = new StudentManagement();
		std.addStudent(s);
	}
}
