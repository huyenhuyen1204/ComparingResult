class StudentManagement {
    Student[] students = new Student[100];
    int length = 0;
    /**
     * Ham kiem tra hai sinh vien cung lop
     * @param s1
     * @param s2
     * @return true hoac false
     */
    public static boolean sameGroup(Student s1, Student s2){
       return s1.getGroup().equalsIgnoreCase(s2.getGroup());
    }

    /**
     * Them hoc sinh vao danh sach hoc sinh
     * @param newStudent
     */
    public void addStudent(Student newStudent){
        students[length++] = newStudent;
    }

    /**
     * Sap xep hoc sinh theo lop va thu tu
     * @return danh sach hoc sinh
     */
    public String studentsByGroup(){
        String temp = new String();
        boolean visited[] = new boolean[length];
        for (int i = 0; i < length; i++){
            visited[i] = false;
        }

        for (int i = 0; i < length; i++){
            if(!visited[i])
                temp += students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
            
for (int j = i + 1; j < length; j++){
                if(sameGroup(students[i], students[j])) {
                    visited[j] = true;
                    temp += students[j].getInfo() + "\n";
                }
            }
        
}

        return temp;
    }

    /**
     * Xoa hoc sinh ra khoi danh sach
     * @param id
     */
    public void removeStudent(String id){
        for (int i = 0; i < length; i++){
            if(students[i].getId().equals(id)) {
                length = length - 1;
                for (int j = i; j < length; j++) {
                    i++;
                    students[j] = students[i];
                }
                break;
            }
        }
    }

    public static void main(String[] args) {

    }
}
