public class StudentManagement {
    Student[] students = new Student[100];
    int n = 0;

    public static 
boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().equals(s2.getGroup()))return true;
        else return false;
    }

    public void addStudent(Student newStudent){
        students[n] = newStudent;
        n++;
    }

    public String studentsByGroup(){
        String[] group = new String[100];

        group[0] = students[0].getGroup();
        int k = 1;
        for(int i = 1; i < n; i++){
            boolean giongnhau = false;
            for(int j = 0; j < k; j++){
                if(students[i].getGroup().equals(group[j])) {
                    giongnhau = true;
                    break;
                }
            }
            if(giongnhau == false) {
                group[k] = students[i].getGroup();
                k++;
            }
        }

        String s = "";
        for(int i = 0; i < k; i++){
            s = s + group[i];
            for(int j = 0; j < n; j++) {
                if (group[i].equals(students[j].getGroup())) {
                    s =s + "\n";
                    s = s + students[j].getInfo();
                }
            }
            s =s + "\n";
        
}
        return s;
    }

    public void removeStudent(String id) {
    for(int i = 0; i < n; i++){
        if(id.equals(students[i].getId())) {
            n--;
            for(int j = i; j < n; j++) {
                students[j].setName(students[j + 1].getName());
                students[j].setId(students[j + 1].getId());
                students[j].setGroup(students[j + 1].getGroup());
                students[j].setEmail(students[j + 1].getEmail());
            }
        }
    }
    }

    public static void main(String[] args) {
        Student s1 = new Student();
        s1.setName("Nguyen Van An");
        s1.setId("17020001");
        s1.setGroup("K62CC");
        s1.setEmail("17020001@vnu.edu.vn");
        Student s2 = new Student();
        s2.setName("Nguyen Van B");
        s2.setId("17020002");
        s2.setGroup("K62CC");
        s2.setEmail("17020002@vnu.edu.vn");
        Student s3 = new Student();
        s3.setName("Nguyen Van C");
        s3.setId("17020003");
        s3.setGroup("K62CB");
        s3.setEmail("17020003@vnu.edu.vn");
        Student s4 = new Student();
        s4.setName("Nguyen Van D");
        s4.setId("17020004");
        s4.setGroup("K62CB");
        s4.setEmail("17020004@vnu.edu.vn");

        StudentManagement sv = new StudentManagement();
        sv.addStudent(s1);
        sv.addStudent(s2);
        sv.addStudent(s3);
        sv.addStudent(s4);

        sv.removeStudent("17020001");
        System.out.print(sv.studentsByGroup());
    }

}
