import  java.util.*;
public class StudentManagement {
    public static void main(String[] args) {
        Student a = new Student("Dat", "121231", "thanhdat");
        Student b = new Student("Nam", "1212", "dasdas");
        Student c = new Student("Ngoc", "asas", "aasasdas");
        Student d = new Student("Hung", "12121", "thsdfsd");
        Student e = new Student("Phong", "213", "efsdfsd");
        Student f = new Student("thanh", "1212" ,"asdasd");
        Student asd = new Student("asdas", "asdasd", "asdasdasd");
        e.setGroup("K63CD");
        d.setGroup("K63CC");
        f.setGroup("K63CC");
        StudentManagement t = new StudentManagement();
        t.addStudent(a);
        t.addStudent(b);
        t.addStudent(c);
        t.addStudent(d);
        t.addStudent(e);
        t.addStudent(asd);
        t.addStudent(f);
        System.out.println(t.studentsByGroup());
    }

    private  Student[] students = new Student[100];
    private int count = 0;
    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public String studentsByGroup() {
        Set<String> store = new LinkedHashSet<>();
        for(Student a : students){
            if(a != null && store.contains(a.getGroup()) == false){
                store.add(a.getGroup());
            }
        }
        String Print = "";
        for(String s : store){
            String AGroup = s;
            for(Student a : students){
                if(a!= null && a.getGroup().equals(s)){
                    AGroup = AGroup + "\n" + a.getInfo();
                }
            }
            Print = Print + AGroup + "\n";
        }
        return Print;
    }
    public void addStudent(Student newStudent) {
        students[count] = newStudent;
        count++;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < count; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j < count; j++) {
                    students[j] = students[j + 1];
                }
                count--;
                break;

            }

        }
    }
}
