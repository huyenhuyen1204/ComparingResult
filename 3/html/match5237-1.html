class StudentManagement {
    private Student[] students = new Student[100];
    private int numStudents = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals( s2.getGroup() );
    }

    public void addStudent(Student newStudent) {
        students[numStudents++] = newStudent;
    }

    public void removeStudent(String id) {
        //remove Student with id = id
        for (int i = 0; i < numStudents; i++)
            if (students[i].getId().equals( id )) {
                //remove this student from array
                for (int j = i; j < numStudents - 1; j++) students[j] = students[j + 1];
                numStudents--;
                break;
            }
    }

    public String studentsByGroup() {
        String res = ""; //where we store the sorted list
        boolean[] check = new boolean[100]; //has been put to the list?
        for (int i = 0; i < numStudents; i++)
            if (check[i] == false) {
                //store all student with group equal group of student i
                String curGroup = students[i].getGroup();
                res += curGroup + "\n"; //add that group to the list

                for (int j = i; j < numStudents; j++)
                    if (students[j].getGroup().equals( curGroup )) {
                        check[j] = true;
                        res += students[j].getInfo() + '\n'; //add info of that student to the list
                    }
            }
        return res;
    }
}
