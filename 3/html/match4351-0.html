public class StudentManagement {
    public static Student students[]=new Student[100];
    public static int dem=0;



    public static boolean sameGroup(Student s1, Student s2) {

        String l1=s1.getGroup();
        String l2=s2.getGroup();
        if(l1.equals(l2))
        {
            return true;
        }
        return false;
    }

    public static void addStudent(Student newStudent){

        students[dem]=newStudent;
        dem++;
        return;

    }



    public static  String studentsByGroup() {

        String danhsach = "";
        int kiemtra[] = new int[dem];
        for (int i=0; i<dem; ++i) kiemtra[i]=1;

        for (int i=0; i<dem; ++i)
        {
            if (kiemtra[i] == 1)
            {
                String group = students[i].getGroup();
                danhsach += group + "\n" + students[i].getInfo() + "\n";
                kiemtra[i] = 0;

                for (int j=i+1; j<dem; ++j)
                    if (students[j].getGroup().equals(group))
                    {
                        danhsach += students[j].getInfo() + "\n";
                        kiemtra[j] = 0;
                    }
            }
        }
        return danhsach;

    }



    public static void removeStudent(String id) {

        int i=0;
        while (i<dem)
        {
            if (students[i].getId().equals(id))
            {
                for (int j=i ; j < dem-1; ++j)
                {
                    students[j] = students[j+1];
                }
                --dem;
            }
            else ++i;
        }

        i = 0;

        while (i < dem - 1)
        {
            id = students[i].getId();
            int k=i+1;
            while (k<dem)
            {
                if (students[k].getId().equals(id))
                {
                    for (int j=k ; j<dem-1; ++j)
                    {
                        students[j]=students[j+1];
                    }
                    dem--;
                }
                else k++;
            }
            i++;
        }

        return;

    }



    public static void main(String[] args) {

        Student s1 = new Student("NVA", "17020000", "1701@vnu.edu.vn");
        Student s2 = new Student("XYZ", "17020001", "1702@vnu.edu.vn");
        Student s3 = new Student("ABC", "17020002", "1703@vnu.edu.vn");
        Student s4 = new Student("BBB", "17020003", "1704@vnu.edu.vn");
        s1.setGroup("K62CC");
        s2.setGroup("K62CC");
        addStudent(s1);
        addStudent(s2);
        addStudent(s3);
        addStudent(s4);
        addStudent(s4);
        addStudent(s3);
        removeStudent("17020001");
        System.out.println(studentsByGroup());


    }
}
