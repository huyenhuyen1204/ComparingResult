import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;

public class StudentManagement {
	//private ArrayList<Student> students = new ArrayList<Student>();
	private int quantity;
	private Student[] students;
	
	public StudentManagement() {
		 students = new Student[100];
		 quantity = 0;
	
}
	
	public static boolean sameGroup(Student s1, Student s2) {
		return s1.getGroup().equals(s2.getGroup());
	}
	
	public void addStudent(Student newStudent) {
		//students.add(newStudent);
		students[quantity] = newStudent;
		quantity++;
	}
	
	public void removeStudent(String id) {
		int index = -1;
		for (int i = 0; i < quantity; i++) {
			if (students[i].getId().equals(id)) {
				index = i;
				break;
			}
		}
		if (index != -1) {
			for (int i = index; i < quantity; i++) {
				students[i] = students[i+1];
			}
			students[quantity-1] = null;
			quantity--;
		}
	}
	
	public String studentsByGroup_HashMap() {
		StringBuilder result = new StringBuilder();
		HashMap<String, ArrayList<Student>> listByGroup = new HashMap<String, ArrayList<Student>>();
		for (Student student : students) {
			if (listByGroup.containsKey(student.getGroup())) {
				listByGroup.get(student.getGroup()).add(student);
			}
			else {
				listByGroup.put(student.getGroup(), new ArrayList<Student>());
				listByGroup.get(student.getGroup()).add(student);
			}
		}
		
		for (String group : listByGroup.keySet()) {
			result.append(group + "\n");
			for (Student student : listByGroup.get(group)) {
				result.append(student.getInfo() + "\n");
			}
		}
		return result.toString();
	
}
	
	public String studentsByGroup() {
		StringBuilder result = new StringBuilder();
		ArrayList<String> GroupsByInputOrder = new ArrayList<String>();
		HashMap<String, ArrayList<Student>> listByGroup = new HashMap<String, ArrayList<Student>>();
		for (int i = 0; i < quantity; i++) {
			Student student = students[i];
			if (listByGroup.containsKey(student.getGroup())) {
				listByGroup.get(student.getGroup()).add(student);
			}
			else {
				listByGroup.put(student.getGroup(), new ArrayList<Student>());
				listByGroup.get(student.getGroup()).add(student);
				GroupsByInputOrder.add(student.getGroup());
			}
		}
		
		for (String group : GroupsByInputOrder) {
			result.append(group + "\n");
			for (Student student : listByGroup.get(group)) {
				result.append(student.getInfo() + "\n");
			}
		}
		return result.toString();
	}
	
	public static void main(String args[]) {
		//Scanner sc = new Scanner(System.in);
		StudentManagement sm = new StudentManagement();
		Student a = new Student("Nguyen Van An", "17020001", "K62CC", "17020001@vnu.edu.vn");
		Student b = new Student("Nguyen Van B", "17020002", "K62CC", "17020002@vnu.edu.vn");
		Student c = new Student("Nguyen Van C", "17020003", "K62CB", "17020003@vnu.edu.vn");
		Student d = new Student("Nguyen Van D", "17020004", "K62CB", "17020004@vnu.edu.vn");
		sm.addStudent(a);
		sm.addStudent(b);
		sm.addStudent(c);
		sm.addStudent(d);
		//sm.removeStudent("17020003");
		
		System.out.println(sm.studentsByGroup());
		//System.out.println(sameGroup(a, b));
	}
}
