public class Student {
    private String name, id, group, email;
    Student(){
        name = "Student";
        id = "000";
        group = "K62CB";
        email = "uet@vnu.edu.vn";
    }
    Student (String name, String id, String email){
        this.name = name;
        this.id = id;
        this.email = email;
        this.group = "K62CB";
    }
    Student(Student s){
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }
    public void setName(String name) {
        this.name = name;
    }
    public String getName(){
        return name;
    }
    public void setId(String id) {
        this.id = id;
    }
    public String getId() {
        return id;
    }

    public void setGroup(String group) {
        this.group = group;
    }

    public String getGroup() {
        return group;
    }

    public void setEmail(String email) {
        this.email = email;
    }

    public String getEmail() {
        return email;
    }
    
String getInfo(){
        String info;
        info = this.getName() + " - " + this.getId() + " - " + this.getGroup() + " - " + this.getEmail();
        return info;
    }

class StudentManagement {
    public static boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }
    public Student [] students = new Student[100];
    public int dem (){
        int i=0;
        while (students[i] != null){
            i++;
        }
        return i;
    }
    public void addStudent (Student newStudent){

        int n = dem();
        students[n] = newStudent;
    }
    public String studentsByGroup(){
        String [] g = new String[10];
        int n=0;
        while(students[n] != null){
            n++;
        }
        int k = 1;
        if(students[0] != null) {
            g[0] = students[0].getGroup();
            for (int i = 1; i < n; i++) {
                int tmp = 0;
                for (int j = 0; j < k; j++) {
                    if (students[i].getGroup().equals(g[j])) {
                        tmp = 1;
                        break;
                    }
                }
                if (tmp == 0) {
                    
g[k] = students[i].getGroup();
                    k++;
                }
            }
        }
        String DS ="";
        for(int i=0; i<k-1; i++){
            DS +=g[i];
            for(int j=0; j<n; j++) {
                if (g[i].equals(students[j].getGroup())) {
                    DS += "\n";
                    DS += students[j].getInfo();
                }
            }
            DS += "\n";
        }
        DS += g[k-1];
        for(int j = 0; j<n; j++){
            if(g[k-1].equals(students[j].getGroup())){
                DS += "\n";
                DS += students[j].getInfo();
            }
        
}
        return DS;
    }
    public void removeStudent(String id){
        int n = dem();
        int vt = 0;
        for(int i = 0; i<n; i++){
            if((students[i].getId()).equals(id)){
               vt = i;
            }
        }
        for(int j = vt ; j<n-1; j++){
            students[j] = students[j+1];
        }
        students[n-1] = null;
    }
