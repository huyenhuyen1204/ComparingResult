public class StudentManagement {
    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    Student[] students = new Student[100];
    Boolean[] gr = new Boolean[100];
    int leng = 0;

    public void addStudent(Student newStudent) {
        students[leng] = newStudent;
        gr[leng] = false;
        leng++;
    }

    public String studentsByGroup() {
        String print = "";
        for (int i = 0; i < leng; i++) {
            boolean b = false;
            for (int j = 0; j < i; j++)
                if (sameGroup(students[i], students[j])) {
                    b = true;
                    break;
                }
            if (b == true) continue;
            print = print + students[i].getGroup() + "\n";
            for (int k = i; k < leng; k++) {
                if (sameGroup(students[k], students[i]))
                    print = print + students[k].getInfo() + "\n";
            }
        }
        return print;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < leng; i++) {
            int h = 0;
            if (students[i].getId().equals(id)) {
                h = 1;
                for (int j = i; j < leng - 1; j++) {
                    students[j] = students[j + 1];
                }
                //students[leng - 1] = null;
                if (h == 1) leng--;
                i--;
            }
        }
    }

}
