public class StudentManagement {
    static int i = 0;
    static Student[] students = new Student[100];

    public StudentManagement() {
    }

    static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public static void addStudent(Student newStudent) {
        students[i] = newStudent;
        ++i;
    }


    public static String studentsByGroup() {
        String print = students[0].getGroup()+"\n"+ students[0].getInfo()+"\n";
        int[] flag = new int[100];
        flag[0]= 1;
        for(int j=0; j<i;j++)
        {
            for(int k =j+1; k<i;k++){
                if(students[j].getInfo().equals(students[k].getInfo())){
                    flag[k]=1;
                }
            }
        }
        for(int j =0; j<i;j++)
        {
            if(flag[j]==0)
            {
                print =print+ students[j].getGroup() + "\n" + students[j].getInfo() + "\n";
                flag[j] = 1;
            }
            for (int k = 1; k < i; k++)
            {
                if (sameGroup(students[j], students[k])&& flag[k]==0) {
                    print =print+ students[k].getInfo() + "\n";
                    flag[k] = 1;
                }
            }
        }
        return print;
    }
    public static void removeStudent(String id) {
        for(int j = 0; j < i; ++j) {
            if (students[j].getId().equals(id)) {
                for(int k = j; k < i - 1; ++k) {
                    students[k] = students[k + 1];
                }

                --i;
            }
        }

    }

    public static void main(String[] args) {

    }
}
