class StudentManagement {
    public Student[] students=new Student[100];
    public int count=0;

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        if(count+1<100) {
            students[count] = new Student(newStudent);
            count = count + 1;
        }
    
}
    public void removeStudent(String id) {
        for (int i = 0; i < count; i++) {
            if (id.equals(students[i].getId())) {
                count = count - 1;
                
for (int k=i; k < count; k++) {
                    students[k] = students[k + 1];
                }
            }
        }
    }

    public String studentsByGroup(){
        if (count==0) return "";
        String result="";
        String first=students[0].getGroup();
        boolean[] exist = new boolean[count];
        for (int i=0; i<count; i++){
            if(exist[i]==false){
                first=students[i].getGroup();
                result=result+first+"\n";
                for (int j=i; j<count; j++){
                    if(students[j].getGroup().equals(first)){
                        exist[j]=true;
                        result=result+students[j].getInfo()+"\n";
                    }
                }
            }
        }
        return result;
    }
}
