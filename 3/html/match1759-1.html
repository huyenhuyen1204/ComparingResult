class StudentManagement {

    private static Student students[] = new Student[100];
    private static int studentCount = 0;
    
    public static boolean sameGroup(Student s1, Student s2) {
        String g1 = s1.getGroup();
        String g2 = s2.getGroup();
        		
    	if (g1.equals(g2)) 
        {
        	return true;
        }
        return false; 
    }

    public static void addStudent(Student newStudent){
    	students[studentCount] = newStudent;
    	++studentCount;
    	return;
    }

    public static String studentsByGroup() {
    	String list = "";
    	int check[] = new int[studentCount];
    	for (int i = 0; i < studentCount; ++i) check[i] = 1;
    	
    	for (int i = 0; i < studentCount; ++i)
    	{
    		if (check[i] == 1)
    		{
    			String 
group = students[i].getGroup();
    			list += group + "\n" + students[i].getInfo() + "\n";
    			check[i] = 0;
    			
    			for (int j = i + 1; j < studentCount; ++j)
    				if (students[j].getGroup().equals(group))
    					{
    						list += students[j].getInfo() + "\n";
    						check[j] = 0;
    					}
    		}
    	}
	return list;
    }

    public static void removeStudent(String id) {
        int i = 0;
        
while (i < studentCount)
        {
        	if (students[i].getId().equals(id))
        	{
        		for (int j = i ; j < studentCount - 1; ++j)
        		{
        			students[j] = students[j + 1];
        		}
        		--studentCount;
        	}
        	else ++i;
        }
        
        i = 0;
        
        while (i < studentCount - 1)
        {
        	id = students[i].getId();
        	int t = i + 1;
        	while (t < studentCount)
            {
            	if (students[t].getId().equals(id))
            	{
            		for (int j = t ; j < studentCount - 1; ++j)
            		{
            			students[j] = students[j + 1];
            		}
            		--studentCount;
            	}
            	else ++t;
            }
        	++i;
        }
        
        return;
    }

    public static void main(String[] args) 
    {
    	Student s1 = new Student("NVA", "1701", "1701");
    	Student s2 = new Student("NVB", "1702", "1702");
    	Student s3 = new Student("NVC", "1703", "1703");
    	Student s4 = new Student("NVD", "1704", "1704");
    	
    	s1.setGroup("K62CC");
    	s2.setGroup("K62CC");
    	
    	addStudent(s1);
    	addStudent(s2);
    	addStudent(s3);
    	addStudent(s2);
    	
    	addStudent(s4);
    	addStudent(s1);
    	
    	removeStudent("1702");
    	
    	System.out.println(studentsByGroup());
    }
}
