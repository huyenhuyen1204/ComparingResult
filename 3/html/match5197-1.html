class StudentManagement {

    // Tao mang student (max 100)
    private Student[] students = new Student[100];
    private int numOfStu = 0;

    // Kiem tra s1 va s2 cung nhom
    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }
    //them student vao
    public void addStudent(Student newStudent) {
        students[numOfStu] = new Student(newStudent);
        numOfStu++;
    }

    public String studentsByGroup() {
        String rs = "";
        boolean[] xArray = new boolean[numOfStu];
        for (int i = 0; i < numOfStu; i++) {
            xArray[i] = false;
        }
        for (int i = 0; i < numOfStu; i++) {
            if (!xArray[i]) {
                if (i > 0)
                    rs += "\n";
                rs += students[i].getGroup();
                
for (int j = i; j < numOfStu; j++) {
                    if (sameGroup(students[i], students[j])) {
                        xArray[j] = true;
                        rs += "\n" + students[j].getInfo();
                    }
                }
            }
        }
        return rs;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < numOfStu; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i + 1; j < numOfStu; j++) {
                    students[j - 1] = students[j];
                }
                numOfStu--;
                break;
            }
        }
    }

    public static void main(String[] args) {
        StudentManagement studentManagement = new StudentManagement();
        // Add student 1
        Student student1 = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        student1.setGroup("K62CC");
        studentManagement.addStudent(student1);

        // Add student 2
        studentManagement.addStudent(new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn"));

        // Add student 3
        Student student3 = new Student("Nguyen Van B", "17020002", "17020002@vnu.edu.vn");
        student3.setGroup("K62CC");
        studentManagement.addStudent(student3);

        // Add student 4
        studentManagement.addStudent(new Student("Nguyen Van D", "17020004", "17020004@vnu.edu.vn"));

        // Test remove
//        studentManagement.removeStudent(student1.getId());

        // Test sameGroup
//        System.out.println(studentManagement.sameGroup(student1, student3));

        System.out.println(studentManagement.studentsByGroup());
    }
}
