    public void addStudent(Student newStudent){
//        Student d = new Student(newStudent);
        students[quantity] = newStudent;

    public String studentsByGroup(){
        String result ="";
        Student[] sub_students = new Student[100];
        sub_students = students;
        for (int i=0; i<quantity; i++){
            if(sub_students[i].getGroup() != "0"){
                String groupnow = sub_students[i].getGroup();
                result += groupnow + "\n";
                for(int j=0; j<quantity; j++){
                    if(sub_students[j].getGroup().equals(groupnow)){
                        result += sub_students[j].getInfo() + "\n";
                        sub_students[j].setGroup("0");
                    }
                }
            
}
        }
        return result;


    }
    public void removeStudent(String id) {
        for(int i=0; i<quantity; i++){
            if(students[i].getId().equals(id)){
                for(int j=i; j<quantity; j++){
                    students[j]= students[j+1];
                }
                quantity--;
                break;
            }
        }
    }
