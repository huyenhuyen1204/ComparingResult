public class Student
{
    private String name,id,group,email;

            }
    public Student(String name ,String id ,String email)
    {
        this.name=name;
        this.id=id;
        this.email=email;
        this.group="K62CB";
    }
    public Student(Student s)
    {
        this.name = s.name;
        this.id = s.id;
        this.email = s.email;
        this.group = s.group;
    }
    public void setName(String n)
    {
        this.name=n;
    }
    public void setId(String n)
    {
        this.id=n;
    }
    public void setGroup(String n)
    {
        this.group=n;
    }
    public void setEmail(String n) 
    {
        this.email=n;
    }
    public String getName()
    {
        return name;
    }
    public String getId()
    {
        return id;
    }
    public String getGroup()
    {
        return group;
        
    }
    public String getEmail()
    {
        return email;
    }
  public String getInfo()
  {
     String s;
     s=name+" - " + id + " - " + group + " - " + email ;
     return s;
  }
}

public class StudentManagement
{
    private Student[] students = new Student[100];
    private int top=0;

    public static  boolean sameGroup(Student s1, Student s2)
   {
       return s1.getGroup().equals(s2.getGroup());
   }
   public void addStudent(Student newStudent)
   {
        top++;
        students[top] = new Student(newStudent);
   }
    public String studentsByGroup()
    {
       String s = "";
       int[] arr = new int[100000];
       for (int i=1;i<=top;i++)  arr[i]=0;
       for (int i=1;i<=top;i++)
       {
           if (arr[i]==0)
           {
             s=s+students[i].getGroup()+"\n" + students[i].getInfo()+"\n";
              arr[i]=1;
              for (int j=i+1;j<=top;j++)
              {
                  if (arr[j]==0 && students[i].getGroup().equals(students[j].getGroup()))
                  {
                      s=s+students[j].getInfo()+"\n";
                      arr[j]=1;
                  }
              }
           }
       }
       return s;
    }
    public void removeStudent(String id)
    {
        for (int i=1;i<=top;i++)
        {
            if (id==students[i].getId())
            {
                for (int j=i;j<=top;j++)
                {
                    students[j]=students[j+1];
                }
                top--;
                break;
            }
        }
    }
    public static void main(String[] args)
    {
        StudentManagement sb = new StudentManagement();
        Student a = new Student();
        a.setName("Nguyen Van An");
        a.setEmail("17020001@vnu.edu.vn");
        a.setGroup("K62CC");
        a.setId("17020001");
        sb.addStudent(a);
        
        a.setName("Nguyen Van B");
        a.setEmail("17020002@vnu.edu.vn");
        a.setGroup("K62CC");
        a.setId("17020002");
        sb.addStudent(a);
        
        a.setName("Nguyen Van C");
        a.setEmail("17020003@vnu.edu.vn");
        a.setGroup("K62CB");
        a.setId("17020003");
        sb.addStudent(a);
        
        a.setName("Nguyen Van D");
        a.setEmail("17020004@vnu.edu.vn");
        a.setGroup("K62CB");
        a.setId("17020004");
        sb.addStudent(a);
        
        System.out.println(sb.studentsByGroup());
        
        
        sb.removeStudent("17020004");
        System.out.println(sb.studentsByGroup());
    }
}
