public class StudentManagement {
    Student[] students;
    private int n  = 0; // so sinh vien;
    StudentManagement(){
        students = new Student[1000];
    }
    public static 
boolean sameGroup(Student s1, Student s2) {
        if(s1.getGroup().compareTo(s2.getGroup()) == 0) return true;
        return false;
    }
    public void addStudent(Student newStudent){
            if(n < 100) {
                n++;
                students[n - 1] = newStudent;
            } else System.out.println("ERROR");
    
}
    public String studentsByGroup() {
        int[] a = new int[10001];
        String result = "";
        for(int i = 0; i < n ; i ++)
        {
            if(a[i] == 1) continue;
            a[i] = 1;
            result +=  students[i].getGroup();
            result += "\n";
            result += students[i].getInfo();
            result += "\n";
            for(int j =  i + 1; j < n ; j ++)
            {
                if(a[j] == 1) continue;
                if(students[i].getGroup().compareTo(students[j].getGroup()) == 0)
                {
                    a[j] = 1;
                    result += students[j].getInfo();
                    result += "\n";
                }
            }
        }
        return result;
    }
    public void removeStudent(String id) {
        n --;
        int index = -1;
        for(int i = 0; i < n ; i ++)
        {
            if(students[i].getId().compareTo(id) == 0)
            {
                index = i;
                break;
            }
        }
        if(index >= 0) for(int i = index; i < n; i ++)
        {
            students[i] = students[i+1];
        }
        else n ++;
    }
