public class Student{

    private String name;
    private String id;
    private String group;
    private String email;

    public Student(){
        this.name = "Student";
        this.id = "000";
        this.group = "K62CB";
        this.email = "uet@vnu.edu.vn";
    }

    public Student(String name, String id, String email){
        this.name = name;
        this.id = id; 
        this.email = email;
        this.group = "K62CB";
    }

    public Student(Student s){
        this.name = s.name;
        this.id = s.id;
        this.group = s.group;
        this.email = s.email;
    }

    public Student(String name, String id, String group, String email){
        this.name = name;
        this.id = id;
        this.group = group;
        this.email = email;
    }

    public String getName(){
        return name;
    }

    public void setName(String n){
        this.name = n;
    }

    public String getGroup(){
        return group;
    }

    public void setGroup(String gr) {
        this.group = gr;
    }

    public String getId() {
        return id;
    }

    public void setId(String id) {
        this.id = id;
    }

    public 
void setEmail(String email) {
        this.email = email;
    }

    public String getEmail() {
        return email;
    }

    public String getInfo(){
        return this.name + " - " + this.id + " - " + this.group + " - " + this.email;
    }    
}

public class StudentManagement{

    public Student[] students = new Student[100];
    public int curr = 0;
    private String[] groups = new String[100];
    private int currGr = 0;

    public static boolean sameGroup(Student s1, Student s2){
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        for (int i = 0; i < curr; i++) {
            
if (newStudent.getId().equals(students[i].getId())) {
                return;
            }
        }
        students[curr] = newStudent;
        boolean equal = false;
        for (int i = 0; i < currGr && !equal; i++) {
            if (newStudent.getGroup().equals(groups[i])) {
                equal = true;
            }
        }
        if (!equal) {
            groups[currGr] = newStudent.getGroup();
            currGr++;
        }
        curr++;
    }

    public String studentsByGroup() {
        String res = "";

        for (int i = 0; i < currGr; i++)
        {
            String smallGroup = "";
            boolean hasStudent = false;
            smallGroup += groups[i] + "\n";
            for (int j = 0; j < curr; j++) {
                if (students[j].getGroup().equals(groups[i])) {
                    
                    hasStudent = true;
                    smallGroup += students[j].getInfo() + "\n";
                }
            }
            if (hasStudent) {
                res += smallGroup;
            }
        }
        if (res.length()==0) return null;
	    return res;
    }

    public void removeStudent(String id) {
        int foundIndex = -1;
        for (int i = 0; i < curr && foundIndex == -1; i++) {
            if (students[i].getId().equals(id)) {
                foundIndex = i;
            
}
        }

        if (foundIndex != -1){
            for (int i = foundIndex; i < curr-1; i++) {
                students[i] = students[i+1];
            }
            curr--;
        }
    }
