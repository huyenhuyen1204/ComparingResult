public class StudentManagement {
	   Student [] students = new Student[100];
	   public static boolean sameGroup(Student s1,Student s2)
	   {
		   return ((s1.getGroup()).equals(s2.getGroup()));
	   }
	   public void addStudent(Student newStudent)
	   {
		   int i=0;
		   while(i<100)
		   {
			   if(students[i]==null)
			   {
				   students[i] = new Student(newStudent);
				   return;
			   }
			   i++;
		   }
	   }
	   public void removeStudent(String id)
	   {
		   int index=0;
		   boolean check = false;
		   for(int i=0;i<100;i++)
		   {
			 if(id.equals(students[i].getId()))
			 {
				 index = i;
				 check = true;
			 }
		   }
		   if(check==false) return;
		   int n=0;
		   for(int i=0;i<100;i++)
		   {
			   if(students[i]!=null) n++;
			   else break;
		   }
		   for(int i=index;i<n-1;i++)
		   {
			   students[i] = new Student(students[i+1]);
		   }
		   students[n-1]=null;
	   }
	   public String studentsByGroup()
	   {
		   String refest = "",group = "";
		   int []a = new int[100];
		   int n=0;
		   for(int i=0;i<100;i++)
		   {
			   if(students[i]!=null) n++;
			   else break;
		   }
		   for(int i=0;i<n-1;i++)
		   {
			   if(a[i]==0)
			   {
				   group = students[i].getGroup();
				   refest +=group + "\n";
				   refest+=students[i].getInfo() +"\n";
				   a[i]=1;
			   }
			   for(int j=i+1;j<n;j++)
			   {
				   if(a[j]==0&&sameGroup(students[i],students[j])) {
					   refest+=students[j].getInfo() +"\n";
					   a[j]=1;
				   }
			   }
		   }
		   return refest;
	   }
	}
