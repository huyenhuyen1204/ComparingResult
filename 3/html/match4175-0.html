import java.lang.String;
public class StudentManagement {

    private Student[] students;
    private int size;

    StudentManagement()
    {
        students = new Student[100];
        size = 0;
    }

    public static boolean sameGroup(Student s1,Student s2)
    {
        return (s1.getGroup().equals(s2.getGroup()));
    }

    public void addStudent(Student s)
    {
        if(size == students.length) return;
        students[size] = s;
        size++;
    }

    public String studentsByGroup()
    {
        String ret = new String();
        boolean[] check = new boolean[size];
        for(int i=0;i<size;i++) check[i] = true;
        for(int i=0;i<size;i++) if(check[i]) {
            check[i] = false;
            ret = ret + students[i].getGroup() + "\n" + students[i].getInfo() + "\n";
            for (int j = i + 1; j < size; j++)
                if (students[j].getGroup() == students[i].getGroup()) {
                    check[j] = false;
                    ret = ret + students[j].getInfo() + "\n";
            }
        }
        return ret;
    }

    public void removeStudent(String id)
    {
        for(int i=0;i<size;i++)  if(id.equals(students[i].getId()))
        {
            for(int j=i+1;j<size;j++) students[j-1] = students[j];
            size--;
        }
    }

    public static void main(String[] args)
    {
        StudentManagement manager = new StudentManagement();
        for(int i=0;i<10;i++)
        {
            Student t = new Student("Viet","180200"+i,"uet@vnu.edu.vn");
            manager.addStudent(t);
        }
        String id = "1802000";
        manager.removeStudent(id);
        System.out.print(manager.studentsByGroup());
    }
}
