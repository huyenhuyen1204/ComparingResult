public class StudentManagement {
    public static Student[] students= new Student[100];
    
    public static boolean sameGroup(Student s1, Student s2){
        return (s1.getGroup().equals(s2.getGroup()));
    }
    
    static int size=0;
    public static  void addStudent(Student newStudent){
        students[size]=newStudent;
        size++;
    }
    
    public static  String studentsByGroup() {
        // TODO:
        int [] check= new int[size];
        String result="";
        // ktra sinh vien co thong tin trung nhau
        for(int i=0; i<size; i++)
        {
            for(int j=0; j<i; j++)
            {
                if(students[i].getInfo().equals(students[j].getInfo())) check[i]=1;
            }
        }
        
        for(int i=0; i<size;i++ ){
            if(check[i]==0)
            {
                result+= (students[i].getGroup()+"\n");
                for(int j=i; j<size; j++)
                {
                    if(sameGroup(students[j], students[i]))
                    {   
                        if(check[j]==0){
                            result +=(students[j].getInfo()+"\n");
                            check[j]=1;    
                        }
                    }
                }
            }
        }
	return result;
    }

    public static void removeStudent(String id) {
        // TODO:
        
        for(int i=0; i<size; i++)
        {
            if(students[i].getId().equals(id))
               {
                for(int j=i; j<size; j++)
                {
                    students[j]=students[j+1];
                }
                size--;
                i--;
            }
        }
    }

    }

}
