public class StudentManagement {
    private Student[] students = new Student[100];
    private String[] groupList = new String[100];
    private int studentNum = 0, 
groupNum = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        if (s1.getGroup().equals(s2.getGroup())) return true;
        return false;
    }

    public void addStudent(Student newStudent) {
        students[studentNum] = new Student(newStudent);
        studentNum++;
        boolean add = 
false;
        for (int i = 0; i < groupNum; i++) {
            if (groupList[i] == newStudent.getGroup()) {
                add = true;
                break;
            }
        }
        if (!add) {
            groupList[groupNum] = newStudent.getGroup();
            groupNum++;
        }
    }

    public String studentsByGroup() {
        String 
result = "";
        for (int i = 0; i < groupNum; i++) {
            result += groupList[i] + "\n";
            for (int j = 0; j < studentNum; j++) {
                if (
students[j].getGroup() == groupList[i]) {
                    result += students[j].getInfo() + "\n";
                }
            }
        }
        return result;
    }

    public void removeStudent(String id) {
        for (int i = 0; i < studentNum; i++) {
            if(students[i].getId().equals(id)){
                for(int j=i; j<studentNum-1; j++){
                    students[j] = students[j+1];
                }
                studentNum--;
            }
        }
    }
}
