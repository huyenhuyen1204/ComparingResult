import java.util.ArrayList;

import java.util.List;

public class StudentManagement {
    private Student[] students = new Student[100];
    private byte numOfStudents = 0;

    public static boolean sameGroup(Student s1, Student s2) {
        return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent) {
        if (numOfStudents < 100) students[numOfStudents] = newStudent;
        numOfStudents++;
    }

    public String studentsByGroup() {
        String result = "";
        List<String> groups = new ArrayList<String>();

        for (int i = 0; i < numOfStudents; i++) {
            if (groups.indexOf(students[i].getGroup()) == -1) groups.add(students[i].getGroup());
        }

        for (int i = 0; i < groups.size(); i++) {
            result += groups.get(i) + "\n";

            for (int j = 0; j < numOfStudents; j++) {
                if (students[j].getGroup().equals(groups.get(i))) result += students[j].getInfo() + "\n";
            }
        }

        return result.substring(0,result.length() - 1);
    
}

    public void removeStudent(String id) {
        for (int i = 0; i < numOfStudents; i++) {
            if (students[i].getId().equals(id)) {
                for (int j = i; j + 1 < numOfStudents; j++)
                    students[j] = students[j + 1];
                break;
            }
        }
        numOfStudents--;
    }

    public static void main(String[] args) {
    }
}
