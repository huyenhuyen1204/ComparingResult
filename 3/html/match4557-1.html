import java.io.*;
import java.lang.reflect.Array;
import java.util.*;

public class StudentManagement {
     Student[] students = new Student[100];
     int dem = 0;


    public static 
boolean sameGroup(Student s1, Student s2){
        if(s1.getGroup().equals(s2.getGroup())) {
            return true;
        }
        return false;
    }


    public void addStudent(Student newStudent){
        if(dem<100){
            students[dem] = new Student(newStudent);
            dem++;
        }
    
}


    public String studentsByGroup(){
        boolean[] test = new boolean[dem];
        for(int i = 0; i<dem; i++){
            test[i] = false;
        }

        String kq = "";
        for(int i = 0; i<dem; i++){
            if(test[i] == false){
                String temp = students[i].getGroup();
                kq = kq + temp + "\n";
                for(int j = i; j<dem; j++){
                    if(students[j].getGroup().equals(temp)){
                        kq = kq +students[j].getInfo() + "\n";
                        test[j] = true;
                    }
                }
            }
        }
        return kq;
    }


    public void removeStudent(String id){
        
for(int i = 0; i<dem; i++){
            if(students[i].getId().equals(id)){
                for(int j = i; j<dem-1; j++){
                    students[j] = students[j+1];
                }
                dem--;
            }
        }
    }


    public static void main(String[] args){
        Student studentAn = new Student("Nguyen Van An", "17020001", "17020001@vnu.edu.vn");
        Student studentB = new Student(studentAn);
        studentB.setId("17020002");
        studentB.setEmail("17020002@vnu.edu.vn");
        studentB.setName("Nguyen Van B");
        Student studentC = new Student("Nguyen Van C", "17020003", "17020003@vnu.edu.vn");
        Student studentD = new Student();
        studentD.setName("Nguyen Van D"); studentD.setGroup("K62CC"); studentD.setId("17020004"); studentD.setEmail("17020004@vnu.edu.vn");
        StudentManagement arr = new StudentManagement();
        arr.addStudent(studentAn); arr.addStudent(studentB); arr.addStudent(studentC); arr.addStudent(studentD);
        System.out.println(arr.studentsByGroup());
        arr.removeStudent("17020001");
        arr.removeStudent("17020002");
        arr.removeStudent("17020003");
        arr.removeStudent("");
        System.out.println(arr.studentsByGroup());
    }
}
