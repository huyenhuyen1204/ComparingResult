public class StudentManagement {
    public Student[] students = new Student[100];
    public int studentsize = 0;
    public static boolean sameGroup(Student s1, Student s2) {
       return s1.getGroup().equals(s2.getGroup());
    }

    public void addStudent(Student newStudent){
        students[studentsize++]= newStudent;

    }

    public String studentsByGroup() {
        String String = new String();
        String[] groups = new String[100];
        groups[0] = students[0].getGroup();
        int groupsize = 1;
        int groupCount;
        for (int i = 1; i <studentsize; i++) {
            groupCount = 0;
            for (int j = 0; j <groupsize; j++) {
                if (students[i].getGroup().equals(groups[j])) {
                    groupCount++;

                }
            }
            if (groupCount == 0) {
                groupsize++;
                groups[groupsize-1] = students[i].getGroup();
            }
        }
        for (int i = 0; i <groupsize; i++) {
            String += groups[i]+ "\n";
            for (int j = 0; j < studentsize; j++) {
                if (students[j].getGroup().equals(groups[i])) {
                  String += students[j].getInfo() +"\n";
                }
            }


        }
        return String;
    }
    public void removeStudent(String id) {
        for(int i =0; i<studentsize;i++){
            if (students[i].getId().equals(id)){
                studentsize--;
                for (int j=i;j<studentsize;j++){
                    i++;
                    students[j]=students[i];
                }
                break;
        }

        }
    }

    public static void main(String[] args) {
        Student s1 = new Student("Nguyen Van An", "17020001","A2", "17020001@vnu.edu.vn");

        Student s4 = new Student("C","003","A1","3@gmail");

}
