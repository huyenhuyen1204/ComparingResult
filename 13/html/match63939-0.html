import java.io.FileNotFoundException;
import java.io.*;
import java.util.*;
public class Week8_Task2 {
    public void NPEx () throws NullPointerException
    {
        Object a = null;
        a.toString();
    }
    public String NPExTest()
    {
        try {
            NPEx();
        }
        catch (NullPointerException e)
        {
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }
    public void AIOoBEx() throws ArrayIndexOutOfBoundsException
    {
        int[] a = {1,2,3,4,5};
        System.out.print(a[7]);
    }
    public String AIOoBExTest()
    {
        try
        {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e)
        {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }
    public void ArithEx() throws ArithmeticException
    {
        int v = 1/0;
    }
    public String ArithExTest()
    {
        try
        {
            ArithEx();
        }
        catch (ArithmeticException e)
        {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }
    public void FNFEx() throws FileNotFoundException
    {
        FileInputStream a = new FileInputStream("sdfhap.text");
    }
    public String FNFExTest()
    {
        try
        {
            FNFEx();
        }
        catch (FileNotFoundException e)
        {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }
    public void IOEx() throws IOException
    {
        FileInputStream a = new FileInputStream("gib.text");
    }
    public String IOExTest()
    {
        try
        {
            IOEx();
        }
        catch (IOException e)
        {
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
}
