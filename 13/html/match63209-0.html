import java.lang.*;
import java.io.*;
import java.util.Scanner;

public class Week8_Task2 {
    public void NPEx() throws NullPointerException{
        String a = null;
        System.out.println(a.charAt(4));
    }
    public String NPExTest(){
        try {
            NPEx();
        }catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }
        return  "Không có lỗi";
    }

    public void AIOoBEx() throws ArrayIndexOutOfBoundsException{
        int[] a = new int[10];
        System.out.println(a[10000]);
    }
    public String AIOoBExTest(){
        try {
            AIOoBEx();
        }catch (ArrayIndexOutOfBoundsException e){
            return "Lỗi Array Index Out of Bounds";
        }
        return  "Không có lỗi";

    }
    public void ArithEx() throws ArithmeticException{
        int b =5/0;
    }
    public String ArithExTest(){
        try {
            ArithEx();
        }catch (ArithmeticException e){
            return "Lỗi Arithmetic";
        }
        return  "Không có lỗi";

    }

    public void FNFEx() throws FileNotFoundException{
        FileReader fr = new FileReader("asdfqwe.txt");
    }
    public String FNFExTest(){
        try {
            FNFEx();
        }catch (FileNotFoundException e){
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    public void IOEx() throws IOException{
        File file = new File("test.txt");
        FileInputStream f = new FileInputStream(file);
        f.read();
    }
    public String IOExTest(){
        try {
            IOEx();
        }catch (IOException e){
            return "Lỗi IO";
        }
        return  "Không có lỗi";

    }


    public static void main(String args[]){
        Week8_Task2 task2 = new Week8_Task2();
        System.out.println(task2.NPExTest());
        System.out.println(task2.AIOoBExTest());
        System.out.println(task2.ArithExTest());
        System.out.println(task2.FNFExTest());
        System.out.println(task2.IOExTest());
    }

}
