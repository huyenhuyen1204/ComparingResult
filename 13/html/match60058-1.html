import java.io.File;

public class Week8_Task2 {
    //NullPointerException
    public boolean NPEx(Object o) throws NullPointerException {
        return o.equals("A");
    }
    public String NPExTest() {
        Object o = null;
        try {
            NPEx(o);
            return "Không có lỗi";
        } catch (NullPointerException npex) {
            return "Lỗi Null Pointer";
        }
    }

try {
            AIOoBEx(o);
            return "Không có lỗi";
