import java.io.FileNotFoundException;
import java.io.FileReader;

import java.io.IOException;
import java.io.LineNumberReader;
import java.util.Scanner;

public class Week8_Task2 {
    public static void NPEx() throws NullPointerException {
        String string = null;
        System.out.println(string.length());
    }

    public static String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }

    public static void AIOoBEx() throws ArrayIndexOutOfBoundsException {
        int arr[] = new int[5];
        arr[5] = 100;
    }

    public static String AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public static double ArithEx(double a, double b) {
        if (b == 0) throw new ArithmeticException("Lỗi Arithmetic");
        else return a/b;
    
}

    public static String ArithExTest() {
        try {
            ArithEx(10,0);
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public static void FNFEx() throws FileNotFoundException {
        FileReader fileReader = new FileReader("FNFEx.txt");
    }

    public static String FNFExTest() {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    public static void IOEx(String file) throws IOException {
        FileReader fileName = new FileReader(file);
        LineNumberReader reader = new LineNumberReader(fileName);
        reader.close();
    
}

    public static String IOExTest() {
        try {
            IOEx("IOEx.txt");
        } catch (IOException e) {
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
}
