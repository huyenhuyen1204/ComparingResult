import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class Week8_Task2 {
    public void NPEx() {
        Object s = null;
        if (s == null){
            throw new NullPointerException();
        }
    }

    public String NPExTest() {
        try{
            NPEx();
        }catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }

        return "Không có lỗi";
    }

    public void AIOoBEx() {
        int[] a = new int [9];
        int k = 20;
        if (k > a.length - 1) {
            throw new ArrayIndexOutOfBoundsException();
        }
    }

    public String AIOoBExTest(){
        try {
            AIOoBEx();
        }catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public void ArithEx() {
        int x = 10;
        int y = 0;
        if (y == 0){
            throw new ArithmeticException();
        }
    }

    public String ArithExTest() {
        try{
            ArithEx();
        }catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public void FNFEx() throws FileNotFoundException{
        FileReader fileReader = new FileReader("ndsuhd");
        throw new FileNotFoundException();
    }

    public String FNFExTest() {
        try{
            FNFEx();
        }catch (FileNotFoundException f) {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    public void IOEx() throws IOException{
        FileReader fileReader = new FileReader("path");
        throw new IOException();
    }

    public String IOExTest() {
        try{
            IOEx();
        }catch (IOException e) {
            return "Lỗi IO";
        }

        return "Không có lỗi";
    }
}
