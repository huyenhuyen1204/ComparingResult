import java.io.*;
import java.lang.NullPointerException;
import java.lang.ArrayIndexOutOfBoundsException;
import java.lang.ArithmeticException;

public class Week8_Task2 {
    public int NPEx(){
        String a = null;
        return a.length();
    }
    public String NPExTest(){
        try{
            NPEx();
        }catch(NullPointerException e){
            return "Lỗi Null Pointer";
        }
        return "";
    }
    public int AIOoBEx(){
        int[] arr = {1,2,3};
        return arr[3];
    }
    public String AIOoBExTest(){
        try{
            AIOoBEx();
        }catch(ArrayIndexOutOfBoundsException e){
            return "Lỗi Array Index Out of Bounds";
        }
        return "";
    }
    public int ArithEx(){
        return 5/0;
    }
    public String ArithExTest(){
        try{
            ArithEx();
        }catch(ArithmeticException e){
            return "Lỗi Arithmetic";
        }
        return "";
    }
    public void FNFEx() throws FileNotFoundException {
        BufferedReader br = new BufferedReader(new FileReader(new File("file.txt")));

    }
    public String FNFExTest(){
        try{
            FNFEx();
        }catch(FileNotFoundException e){
            return "Lỗi File Not Found";
        }
        return "";
    }
    public String IOEx() throws IOException {
        BufferedReader rd = new BufferedReader(new FileReader(new File("file.txt")));
        String inputLine = null;
        inputLine = rd.readLine();
        return inputLine;

    }
    public String IOExTest(){
        String string = new String();
        try{
            string = IOEx();
        }catch(IOException e){
            return "Lỗi IO";
        }

    }
}
