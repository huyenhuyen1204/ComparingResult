import java.io.*;

public class Week8_Task2 {
    //chia 0
    public String ArithExTest() {
        String a="";
        try {
            ArithEx();
            a="Không có lỗi";
        } catch (ArithmeticException e) {
            a= "Lỗi Arithmetic";
        }
        return a;
    }

    public static void ArithEx() throws ArithmeticException {
        int i = 5 / 0;
    }

    //loi mang
    public String AIOoBExTest() {
        String a="";
        try {
            AIOoBEx();
            a="Không có lỗi";
        } catch (ArrayIndexOutOfBoundsException e) {
           a="Lỗi Array Index Out of Bounds";
        }
        return a;
    }

    public static void AIOoBEx() throws ArrayIndexOutOfBoundsException {
        int[] a = new int[3];
        a[3] = 4;
    }

    //loi null
    public String NPExTest() {
        String b="";
        try {
            NPEx();
            b="Không có lỗi";
        } catch (NullPointerException e) {
           b="Lỗi Null Pointer";
        }
        return b;
    }

    public static void NPEx() throws NullPointerException {
        String a = null;
        a.equals("7777");
    }

    //loi file
    public String FNFExTest() {
        String b="";
        try {
            FNFEx();
            b="Không có lỗi";
        } catch (java.io.FileNotFoundException e) {
            b="Lỗi File Not Found";
        }
        return b;
    }

    public static void FNFEx() throws java.io.FileNotFoundException {
        File a = new File("aa.txt");
        InputStream a1 = new FileInputStream(a);
    }

    // loi io
    public String IOExTest() {
        String a1="";
        try {
            IOEx();
            a1="Không có lỗi";
        } catch (IOException e) {
            a1="Lỗi IO";
        }
        return a1;
    }

    public static void IOEx() throws IOException {
        File a = new File("aa.txt");
        InputStream a1 = new FileInputStream(a);
    }
}
