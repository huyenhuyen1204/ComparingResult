import java.io.File;

import java.io.FileNotFoundException;
import java.io.IOException;
import java.util.Arrays;

public class Week8_Task2 {
    public void NPEx(Object object) throws NullPointerException
    {
        if (object == null)
        {
            throw new NullPointerException("Lỗi Null Pointer");
        }
    }

    public String NPExTest()
    {
        Object object = null;
        try
        {
            NPEx(object);
            return "Không có lỗi";
        }
        catch (NullPointerException e)
        {
            return e.getMessage();
        }
    }

    public void AIOoBEx(int arr[], int i) throws ArrayIndexOutOfBoundsException
    {
        if (i > arr.length)
        {
            throw new ArrayIndexOutOfBoundsException("Lỗi Array Index Out of Bounds");
        }
    }

    public String AIOoBExTest()
    {
        int[] arr = new int[10];
        int i = 11;
        try
        {
            AIOoBEx(arr, i);
            return "Không có lỗi";
        }
        catch (ArrayIndexOutOfBoundsException e)
        {
            return e.getMessage();
        }
    }

    public void ArithEx(int a, int b) throws ArithmeticException
    {
        if (b == 0) throw new ArithmeticException("Lỗi Arithmetic");
    }

    public String ArithExTest()
    {
        int a = 10;
        int b = 0;
        try
        {
            ArithEx(a, b);
            return "Không có lỗi";
        }
        catch (ArithmeticException e)
        {
            return e.getMessage();
        }
    }

    public void FNFEx(File file) throws FileNotFoundException
    {
        if (file == null) throw new FileNotFoundException("Lỗi File Not Found");
    }

    public String FNFExTest()
    {
        File file = null;
        try
        {
            FNFEx(file);
            return "Không có lỗi";
        }
        catch (FileNotFoundException e)
        {
            return e.getMessage();
        }
    }

    public void IOEx(File file) throws IOException
    {
        if (file == null) throw new IOException("Lỗi IO");
    }

    public String IOExTest()
    {
        File file = null;
        try
        {
            IOEx(file);
            return "Không có lỗi";
        }
        catch (IOException e)
        {
            return e.getMessage();
        }
    }
}
