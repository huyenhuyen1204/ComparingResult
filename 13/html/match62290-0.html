import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

/**
 * Created by CCNE on 04/11/2019.
 */
public class Week8_Task2         {
    static String str = null;
    static int[] arr = new int[100];
    static int denominator = 0;

    public Week8_Task2() {

    }

    public static void NPEx() throws NullPointerException {
        int a = str.length();
    }

    public static String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }


    public static void AIOoBEx() throws ArrayIndexOutOfBoundsException {
        int someval = arr[12341241];
    }

    public static String AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public static void ArithEx() throws ArithmeticException {
        int someval = 1 / denominator;
    }

    public static String ArithExTest() {
        try {
            ArithEx();
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public static void FNFEx() throws FileNotFoundException {
        FileReader f = new FileReader("somefile.txtzv");
    }

    public static String FNFExTest() {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            return ("Lỗi File Not Found");
        }
        return "Không có lỗi";
    }

    public static void IOEx() throws IOException {
        FileReader f = new FileReader("hellothreeee");
    }

    public static String IOExTest() {
        try {
            IOEx();
        } catch (IOException e) {
            return ("Lỗi IO");
        }
        return "Không có lỗi";
    }

    public static void main(String[] args) {
        System.out.println(NPExTest());
        System.out.println(IOExTest());
        System.out.println(FNFExTest());
        System.out.println(ArithExTest());
        System.out.println(AIOoBExTest());


    }

}
