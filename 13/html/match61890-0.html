import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class Week8_Task2 {

    /**
     * Phuong thuc nem ngoai le: NullPointerException
     *
     * @return mot ngoai le
     */
    public boolean NPEx() {
        //TODO: Truy cap vao mot doi tuong co gia tri null
        String test = null;
        return test.equals(null);
    }

    /**
     * Phuong thuc nem ngoai le: ArrayIndexOutOfBoundsException
     *
     * @return mot ngoai le
     */
    public int AIOoBEx() {
        //TODO: Truy cap vao vi tri chi muc khong ton tai trong mot mang
        int[] test = {1, 2, 3};
        return test[3];
    }

    /**
     * Phuong thuc nem ngoai le: ArithmeticException
     *
     * @return mot ngoai le
     */
    public int ArithEx() {
        //TODO: Thuc hien phep chi cho 0
        int a = 1;
        int b = 0;
        return a / b;
    }

    /**
     * Phuong thuc nem ngoai le: FileNotFoundException
     *
     * @return mot ngoai le
     */
    public void FNFEx() throws FileNotFoundException {
        //TODO: Su dung lop FileReader doc noi dung mot tep khong ton tai
        File file = new File("");
        FileReader fr = new FileReader(file);
    }

    /**
     * Phuong thuc nem ngoai le: IOException
     *
     * @return mot ngoai le
     */
    public void IOEx() throws IOException {
        //TODO: Cac ngoai le lien quan den van de Vao/Ra cua file
        File file = new File("");
        FileReader fr = new FileReader(file);
    }

    /**
     * Phuong thuc bat ngoai le: NullPointerException
     *
     * @return ten ngoai le
     */
    public String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return "Lỗi Null Pointer";
        }

        return "Không có lỗi";
    }

    /**
     * Phuong thuc bat ngoai le: ArrayIndexOutOfBoundsException
     *
     * @return ten ngoai le
     */
    public String AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }

        return "Không có lỗi";
    }

    /**
     * Phuong thuc bat ngoai le: ArithmeticException
     *
     * @return ten ngoai le

    public String ArithExTest() {
        try {
            ArithEx();
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }

        return "Không có lỗi";
    }

    /**
     * Phuong thuc bat ngoai le: FileNotFoundException
     *
     * @return ten ngoai le

    public String FNFExTest() {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    /**
     * Phuong thuc bat ngoai le: IOException
     *

    public String IOExTest() {
        try {
            IOEx();
        } catch (IOException e) {

        }
        return "Không có lỗi";

}
