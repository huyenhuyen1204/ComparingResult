import java.io.*;
import java.util.Objects;

public class Week8_Task2 {
    public static 
String NPExTest () {
        String s = "Không có lỗi";
        try {
            NPEx();
        } catch (NullPointerException e){
            s = "Lỗi Null Pointer";
        }
        return s;
    }
    public static 
void NPEx() {
        Objects p = null;
        System.out.println(p.toString());
    }
    public static String AIOoBExTest () {
        String s = "Không có lỗi";
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e){
            s = "Lỗi Array Index Out of Bounds";
        }
        return s;
    }
    public static void AIOoBEx() {
        int[] p = {1,2};
        System.out.println(p[3]);
    
}

    public static String ArithExTest () {
        String s = "Không có lỗi";
        try {
            ArithEx();
        } catch (ArithmeticException e){
            s = "Lỗi Arithmetic";
        }
        return s;
    }
    public static void ArithEx() {
        System.out.println(4/0);
    
}

    public static String FNFExTest(){
        String s = "Không có lỗi";
        try {
            FNFEx();
        } catch (Exception e) {
            s = "Lỗi File Not Found";
        }
        return s;
    }
    public static void FNFEx() throws IOException{
        BufferedReader  rd = new BufferedReader(new FileReader(new File("file.txt")));
    }

    public static String IOExTest() {
        String s = "Không có lỗi";
        try {
            IOEx();
            System.out.println("Không có lỗi");
        } catch (Exception e) {
            s = "Lỗi IO";
        }
        return s;
    }
    public static void IOEx() throws IOException{
        BufferedReader rd = null;
        rd.read();
    }
}
