import java.io.*;
import java.util.Scanner;

public class Week8_Task2 {
    public String NPExTest() {
        String str = null;
        try {
            str.toUpperCase();
        } catch (Exception e) {
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi ";
    }

    public String AIOoBExTest() {
        int[] arr = new int[5];
        try {
            arr[5] = 9;
        } catch (Exception e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public String ArithExTest() {
        try {
            int n = 2 / 0;
        } catch (Exception e) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public String FNFExTest() {
        try {
            FileReader file = new FileReader("out.exe");
        } catch (Exception e) {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    public static String IOExTest() {
        Scanner sc = new Scanner(System.in);
        try {
            int n = sc.nextInt();
        } catch (Exception e) {
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
    }
