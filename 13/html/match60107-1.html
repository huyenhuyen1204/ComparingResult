import java.io.FileReader;
import java.util.Scanner;

public class Week8_Task2 {

    public String NPExTest () {
        String s1 = null;
        String s2 = null;
        try {
            boolean l = s1.equals(s2);
        } catch ( Exception e ){
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }

    public String AIOoBExTest(){
        int[] a = new int[5];
        try {
            int k = a[10];
        } catch ( Exception e ){
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public String ArithExTest () {
        
try {
            int c = 10 / 0;
        } catch ( Exception e ){
            return  "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public String FNFExTest (){
        try {
            FileReader fr = new FileReader("test.inp");
        } catch (Exception e){

        }
        return "Không có lỗi";
    }

    public String IOExTest (){

            int n = sc.nextInt();
        } catch (Exception e){

        }
        return "Không có lỗi";
    }
}
