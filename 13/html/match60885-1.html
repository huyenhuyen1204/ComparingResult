import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class Week8_Task2 {
    public boolean NPEx(Boolean b) throws NullPointerException{
        return b;
    }
    public String NPExTest(){
        Boolean b = null;
        try{
            NPEx(b);

        }catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }
    public int AIOoBEx(int[] a) throws ArrayIndexOutOfBoundsException{
        return a[a.length];
    }
    public String AIOoBExTest(){
        int[] a = new int[1];
        try {
            AIOoBEx(a);
        }catch (ArrayIndexOutOfBoundsException e){
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }
    public int ArithEx(int a, int b) throws ArithmeticException{
        return a / b;
    }
    public String ArithExTest(){
        int a = 1, b = 0;
        try {
            ArithEx(a, b);
        }catch (ArithmeticException e){
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }
    public String FNFEx(String path) throws FileNotFoundException{
        FileReader fileReader = new FileReader(path);
        return fileReader.toString();
    }
    public String FNFExTest(){
        String path = "/1213.txt";
        try {
            FNFEx(path);
        }catch (FileNotFoundException e){
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }
    public String IOEx(String path) throws IOException {
        FileReader fileReader = new FileReader(path);
        return fileReader.toString();
    }
    public String IOExTest(){
        String path = "/1213.txt";
        try {
            IOEx(path);
        }catch (IOException e){
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
}
