import java.io.*;
import java.text.*;

/**
 * Created by CCNE on 22/10/2019.
 */
public class Week8_Task2 {
    public String NPExTest(){
        String s = "Lỗi Null Pointer";;
        try{
            String obj = null;
            int l = obj.length();
        }catch(NullPointerException e){
            return s;
        }
        return s;
    }

    public String AIOoBExTest(){
        String s = "Lỗi Array Index Out of Bounds";
        try{
            int[] a = new int[4];
            int n = a[5];
        }catch(ArrayIndexOutOfBoundsException e){
            return s;
        }
        return s;
    }

    public String ArithExTest(){
        String s = "Lỗi Arithmetic";
        try{
            int n = 5/0;
        }catch(ArithmeticException e){
            return s;
        }
        return s;
    }

    public String FNFExTest(){
        String s = "Lỗi File Not Found";
        try {
            FileReader fr;
            fr = new FileReader("data.txt");
        } catch (FileNotFoundException e) {
            return s;
        }
        return s;
    }

    public String IOExTest(){
        String s = "Lỗi IO";
        try {
            FileWriter fw = null;
            fw = new FileWriter("data.txt");
        } catch (IOException e) {
            return s;
        }
        return s;
    }
}
