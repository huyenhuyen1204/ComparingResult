import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
import java.util.Scanner;

public class Week8_Task2 {
    public void NPEx() throws NullPointerException
    {
        String s = null;
        System.out.println(s.charAt(1));
    }
    public String NPExTest()
    {
        String res = "No Problem";
		try
        {
            NPEx();
        }
        catch (NullPointerException e)
        {
            res = "Lỗi Null Pointer";
        }
		return res;
    }
    public void AIOoBEx() throws ArrayIndexOutOfBoundsException
    {
        int[] a = new int[2];
        System.out.println(a[2]);
    }
    public String AIOoBExTest()
    {
        String res = "No Problem";
		try
        {
            AIOoBEx();
        }
        catch (ArrayIndexOutOfBoundsException e)
        {
            res = "Lỗi Array Index Out of Bounds";
        }
		return res;
    }
    public void ArithEx() throws ArithmeticException
    {
        System.out.println(17/0);
    }
    public String ArithExTest()
    {
        String res = "No Problem";
		try
        {
            ArithEx();
        }
        catch (ArithmeticException e)
        {
            res = "Lỗi Arithmetic";
        }
		return res;
    }
    public void FNFEx() throws FileNotFoundException
    {
        Scanner scanner = new Scanner(new File("dung.txt"));
        scanner.close();
    }
    public String FNFExTest()
    {
        String res = "No Problem";
		try
        {
            FNFEx();
        }
        catch (FileNotFoundException e)
        {
            res = "Lỗi File Not Found";
        }
		return res;
    }
    public void IOEx() throws IOException
    {
        FileReader fileReader = new FileReader("new.org");
        int i = fileReader.read();
    }
    public String IOExTest()
    {
        String res = "No Problem";
		try
        {
            IOEx();
        }
        catch (IOException e)
        {
            res = "Lỗi IO";
        }
		return res;
    }
    public static void main(String[] args)
    {
        Week8_Task2 week8_task2 = new Week8_Task2();
        week8_task2.NPExTest();
        week8_task2.AIOoBExTest();
        week8_task2.ArithExTest();
        week8_task2.FNFExTest();
        week8_task2.IOExTest();
    }
}
