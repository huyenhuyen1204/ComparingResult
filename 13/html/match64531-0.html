import java.io.FileNotFoundException;
import java.io.FileInputStream;
import java.lang.NullPointerException;
import java.lang.ArrayIndexOutOfBoundsException;
import java.lang.ArithmeticException;
import java.io.IOException;


public class Week8_Task2 {
    public static void NPEx() throws NullPointerException {
        String a = null;
    }

    public static String NPExTest() {
        String string = "Không có lỗi";
        try {
            NPEx();
        } catch (NullPointerException a) {
            string = "Lỗi Null Pointer";
        }
        return string;
    }

    public static void AIOoBEx() throws ArrayIndexOutOfBoundsException {
        int a[] = new int[5];
        int b = a[100];
    }

    public static String AIOoBExTest() {
        String string = "Không có lỗi";
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            string = "Lỗi Array Index Out of Bounds";
        }
        return string;
    }

    public static void ArithEx() throws ArithmeticException {
        int a = 2 / 0;
    }

    public static String ArithExTest() {
        String string = "Không có lỗi";
        try {
            ArithEx();
        } catch (ArithmeticException a) {
            string = "Lỗi Arithmetic";
        }
        return string;
    }

    public static void FNFEx() throws FileNotFoundException{
        FileInputStream in = new FileInputStream("input.txt");
    }

    public static String FNFExTest(){
        String string = "Không có lỗi";
        try {
            FNFEx();
        } catch (FileNotFoundException a) {
            string = "Lỗi File Not Found";
        }
        return string;
    }

    public static void IOEx() throws IOException {}

    public static String IOExTest(){
        String string = "Không có lỗi";
        try {
            IOEx();
        } catch (IOException a) {
            string = "Lỗi IO";
        }
        return string;
    }

    public static void main(String[] args) {
        System.out.println(ArithExTest());
    }
}
