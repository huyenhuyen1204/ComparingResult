import java.io.FileNotFoundException;
import java.io.IOException;
import java.io.File;

public class Week8_Task2 {
    public static void NPEx()
    {
        String a = null;
        if(a == null) throw new NullPointerException("Lỗi Null Pointer");
    }

    public String NPExTest()
    {
        try{
            NPEx();
        } catch (NullPointerException N) {
            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }
    public static void AIOoBEx()
    {
        int i = 11;
        int[] arr = new int[10];
        if( i<0 || i >= arr.length) throw new ArrayIndexOutOfBoundsException("Lỗi Array Index Out of Bounds");
    }

    public String AIOoBExTest()
    {
        try{
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException N) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }
    public static void ArithEx()
    {

        int b = 0;
        if(b == 0) throw new ArithmeticException("Lỗi Arithmetic");
    }

    public String ArithExTest()
    {
        try{
            ArithEx();
        } catch (ArithmeticException N) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }
    public static void FNFEx() throws IOException
    {
        String path = "asdfghjkl";
        File f = new File(path);
        if(!f.exists()) throw new FileNotFoundException("Lỗi File Not Found");
    }

    public String FNFExTest()
    {
        try{
            FNFEx();
        } catch (IOException N) {
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }
    public static void IOEx() throws IOException
    {
        String path = "asdfghjkl";
        File f = new File(path);
        if(!f.exists())  throw new IOException("Lỗi IO");
    }

    public String IOExTest()
    {
        try{
            IOEx();
        } catch (IOException N) {
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
}
