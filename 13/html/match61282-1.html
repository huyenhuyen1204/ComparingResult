import java.io.File;

import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Scanner;

public class Week8_Task2 {

    public static String ArithExTest() {
        try{
            ArithEx();
            return "Không có lỗi";
        }catch(ArithmeticException e){
            return "Lỗi Arithmetic";
        }
    }

    public static int ArithEx() throws ArithmeticException{
        int x = 1 / 0;
        return x;
    }

    public static int AIOoBEx() throws ArrayIndexOutOfBoundsException{
        int[] arr = {1};
        return arr[2];
    }

    public static String AIOoBExTest(){
        try{
            AIOoBEx();
            return "Không có lỗi";
        }
        catch (ArrayIndexOutOfBoundsException e){
            return "Lỗi Array Index Out of Bounds";
        }
    }

    public static int NPEx() throws NullPointerException{
        String name = null;
        return name.length();
    }

    public static String NPExTest(){
        try{
            NPEx();
            return "Không có lỗi";
        }
        catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }
    }

    public static Scanner FNFEx() throws FileNotFoundException {
        Scanner x = new Scanner(new File("doc.txt"));
        return x;

    }

    public static String FNFExTest(){
        try{
            FNFEx();
            return "Không có lỗi";
        }
        catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
    }

    public static int IOEx() throws IOException{
        FileReader x = new FileReader("doc.txt");
        return x.read();
    }

    public static String IOExTest(){
        try{
            IOEx();
            return "Không có lỗi";
        }
        catch(IOException e){
            return "Lỗi IO";
        }
    }

    public static void main(String[] args) {
        String a = ArithExTest();
        String b = AIOoBExTest();
        String c = NPExTest();
        String d = FNFExTest();
        String e = IOExTest();

        System.out.println(a);
        System.out.println(b);
        System.out.println(c);
        System.out.println(d);
        System.out.println(e);
    }
}
