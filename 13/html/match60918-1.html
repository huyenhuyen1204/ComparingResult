import java.util.Scanner;
import java.io.*;
import java.lang.ArrayIndexOutOfBoundsException;
import java.lang.NullPointerException;
import java.lang.ArithmeticException;
import java.io.FileNotFoundException;
import java.io.IOException;

public class Week8_Task2 {

    public 
double ArithEx() throws ArithmeticException{
        double a =5;
        double b =0;
        if(b==0) {
            throw new ArithmeticException();
        }else {
            return a/b;
        
}
    }

    public String ArithExTest() {
        try{
            double a = ArithEx();
            return "Không có lỗi";
        }catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
    }

    public 
String NPEx() throws NullPointerException{
        String str = null;
        if(str != null) {
            return str;
        }else {
            throw new NullPointerException();
        }
    }
    public String NPExTest() {
        try{
            String string = NPEx();
            return "Không có lỗi";
        }catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }
    }
    public int AIOoBEx() throws ArrayIndexOutOfBoundsException{
        int [] arrayInt = new int[5];
        int index = 6;
        if(0<=index && index<5){
            return arrayInt[index];
        }else {
            throw new ArrayIndexOutOfBoundsException();
        
}
    }
    public String AIOoBExTest() {
        try{
            int a = AIOoBEx();
            return "Không có lỗi";
        }catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
    }

    public String FNFEx()throws FileNotFoundException{
        File file = new File("file.txt");
        if(file.exists()) {
            return "";
        }else {
            throw new FileNotFoundException();
        }
    }
    public String FNFExTest(){
        try{
            String file = FNFEx();
            return "Không có lỗi";
        }catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
    }
    public int IOEx() throws IOException{
        int age = -3;
        if(age < 0) {
            throw new IOException();
        }else {
            return age;
        }
    }
    public String IOExTest() {
        try{
            int a = IOEx();
            return "Không có lỗi";
        }catch (IOException e) {
            return "Lỗi IO";
        }
    }
}
