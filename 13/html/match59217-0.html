import java.io.FileNotFoundException;
import java.io.IOException;

public class Week8_Task2 {
    public void NPEx() {
        throw new NullPointerException("Lỗi Null Pointer");
    }

    public String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return e.getMessage();
        };
        return "Không có lỗi";
    }

    public void AIOoBEx() {
        throw new ArrayIndexOutOfBoundsException("Lỗi Array Index Out of Bounds");
    }

    public String AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return e.getMessage();
        };
        return "Không có lỗi";
    }

    public void ArithEx() {
        throw new ArithmeticException("Lỗi Arithmetic");
    }

    public String ArithExTest() {
        try {
            ArithEx();
        } catch (ArithmeticException e) {
            return e.getMessage();
        };
        return "Không có lỗi";
    }

    public void FNFEx() throws FileNotFoundException {
        throw new FileNotFoundException("Lỗi File Not Found");
    }

    public String FNFExTest() {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            return e.getMessage();
        };
        return "Không có lỗi";
    }

    public void IOEx() throws IOException {
        throw new IOException("Lỗi IO");
    }

    public String IOExTest() {
        try {
            IOEx();
        } catch (IOException e) {
            return e.getMessage();
        };
        return "Không có lỗi";
    }

}
