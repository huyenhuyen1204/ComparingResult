import java.io.*;
import java.util.*;
public class Week8_Task2 {
    static String res;
    static String NPEx(String s) throws NullPointerException {
        if (s == null) throw new NullPointerException("Lỗi Null Pointer");
        else {
            if (s.equals("OK"))return("OK");
            else return("not OK");
        }

    }

    static String NPExTest() {
        try {
            res = NPEx(null);
        } catch (Exception e) {return(e.getMessage());};
        return res;
    }

    static String AIOoBEx(int[] arr, int x) throws NullPointerException {
        if (x >= arr.length) throw new ArrayIndexOutOfBoundsException("Lỗi Array Index Out of Bounds");
        else {
            return(String.valueOf(arr[x]));
        }

    }

    static String AIOoBExTest() {
        try {
            int a[] = {1,2,3,4,5};
            res = AIOoBEx(a, 5);
        } catch (Exception e) {return(e.getMessage());};
        return res;
    }

    static String ArithEx(int a, int b) throws NullPointerException {
        if (b == 0) throw new ArithmeticException("Lỗi Arithmetic");
        else {
            return(String.valueOf(a/b));
        }

    }

    static String ArithExTest() {
        try {
            res = ArithEx(4,0 );
        } catch (Exception e) {return(e.getMessage());};

        return res;
    }


    static String FNFEx(String path) throws FileNotFoundException {
        try {
            FileReader in = new FileReader(path);
            res = String.valueOf(in.read());

        } catch (FileNotFoundException e) {
            throw new FileNotFoundException("Lỗi File Not Found");
        } catch (IOException e) {
            e.printStackTrace();
        }
        return res;
    }

    static String FNFExTest() {
        try {
            res = FNFEx("abc/def");
        } catch (Exception e) {return(e.getMessage());};

        return res;
    }

    static String IOEx() throws IOException {
        File file;
        BufferedReader fs = new BufferedReader(new InputStreamReader(System.in));
        try {
            fs.close();
            res = String.valueOf(fs.readLine());
            res = String.valueOf(fs.readLine());

        } catch (IOException e) {
            throw new IOException("Lỗi IO");
        }
        finally {
            fs.close();
        }
        return res;
    }

    static String IOExTest() {
        try {
            res = IOEx();
        } catch (Exception e) {return(e.getMessage());};

        return res;
    }

    public static void main(String[] args) {
        System.out.println(NPExTest());
        System.out.println(AIOoBExTest());
        System.out.println(ArithExTest());
        System.out.println(FNFExTest());
        System.out.println(IOExTest());
    }
}
