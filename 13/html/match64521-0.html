import java.io.BufferedReader;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
    public class Week8_Task2 {
        public void NPEx() throws NullPointerException
        {
            String s=null;
            System.out.println(s.length());
        }
        public String NPExTest()
        {
            try
            {
                NPEx();
            }catch (NullPointerException e)
            {
                return "Lỗi Null Pointer";
            }
            return "Không có lỗi";
        }
        public void ArithEx() throws ArithmeticException
        {
            System.out.println(10/0);
        }
        public String ArithExTest()
        {
            try
            {
                ArithEx();
            }catch (ArithmeticException e)
            {
                return "Lỗi Arithmetic";
            }
            return "Không có lỗi";
        }

        {
            try
            {

            {

            }return"Không có lỗi";
        }

        {
            int[] a={1,3};
            System.out.println(a[5]);
        }
        public String AIOoBExTest()
        {
            try
            {
                AIOoBEx();
            }catch (ArrayIndexOutOfBoundsException e)
            {
                return "Lỗi Array Index Out of Bounds";
            }
            return "Không có lỗi";
        }
        public void IOEx()throws IOException
        {
            BufferedReader bufferedReader=new BufferedReader(new FileReader("bao.txt"));
            System.out.println(bufferedReader.readLine());
        }
        public String IOExTest() {
            try {
