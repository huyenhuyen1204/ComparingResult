import java.util.Scanner;

import java.io.*;
import java.lang.ArrayIndexOutOfBoundsException;
import java.lang.NullPointerException;
import java.lang.ArithmeticException;
import java.io.FileNotFoundException;
import java.io.IOException;
public class Week8_Task2 {
    public String NPEx() throws NullPointerException{
        String x = null;
        if(x != null) {
            return x;
        }else {
            throw new NullPointerException();
        }
    }
    public String NPExTest() {
        try{
            String X = NPEx();
            return "Không có lỗi";
        }catch (NullPointerException e){
            return "Lỗi Null Pointer";
        }
    }
    public int AIOoBEx() throws ArrayIndexOutOfBoundsException{
        int [] myArray = new int[10];
        int a = 11;
        if(0<=a && a<10){
            return myArray[a];
        }else {
            throw new ArrayIndexOutOfBoundsException();
        }
    }
    public String AIOoBExTest() {
        try{
            int A = AIOoBEx();
            return "Không có lỗi";
        }catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
    }
    public double ArithEx() throws ArithmeticException{
        double x =2;
        double y =0;
        if(y==0) {
            throw new ArithmeticException();
        }else {
            return x/y;
        }
    }

    public String ArithExTest() {
        try{
            double X = ArithEx();
            return "Không có lỗi";
        }catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
    }
    public String FNFEx()throws FileNotFoundException{
        File file = new File("file.txt");
        if(file.exists()) {
            return "";
        }else {
            throw new FileNotFoundException();
        }
    }
    public String FNFExTest(){
        try{
            String file = FNFEx();
            return "Không có lỗi";
        }catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
    }
    public int IOEx() throws IOException{
        int c = -1;
        if(c < 0) {
            throw new IOException();
        }else {
            return c;
        }
    }
    public String IOExTest() {
        try{
            int C = IOEx();
            return "Không có lỗi";
        }catch (IOException e) {
            return "Lỗi IO";
        }
    }
}
