import java.io.*;

public class Week8_Task2 {
    public static void NPEx() throws NullPointerException{
        Integer integer = 1;
        integer = null;
        int a = integer + 1;
    }

    public static String NPExTest() {
        String ex = "";
        try {
            NPEx();
        }
        catch (NullPointerException e){
            ex = "Lỗi Null Pointer";
        }
        return ex;
    }

    public static void AIOoBEx() throws ArrayIndexOutOfBoundsException{
        int [] arr = new int[1];
        arr[1] = 1;
    }

    public static String AIOoBExTest(){
        String ex = "";
        try {
            AIOoBEx();
        }
        catch (ArrayIndexOutOfBoundsException e){
            ex = "Lỗi Array Index Out of Bounds";
        }
        return ex;
    }

    public static void ArithEx() throws ArithmeticException {
        int a = 5 / 0;
    }

    public static String ArithExTest() {
        String ex = "";
        try {
            ArithEx();
        }
        catch (ArithmeticException e) {
            ex = "Lỗi Arithmetic";
        }
        return ex;
    }

    public static void FNFEx() throws FileNotFoundException {
        FileReader fileReader = new FileReader("a.txt");
    }

    public static String FNFExTest() {
        String ex = "";
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            ex = "Lỗi File Not Found";
        }
        return ex;
    }

    public static void IOEx() throws IOException {
        FileInputStream in = new FileInputStream("a.txt");
    }

    public static String IOExTest() {
        String ex = "";
        try {
            IOEx();
        }
        catch (IOException e) {
            ex = "Lỗi IO";
        }
        return ex;
    }


    public static void main(String[] args) {
        System.out.println(NPExTest());
        System.out.println(AIOoBExTest());
        System.out.println(ArithExTest());
        System.out.println(FNFExTest());
        System.out.println(IOExTest());
    }
}
