import java.io.*;
public class Week8_Task2 {
    public void NPEx() throws java.lang.NullPointerException{
        throw new java.lang.NullPointerException();
    }
    public String NPExTest(){
        String str = null;
        try{
            System.out.println(str.length());
        }catch (java.lang.NullPointerException e){
            return "Lỗi Null Pointer";
        }return "Không có lỗi";
    }


    public void AIOoBEx() throws java.lang.ArrayIndexOutOfBoundsException{
        throw new java.lang.ArrayIndexOutOfBoundsException();
    }
    public String AIOoBExTest(){
        int[] arr = new int[5];
        try {
            arr[27] = 12;
        }catch (java.lang.ArrayIndexOutOfBoundsException e){
            return "Lỗi Array Index Out of Bounds";
        }return "Không có lỗi";
    }



    public void ArithEx() throws java.lang.ArithmeticException{
        throw new java.lang.ArithmeticException();
    }
    public String ArithExTest(){
        int i = 0;
        try{
            System.out.println(27/i);
        }catch (java.lang.ArithmeticException e){
            return "Lỗi Arithmetic";
        }return "Không có lỗi";
    }



    public void FNFEx() throws java.io.FileNotFoundException{
        throw new java.io.FileNotFoundException();
    }
    public String FNFExTest(){
        FileInputStream file;
        try{
            file = new FileInputStream("27.txt");
        }catch (FileNotFoundException e){
            return "Lỗi File Not Found";
        }return "Không có lỗi";
    }


    public void IOEx() throws java.io.IOException{
        throw new java.io.IOException();
    }
    public String IOExTest(){
        FileInputStream file;
        try{
            file = new FileInputStream("27.txt");
        }catch (IOException e){
            return "Lỗi IO";
        }return "Không có lỗi";
    }

    public static void main(String[] args) {

    }




}
