import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class Week8_Task2 {
    void NPEx() throws NullPointerException {
        String s = null;
        if (s == null) throw new NullPointerException("Lỗi Null Pointer");
    }

    String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return e.getMessage();
        }
        return "Không có lỗi";
    }

    void AIOoBEx() throws ArrayIndexOutOfBoundsException {
        int[] arr = new int[1];
        if (arr.length <= 1) throw new ArrayIndexOutOfBoundsException("Lỗi Array Index Out of Bounds");
    }

    String AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return e.getMessage();
        }
        return "Không có lỗi";
    }

    void ArithEx() throws ArithmeticException {
        int dev = 0;
        if (dev == 0) throw new ArithmeticException("Lỗi Arithmetic");
    }

    String ArithExTest() {
        try {
            ArithEx();
        } catch (ArithmeticException e) {
            return e.getMessage();
        }
        return "Không có lỗi";
    }

    void FNFEx() throws FileNotFoundException {
        File file = new File("text.txt");
        if (!file.exists()) throw new FileNotFoundException("Lỗi File Not Found");
    }

    String FNFExTest() {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            return e.getMessage();
        }
        return "Không có lỗi";
    }

    void IOEx() throws IOException {
        throw new IOException("Lỗi IO");
    }

    String IOExTest() {
        try {
            IOEx();
        } catch (IOException e) {
            return e.getMessage();
        }
        return "Không có lỗi";
    }
}
