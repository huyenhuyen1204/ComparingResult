import java.io.FileNotFoundException;

    public void NPEx(){

        System.out.println(name.length());
    
}

        try {
            NPEx();

            return "Lỗi Null Pointer";
        }
        return "Không có lỗi";
    }

    public void AIOoBEx(){
        int arr[] = new int[5];
