import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;

public class Week8_Task2 {
    public String NPEx(Object o) throws NullPointerException {
        return o.toString();
    }

    public String NPExTest() {
        
try {
            NPEx(null);
            return "";
        } catch (NullPointerException e) {
            return "Lỗi Null Pointer";
        }
    }

    public int AIOoBEx(int[] arr, int index) throws ArrayIndexOutOfBoundsException {
        return arr[index];
    }

    public String AIOoBExTest() {
        int[] arr = {1, 2, 3};
        
try {
            AIOoBEx(arr, 4);
            return "";
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
    }

    public int ArithEx(int a, int b) throws ArithmeticException {
        return a / b;
    }

    public String ArithExTest() {
        
try {
            ArithEx(1, 0);
            return "";
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
    }

        FileReader fileReader  = new FileReader(f);
        return;
    }

        try{
            FNFEx(new File(""));

        }

        FileReader fileReader  = new FileReader(f);
        return;
    }
    public String IOExTest(){
        try{
            IOEx(new File(""));
            return "";
        } catch (IOException e){

        }

}
