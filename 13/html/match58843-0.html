import java.io.*;
import java.util.Scanner;

public class Week8_Task2 {

    public void NPEx() throws NullPointerException {
        String s = null;
        s.length();
    }

    public String NPExTest() {
        try {
            NPEx();
        } catch (NullPointerException e) {
            return "Lỗi Null Pointer";
        }
        return "";
    }

    public void AIOoBEx(int arr[], int n) throws ArrayIndexOutOfBoundsException {
        if (n >= arr.length) {
            throw new ArrayIndexOutOfBoundsException();
        }
    }

    public String AIOoBExTest() {
        int arr[] = new int[100];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = 0;
        }
        try {
            AIOoBEx(arr, arr.length + 10);
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "";
    }

    public void ArithEx(int a, int b) throws ArithmeticException {
        if (b == 0)
            throw new ArithmeticException();
    }

    public String ArithExTest() {
        try {
            ArithEx(2,0);
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
        return "";
    }

    public void FNFEx(String fileName) throws FileNotFoundException {
        Scanner scanner = new Scanner(new File(fileName));
    }

    public String FNFExTest() {
        try {
            FNFEx("FNFEx.txt");
        } catch (FileNotFoundException e) {
            return "Lỗi File Not Found";
        }
        return "";
    }

    public void IOEx(String fileName) throws IOException {
        Scanner scanner = new Scanner(new FileReader(fileName));
    }

    public String IOExTest() {
        try {
            IOEx("IOEEx.txt");
        } catch (IOException e) {
            return "Lỗi IO";
        }
        return "";
    }
}
