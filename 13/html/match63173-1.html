import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
import java.lang.*;
import java.io.*;

public class Week8_Task2 {
    public void NPEx() throws java.lang.NullPointerException {
        String s = null;
        System.out.println(s.charAt(5));
    }
    public String NPExTest() {
       try {
           NPEx();
       } catch (NullPointerException e) {
           return "Lỗi Null Pointer";
       }
        return "Không có lỗi";
    }

    public void AIOoBEx() throws java.lang.ArrayIndexOutOfBoundsException {
        int[] arr = new int[10];
        System.out.println(arr[13]);
    }
    public String  AIOoBExTest() {
        try {
            AIOoBEx();
        } catch (ArrayIndexOutOfBoundsException e) {
            return "Lỗi Array Index Out of Bounds";
        }
        return "Không có lỗi";
    }

    public 
void ArithEx() throws java.lang.ArithmeticException {
        int res = 15 / 0;
        System.out.println(res);
    }
    public String ArithExTest() {
        try {
            ArithEx();
        } catch (ArithmeticException e) {
            return "Lỗi Arithmetic";
        }
        return "Không có lỗi";
    }

    public void FNFEx () throws java.io.FileNotFoundException {
        String fileName = "toangcmnr.txt";
        FileReader fir = new FileReader(fileName);
    }
    public String FNFExTest () {
        try {
            FNFEx();
        } catch (FileNotFoundException e) {
            e.printStackTrace();
            return "Lỗi File Not Found";
        }
        return "Không có lỗi";
    }

    public void IOEx() throws java.io.IOException {
        File file = new File("toang.txt");
        FileInputStream f = new FileInputStream(file);
        f.read();
    }
    public String IOExTest() {
        try {
            IOEx();
        } catch (IOException e) {
            return "Lỗi IO";
        }
        return "Không có lỗi";
    }
    public static void main(String args[]){
        Week8_Task2 a = new Week8_Task2();
        System.out.println(a.NPExTest());
        System.out.println(a.AIOoBExTest());
        System.out.println(a.ArithExTest());
        System.out.println(a.FNFExTest());
        System.out.println(a.IOExTest());
    }
}
