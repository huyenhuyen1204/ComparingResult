import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.*;

public class Week8_Task2
{
    public static int NPEx(String str) throws NullPointerException
    {
        return str.length();
    }

    public static String NPExTest()
    {
        try {
            String str = null;
            int len = NPEx(str);
            return "Không có lỗi";
        } catch (NullPointerException ex) {
            return "Lỗi Null Pointer";
        }
    }

    public static int AIOoBEx(int arr[]) throws ArrayIndexOutOfBoundsException
    {
        return arr[-1];
    }

    public static String AIOoBExTest()
    {
        try {
            int arr[] = {0};
            int x = AIOoBEx(arr);
            return "Không có lỗi";
        } catch (ArrayIndexOutOfBoundsException ex) {
            return "Lỗi Array Index Out of Bounds";
        }
    }

    public static void ArithEx(double x, double y) throws ArithmeticException
    {
        double result = x/y;
        throw new ArithmeticException();
    }

    public static String ArithExTest()
    {
        try {
            ArithEx(1,0);
            return "Không có lỗi";
        } catch (ArithmeticException ex) {
            return "Lỗi Arithmetic";
        }
    }

    public static void FNFEx(String file) throws FileNotFoundException
    {
        FileReader f = new FileReader(file);
        throw new FileNotFoundException();
    }

    public static String FNFExTest()
    {
        try {
            FNFEx("file.txt");
            return "Không có lỗi";
        } catch (FileNotFoundException ex) {
            return "Lỗi File Not Found";
        }
    }

    public static void IOEx(String file) throws IOException
    {
        FileReader fileReader = new FileReader(file);
        throw new FileNotFoundException();
    }

    public static String IOExTest()
    {
        try {
            IOEx("file.txt");
            return "Không có lỗi";
        }
        catch (IOException e){
            return "Lỗi IO";
        }
    }

    public static void main(String[] args)
    {
        System.out.println(NPExTest());
        System.out.println(AIOoBExTest());
        System.out.println(ArithExTest());
        System.out.println(FNFExTest());
        System.out.println(IOExTest());
    }
}
