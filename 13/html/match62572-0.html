import java.io.BufferedReader;
import java.io.File;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.IOException;
import java.nio.Buffer;
import java.util.Scanner;

public class Week8_Task2 {

	public static int NPEx() throws NullPointerException
	{
		String str = null;
		return str.length();
	}
	
	public static String NPExTest()
	{
		try {
			int x = NPEx();
			return "Không có lỗi";
		} catch (NullPointerException e) {
			return "Lỗi Null Pointer";
		}
	}
	
	public static int AIOoBEx() throws ArrayIndexOutOfBoundsException
	{
		int []arr = new int[5];
		return arr[5];
	}
	
	public static String AIOoBExTest()
	{
		try {
			int x = AIOoBEx();
			return "Không có lỗi";
		} catch (ArrayIndexOutOfBoundsException e) {
			return "Lỗi Array Index Out of Bounds";
		}
	}
	
	public static double ArithEx() throws ArithmeticException
	{
		return 4 / 0;
	}
	
	public static String ArithExTest()
	{
		try {
			double x = ArithEx();
			return "Không có lỗi";
		} catch (ArithmeticException e) {
			return "Lỗi Arithmetic";
		}
	}
	
	public static void FNFEx() throws FileNotFoundException 
	{
		BufferedReader br = new BufferedReader(new FileReader("OOP.txt"));
	}
	
	public static String FNFExTest()
	{
		try {
			FNFEx();
			return "Không có lỗi";
		} catch (FileNotFoundException e) {
			return "Lỗi File Not Found";
		}
	}
	
	public static void IOEx() throws IOException
	{
		FNFEx();
	}
	
	public static String IOExTest()
	{
		try {
			IOEx();
			return "Không có lỗi";
		} catch (IOException e) {
			return "Lỗi IO";
		}
	}
}
