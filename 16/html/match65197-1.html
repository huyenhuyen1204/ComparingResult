import java.util.List;
import java.util.Random;

public class Week11 {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> arr) {
        if (arr.size() <= 1) return arr;
        Random rand = new Random();
        int pivot = rand.nextInt(arr.size());
        int left = 0, right = arr.size() - 1;
        while (true) {
            while (left < arr.size() && arr.get(left).compareTo(arr.get(pivot)) <= 0) left++;
            while (right >= 0 && arr.get(right).compareTo(arr.get(pivot)) > 0) right--;
            if (left < right) {
                T tmp = arr.get(left); arr.set(left, arr.get(right)); arr.set(right, tmp);
                left++; right--;
            }

            if (left > right) break;
        }
        sortGeneric(arr.subList(0, left)); sortGeneric(arr.subList(left, arr.size()));
        return arr;
    }
}
