import java.util.*;

public class Person<T> implements Comparable<T>{
        private String name;
        private int age;
        private String address;

        public Person() {
        }

        public Person(String name, int age, String address) {
            this.name = name;
            this.age = age;
            this.address = address;
        }

        public String getName() {
            return name;
        }

        public void setName(String name) {
            this.name = name;
        }

        public int getAge() {
            return age;
        }

        public void setAge(int age) {
            this.age = age;
        }

        public String getAddress() {
            return address;
        }

        public void setAddress(String address) {
            this.address = address;
        }


        @Override
        public int compareTo(T obj) {
            if (obj instanceof Person) {
                Person p = (Person) obj;
                int n = this.name.compareTo(p.name);
                if (n != 0) return n;
                if (this.age < p.age) return -1;
                else if (this.age > p.age) return 1;
                else return 0;
            }
            else throw new ClassCastException();
        }
    }

import java.util.List;
import java.util.Random;

public class Week11 {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> a) {
        if (a.size() <= 1) return a;
        Random rand = new Random();
        int p = rand.nextInt(a.size());
        int left = 0, right = a.size() - 1;
        while (true) {
            while (left < a.size() && a.get(left).compareTo(a.get(p)) <= 0) left++;
            while (right >= 0 && a.get(right).compareTo(a.get(p)) > 0) right--;
            if (left < right) {
                T tmp = a.get(left); a.set(left, a.get(right)); a.set(right, tmp);
                left++; right--;
            }

            if (left > right) break;
        }
        sortGeneric(a.subList(0, left)); sortGeneric(a.subList(left, a.size()));
        return a;
    }
}
