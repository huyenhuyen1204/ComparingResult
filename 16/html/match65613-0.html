import java.util.List;

public class Week11 {

    public static <T extends Comparable<T>>List<T> sortGeneric (List<T> a) {
        for (int i=0;i<a.size();i++) {
            for (int j=i+1;j<a.size();j++) {
                if (a.get(i).compareTo(a.get(j)) > 0) {
                    T tg = a.get(i);
                    a.set(i,a.get(j));
                    a.set(j,tg);
                }
            }
        }
        return a;
    }

}
