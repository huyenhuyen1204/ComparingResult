import java.util.ArrayList;
import java.util.List;

public class Week11 {
    public static <t extends Comparable<t>> List<t> sortGeneric(List<t> arr){

        for (int i = 0; i <arr.size()-1 ; i++) {
            for (int j = i+1; j <arr.size() ; j++) {
                if (arr.get(i).compareTo(arr.get(j))>0)
                {
                    t temple=arr.get(j);
                    arr.set(j,arr.get(i));
                    arr.set(i,temple);
                }
            }
        }
        return arr;
    }

    public static void main(String[] args) {
        Person a=new Person("le",19,"113");
        Person b=new Person("le",18,"113");
        Person c=new Person("a",19,"113");
        Person d=new Person("a",20,"113");

        List<Person> people=new ArrayList<Person>();

        people.add(a);
        people.add(b);
        people.add(c);
        people.add(d);

        people=sortGeneric(people);

        for (Person person:people)
        {
            System.out.println(person.getName()+"====="+ person.getAge());
        }

    }


}
