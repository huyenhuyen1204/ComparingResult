import java.util.*;

class StudentComparator<T> implements Comparator<T> {
    
    @Override
    public int compare(T o3, T o4) {
        Person o1= (Person)o3;
        Person o2= (Person)o4;
        if(o1.getName().equals(o2.getName())){
            return o1.getAge()-o2.getAge();
        }
        else return o1.getName().compareTo(o2.getName());
    }
}
public class Week11 implements Comparable {

    public static  <T extends Comparable> List<T> sortGeneric(List<T> arr) {
        if(arr.get(0) instanceof Person){
            Collections.sort(arr,new StudentComparator());
        }else{
            for (int i = 
0; i < arr.size() - 1; i++) {
                for (int j = i + 1; j < arr.size(); j++) {
                    int b = arr.get(i).compareTo(arr.get(j));
                    if (b > 0) {
                        T a = arr.get(j);
                        arr.set(j, arr.get(i));
                        arr.set(i, a);
                    }
                }
            }
        }

        return arr;
    }

class Person  implements Comparable{
    private String 
name;
    private  int age;
    private  String address;

    public Person(){}

    public Person(String name, int age, String address) {
        this.name = name;
        this.age = age;
        this.address = address;
    }


    public String getName() {
        return name;
    }

    public int getAge() {
        return age;
    }

    public String getAddress() {
        return address;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public void setAddress(String address) {
        this.address = address;
    }


    @Override
    public int compareTo(Object o) {
        return 0;
    }
}
