//package Generic;

import java.util.List;
import java.util.Collections;

public class Week11
{
    public static <Type> List SwapElement(List<Type> list,int i,int j)
    {
        if (i != j)
        {
            Collections.swap(list,i,j);
        }
        return list;
    }
    public static <Type extends Comparable<Type>> List sortGeneric(List<Type> arr)
    {
        int i,j;
        for (i = 0;i < arr.size() - 1;i++)
        {
            for (j = i + 1;j < arr.size();j++)
            {
                if (arr.get(i).compareTo(arr.get(j)) > 0)
                {
                    SwapElement(arr,i,j);
                }
            }
        }
        return arr;
    }
}
