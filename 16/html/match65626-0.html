import java.util.ArrayList;
import java.util.List;

public class Person  implements Comparable<Person>{
    private String name;
    private int age;
    private String address;

    public Person(){}

    public Person(String name, int age, String address) {
        this.name = name;
        this.age = age;
        this.address = address;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getName() {
        return name;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public int getAge() {
        return age;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public String getAddress() {
        return address;
    }

    @Override
    public int compareTo(Person person){
        if(this.getName().equals(person.getName())){
            if(this.getAge() > person.getAge()) return 1;
            else if(this.getAge() < person.getAge()) return -1;
        }
        else return this.getName().compareTo(person.getName());
        return 0;
    }

  




}

import java.util.ArrayList;
import java.util.List;

public class Week11 <T> {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> input){
        T t;
        for(int i = 0; i < input.size() - 1; i++) {
            for (int j = i + 1; j <= input.size() - 1; j++) {
                if(input.get(i).compareTo(input.get(j)) > 0){
                    t = input.get(i);
                    input.set(i, input.get(j));
                    input.set(j, t);
                }
            }
        }
        return input;
    }

    public static void main(String[] args) {
        List<String> s = new ArrayList<>();
        s.add("s");
        s.add("as");
        List<Integer> in = new ArrayList<>();
        in.add(4);
        in.add(1);
        s = sortGeneric(s);
        in = sortGeneric(in);
        for(String i: s){
            System.out.println(i + " ");
        }
        System.out.println("\n");
        for(Integer i: in){
            System.out.println(i + " ");
        }
    }


}
