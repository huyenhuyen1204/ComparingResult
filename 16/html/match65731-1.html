import java.util.List;

public class Week11<T extends Comparable<T>> {
    public static 
<T extends Comparable<T>> List<T> sortGeneric(List<T> arr){
        for (int i = 0; i < arr.size(); i++)
            for (int j = i + 1; j < arr.size(); j++)
                if (arr.get(i).compareTo(arr.get(j)) > 0){
                    T tmp = arr.get(i);
                    arr.set(i, arr.get(j));
                    arr.set(j, tmp);
                }
        return arr;
    }
}
