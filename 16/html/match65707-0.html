import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public class Person implements Comparable<Person>{
    private String name;
    private int age;
    private String address;
    public Person(){}

    public Person(String  name , int age, String address) {
        this.name = name;
        this.age = age;
        this.address = address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public String getAddress() {
        return address;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public int getAge() {
        return age;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getName() {
        return name;
    }

    public int compareTo(Person person){
        if(name.equals(person.name)){
            return age - person.age;
        }
        else return name.compareTo(person.name);
    }

    public String toString()
    {
        return name+ " , "+ age +" , "+ address;
    }

    public static void main(String [] args)
    {
        Person e1 = new Person("aTestName", 34,  "dLastName");
        Person e2 = new Person( "aTestName", 31, "pLastName");
        Person e3 = new Person( "aTestName", 31,"sLastName");
        Person e4 = new Person("dTestName", 25 , "zLastName");
        List<Person> employees = new ArrayList<Person>();
        employees.add(e2);
        employees.add(e3);
        employees.add(e1);
        employees.add(e4);
        employees = Week11.sortGeneric(employees);
        for(int i=0; i<employees.size(); i++){
            System.out.println(employees.get(i).toString());
        }
    }
}
