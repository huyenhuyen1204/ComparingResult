public class Person implements Comparable{
    private String name;
    private int age;
    private String address;

    public Person() {}

    public Person(String name, int age, String address) {
        this.name = name;
        this.address = address;
        this.age = age;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public int compareTo(Person other) {
        if (this.getName().equals(other.getName())) {
            if (this.getAge() > other.getAge()) return 1;
            else return -1;
        } else {
            if (this.getName().compareTo(other.getName()) > 0) return 1;
            else return -1;
        }
    }

    @Override
    public int compareTo(Object o) {
        if (o instanceof Person) {
            Person other = (Person)o;
            if (this.getName().equals(other.getName())) {
                if (this.getAge() > other.getAge()) return 1;
                else return -1;
            } else {
                if (this.getName().compareTo(other.getName()) > 0) return 1;
                else return -1;
            }
        }
        return -1;
    }
}

import java.util.ArrayList;
import java.util.List;

public class Week11 {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> arr) {
        if (arr.isEmpty()) return arr;
        for (int i = 0; i < arr.size()-1; i++) {
            for (int j = i + 1; j < arr.size() ; j++) {
                if (arr.get(i).compareTo(arr.get(j)) > 0) {
                    T temp = arr.get(i);
                    arr.set(i, arr.get(j));
                    arr.set(j, temp);
                }
            }
        }
        return arr;
    }

    public static void main(String[] args) {
        List<Person> arr = new ArrayList<>();
        arr.add(new Person("Nguyen A", 22, "1"));
        arr.add(new Person("Nguyen A", 20,"2"));
        arr.add(new Person("Le B", 20, "3"));
        sortGeneric(arr);
        for (int i = 0; i < arr.size() ; i++) {
            System.out.println(arr.get(i).getName() + " " + arr.get(i).getAge());
        }
    }
}
