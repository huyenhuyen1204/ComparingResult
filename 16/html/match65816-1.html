import java.util.Collections;

import java.util.List;

public class Week11 {
	public static <T extends Comparable <T>> List<T> sortGeneric (List <T> arr) {
		for (int i = 0; i < arr.size(); ++i){
			for (int j = 0; j < i; ++j) {
				if (
arr.get(i).compareTo(arr.get(j)) < 0) {
					Collections.swap(arr, i, j);
				}
			}
		}
		return arr;
	}
}
