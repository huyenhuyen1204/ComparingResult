import java.util.List;

public class Week11 {
    public static <T extends Comparable> List<T> sortGeneric(List<T> t){
        for (int i = 0; i < t.size() - 1; i++){
            for (int j = i + 1; j < t.size(); j++){
                if (t.get(i).compareTo(t.get(j)) > 0){
                    T temp = t.get(i);
                    t.set(i, t.get(j));
                    t.set(j, temp);
                }
            }
        }
        return t;
    }
}
