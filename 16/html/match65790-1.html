import java.util.*;
public class Week11 {
    public static <t extends Comparable> List<t> sortGeneric(List<
t> T) {
        for (int i = 0; i < T.size(); i++) {
            for (int j = i + 1; j < T.size(); j++) {
                if (T.get(i).compareTo(T.get((j))) > 0) {
                    t temp = T.get(i);
                    T.set(i, T.get(j));
                    T.set(j, temp);
                }
            }
        }
        return T;
    }
}
