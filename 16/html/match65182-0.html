Person.java

public class Person implements Comparable<Person>{
	private String name;
	private int age;
	private String address;
	
	public void setName(String name) {
		this.name = name;
	}
	
	public void setAge(int age) {
		this.age = age;
	}
	
	public void setAddress(String address){
		this.address = address;
	}
	
	public String getName() {
		return name;
	}
	
	public int getAge() {
		return age;
	}
	
	@Override
	public int compareTo(Person o) {
		Person p2 = (Person)o;
		int value = name.compareTo(p2.getName()); 
		if (value == 0) {
			if (age > p2.getAge()) return 1;
			else if (age < p2.getAge()) return -1;
		} 
		return value;
	}
}

import java.util.List;
public class Week11 {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> list){
        for (int i = 0; i < list.size() - 1; i++) {
            int min = i;
            for (int j = i + 1; j < list.size(); j++) {
                if (list.get(min).compareTo(list.get(j)) > 0) min = j;
            }
            T temp = list.get(min);
            list.set(min, list.get(i));
            list.set(i, temp);
        }
        return list;
    }
}
