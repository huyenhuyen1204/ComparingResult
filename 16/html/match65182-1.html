	public static <T extends Comparable<T>> List<T> sortGeneric(List<T> list){
		for (int i = 0; i < list.size() - 1; i++) {
			int min = i;
			for (int j = i + 1; j < list.size(); j++) {
				if (list.get(min).compareTo(list.get(j)) > 0) min = j;
			}
			
			T temp = list.get(min);
			list.set(min, list.get(i));
			list.set(i, temp);
		}
		return list;
	}
