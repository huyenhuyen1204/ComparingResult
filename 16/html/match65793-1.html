import java.util.ArrayList;
import java.util.Collections;

import java.util.LinkedList;
import java.util.List;

public class Person implements Comparable<Person>{
    public String name;
    public int age;
    public String address;

    public Person(String name, int age, String address){
        this.name = name;
        this.age = age;
        this.address = address;
    }
    public Person(){};

    public int getAge() {
        return age;
    }

    public String getAddress() {
        return address;
    }

    public String getName() {
        return name;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int compareTo(Person s){
        if(name.compareTo(s.name) == 0){
            if(age == s.age){
                if(address.compareTo(s.address) == 0)
                    return 0;
                else if(address.compareTo(s.address) == 1)
                    return 1;
                else
                    return -1;
            }
            else return (age - s.age );
        }
        else if(name.compareTo(s.name) < 0)
            return -1;
        else
            return 1;
    }

    public String toString(){
        String res;
        res = this.name + ", " + this.age + ", " + this.address;
        return res;
    
}

        Person p1 = new Person("nguyena", 22, "namdinh");
        Person p2 = new Person("nguyena", 21, "hanoi");
        Person p3 = new Person("trandan", 22, "thaibinh");
        List<Person> list = new ArrayList<Person>();
        list.add(p1);
        list.add(p2);
        list.add(p3);

        for(Person i : list){
            System.out.println(i.toString());
        }
    }
}
