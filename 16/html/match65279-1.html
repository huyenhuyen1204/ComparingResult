import java.util.Comparator;

public 
class Person implements Comparable<Person> {
    public String name;
    public int age;
    public String address;
    public Person(){

    }
    public Person(String name, int age, String address){
        this.name = name;
        this.age = age;
        this.address =address;
    }

    public String getName() {
        return name;
    }

    public int getAge() {
        return age;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public void setName(String name) {
        this.name = name;
    }
    @Override
    public int compareTo(Person y) {
        if(this.getName().compareTo(y.getName()) > 0)
            return 1;
        else if(this.getName().compareTo(y.getName()) < 0)
            return -1;
        else
        if(this.age < y.age) return -1;
        else if(this.age > y.age) return 1;
        else
            return 0;
    }



}

import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;

public class Week11 {

    public static <t extends Comparable<t>>  List<t>  sortGeneric(List<t> arr){
        for(int i = 0; i < arr.size() - 1; i ++){
            for(int j = i + 1; j < arr.size(); j ++){
                if(arr.get(i).compareTo(arr.get(j)) > 0){
                    t tmp = arr.get(i);
                    arr.set(i,arr.get(j));
                    arr.set(j,tmp);
                }
            }
        }
        return  arr;
    }

    public static void main(String[] args) {
        List<Person> arr = new ArrayList<>();
        Person p1 = new Person("F",22,"ThanhHoa1");
        Person p2 = new Person("F",25,"ThanhHoa2");
        Person p3 = new Person("F",20,"ThanhHoa3");
        arr.add(p1);
        arr.add(p3);
        arr.add(p2);

        sortGeneric(arr);

        for (int i = 0; i < arr.size(); i ++){
            System.out.println(arr.get(i).address);
        }
    }
}
