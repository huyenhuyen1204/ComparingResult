src/Person.java

public class Person implements Comparable<Person> {
    private String name;
    private int age;
    private String Address;

    public Person() {
    }

    public Person(String name, int age, String address) {
        this.name = name;
        this.age = age;
        Address = address;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getAddress() {
        return Address;
    }

    public void setAddress(String address) {
        Address = address;
    }

    @Override
    public int compareTo(Person o) {
        if (this.getName().toLowerCase().equals(o.getName().toLowerCase())) {
            return this.getAge() > o.getAge() ? 1 : -1;
        }

        if (!this.getName().toLowerCase().equals(o.getName().toLowerCase()))
            return -1;
        else return 1;


    }
}

import java.util.List;

public class Week11 {
    public static <T extends Comparable<T>> List<T> sortGeneric(List<T> arr) {
        for (int i = 0; i < arr.size(); i++) {
            for (int j = i + 1; j < arr.size(); j++) {
                if (arr.get(j).compareTo(arr.get(i)) < 0) {
                    T temp = arr.get(i);
                    arr.set(i, arr.get(j));
                    arr.set(j, temp);
                }
            }
        }
        return arr;
    }
}
