import java.util.*;
public class Week11 {
    public static<T extends Comparable> List sortGeneric(List<T> arr){
          for(int i=0;i<arr.size();i++){
              for(int j=i+1;j<arr.size();j++){
                  if(arr.get(i).compareTo(arr.get(j))>0){
                      T key1 = arr.get(i);
                      T key2 = 
arr.get(j);
                      arr.set(i,key2);
                      arr.set(j,key1);
                  }
              }
          }
          return arr;
    }
}
