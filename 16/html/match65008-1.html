import java.util.List;

public class Week11{
    public static <t extends Comparable<t>> List<
t> sortGeneric(List<t> list){
        for (int i = 0; i < list.size() - 1; i++)
            for (int j = i + 1; j < list.size(); j++)
                if(list.get(i).compareTo(list.get(j)) > 0) {
                    t temp = list.get(i);
                    list.set(i, list.get(j));
                    list.set(j, temp);
                }
        return list;
    }
}
