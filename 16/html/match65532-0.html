import java.util.ArrayList;
import java.util.List;

public class Person implements Comparable {
    private String name;
    private int age;
    private String address;

    public static void sort(List<Person> personList) {
        Week11.sortGeneric(personList);
    }


    public Person(String name, int age, String address) {
        this.name = name;
        this.age = age;
        this.address = address;
    }

    public Person() {
        this.name = "";
        this.age = 0;
        this.address = "";
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    @Override
    public int compareTo(Object o) {
        if (o instanceof Person) {
            Person person = (Person) o;
            if (this.getName().compareTo(person.getName()) > 0) return 1;
            if (this.getName().compareTo(person.getName()) == 0 && this.getAge() > person.getAge()) return 1;
            if (this.getName().compareTo(person.getName()) < 0) return -1;
            if (this.getName().compareTo(person.getName()) == 0 && this.getAge() < person.getAge()) return -1;
            if (this.getName().compareTo(person.getName()) == 0 && this.getAge() == person.getAge()) return 0;
        }
        return Integer.MIN_VALUE;
    }

    public static void printPersonList(List<Person> personList) {
        for (Person person : personList)
            System.out.println("(" + person.getName() + " " + person.getAge() + ")");
        System.out.println();
    }

    }

import java.util.ArrayList;
import java.util.List;

public class Week11 {
    public static <T extends Comparable> List<T> sortGeneric(List<T> list) {
        for (int i = 0 ; i < list.size(); i++) {
            for (int j = i + 1; j < list.size(); j++) {
                if (list.get(i).compareTo(list.get(j)) > 0) {
                    T temp = list.get(i);
                    list.set(i, list.get(j));
                    list.set(j, temp);
                }
            }
        }
        return list;
    }

    public static <T> void printArray(List<T> inputArray ) {
        for(T a : inputArray) {
            System.out.print(a + " ");
        }
        System.out.println();
    }

    public static void main(String[] args) {

    }
}
