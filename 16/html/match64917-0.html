public class Person implements Comparable<Person>{

    private String name;
    private int age;
    private String address;

    public Person(String name, int age, String address) {
        this.name = name;
        this.age = age;
        this.address = address;
    }

    public Person() {
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getAge() {
        return age;
    }

    public void setAge(int age) {
        this.age = age;
    }

    public String getAddress() {
        return address;
    }

    public void setAddress(String address) {
        this.address = address;
    }

    @Override
    public String toString(){
        return name + ',' + age + ',' + address;
    }

    @Override
    public int compareTo(Person obj) {
        if(name.equals(obj.name)){
            return Integer.compare(age, obj.age);
        }else{
            int temp = name.compareTo(obj.name);
            return temp > 0 ? 1 : temp == 0 ? 0 : -1;
        }
    }

}

import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;

public class Week11<T> {

    public static <T> List<T> sortGeneric(List<T> arr){
        arr.sort(new Comparator<T>() {
            @Override
            public int compare(T o1, T o2) {
                if(o1 instanceof Integer){
                    return Integer.compare(((Integer) o1).intValue(), ((Integer)o2).intValue());
                }
                if(o1 instanceof Double){
                    return Double.compare(((Double) o1).doubleValue(), ((Double)o2).doubleValue());
                }
                if(o1 instanceof Character){
                    return Character.compare(((Character) o1).charValue(), ((Character)o2).charValue());
                }
                if(o1 instanceof Person){
                    return ((Person) o1).compareTo((Person)o2);
                }
                if(o1 instanceof Byte){
                    return Byte.compare(((Byte) o1).byteValue(), ((Byte)o2).byteValue());
                }
                if(o1 instanceof Short){
                    return Short.compare(((Short) o1).shortValue(), ((Short)o2).shortValue());
                }
                if(o1 instanceof Long){
                    return Long.compare(((Long) o1).longValue(), ((Long)o2).longValue());
                }
                if(o1 instanceof Float){
                    return Float.compare(((Float) o1).floatValue(), ((Float)o2).floatValue());
                }
                if(o1 instanceof String){
                    return ((String) o1).compareTo((String)o2);
                }
                return 0;
            }
        });
        return arr;
    }

}
