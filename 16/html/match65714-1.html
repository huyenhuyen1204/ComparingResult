import java.util.*;

public class Week11 {
	public static <T extends Comparable<? super T>> List<T> sortGeneric(List<T> old_list) {
		for (int i = 0; i < old_list.size()-1; i++)
			for (int j = i+1; j < old_list.size(); j++) {
				T item_i = old_list.get(i);
				T item_j = old_list.get(j);
				if (
item_i.compareTo(item_j) > 0)
					Collections.swap(old_list, i, j);
			}

		return old_list;
	}

	public static void main(String[] args) {
		Person p1 = new Person("Nguyen A", 22, "earth");
		Person p2 = new Person("Nguyen A", 20, "earth");
		Person p3 = new Person("Le B", 20, "earth");
		List<Person> p = new ArrayList<Person>();
		
p.add(p1);
		p.add(p2);
		p.add(p3);
		p = Week11.<Person>sortGeneric(p);
		for (Person x: p)
			System.out.println(x.getName() + " " + x.getAge() + " " + x.getAddress());
	}
}
