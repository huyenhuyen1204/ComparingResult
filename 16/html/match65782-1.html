import java.util.ArrayList;
import java.util.Arrays;

import java.util.List;

public class Week11 {
    public static <E extends Comparable<E>> List<E> sortGeneric (List<E> array) {
        for(int i = 0 ; i < array.size() ; i++)
            for(int j = 0 ; j < array.size() ; j++)
                if (i < j && (array.get(i)).compareTo(array.get(j)) > 0) {
                    E tmp = array.get(i);
                    array.set(i , array.get(j));
                    array.set(j , tmp);
                }
        return array;
    }
}
