public class Week4 {

    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int min=arr[0];
        for (int i=1 ; i<arr.length ; i++){
            if (arr[i]<min) min=arr[i];
        }
        return min;
    }


    public static String calculateBMI(double weight, double height) {
        String BMI;
        float bmi= (float) (weight/(height*height));
        
if(bmi<18.5000) BMI="Thiếu cân";
        else if(bmi >= 18.5000 && bmi <= 22.9000) BMI="Bình thường";
        else if(bmi > 22.9000 && bmi <= 24.9000) BMI="Thừa cân";
        else BMI="Béo phì";
        return BMI;
    }


}
