public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a > b) return a ;
        return b ;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int minNum = arr[0];
        for (int i = 1; i < arr.length ; i++)
        {
            if (arr[i] < minNum) minNum = arr[i] ;
        }
        return minNum ;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double doubleResult = weight / height / height;
        double doubleResult1 = (double) Math.round(doubleResult * 10) / 10;

        String s = new String();
        if (doubleResult1 < 18.5) s = "Thiếu cân";
        else if (doubleResult1 < 22.9) s = "Bình thường";
        else if (doubleResult1 < 24.9) s = "Thừa cân";
        else if (doubleResult1 >= 25) s = "Béo phì";

        return s ;
    }

    public static void main(String[] args) {
      //  System.out.println("Hello Java");
    }
}
