import static java.lang.Math.max;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return(max(a,b));
    }
    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min=2147483647;
        for (int i=0;i<arr.length;i++)
        {
            if (min>arr[i]) min=arr[i];
        }
        if (min==2147483647) return -1;
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double ans=weight/(height*height);
        if (ans<18.45) return ("Thiếu cân");
        if (ans<22.95) return ("Bình thường");
        if (ans<24.95) return ("Thừa cân");
        return ("Béo phì");
    }
}
