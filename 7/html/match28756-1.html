public class Week4 {
    /**
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return (a >= b) ? a : b;
    }

    /**
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int length = arr.length;
        int Num_min = arr[0];
        for (int i = 1; i < length; i++) {
            Num_min = (Num_min > arr[i]) ? arr[i] : Num_min;
        }
        return Num_min;
    }

    /**
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        String str = "";
        if (weight > 0 && height > 0) {
            double value = weight / (height * height);
            // System.out.println(deta);
            // System.out.println(" Value = " + value);
            double BMI = 0;
            BMI = Math.round(value * 10) / 10.0;

            System.out.println("BMI= " + BMI);

            
if (BMI < 18.5) {
                str = "Thiếu cân";
            } else if (BMI < 23) {
                str = "Bình thường";
            } else if (BMI < 25) {
                str = "Thừa cân";
            } else {
                str = "Béo phì";
            }
            // return Double.toString(BMI) + str;
        }

        return str;
    }
}
