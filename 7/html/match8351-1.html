public class Week4 {


    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int minArray = 999;
        for(int i:arr){
            if(i<minArray) minArray=i;
        }
        return minArray;
    }
