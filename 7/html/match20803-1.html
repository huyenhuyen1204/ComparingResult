public class Week4 {

    public static int max2Int(int a, int b) {
        if(a>b)
            return a;
        else return b;
    }


    public static int minArray(int[] arr) {
        int min = arr[0];
        for( int i = 0 ; i < arr.length ; i++){
            if(arr[i]<min)
                min = arr[i];
        }
        return min;
    }

        String str = new 
String();

        double bmi = (double) Math.round(t*10)/10;

            str = "Thiếu cân";

        }
        if(bmi>=18.5&&bmi<23) {
            str = "Bình thường";
            return str;
        }
        if(bmi>=23&&bmi<25) {

            return str;
        }
        if(bmi>=25) {
            str = "Béo phì";
            return str;
        }
        return str;
