class Week4 {
    /**
     * 
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    /**
     * 
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
            int i=0;
        int min=arr[0];
        for(i=1;i<arr.length;i++){
            if(arr[i]<min) min=arr[i];
        }
        return min;
    }

    /**
     * 
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
    	String thieucan="Thiếu cân";
    	String binhthuong="Bình thường";
    	String thuacan="Thừa cân";
    	String beophi="Béo phì";
        double a=height*height;
        double BMI=weight/a;
        if(BMI<18.5) return thieucan;
        else if(BMI>=18.5&&BMI<=22.9) return binhthuong;
        else if(BMI>=23&&BMI<=24.9) return thuacan;
        else return beophi;
    }
}
