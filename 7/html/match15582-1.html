    public static int max2Int(int a,int b){
        if (a>b) return a;
        else return b;
    }
    public static int minArray (int [] arr){
        int min=arr[0];
        if (arr.length==0) return Integer.MIN_VALUE;
        for(int i=0;i<arr.length-1;i++){
            if(min>arr[i+1]) min=arr[i+1];
        }
        return min;
    }

    double BMI=(double)
