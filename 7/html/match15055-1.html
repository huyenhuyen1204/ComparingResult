class Week4
{
    public static int max2Int(int a, int b)
    {
        if(a>=b) return a;
        else return b;
    }
    public static int minArray(int[]a)
    {
        int min=a[0];
        for(int k:a) {if(k<min)min=k;}
        return min;
    }
    public static String calculateBMI(double weight, double height)
    {
        double BMI= weight/(height*height);
        double a=(double)Math.round(BMI*10)/10;
        if(a<18.5) return "Thiếu cân";
        else if(a>=18.5&&a<=22.9) return "Bình thường";
        else if(a>=23&&a<=24.9) return"Thừa cân";
        return "Béo phì";
    }
}
