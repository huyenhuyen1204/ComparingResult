public class Week4 {
    /**
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a > b) return a;
        else return b;
    }

    /**
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int i = 0; i < arr.length; i++) {
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < min) {
                    min = arr[j];
                }
            }
        }
        return min;
    }

    /**
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = weight / (height * height);
        int scale = (int) Math.pow(10, 1);
        double BMIRounded = (double) Math.round(BMI * scale) / scale;
        String result = "";
        if (BMIRounded < 18.5) {
            result = "Thiếu cân";
        } else if (BMIRounded >= 18.5 && BMIRounded <= 22.9) {
            result = "Bình thường";
        } else if (BMIRounded >= 23 && BMIRounded <= 24.9) {
            result = "Thừa cân";
        } else if (BMIRounded >= 25) {
            result = "Béo phì";
        }
        return result;
    }
}
