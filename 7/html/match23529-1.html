public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a>b) return a;
        return b;

    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int n=arr.length;
        int min=2000000;
        for (int i=0;i<n;i++){
            if (min>arr[i]) min=arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double bmi=(double) Math.round((weight/(height*height))*10)/10;

        
if  (bmi>=25.0)
        {return "Béo phì";}
        if  (bmi<25 && bmi>=23)
        {return "Thừa cân";}
        if  (bmi>=18.5 && bmi<23)
        {return "Bình thường";}
        return "Thiếu cân";
    }
}
