public class Week4 {

    /**
     * Ham tim gia tri lon nhat cua hai so nguyen
     * @param a
     * @param b
     * @return gia tri lon nhat cua a va b
     */
    public static int max2Int(int a, int b){
        return Math.max(a, b);
    }

    /**
     * Ham tim gia tri nho nhat cua mang so nguyen
     * @param arr
     * @return gia tri nho nhat cua mang
     */
    public static int minArray(int[] arr){
        int min = arr[0];
        for (int i = 1; i < arr.length; i++){
            if(arr[i] < min)
                min = arr[i];
        }

        return min;
    }

    /**
     * Ham tinh chi so BMI
     * @param weight (don vi kg)
     * @param height (don vi kg)
     * @return chi so BMI
     */
    public static String calculateBMI(double weight, double height){
        double index = weight / (height * height);
        index = Double.parseDouble(String.format("%.1f", index));
        if(index < 18.5)    return ("Thiếu cân");
        else if(index >= 18.5 && index <= 22.9)    return ("Bình thường");
        else if(index >= 23 && index <= 24.9)    return ("Thừa cân");
        else if (index > 25)    return ("Béo phì");

        return null;
    }

}
