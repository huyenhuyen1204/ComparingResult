public class Week4 {
    int a;
    int b;
    public static int max2Int(int a, int b){
        if (a>=b) return a;
        return b;
    }
    int [] arr = new int[100];
    public static int minArray(int [] arr){
        int min = arr[0];
        for (int i=1; i<arr.length; i++){
            if (arr[i]<min) min = arr[i];
        }
        return min;
    }
    double weight;
    double height;
    public static String calculateBMI(double weight, double height){
        double BMI;
        BMI =(double) weight/(height*height);
        double n=Math.ceil(BMI*10)/10;
        if (n<18.5) return "Thiếu cân";
        else if (18.5<=n && n<=22.9) return "Bình thường";
        else if (22.9<=n && n<=24.9) return "Thừa cân";
        else if (n>=25) return "Béo phì";
        return null;
    }

}
