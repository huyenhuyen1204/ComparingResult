src1/Week4.java

public class Week4 {
    public static int max2Int(int a, int b){
        if(a>b) return a;
        return b;
    }

    public static int minArray(int[] arr){
        int min=arr[0];
        for(int i=1;i<arr.length;i++){
            if(arr[i]<min){
                min = arr[i];
            }
        }
        return min;
    }

    public static String calculateBMI(double weight, double height){
        double BMI;
        String result = "";
        BMI = (double)Math.round(weight/(height*height) * 10) / 10;
        if(BMI < 18.5) result = "Thiếu cân";
        else if(BMI>=18.5 && BMI<=22.9) result = "Bình thường";
        else if(BMI>=23 && BMI<24.9) result = "Thừa cân";
        else if(BMI>=25) result = "Béo phì";
        return result;
    }
}
