class Week4 {
	/**
	 *
	 * @param a
	 * @param b
	 * @return max of $a and $b
	 */
	public static int max2Int(int a, int b) {
		return Math.max(a, b);
	}

	/**
	 *
	 * @param arr
	 * @return min of array $arr
	 */
	public static int minArray(int[] arr) {
		int minResult = Integer.MAX_VALUE;
		for(int index = 0; index < arr.length; index++){
			minResult = Math.min(minResult, arr[index]);
		}
		return minResult;
	}

	/**
	 *
	 * @param weight: in kg
	 * @param height: in m
	 * @return BMI index
	 */
	public static String calculateBMI(double weight, double height) {
		double bodyMassIndex = (double) Math.round (weight * 10 / (height * height)) / 10 ;
		if (bodyMassIndex < 18.5) return "Thiếu cân";
		else if (bodyMassIndex < 22.9) return "Bình thường";
		else if (bodyMassIndex < 24.9) return "Thừa cân";
		else return "Béo phì";
	}

	public static void main(String[] args) {
		System.out.println(calculateBMI(70, 1.7));
	}
}
