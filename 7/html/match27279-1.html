class Week4 {

    public static int max2Int(int a, int b){
        boolean bol = (a >= b) ;
        if (bol) return a ;
        else return b ;
    }

    public static int minArray(int[] arr){
        int n = arr.length ;
            int kmin = arr[0] ;
            for(int i = 0 ; i < n; i++){
                if (kmin > arr[i]) kmin = arr[i] ;
            }
            return kmin ;
    }

    public static String calculateBMI(double weight, double height){
            double BMI = (double) Math.round((weight / (height * height)) * 10) / 10;
            System.out.println(BMI);
            if (BMI < 18.5) return "Thiếu cân" ;
            else if (BMI >= 18.5 && BMI <= 24.9) return "Bình thường";
            else if (BMI >= 23 && BMI <= 24.9) return  "Thừa cân";
            else return "Béo phì";
    }

}
