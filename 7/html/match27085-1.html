import java.text.DecimalFormat;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return a > b ? a : b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] < min) {
                min = arr[i];
            }
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double bmi = weight / (height*height);
        String result = "";
        DecimalFormat df = new DecimalFormat("#.#");
        String change = df.format(bmi);
        bmi = Double.parseDouble(change);
        
if (bmi < 18.5) {
            result = "Thiếu cân";
        } else if (bmi <= 22.9) {
           result = "Bình thường" ;
        } else if ( bmi <= 24.9) {
            result = "Thừa cân";
        } else {
            result = "Béo phì";
        }
        return result;
    }
}
