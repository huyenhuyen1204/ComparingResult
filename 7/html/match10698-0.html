public class Week4 {
    public static int max2Int(int a, int b) {
        if (a  >b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int len = arr.length;
        int min = arr[0];
        for (int i = 1; i < len; ++i)
            if (arr[i] < min) min = arr[i];
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI = (double) weight / (height * height);
        String str = null;
        double BMI1 = (double) Math.round(BMI * 10) / 10;
        if (BMI1 < 18.5) str = "Thiếu cân";
        else if (BMI1 < 23) str = "Bình thường";
        else if (BMI1< 25) str = "Thừa cân";
        else str = "Béo phì";
        return str;
    }
}
