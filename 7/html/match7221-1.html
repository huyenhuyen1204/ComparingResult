public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a > b){
            return a;
        }else{
            return b;
        }
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for(int i = 0; i < arr.length; i++){
            if(min > arr[i]){
                min = arr[i];
            }
        }

        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = weight / (height * height);

        long factor = (long) Math.pow(10, 1);
        BMI = BMI * factor;
        BMI = Math.round(BMI);
        BMI = BMI / factor;
        String res = "";
        
if(BMI < 18.5){
            res = "Thiếu cân";
        }else if(BMI >= 18.5 && BMI <= 22.9){
            res = "Bình thường";
        }else if(BMI >= 23 && BMI <= 24.9){
            res = "Thừa cân";
        }else if(BMI >= 25){
            res = "Béo phì";
        }
        return res;
    }

}
