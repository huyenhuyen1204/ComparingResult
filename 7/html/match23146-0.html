import java.util.Arrays;
public class Week4 {

    public static int max2Int(int a, int b) {
        if (a>=b) return a;
        else return b;

    }

    public static int minArray(int[] arr) {

        Arrays.sort(arr);
        return arr[0];

    }

    /*public static double LamTron(double a) {
        a = (double)Math.round(a*10)/10;
        return a;
    }*/

    public static String calculateBMI(double weight, double height) {

        String sol;
        Double BMI = weight/(height*height);
        BMI = (double)Math.round(BMI*10.0)/10.0;
        if(BMI < 18.5 ) sol = "Thiếu cân";
        else {
            if(BMI >= 18.5 && BMI <= 22.9) sol = "Bình thường";
            else {
                if(BMI >= 23 && BMI <= 24.9) sol = "Thừa cân";
                else {
                    sol = "Béo phì";
                }
            }
        }

        return sol;

    }
    /*public static void main(String[] args) {
        System.out.println(calculateBMI(54,1.65));
    }*/
}
