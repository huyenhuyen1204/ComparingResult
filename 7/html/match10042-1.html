public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */

    public static int max2Int(int a, int b) {
        if (a > b ) return a;
        else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int l = arr.length;
        int ret = arr[0];
        for (int i = 1; i < l;i++){
            if (arr[i] <= ret) ret =arr[i];
        }
        return ret;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double cur = (double) Math.round(weight/(height*height)*100)/100;
        
if (cur  < 18.5) return "Thiếu cân";
        if (cur >= 18.5 && cur <= 22.9) return "Bình thường";
        if (cur >= 23 && cur <= 24.9) return "Thừa cân";
        return "Béo phì";
    }
