//package school.mavenjunit;
import java.text.DecimalFormat;
public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        int max=a;
        if(b>a) max=b;
        return max;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for(int i=1; i<arr.length; i++){
            if(arr[i]<min) min=arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI1= weight/height/height;
        DecimalFormat 
df = new DecimalFormat("0.0");
        String BMI2 = df.format(BMI1);
        double BMI = Double.parseDouble(BMI2);
        //double BMI = df.format(value);
        if(BMI-18.5<0) return "Thiếu cân";
        else if(18.5-BMI<=0 && BMI-22.9<=0) return "Bình thường";
        else if(23-BMI<=0 && BMI-24.9<=0) return "Thừa cân";
        else return "Béo phì";
    }
}
