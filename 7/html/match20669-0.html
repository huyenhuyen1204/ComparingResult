import java.text.DecimalFormat;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        int max = a;
        if (b > max) max = b;
        return max;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int i = 1; i < arr.length; i++){
            if (arr[i] < min) min = arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double d;
        String result = null;
        d = weight / (height*height);
        DecimalFormat dcf = new DecimalFormat("#.##");
        double BMI = Double.parseDouble(dcf.format(d));

        if (BMI < 18.5)
            result = "Thiếu cân";
        if (BMI >= 18.5 && BMI < 23)
            result = "Bình thường";
        if (BMI >= 23 && BMI < 25)
            result = "Thừa cân";
        if (BMI >= 25)
            result = "Béo phì";

        return result;
    }

}
