class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a > b) return a;
        return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int res = Integer.MAX_VALUE;
        for (int i : arr) if (i < res) res = i;
        return res;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        int BMI = (int) Math.round(10.00 * weight / (height * height));
        if (BMI < 185) return "Thiếu cân";
        else if (BMI <= 229) return "Bình thường";
        else if (BMI <= 249) return "Thừa cân";
        else return "Béo phì";
    }
}
