import java.util.Arrays;

public 
class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a >= b)return a;
        else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */

   public static int minArray(int[] arr) {
       Arrays.sort(arr);
       return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */

    public static String calculateBMI(double weight, double height) {
        
if(25 * height * height <= weight) return "Béo phì";
        if(23 * height * height <= weight) return "Thừa cân";
        if(37 * height * height <= weight * 2 + 1e-7) return "Bình thường";
        return "Thiếu cân";
    }

}
