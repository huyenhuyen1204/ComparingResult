public class Week4 {

    public static int max2Int(int a, int b) {
        if( a - b >= 0 ) return a;
        else return b;
    }

   // public int[] arr = new int[100];
    public static int minArray(int[] arr) {
        int min = arr[0];
        for( int i = 0; i < arr.length; i++){
            if(min > arr[i]) min = arr[i];
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        long BMI = Math.round(10.0*weight/(height*height));
        if ( BMI < 185 ) return "Thiếu cân";

        
if ( BMI >= 185 && BMI <= 229 ) return  "Bình thường";

        if ( BMI >= 230 && BMI <= 249 ) return "Thừa cân";

        if ( BMI > 250 ) return "Béo phì";

        return "Béo phì";
    }

}
