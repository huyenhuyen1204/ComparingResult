import java.util.Arrays;

public class Week4 {

    public static int max2Int(int a, int b) {
        return a>b ? a:b;
    }
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }
    public static String calculateBMI(double weight, double height) {
        double BMI = (double) weight/(height*height);
        String temp = null;
        double 
BMI1 = (double) Math.round(BMI*10)/10;
        if (BMI1 < 18.5) temp = "Thiếu cân";
        else if (BMI1 < 23) temp = "Bình thường";
        else if (BMI1 < 25) temp = "Thừa cân";
        else temp = "Béo phì";
        return temp;
    }

}
