class Week4 {
    public static int max2Int(int a, int b) {
        return Math.max(a,b);
    }

    public static int minArray(int[] arr) {
        int result = arr[0];
        for(int i = 0;i < arr.length;i++) result = Math.min(result,arr[i]);
        return result;
    }

    public static String calculateBMI(double weight, double height) {
        Double k = (double) (Math.round(weight * 10 / (height * height)) )/ 10;
        if (k < 18.5) return "Thiếu cân";
        if (k < 22.9) return "Bình thường";
        if (k < 24.9) return "Thừa cân";
        return "Béo phì";
    }
}
