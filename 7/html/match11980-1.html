class Week4 {
    int a;
    int b;

    public  static int max2Int(int a, int b){
        if(a >= b){
            return a;
        }else {
            return b;
        }
    }

    public static int minArray(int[] arr){
        int min = arr[0];
        for(int i = 1; i < arr.length; i++){
            if(min > arr[i]){
                min = arr[i];
            }
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double IBM_index = weight / Math.pow(height, 2);
        IBM_index = (double) Math.round(IBM_index * 10) / 10;
        if(IBM_index < 18.5){
            return "Thiếu cân";
        } else if(IBM_index >= 18.5 && IBM_index <= 22.9){
            return "Bình thường";
        } else if(IBM_index >= 23.0 && IBM_index <= 24.9) {
            return "Thừa cân";
        } else {
            return "Béo phì";
        }
    }


}
