public class Week4{
	public static int max2Int(int a, int b){
		if(a>b)
			return a;
		else 
			return b;
	}

	public static int minArray(int []arr){
		int minNumber = arr[0];
		for(int i : arr){
			if(i < minNumber)
				minNumber = i;
		}
		return minNumber;
	}

	public static String calculateBMI(double weight, double height){
		double bmi = weight/height/height;
		if(bmi < 18.4)
			return "Thiếu cân";
		else if( bmi <= 22.9)
			return "Bình thường";
		else if(bmi <= 24.9)
			return "Thừa cân";
		else 
			return "Béo phì";
	}
	
}
