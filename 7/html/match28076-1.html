import java.util.Arrays;

public class Week4 {
    /**
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return a > b ? a : b;
    }

    /**
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }

    /**
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = weight / (height * height);
        double BMI_Round = Math.round(BMI * 10.0) / 10.0;
        if (BMI_Round < 18.5) return "Thiếu cân";
        else if (BMI_Round >= 18.5 && BMI_Round <= 22.9) return "Bình thường";
        else if (BMI_Round >= 23 && BMI_Round <= 24.9) return "Thừa cân";
        else return "Béo phì";
    }
}
