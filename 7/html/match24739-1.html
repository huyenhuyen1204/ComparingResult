    public static int max2Int(int a, int b){
        if(a>b) return a;
        return b;
    }
    public static int minArray(int[]a){
        int n = a.length;
        int min = a[0];
        for(int i=1; i<n; i++){
            if(a[i]<min) min = a[i];
        }
        return min;
    }
    public static String calculateBMI(double canNang, double chieuCao){
        String a;
        double BMI = canNang/(chieuCao* chieuCao);
        BMI = (double) Math.ceil(BMI*10)/10;
        
if(BMI<18.5) a = "Thiếu cân";
        else if(BMI<=22.9) a = "Bình thường";
        else if(BMI<=24.9) a = "Thừa cân";
        else a = "Béo phì";
        return a;
    }
