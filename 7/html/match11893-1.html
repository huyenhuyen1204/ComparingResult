class Week4{
    public static int max2Int(int a, int b)
    {
        if (a > b) return a;
        else return b;
    }

    public static int minArray(int[] arr)
    {
        int min = 999999999;
        for (int i : arr)
        {
            if (min > i) min = i;
        }
        return min;
    }

    public static String calculateBMI(double weight, double height)
    {
        double BMI_d = weight / (height * height);
        Double BMI = Math.round(BMI_d * 10) / 10.0;
        if (BMI < 18.5) return "Thiếu cân";
        else if (BMI < 22.9) return "Bình thường";
        else if (BMI <= 24.9) return "Thừa cân";
        else return "Béo phì";
    }

    public static void main(String[] args)
    {
        double w = 63.7696;
        double h = 1.6;
        double bmi = w / (h * h);
        System.out.println(Math.round(bmi * 10) / 10.0);
    }

}
