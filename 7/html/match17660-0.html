public class Week4 {
	public static int max2Int(int a, int b) {
		return Math.max(a, b);
	}

	public static int minArray(int[] arr) {
		int leng = arr.length;
		int min = 99999999;
		for (int i = 0; i < leng; ++i) {
			if (arr[i] < min)
				min = arr[i];
		}
		return min;
	}

	public static String calculateBMI(double weight, double height) {
		double h = Math.pow(height, 2);
		double BMI = weight / h; // kq , m
		BMI = (double) Math.round(BMI * 10) / 10;

		String kq = "";
		if (BMI < 18.5) {
			kq = "Thiếu cân";
		}
		if (BMI >= 18.5 && BMI <= 22.9) {
			kq = "Bình thường";
		}
		if (BMI >= 23 && BMI <= 24.9) {
			kq = "Thừa cân";
		}
		if (BMI >= 25) {
			kq = "Béo phì";
		}
		return kq;
	}
}
