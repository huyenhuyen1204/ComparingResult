class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return Math.max(a, b);
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int minOfArr = arr[0];
        for(int i = 0; i < arr.length; i++) minOfArr = Math.min(minOfArr, arr[i]);
        return minOfArr;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = Math.round(((weight / (height * height))*10)/10);
        if(BMI < 18.5) return "Thiếu cân";
        else if(BMI >= 18.5 && BMI <= 22.9) return "Bình thường";
        else if(BMI >= 23 && BMI <= 24.9) return "Thừa cân";
        else return "Béo phì";
    }
}
