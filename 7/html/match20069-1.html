class Week4 {
    public static int max2Int(int a, int b) {
        if(a>b)return a;
        else return b;
    }
    public static int minArray(int[] arr) {
        int min=arr[0];
        int a=arr.length;
        for(int i=0;i<a;i++){
            if(arr[i]<min){min=arr[i];}
        }
        return min;
    }
    public static String calculateBMI(double weight, double height) {
        double a=weight/(height*height);
        double b=((double)Math.round(a * 10) / 10);
        String c="Thiếu cân";
        String d="Bình thường";
        String e="Thừa cân";
        String f="Béo phì";
        if(b<18.5){return c;}
        else if(b>=18.5&&b<=22.9){return d;}
        else if(b>=23&&b<=24.9){return e;}
        else return f;
    }    

    
}
