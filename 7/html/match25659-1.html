class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a>b){
            return a;
        }
        else {
            return b;
        }
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        for (int i=0;i<arr.length;i++){
            for (int j=0;j<i;j++){
                if(arr[i]<arr[j]){
                    int tam = arr[i];
                    arr[i] = arr[j];
                    arr[j] = tam;
                }
            }
        }
        return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = weight/(height*height);
        if(BMI<18.5){
            return "Thiếu cân";
        }
        else if(BMI<23){
            return "Bình thường";
        }
        else if(BMI<25){
            return "Thừa cân";
        }
        else {
            return "Béo phì";
        }
    }
}
