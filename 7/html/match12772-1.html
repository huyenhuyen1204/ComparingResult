import java.lang.Math;
public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int n=arr.length;
        int min=arr[0];
        for(int i=1; i<n; i++){
            if(min>arr[i]) min=arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        String bim="";
        double idx=Math.round((weight/(height*height))*10)/10.0;
        if(idx < 18.5) bim="Thiếu cân";
        else if(idx >= 18.5 && idx <= 22.9) bim="Bình thường";
        else if(idx >=23 && idx <=24.9) bim= "Thừa cân";
        else bim = "Béo phì";
        return bim;
    }
}
