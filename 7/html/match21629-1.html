class Week4 {

        public static int max2Int(int a, int b) {
            return Math.max(b, a);
        }


        public static int minArray(int[] arr) {
            int min = arr[0];
            for (int value : arr)
                if (min > value)
                    min = value;
            return min;
        }

        public static String calculateBMI(double weight, double height) {
            double BMI = weight / (height * height);
            BMI = Math.round(BMI * 10.0) / 10.0;
            String s;
            if ( BMI < 18.5)
                s = "Thiếu cân";
            else if ( BMI < 23)
                s = "Bình thường";
            else if ( BMI < 25)
                s = "Thừa cân";
            else
                s = "Béo phì";
            return s;
        }

}
