class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a > b) return a;
        else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        if (arr.length == 0)
            return Integer.MAX_VALUE;
        int min = arr[0];
        for ( int i = 0; i < arr.length; i++ ){
            if ( arr[i] < min )
                min = arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight : in kg
     * @param height : in m
     */
    public static String calculateBMI(double weight, double height) {
        double res = weight/ (height * height);
        double BMI = (double) Math.round(res*10)/10;
        if ( BMI + 1e-8 < 18.5 )
            return "Thiếu cân";
        else if ( BMI + 1e-8 >= 18.5 && BMI + 1e-8 <= 22.9 )
            return "Bình thường";
        else if (BMI + 1e-8 >= 23 && BMI + 1e-8 <= 24.9)
            return "Thừa cân";
        else if ( BMI + 1e-8 >= 25 )
            return "Béo phì";
        return null;
    }

}
