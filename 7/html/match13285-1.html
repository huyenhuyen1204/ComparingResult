import java.util.*;
import java.lang.*;
public 
class Week4 {
    /**
     * 
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    /**
     * 
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int n = arr.length;
        int min = arr[0];
        for(int i=1;i<n;i++) min = min>arr[i]? arr[i]:min;
        return min;
    }

    /**
     * 
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double ans = (double) Math.round(weight/(height * height) * 10)/10;
        if(ans<18.5) return new String("Thiếu cân");
        else if(ans<22.9) return new String("Bình thường");
        else if(ans<24.9) return new String("Thừa cân");
        else return new String("Béo phì");

    }
}
