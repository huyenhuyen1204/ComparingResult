import java.util.*;
public 
class Week4 {
        /**
     *
 * @param a
     * @param b
     * @return max of $a and $b
     */

    public static int max2Int(int a, int b) {
            if(a >= b) return  a;
            else  return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
            int min_num = arr[0];
          int len = arr.length;
            for(int i=1; i<len; i++){
                if(arr[i] <= min_num) min_num = arr[i];
            }
        return  min_num;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double bmi;
        bmi = weight/(height* height);
        bmi = Math.round(bmi*10.0)/10.0;
        String result;
        
if(bmi< 18.5)
            result = "Thiếu cân";
        else if(bmi >= 18.5 && bmi <= 22.9)
            result = "Bình thường";
        else if(bmi >= 23 && bmi <= 24.9)
            result = "Thừa cân";
        else
            result = "Béo phì";
        return result;
    }

}
