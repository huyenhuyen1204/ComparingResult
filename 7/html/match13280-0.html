/**

 */
public class Week4 {
    public static int max2Int(int a, int b){
        if (a > b) return a;
        else return b;
    }

    public static int minArray(int[] a){
        if(a.length == 0) return Integer.MAX_VALUE;
        int min=a[0];
        for(int i=0; i<a.length; i++){
            if(a[i]<min) min = a[i];
        }
        return min;
    }

    public static String calculateBMI(double weight, double height){
        double res = weight/ (height * height);
        double BMI = (double) Math.round(res*10)/10;
        if ( BMI + 1e-8 < 18.5 )
            return "Thiếu cân";
        else if ( BMI + 1e-8 >= 18.5 && BMI + 1e-8 <= 22.9 )
            return "Bình thường";
        else if (BMI + 1e-8 >= 23 && BMI + 1e-8 <= 24.9)
            return "Thừa cân";
        else if ( BMI + 1e-8 >= 25 )
            return "Béo phì";
        return null;
    }
}
