import java.util.Arrays;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return Math.max(a, b);
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double bmiIndex = weight / (height * height);
        int num = (int)bmiIndex;
        if((bmiIndex - num) *100.0 % 10.0 > 5) bmiIndex = Math.ceil(bmiIndex *10.0) /10.0;
        else bmiIndex = Math.floor(bmiIndex *10.0) /10.0;

        String res = "";
        
if(bmiIndex < 18.5) res = "Thiếu cân";
        else if(bmiIndex >= 18.5 && bmiIndex <= 22.9) res = "Bình thường";
        else if(bmiIndex >= 23 && bmiIndex <= 24.9) res = "Thừa cân";
        else if(bmiIndex >= 25) res = "Béo phì";

        return res;
    }
}
