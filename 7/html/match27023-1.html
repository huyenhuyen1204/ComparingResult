public class Week4 {

    public static int max2Int(int a, int b) {
        return a < b ? b : a;
    }

    public static int minArray(int[] arr) {
        int index = 0;
        for( int i = 0 ; i < arr.length ; i++) {
            if( arr[index] > arr[i] )
                index = i;
        }
        return arr[index];
    }

    public static String calculateBMI(double weight, double height) {
        double chiSoBMI = (double)Math.round((weight / ( height * height)) *10) / 10;
        String thongBao = "";
        
if(chiSoBMI < 18.5)
            thongBao += "Thiếu cân";
        else if (chiSoBMI < 22.9)
            thongBao += "Bình thường";
        else if (chiSoBMI <24.9)
            thongBao += "Thừa cân";
        else
            thongBao += "Béo phì";
        return thongBao;
    }
    public static  void main(String[] args) {
    }
}
