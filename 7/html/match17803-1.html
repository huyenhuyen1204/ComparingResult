public class Week4 {
    int a;
    int b;

    public static int max2Int(int a, int b)
    {
        if(a>b) return a;
        else return b;
    }

    int[] arr;
    public static int minArray(int[] arr)
    {
        int min = arr[0];
        for(int i=1;i<arr.length;i++)
        {
            if(min>arr[i])  min=arr[i];
        }
        return min;
    }

    double weight, height;
    public static String calculateBMI(double weight, double height)
    {
        double BMI;
        String BMIindex="";
        BMI = weight/(height*height);
        BMI = Math.ceil(BMI*100)/100;
        
if(BMI<18.5)    BMIindex = "Thiếu cân" ;
        else if(BMI>=18.5&&BMI<=22.9) BMIindex="Bình thường";
        else if(BMI>=23&&BMI<=24.9)  BMIindex="Thừa cân";
        else if(BMI>=25) BMIindex = "Béo phì";
        return BMIindex;
    }
}
