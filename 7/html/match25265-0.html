import java.*;
public class Week4 {
    public static int max2Int(int a,int b)
    {
        if(a > b) return a;
        return b;
    }
    public static int minArray(int[] Arr)
    {
        int c = Arr[0];
        for(int i=0;i<Arr.length;i++)
        {
            if(c > Arr[i]) c = Arr[i];
        }
        return c;
    }
    public static String calculateBMI(double weight,double height)
    {
        double BMI = weight/(height*height);
        double BMI1 = Math.round(BMI*10.0)/10.0;
        String kq1 = "Thiếu cân";
        String kq2 = "Bình thường";
        String kq3 = "Thừa cân";
        String kq4 = "Béo phì";
        if(BMI1 < 18.5 ) return kq1;
        if(BMI1 >= 18.5 && BMI1 <= 22.9) return kq2;
        if(BMI1 >= 23 && BMI1 <= 24.9) return kq3;
        return kq4;
    }

}
