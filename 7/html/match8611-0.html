import javax.swing.event.MenuDragMouseListener;

public class Week4 {
    public static int max2Int(int a, int b) {
        if(a >= b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int n = arr.length;
        if( n == 0) return 0;
        int min = arr[0];
        int i;
        for( i = 0; i < n; i ++)
        {
            if( arr[i] < min) min = arr[i];
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI = weight / ( height * height);
        if( BMI < 18.45)  return "Thiếu cân";
        else if( BMI <= 22.85) return "Bình thường";
        else if( BMI <= 24.85) return "Thừa cân";
        else return "Béo phì";
    }

}
