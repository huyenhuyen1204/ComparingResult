import java.util.Arrays;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
    if(a>b) return a;
    return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI=Math.round(weight*10)/(Math.pow(height, 2));
        if(BMI<185) return "Thiếu cân";
        if(BMI<229) return "Bình thường";
        if(BMI<249) return "Thừa cân";
        return "Béo phì";
    }
}
