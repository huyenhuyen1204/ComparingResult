import java.util.Arrays;

public class Week4 {

    public static int max2Int(int a , int b){
        return Math.max(a, b);

    }
    public static int minArray(int [] arr){
        Arrays.sort(arr);
        return arr[0];
    }
    public static String calculateBMI(double weight, double height){
        double BMI_f = weight / ( height * height);
        double BMI = (double)Math.round(BMI_f*10)/10; // round return integer
        String answer = "";
        if(BMI < 18.5)
            answer ="Thiếu cân";
        else if(BMI >= 18.5 && BMI <= 22.9)
            answer = "Bình thường";
        else if(BMI >= 23 && BMI <= 24.9)
            answer = "Thừa cân";
        else if(BMI >= 25)
            answer = "Béo phì";
        return answer;
    }
}
