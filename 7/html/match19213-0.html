public class Week4 {
    public static int max2Int(int a, int b) {
        if(a > b ) return a;
        else return b;
    }


    public static int minArray(int[] arr) {
        int min = arr[0];
        int len = arr.length;
        for(int i = 0 ; i < len ; i++){
            if(arr[i] < min) min = arr[i];
        }
        return min;
    }


    public static String calculateBMI(double weight, double height) {
        double BMI = weight/(height*height);
        String temp ;
        temp = "";
        BMI =(double) Math.round(BMI*10)/10;
        if(BMI < 18.5)  temp +="Thiếu cân";
        else if(BMI >= 18.5 && BMI < 23)  temp +="Bình thường";
        else if(BMI >=23 && BMI < 25)  temp +="Thừa cân";
        else if(BMI >= 25)  temp +="Béo phì";
        return temp;
    }

}
