public class Week4 {
    public static int max2Int(int a, int b){
        if(a>b){
            return a;
        }else{
            return b;
        }
    }
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int i = 1; i < arr.length; i++){
            if(arr[i] < min){
                min = arr[i];
            }
        }
        return min;
    }

    public static String calculateBMI(double weight, double height){
        
if(weight >= 0 && height >= 0){
            double BMI = Math.round(((weight/(height*height))*10)/10);
            String Result;
            if(BMI < 18){
                Result = "Thiếu cân";
            }else if(18 <= BMI&& BMI <= 22.9){
                Result = "Bình thường";
            }else if(23 <= BMI&& BMI <= 24.9){
                Result = "Thừa cân";
            }else {
                Result = "Béo phì";
            }
            return Result;
        }
        return null;
    }
    public static void main(String args[]) {

        int[] arr = new int[20];
        for(int i=0 ; i<20 ; i++) {
            arr[i] = 20;
        }

        System.out.println(max2Int(10,20));
        System.out.println(minArray(arr));
        System.out.println(calculateBMI(56, 1.62));

    }
}
