Week4.java

public class Week4 {
    public static int max2Int(int a, int b) {
        return (a > b) ? a : b;
    }

    public static int minArray(int[] arr) {
        int ans = Integer.MAX_VALUE;
        for (int foo : arr) ans = (ans < foo) ? ans : foo;
        return ans;
    }

    public static String calculateBMI(double weight, double height) {
        double value = weight / (height * height);
        
if (value < 18.45) return "Thiếu cân";
        if (value < 22.95) return "Bình thường";
        if (value < 24.95) return "Thừa cân";
        return "Béo phì";
    }
}
