public class Week4 {
    public static int max2Int(int a, int b) {
        if(a>b) {
            return a;
        } else {
            return b;
        }

    }

    public static int minArray(int[] arr) {
        int n = arr.length;
        int dem = arr[0];
        for(int i=1; i<n; i++) {
            if(arr[i]<dem) {
                dem = arr[i];
            }
        }

        return dem;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI = weight/(height*height);
        BMI = Math.round(BMI*10.00)/10.00;
        String[] s = {"Thiếu cân", "Bình thường", "Thừa cân", "Béo phì"};
        if(BMI < 18.5) {
            return s[0];
        } else if(18.5<=BMI && BMI<=22.9) {
            return s[1];
        } else if(23<=BMI && BMI<=24.9) {
            return s[2];
        } else {
            return s[3];
        }

    }


}
