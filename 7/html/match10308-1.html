public class Week4 {
    /**
     * 
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a>b) return a;
        else return b;
    }

    /**
     * 
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int i=1; i< arr.length; i++){
            if(min > arr[i]) min = arr[i];
        } 
        return min;
    }

    /**
     * 
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double t;
        
        t = weight/(height*height);
        
t = (double) Math.round(t*10)/10;
        if(t<18.5) return "Thiếu cân";
        if((t>=18.5) && (t<=22.9)) return "Bình thường";
        if((t>=23) && (t<=24.9)) return "Thừa cân";
        return "Béo phì";
    }
}
