//package com.company;

public class Week4 {
        /**
         *
         * @param a
         * @param b
         * @return max of $a and $b
         */
        public static int max2Int(int a, int b) {
            if(a>b) return a;
            return b;
        }

        /**
         *
         * @param arr
         * @return min of array $arr
         */
        public static int minArray(int[] arr) {
                int minn=100000;

                for(int i=0; i< arr.length; i++ ) {
                    if (arr[i] < minn) minn = arr[i];
                }

                return minn;
        }

        /**
         *
         * @param weight: in kg
         * @param height: in m
         * @return BMI index
         */
        public static String calculateBMI(double weight, double height) {
            double res=weight/((height)*height);
            res=((double) Math.round(res*10)/10);
            
            if(res<18.5) return "Thiếu cân";
            
if(res>=18.5&&res<=22.9) return "Bình thường";
            if(res>=23&&res<=24.9) return "Thừa cân";
            if(res>=25) return "Béo phì";

            return "";
        }
}
