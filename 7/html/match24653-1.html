class Week4 {
    public static int max2Int(int a, int b) {
        if ( a >= b ) return a ;
        else return b ;
    }
    public static int minArray(int[] arr) {
        int num = arr.length ;
        int min = 999999 ;
        for ( int n = 0 ; n < num ; n++ )
        {
            if ( arr[n] < min ) min = arr[n] ;
        }
        return min ;
    }
    public static String calculateBMI(double weight, double height) {
        double BMI = weight / ( Math.pow( height , 2 ) );
        BMI = (double)Math.round(BMI * 10)/10;
        String aaa = null ;
        
if ( BMI < 18.5 ) aaa = "Thiếu cân";
        else if ( BMI >= 18.5 && BMI <= 22.9 ) aaa = "Bình thường" ;
        else if ( BMI >= 23 && BMI <= 24.9 ) aaa = "Thừa cân" ;
        else if ( BMI >= 25 ) aaa = "Béo phì" ;
        return  aaa ;
    }
}
