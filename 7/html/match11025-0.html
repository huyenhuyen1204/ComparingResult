import org.junit.Assert;
import org.junit.Test;

public class Week4Test {

    @Test
    public void testMax2Int1(){
        int numberA = 100;
        int numberB = 20;

        Assert.assertEquals(Week4.max2Int(numberA, numberB), numberA);

    }

    @Test
    public void testMax2Int2(){
        int numberA = 80;
        int numberB = 200;

        Assert.assertEquals(Week4.max2Int(numberA, numberB), numberB);
    }

    @Test
    public void testMax2Int3(){
        int numberA = 10;
        int numberB = 10;

        Assert.assertEquals(Week4.max2Int(numberA, numberB), numberB);
        Assert.assertEquals(Week4.max2Int(numberA, numberB), numberA);
    }

    @Test
    public void testMax2Int4(){
        int numberA = 0;
        int numberB = 1;

        Assert.assertNotEquals(Week4.max2Int(numberA, numberB), numberA);
    }

    @Test
    public void testMax2Int5(){
        int numberA = 55;
        int numberB = 60;

        Assert.assertNotEquals(Week4.max2Int(numberA, numberB), numberB);
    }

    @Test
    public void testMinArray1(){
        int[] arr = new int[] { 1, 3, 4, 5, 6};

        Assert.assertEquals(Week4.minArray(arr), arr[0]);
    }

    @Test
    public void testMinArray2(){

        int[] arr = new int[] { 7, 10, 15, 30, 20};

        Assert.assertEquals(Week4.minArray(arr), arr[0]);

    }

    @Test
    public void testMinArray3(){

        int[] arr = new int[] { 9, 8 , 7 , 6, 5};

        Assert.assertEquals(Week4.minArray(arr), arr[arr.length - 1]);
    }

    @Test
    public void testMinArray4(){
        int[] arr = {123, 345, 567, 789};
        assertEquals(120, Week4.minArray(arr));
    }

    @Test
    public void testMinArray5(){
        int[] arr = {23, 34, 45, 35};
        assertEquals(1, Week4.minArray(arr));
    }
