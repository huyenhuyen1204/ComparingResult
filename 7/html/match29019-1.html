public class Week4 {

    public static int max2Int(int a, int b) {
        if (a>b) return a;
        else a=b;
        return a;
    }

    public static int minArray(int[] arr) {
        int a=arr[0];
        for (int i = 0; i<arr.length; i++)
            if (a > arr[i]) a = arr[i];
        return a;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI;
        String rs = null;
        
BMI=weight/(height*height);
        if(BMI<18.5) rs="Thiếu cân";
        else if((BMI>18.5)&&(BMI<22.9)) rs="Bình thường";
        else if((BMI>23)&&(BMI<24.9))  rs = "Thừa cân";
        else if(BMI>=25) rs="Béo phì";
        return rs;
    }
}
