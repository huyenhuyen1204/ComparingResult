import java.text.*;
public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a >= b ) return  a;
        else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int size = arr.length , min=arr[0];
        for (int i=0;i<size ; i++){
            if (min > arr[i]) min =arr[i];
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = (double)Math.round(weight / Math.pow(height , 2.0) *10)/10 ;

//        double BMI =weight / Math.pow(height , 2.0);
//        DecimalFormat decimalFormat = new DecimalFormat("0.0");
//        decimalFormat.format(BMI);
        if (BMI >=25.0) return "Béo phì";
        else if (BMI >=23.0 && BMI <=24.9 ) return "Thừa cân";
        else if (BMI >=18.5 && BMI <= 22.9) return  "Bình thường";
        else if (BMI <18.5) return "Thiếu cân";
        return "";
    }
}
