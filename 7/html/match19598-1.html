class Week4 {

    public static int max2Int(int a, int b){
        if(a>b) return a;
        else  return b;
    }
    public static int  minArray(int[] arr){
        if(arr.length==0) return 0;
        int min= arr[0];
        for(int i=0;i<arr.length;i++){
            if(min>arr[i]) min= arr[i];
        }
        return min;
    }

    public static String calculateBMI(double w, double h){
        String s="";
        double BMI= w/(h*h);
        if(BMI<18.45) s="Thiếu cân";
        if(BMI>=18.45&&BMI<22.95) s="Bình thường";
        if(BMI>=22.95&&BMI<24.95) s="Thừa cân";
        if(BMI>=24.95) s="Béo phì";
        return s;
    }

}
