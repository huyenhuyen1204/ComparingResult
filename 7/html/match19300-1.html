public class Week4 {
    public static int max2Int(int a, int b) {
        if (a > b ){
            return a;
        }
        return b;
    }
    public static int minArray(int[] arr) {
        int min;
        min = arr[0];
        for (int i = 0 ; i < arr.length ; i++){
            if (arr[i] < min){
                min = arr[i];
            }
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI;
        BMI = weight/(height*height);
        double bmiRound = 
(double) Math.round(BMI *10)/10;
        if (bmiRound < 18.5){
            return "Thiếu cân";
        }
        if (bmiRound >= 18.5 && bmiRound <= 22.9){
            return "Bình thường";
        }
        if (bmiRound >= 23 && bmiRound <= 24.9){
            return "Thừa cân";
        }
        return "Béo phì";
    }
}
