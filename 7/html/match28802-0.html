src/Week4.java

public class Week4 {
    public static int max2Int(int a, int b) {
        return Math.max(a,b);
    }
    public static int minArray(int[] arr) {
        int length = arr.length;
        int kq = arr[0];
        for (int i = 1; i < length; i++)
            kq = Math.min(kq,arr[i]);
        return kq;
    }
    public static String calculateBMI(double weight, double height) {
        double y = weight/(height*height);
        double x = Math.round(y * 10.0) / 10.0 ;
        String kq = "";
        if (x < 18.5) kq += "Thiếu cân";
        if (x >= 25) kq += "Béo phì";
        if (x >= 18.5 && x < 23) kq += "Bình thường";
        if (x >= 23 && x < 25) kq+= "Thừa cân";
        return kq;
    }
}
