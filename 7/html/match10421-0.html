public class Week4 {

    public static int max2Int(int a, int b) {
        if(a>=b) return  a;
        else
            return b;
    }

    public static int minArray(int[] arr) {
        int min = arr[0];
        // for each loop
        for(int a : arr){
            if(a < min ){
                min = a;
            }
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI = weight/(Math.pow(height,2));
        BMI = (double) Math.round(BMI * 10) / 10;
        String S=null;
        if(BMI < 18.5) {
            S = "Thiếu cân";
        }
        else{
            if(BMI < 22.9){
                S = "Bình thường";
            }
            else{
                if(BMI < 24.9){
                    S = "Thừa cân";
                }
                else{
                    S = "Béo phì";
                }
            }
        }
        return S;
    }
}
