public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if(a > b) return a;
        return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int ans = arr[0];
        for (int i = 1; i < arr.length; i++){
            if (ans > arr[i]) ans = arr[i];
        }
        return ans;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI_ = weight /  Math.pow(height,2.0);
        double BMI = Math.round(BMI_ * 10) / 10.0;
        if (BMI < 18.5) return "Thiếu cân"
;
        if (BMI >= 18.5 && BMI <= 22.9) return "Bình thường";
        if (BMI >= 23 && BMI <= 24.9) return "Thừa cân";
        return "Béo phì";
    }
}
