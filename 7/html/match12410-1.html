class Week4
{
    static int max2Int(int a, int b)
    {
        int max;
        if(a>b)
            max = a;

        else
            max = b;
        return max;


    }

    static int minArray(int[] arr)
    {



            int min = arr[0];


            for (int i = 0; i < arr.length; i++)
            {

                if (arr[i] < min)
                    min = arr[i];
            }
            return min;

    }

    static String calculateBMI(double weight, double height)
    {
        double hien = (weight)/ (height*height);
        double BMI = Math.round(hien*100.0)/100.0;
        if (BMI < 18.5)
            return "Thiếu cân";
        else if (BMI >= 18.5 && BMI <= 22.99)
            return "Bình thường";
        else if (BMI >= 23 && BMI <= 24.99)
            return "Thừa cân";
        else
            return "Béo phì";
    }

}
