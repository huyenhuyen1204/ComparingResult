public class Week4 {
    public static int max2Int(int a, int b) {
        if(a>b)
        {
            return a;
        }
        return b;
    }


    public static int minArray(int[] arr) {
        int t=arr.length;
        for(int i=0;i<t;i++)
        {
            for(int j=i+1;j<t;j++)
            {
                if(arr[j]<arr[i])
                {
                    int z=arr[i];
                    arr[i]=arr[j];
                    arr[j]=z;
                }
            }
        }
        return arr[0];
    }


    public static String calculateBMI(double weight, double height) {
        
        double bmi;
        bmi = weight/(height*height);
        bmi=  Math.round(bmi * 10) / 10;
        if(bmi<18.5)
        {
           
            return "Thiếu cân";
        }
        else if(bmi>=18.5&&bmi<=22.9)
        {
            return "Bình thường";
        }
        else if(bmi>=23&&bmi<=24.9)
        {
            return "Thừa cân";
        }
        
        return "Béo phì";
        
        
    }
}
