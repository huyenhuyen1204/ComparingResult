public class Week4 {
    public static int max2Int(int a, int b) {
        int max1 = a;
        if (a<b)
            max1 = b;
        return max1;
    }

    public static int minArray(int[] arr) {
        int i, min = arr[0];
        for (i = 1; i< arr.length; i++){
            if (arr[i]<min)
                min = arr[i];
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BMI;
        String result="";
        BMI = (double) Math.round(weight/(height*height)*10)/10;
        
if (BMI < 18.5)
            result =  "Thiếu cân";
        else if (BMI <= 22.9)
            result = "Bình thường";
        else if (BMI <= 24.9)
            result =  "Thừa cân";
        else
            result =  "Béo phì";
        return result;
    }
}
