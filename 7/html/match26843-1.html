import java.util.Arrays;

public class Week4 {

    public static int max2Int(int a, int b) {
        return Integer.max(a,b);
    }


    public static int minArray(int[] arr) {
        return Arrays.stream(arr).min().getAsInt();
    }


    public static String calculateBMI(double weight, double height) {

        double a=weight*1.0/(height*height)+0.05;
        if(a<18.5) return "Thiếu cân";
        else if(a<23) return "Bình thường";
        else if(a<25) return "Thừa cân";
        else return "Béo phì";
        //return String.format("%.0f",weight*1.0/(height*height));
    }
    public static void main(String[] args) {
        System.out.println(calculateBMI(53.4,1.7));
        System.out.println(max2Int(5,4));

    }
}
