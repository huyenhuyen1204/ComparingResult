public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return Math.max(a,b);
    }
    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int rs=arr[0];
        for (int i=0; i < arr.length; i++)
            rs=Math.min(rs,arr[i]);
        return rs;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double rs=weight/(height*height);
        
rs=(double) Math.round((rs*10)/10);
        if (rs < 18.5){
            return "Thiếu cân";
        }
        if (rs <= 22.9){
            return "Bình thường";
        }
        if (rs <= 24.9){
            return "Thừa cân";
        }
        return "Béo phì";
    }
    public static void main(String[] args){
    }
}
