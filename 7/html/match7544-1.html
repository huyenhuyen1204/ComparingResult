class Week4 {

    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a > b) {
            return a;
        } else {
            return b;
        }
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int min = arr[0];
        for (int n : arr) {
            if (n < min) {
                min = n;
            }
        }
        return min;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        String rs;
        double bmi = Math.round((weight / (height * height)) * 10.0) / 10.0;
        if (bmi < 18.5) {
            rs = "Thiếu cân";
        } else if (bmi < 23) {
            rs = "Bình thường";
        } else if (bmi < 25) {
            rs = "Thừa cân";
        } else {
            rs = "Béo phì";
        }
        return rs;
    }
}
