class Week4 {
    public static int max2Int(int a, int b) {
        if (a > b) return a;
        return b;
    }

    public static int minArray(int[] arr) {
    int size_arr = arr.length;
    int min = arr[0];
    for (int i =1 ;i<size_arr;i++){
        if(min > arr[i]) min = arr[i];
    }
    return min;
    }


    public static String calculateBMI(double weight, double height) {
        double IBM = (Math.round((weight / (Math.pow(height, 2)))*10))/10.0;
        
System.out.println(IBM);
        if (IBM >= 25) return  "Béo phì";
        if (IBM>=23 && IBM<=24.9) return "Thừa cân";
        if (IBM>=18.5 && IBM <=22.9) return "Bình thường";
        return "Thiếu cân";
    }
}
