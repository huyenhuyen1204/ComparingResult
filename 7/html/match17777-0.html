public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return (a>b)? a : b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int result = arr[0];
        for (int t : arr){
            if ( t < result) result = t;
        }
        return result;
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = Math.round(weight * 10 / (height * height))/10.0;
        String result;
        if (BMI < 18.5) result = "Thiếu cân";
        else if (BMI < 23) result = "Bình thường";
        else if (BMI < 25) result = "Thừa cân";
        else result = "Béo phì";
        return result;
    }
}
