Week4.java

import java.util.*;
public class Week4 {
    public static int max2Int(int a, int b) {
        if(a>=b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int min = arr[0];
        for(int i=1; i<arr.length; i++){
            if(arr[i]<min) min=arr[i];
        }
        return min;
    }


    public static String calculateBMI(double weight, double height) {
        double k=weight/(height*height);
        double BMI = (double) Math.round( k * 10 ) / 10 ;
        String result=null;
        if(BMI<18.5){
            result="Thiếu cân";
        }
        if(BMI>=18.5&&BMI<=22.9){
            result="Bình thường";
        }
        if(BMI>=23&&BMI<=24.9){
            result="Thừa cân";
        }
        if(BMI>=25){
            result="Béo phì";
        }
        return result;
    }
}
