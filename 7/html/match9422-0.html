import java.util.Arrays;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        return a>b ? a:b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI = (double) Math.round((weight/(height*height))*10)/10;
        String result="";
        if(BMI<18.5) result = "Thiếu cân";
        else if(BMI>=18.5 && BMI <=22.9) result = "Bình thường";
        else if(BMI>=23.0 && BMI<=24.9) result = "Thừa cân";
        else if(BMI >= 25.0) result = "Béo phì";
        return result;
    }
}
