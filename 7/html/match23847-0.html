Week4.java

public class Week4 {

    public static int max2Int(int a, int b) {
        if(a>b){
            return a;
        }
        return b;

    }


    public static int minArray(int[] arr) {
        int n=arr.length,min=99999;
        for(int i=0;i<n;i++){
            if(arr[i]<min) min=arr[i];
        }
        return min;
    }


    public static String calculateBMI(double weight, double height) {
        double BMI;
        BMI=weight/(height*height);
        BMI=(double)Math.round(BMI*10)/10;
        String a;
        if(BMI<18.9) a = "Thiếu cân";
        else if(BMI>=18.9 && BMI<23) a = "Bình thường";
        else if(BMI>=23 && BMI<24.9) a = "Thừa cân";
        else a = "Béo phì";
        return a;
    }
}
