import java.util.*;

public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        if (a >= b){
            return a;
        } else return b;
    }

    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        Arrays.sort(arr);
        return arr[0];
    }

    /**
     *
     * @param weight: in kg
     * @param height: in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double BMI;
        double BMIs = ( weight ) / ( height * height );
        BMI = (double) Math.round(BMIs * 10) / 10;
        if (BMI < 18.5){
            return "Thiếu cân";
        } else if (BMI >= 18.5 && BMI < 22.9){
            return "Bình thường";
        } else if (BMI >= 23 && BMI < 24.9){
            return "Thừa cân";
        } else return "Béo phì";
    }

}
