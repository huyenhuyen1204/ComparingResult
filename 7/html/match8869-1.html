import java.text.DecimalFormat;

public 
class Week4 {

    public static int max2Int(int a, int b) {
        return Math.max(a ,b);
    }

    public static int minArray(int[] arr) {
        int min = arr[0];
        for(int a : arr){
            if(a < min) min = a;
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        DecimalFormat df = new DecimalFormat("0.0");
        double BMI_index = weight / Math.pow(height , 2);
        BMI_index = Double.valueOf(df.format(BMI_index));
        if(BMI_index < 18.5) return "Thiếu cân";
        else if(BMI_index >= 18.5 && BMI_index <= 22.9) return "Bình thường";
        else if(BMI_index >= 23 && BMI_index <= 24.9) return "Thừa cân";
        else return "Béo phì";
    }
}
