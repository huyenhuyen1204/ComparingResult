Week4.java

import java.util.Arrays;

public class Week4 {

    public static int max2Int(int a, int b) {
        return (a > b ? a : b);
    }

    public static int minArray(int[] arr) {
        return Arrays.stream(arr).min().getAsInt();
    }

    public static String calculateBMI(double weight, double height) {
        double BIM = weight/(height*height);
        
BIM = Math.round(BIM*10) / 10.0;
        if (BIM < 18.5)
            return "Thiếu cân";
        if (BIM < 23)
            return "Bình thường";
        if (BIM < 25)
            return "Thừa cân";
        else
            return "Béo phì";
    }

    public static void main(String[] args) {
        int[] arr = {2, 6, 1, 8, 9};
        System.out.println(minArray(arr));
        System.out.println(max2Int(3, 5));
        System.out.println(calculateBMI(55, 1.65));
        System.out.println(Math.round(18.05*10) / 10.0);
    }
}
