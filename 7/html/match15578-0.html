Week4.java


public class Week4 {
    public static int max2Int(int a, int b) {
        if (a > b) return a;
        else return b;
    }

    public static int minArray(int[] arr) {
        int size = arr.length;
        int min = arr[0];
        for(int i = 0; i < size; i++) {
            if (arr[i] < min) min = arr[i];
        }
        return min;
    }

    public static String calculateBMI(double weight, double height) {
        double BIM = weight/(height*height);
        BIM = (double)Math.round(BIM*10)/10;
        if(BIM<18.5) return "Thiếu cân";
        else if(BIM >= 18.5 && BIM <=22.9) return  "Bình thường";
        else if(BIM >= 23 && BIM <= 24.9)  return "Thừa cân";
        else  return "Béo phì";
    }
}
