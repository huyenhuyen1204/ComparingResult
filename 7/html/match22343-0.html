public class Week4 {
    /**
     *
     * @param a
     * @param b
     * @return max of $a and $b
     */
    public static int max2Int(int a, int b) {
        int k  = a;
        if (b > k) k=b;
        return k;}
    /**
     *
     * @param arr
     * @return min of array $arr
     */
    public static int minArray(int[] arr) {
        int a=arr[0];
        for(int i=0;i<arr.length;i++) if(arr[i]<a) a=arr[i];
     return a;
    }
    /**
     *
     * @param weight : in kg
     * @param height : in m
     * @return BMI index
     */
    public static String calculateBMI(double weight, double height) {
        double x=weight/(height*height);
        double b;
        b = (double) Math . round (x * 10)/10;
        if(b<18.5) return "Thiếu cân";
        if(b>=18.5 && b<=23.9)
            return "Bình thường";
        if(b>=23 && b<=24.9)
            return "Thừa cân";
        if(b>=25) return "Béo phì";
        return null;
    }
}
