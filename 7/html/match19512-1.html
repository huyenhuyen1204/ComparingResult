class Week4 {
    public static int max2Int(int a, int b){
        if (a>=b) return a;
        else return b;
    }
    public static int minArray(int[] arr){
        int min = Integer.MAX_VALUE;
        for (int a:arr){
            if (a<min) min = a;
        }
        return min;
    }
    public static String calculateBMI(double weight, double height){
        double BMI = weight/(height*height);
        String s = null;
        double a = 
(double)Math.round(BMI*10)/10;
        if (a<18.5)
            s = "Thiếu cân";
        else if (18.5<=a&&a<=22.9)
            s = "Bình thường";
        else if (23<=a&&a<=24.9)
            s = "Thừa cân";
        else if (a>=25)
            s = "Béo phì";
        return s;
    }
}
