public class ExpressionTest {
    public static void main(String[] argv) {
        Numeral x1 = new Numeral(10);
        Numeral x2 = new Numeral(-3);
        Numeral x3 = new Numeral(4);
        Numeral x4 = new Numeral(3);
        Square s1 = new Square(x1);
        Addition a1 = new Addition(s1, x2);
        Multiplication m1 = new Multiplication(x3, x4);
        Addition a2 = new Addition(a1, m1);
        Square s2 = new Square(a2);
        System.out.println(s2.toString() + " = " + s2.evaluate());
    }
}
