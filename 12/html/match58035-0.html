BinaryExpression.java

public abstract class BinaryExpression extends Expression {
    protected Expression left;
    protected Expression right;
    public BinaryExpression(Expression left, Expression right) {
        this.left = left;
        this.right = right;
    }

    public abstract String toString();
    public abstract double evaluate();
}

public class ExpressionTest {
    public static void main(String[] args) {
        Expression nu = new Numeral(10);
        Expression nu1 = new Numeral(3);
        Expression sq = new Square(nu);
        Expression su = new Subtraction(sq, nu1);
        Expression nu2 = new Numeral(4);
        Expression mul = new Multiplication(nu2,nu1);
        Expression add = new Addition(su, mul);
        Expression sqt = new Square(add);
        System.out.println(sqt.evaluate());
        System.out.println(sqt.toString());
    }

}
