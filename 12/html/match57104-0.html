public class ExpressionTest {

    public static void main(String[] args){

       Expression numberal = new Numeral(10);
       Expression a = new Square(numberal);
       Expression numberal1 = new Numeral(5);
       Expression addition = new Addition(a, new Numeral(-3));
       Expression mul = new Multiplication(numberal1, new Numeral(-3));
       Expression add = new Addition(addition, mul);
       Expression square = new Square(add);
       System.out.println(square.toString()); 
       Expression division = new Division(numberal, new Numeral(0));
       System.out.println(division);
    }
