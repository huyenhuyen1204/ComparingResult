		super(l, r);
	}

	public double evaluate() {
		return left.evaluate() + right.evaluate();
	}

	public 
String toString() {

		return "(" + left_str + " + " + right_str + ")";

import java.lang.Math;

public class Division extends BinaryExpression {
	public Division(Expression l, Expression r) {
		super(l, r);
	}

	public double evaluate() {
		double left_val = left.evaluate();
		double right_val = right.evaluate();

		if (Math.abs(right_val) < 1e-7)
			throw new java.lang.ArithmeticException("Lá»—i chia cho 0");
		else
			return left_val / right_val;
	}

	public 
String toString() {
		String left_str = left.toString();
		String right_str = right.toString();

		return "(" + left_str + " / " + right_str + ")";
	}
}
