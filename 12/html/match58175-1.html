Numeral.java

public 
class Numeral extends Expression {
    private double value;

    public Numeral (double value) {
        this.value = value;
    }

    public Numeral(){};

    @Override
    public String toString() {
        if (value % 1 == 0) return String.valueOf((int)value);
        return String.valueOf(value);
    }

    @Override
    public double evaluate() {
        return value;
    }


}
