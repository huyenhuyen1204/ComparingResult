Numeral.java

public class Numeral extends Expression{

    private double value;

    public Numeral(double value) {
        this.value = value;
    }
    public Numeral(){ }

    @Override
    public String toString() {
        if(value%1!=0)
            return Double.toString(value);
        else
            return Integer.toString((int)value);
    }

    @Override
    public double evaluate() {
        return value;
    }
}
