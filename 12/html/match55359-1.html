public class Division extends BinaryExpression {
    public Division(Expression left, Expression right) {
        super(left, right);
    }
    @Override
    public String toString() {
        String str = "";
        str += "(" + this.left.toString() + " / " + this.right.toString() + ")";
        
return str;
    }

    @Override
    public double evaluate() {
       if(this.right.evaluate() == 0) {
           throw new ArithmeticException("Lá»—i chia cho 0");
       }
       else return this.left.evaluate()/this.right.evaluate();
    }
}

Square.java

/**
 * Created by hung on 27/10/2019.
 */
public class Square extends Expression {
    private Expression expression;
    public Square(Expression expression) {
        this.expression = expression;
    }
    @Override
    public String toString() {
        String str = "";
        str += "(" + this.expression + ") ^ 2";
        
return str;
    }

    @Override
    public double evaluate() {
        return this.expression.evaluate()*this.expression.evaluate();
    }
}
