Addition.java

public class Addition extends BinaryExpression{
  public Addition(Expression left, Expression right){
    super(left,right);
  }
  public String toString(){
    return "(" + left.toString()+ " + " +right.toString()+")";
  }
  public double evaluate(){
      return left.evaluate()+right.evaluate();
  }
}

public class ExpressionTest{
  public static void main(String args[]){
    Subtraction test = new Subtraction(new Numeral(10), new Numeral(2));
    System.out.println(test.toString() + "=" + test.evaluate());

    Square test1 = new Square(new Addition(new Subtraction( new Square(new Numeral(10)),new Numeral(3)), new Multiplication(new Numeral(4), new Numeral(3))));
    System.out.println(test1.toString() + "=" + test1.evaluate());

    Division test2 = new Division(new Numeral(300), new Numeral(0));
    System.out.println(test2.toString() + "=" + test2.evaluate());

  }
}

Multiplication.java

public class Multiplication extends BinaryExpression{
  public Multiplication(Expression left, Expression right){
    super(left,right);
  }
  public String toString(){
    return "(" + left.toString()+ " * "+right.toString()+")";
  }
  public double evaluate(){
      return left.evaluate()*right.evaluate();
  }
}

Subtraction.java

public class Subtraction extends BinaryExpression{
  public Subtraction (Expression left, Expression right){
    super(left,right);
  }
  public String toString(){
    return "(" +left.toString()+ " + -"+right.toString()+")";
  }
  public double evaluate(){
      return left.evaluate()-right.evaluate();
  }
}
