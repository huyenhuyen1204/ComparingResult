public class Numeral extends Expression {
    private double value;
    public Numeral(double x) {
        value=x;
    }
    public Numeral(){};

    @Override
    public String toString() {
        return (int) value +"";
    }

    @Override
    public double evaluate() {
        return value;
    }
    
}
