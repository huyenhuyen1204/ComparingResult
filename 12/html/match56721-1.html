Addition.java

public 
class Addition extends BinaryExpression{
    public Addition(Expression left, Expression right) {
        super(left, right);
    }
    public double evaluate() {
        return left.evaluate()+right.evaluate();
    }
    public String toString() {
        return "(" +left.toString()+" + "+right.toString()+")";
    }
}

public class ExpressionTest {
    @Test
    public

        Expression ten= new Numeral(10);
        Expression three= new Numeral(3);
        Expression nethree = new Numeral(-3);
        Expression four = new Numeral(4);
        Square square1 = new Square(ten);
        Addition addition1 = new Addition(square1, nethree);
        Multiplication multiplication = new Multiplication(four, three);
        Addition addition2 = new Addition(addition1, multiplication);
        Square square2 = new Square(addition2);
        assertEquals("((((10) ^ 2 + -3) + (4 * 3))) ^ 2", square2.toString());
    }

}

class Multiplication extends BinaryExpression {

    }

    }
}

class Subtraction extends BinaryExpression{
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }
    public double evaluate() {
        return left.evaluate() - right.evaluate();
    }
    public String toString() {
        return "("+left.toString()+" - "+right.toString()+")";
    }
