    public static void main(String[] args){
        Expression number1= new Numeral(10);
        Expression number2= new Numeral(3);
        Expression number3= new Numeral(4);
        Expression number4= new Numeral(0);
        Expression square1 = new Square(number1);
        Expression subtraction1 = new Subtraction(square1, number2);
        Expression multiplication = new Multiplication(number3, number2);
        Expression addition1 = new Addition(subtraction1, multiplication);
        Expression result = new Square(addition1);
        System.out.println(result.toString());
        System.out.println(result.evaluate());
        try{
            Expression divisionTest = new Division(number1, number4);
            double result2= divisionTest.evaluate();
            System.out.println(result2);
        } catch(ArithmeticException e){
            System.out.println("Lá»—i chia cho 0");
        }
    }
}

public class Multiplication extends BinaryExpression{
    public Multiplication(Expression left, Expression right){
        super(left, right);
    }
    public double evaluate(){
        return this.left.evaluate() * this.right.evaluate();
    }

    @Override
    public String toString() {
        return "(" + this.left + " * " + this.right + ")";
    }
}
