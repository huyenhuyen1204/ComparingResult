Numeral.java

public 
class Numeral extends Expression {
    private double value;

    public Numeral() {
        value = 1;
    }

    public Numeral(double value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return String.valueOf(Math.round(value));
    }

    @Override
    public double evaluate() {
        return value;
    }
}
