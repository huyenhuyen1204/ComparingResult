ExpressionTest.java

public class ExpressionTest {
    public static void main(String[] args){
        Expression a1 = new Numeral(10);
        Expression a2 = new Numeral(3);
        Expression a3 = new Numeral(4);
        Expression sq1 = new Square(a1);
        Expression sub1 = new Subtraction(sq1, a2);
        Expression mul = new Multiplication(a3, a2);
        Expression add = new Addition(sub1, mul);
        Expression result = new Square(add);
        System.out.println(result.toString());
        System.out.println(result.evaluate());
    }
}

Numeral.java

public class Numeral extends Expression {
    private double value;
    public Numeral(){}
    public Numeral(double value) {
        this.value = value;
    }
    public String toString(){
        return Integer.toString((int)value);
    }
    public double evaluate(){
        return value;
    }
}
