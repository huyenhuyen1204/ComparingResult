Division.java

//package Week_8;


import java.lang.ArithmeticException;
/**
 * Created by CCNE on 23/10/2019.
 */
public class Division extends BinaryExpression{
    public Division(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        String result = "(" + this.left + " / " + this.right + ")";
        return result;
    }

    @Override
    public double evaluate(){
        double res = this.left.evaluate() / this.right.evaluate();
        if (Double.isNaN(res) || Double.isInfinite(res))
            throw new ArithmeticException("Lá»—i chia cho 0");
        return res;
    }
}

Numeral.java

//package Week_8;



import java.text.DecimalFormat;

/**
 * Created by CCNE on 23/10/2019.
 */
public class Numeral extends Expression{
    private double value;
    public Numeral(){
        this.value = 0;
    }
    public Numeral(double value){
        this.value = value;
    }

    @Override
    public String toString() {
        DecimalFormat decimalFormat = new DecimalFormat();
        String result = decimalFormat.format(this.value);
        return result;
    }

    @Override
    public double evaluate() {
        return this.value;
    }
}
