BinaryExpression.java

public abstract class BinaryExpression extends Expression {

    protected Expression left;
    protected Expression right;

    public BinaryExpression(Expression left, Expression right) {
        this.left = left;
        this.right = right;
    }

    @Override
    public abstract String toString();

    @Override
    public abstract double evaluate();
}

public class ExpressionTest {

    public static void main(String[] args) {
        Expression n1 = new Numeral(10);
        Expression n2 = new Square(n1);
        Expression n3 = new Numeral(-3);

        Expression l1 = new Addition(n2, n3);

        Expression n4 = new Multiplication(new Numeral(4), new Numeral(3));

        Expression l2 = new Addition(l1, n4);
        Expression result = new Square(l2);

        System.out.println(result.toString());;
    }
