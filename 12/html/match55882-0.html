abstract public class BinaryExpression extends Expression {
    protected Expression left;
    protected Expression right;
    BinaryExpression(Expression left,Expression right){
        this.left=left;
        this.right=right;
    }
    abstract public double evaluate();
    abstract public String toString();
    public static void main(String[] args){
        Expression a=new Addition(new Numeral(2),new Numeral(3));
        System.out.println(a.toString());
    }

}

/**
 * Created by CCNE on 22/10/2019.
 */
public class Numeral extends  Expression{
    private  int value;
    Numeral(){}
    Numeral(int value){
        this.value=value;
    }

    public String toString() {
        return String.valueOf((int)value);
    }

    public double evaluate() {
        return value;
    }
}

/**
 * Created by CCNE on 22/10/2019.
 */
public class Square extends Expression {
    private Expression expression;
    Square(Expression expression){
        this.expression=expression;
    }

    public double evaluate() {
        return (double)expression.evaluate()*expression.evaluate();
    }

    public String toString() {
        return "("+String.valueOf(expression)+")^2";
    }
}
