public class ExpressionTest {
    public static void main(String[] args) {
        // (10^2 - 3 + 4*3)^2   // ((((10) ^ 2 + -3) + (4 * 3))) ^ 2
        // (10) ^ 2
        Square square1 = new Square(new Numeral(10));

        // -3
        Numeral numeral1 = new Numeral(-3);

        // ((10) ^ 2 + -3)
        Addition addition1 = new Addition(square1, numeral1);

        // (4 * 3)
        Multiplication multiplication1 = new Multiplication(new Numeral(4), new Numeral(3));

        // (((10) ^ 2 + -3) + (4 * 3))
        Addition addition2 = new Addition(addition1, multiplication1);

        // ((((10) ^ 2 + -3) + (4 * 3))) ^ 2
        Square square2 = new Square(addition2);
        System.out.println(square2.toString());

    }
}

Numeral.java

public class Numeral extends Expression {
    private double value;

    public Numeral() {

    }

    public Numeral(double value) {
        this.value = value;
    }

    public double getValue() {
        return value;
    }

    public void setValue(double value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return Integer.toString((int)this.getValue());
    }

    @Override
    public double evaluate() {
        return this.getValue();
    }
}
