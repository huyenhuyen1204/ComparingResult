src/Numeral.java

public class Numeral extends Expression{
    private double value;
    public Numeral(){

    }
    public Numeral(double value){
        this.value = (int)value;
    }
    @Override
    public String toString() {
        return "Numeral{" +
                "value=" + value +
                '}';
    }

    @Override
    public double evaluate() {
        return value;
    }
}
