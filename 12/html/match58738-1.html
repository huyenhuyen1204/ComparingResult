class Numeral extends Expression {
    private double value;
    public Numeral()
    {

    }

    public Numeral(double value)
    {
        this.value = value;
    }

    @Override
    public double evaluate() {
        if(value == 0)
        {
            return (int) value;
        }
        return value;
    }

    @Override
    public String toString() {
        Double i = this.value;
        Integer a = i.intValue();
        return a.toString();
    }
}
