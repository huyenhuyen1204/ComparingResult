class Division extends BinaryExpression{
    public Division(Expression left, Expression right) {
        super(left, right);
    }


    @Override
    public double evaluate() throws ArithmeticException{
        double numerator = left.evaluate();
        double denominator = right.evaluate();
        if (denominator == 0) throw new ArithmeticException("Lá»—i chia cho 0");
        else return numerator / denominator;
    }

    @Override
    public String toString() {
        return "(" + left.toString() + " / " + right.toString() + ")";
    }
}

class Square extends Expression{
    private Expression expression;

    public Square(Expression expression) {
        this.expression = expression;
    }

    @Override
    public double evaluate() {
        double e = expression.evaluate();
        return e * e;
    }

    @Override
    public String toString() {
        return "(" + expression.toString() + ") ^ 2";
    }

    public static void main(String[] args) {
        double f = 3;
        String s = String.format("%0s", f);
        System.out.println(s);
    }
}
