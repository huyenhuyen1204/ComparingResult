class ExpressionTest {
    public static  void main(String[] args){
        Numeral num1 = new Numeral(10);
        Numeral num2 = new Numeral(-3);
        Numeral num3 = new Numeral(4);
        Numeral num4 = new Numeral(3);
        Square square = new Square(num1);
        Addition add1 = new Addition(square, num2);
        Multiplication multiplication = new Multiplication(num3,num4);
        Addition add2 = new Addition(add1, multiplication);

        Numeral numdiv1 = new Numeral(2);
        Numeral numdiv2 = new Numeral(0);
        Division division = new Division(numdiv1,numdiv2);
        Square result = new Square(add2);
        System.out.println(result);
        System.out.println(division.evaluate());
    }
}

public 
class Square extends Expression {
    private Expression expression;
    public Square(Expression expression){
        this.expression = expression;
    }
    @Override
    public String toString(){
        return "(" + expression + ") ^ 2";
    }
    @Override
    public double evaluate(){
        return expression.evaluate() * expression.evaluate();
    }
}
