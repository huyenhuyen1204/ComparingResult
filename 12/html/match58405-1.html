public 
class ExpressionTest {
    //  Test: (10^2 - 3 + 4*3)^2
    public static void main(String [] args) {
         Numeral n1 = new Numeral(10);
         Numeral n2 = new Numeral(3);
         Numeral n3 = new Numeral(4);
         Square square = new Square(n1); // (10) ^ 2
         Subtraction subtraction = new Subtraction(square, n2); // ((10) ^ 2 + -3)
         Multiplication multiplication = new Multiplication(n3, n2); // (4*3)
         Addition addition = new Addition(subtraction, multiplication); // (((10) ^ 2 + -3) + (4*3))
         Square result = new Square(addition); // ((((10 ^ 2) + -3) + (4*3))) ^ 2
         System.out.println(result.toString()); // in ra ((((10 ^ 2) + -3) + (4*3))) ^ 2
         /*  lệnh result.toString() gồm các lệnh nhỏ sau đây hợp thành
            b1: gọi toString() của Square;
            b2: gọi toString() của Addition
            b3: gọi toString() của Subtraction, Multiplication
            b4: gọi toString() của Square và Numeral
         */
    }

}

week8/Square.java

public 
class Square extends Expression{
    private Expression expression;
    public Square(Expression expression){
        this.expression = expression;
    }
    @Override

    public String toString() {
        return "(" + expression.toString() + ") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate() * expression.evaluate();
    }
}
