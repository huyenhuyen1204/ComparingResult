Division.java

public class Division extends BinaryExpression {
    public Division(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        return "(" + left.toString() + " / " + right.toString() + ")";
    }

    @Override
    public double evaluate() {
        if (right.evaluate() == 0) throw new ArithmeticException("Lá»—i chia cho 0");
        else return left.evaluate()/right.evaluate();
    }

    public static void main(String[] args) {
        Numeral a = new Numeral(2);
        Numeral b = new Numeral(0);
        Division result = new Division(a, b);
        System.out.print(result.evaluate());
    }
}

Numeral.java

public class Numeral extends Expression {
    private double value;

    public Numeral() {}
    public Numeral(double value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return "" + (int)value;
    }

    @Override
    public double evaluate() {
        return this.value;
    }
}
