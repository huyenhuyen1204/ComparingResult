class ExpressionTest {
    public static void main(String[] args) {
        Numeral n1 = new Numeral(10);
        Numeral n2 = new Numeral(3);
        Numeral n3 = new Numeral(4);
        Square s1 = new Square(n1);
        Subtraction d1 = new Subtraction(s1,n2);
        Multiplication m1 = new Multiplication(n3,n2);
        Addition a1 = new Addition(d1,m1);
        Square s = new Square(a1);
        System.out.println(s.toString());
        System.out.println(s.evaluate());
        Numeral n = new Numeral(0);
        Division d = new Division(n1,n);

}

public 
class Square extends Expression{
    private Expression expression;
    public Square(Expression expression)
    {
        this.expression = expression;
    }

    @Override
    public String toString() {
        return "("+this.expression+") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}
