Week7/Addition.java

public 
class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }
    public String toString() {
        return String.format("%s + %s", left, right);
    }
    public double evaluate() {
        return left.evaluate() + right.evaluate();
    }
}

Week7/Division.java

public 
class Division extends BinaryExpression {
        public Division(Expression left, Expression right) {
            super(left, right);
        }

        public Expression left() {
            return super.left();
        }

        public Expression right() {
            return super.right();
        }

        public String toString() {
            return String.format("%s / %s", left, right);
        }

        public double evaluate() {
            return left.evaluate() / right.evaluate();
        }
    }

Week7/Multiplication.java

public 
class Multiplication extends BinaryExpression {
    public Multiplication(Expression left, Expression right){
        super(left,right);
    }
    public Expression left(){
        return super.left();
    }
    public Expression right(){
        return super.right;
    }
    public String toString(){
        return String.format("%s * %s", left, right);
    }
    public double evaluate() {
        return left.evaluate() * right.evaluate();
    }
}

Week7/Numeral.java

public class Numeral extends Expression {
    private double value;

    public Numeral(double value){
        this.value=value;
    }
    public Numeral(){ 
}
    public String toString() {
        return String.format("%d", value);
    }
    public double evaluate() {
        return value;
    }
}

Week7/Square.java

public 
class Square extends  Expression {
    private Expression expression;
    public Square(Expression expression){
       this.expression=expression;
    }
    public String toString() {
        return String.format("(%s)^2", expression);
    }
    public double evaluate() {
        return Math.pow(expression.evaluate(), 2);
    }
}
