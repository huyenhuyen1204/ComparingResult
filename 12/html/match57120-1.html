src/Addition.java

public 
class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right){
        super(left, right);
    }
    public double evaluate(){
        return left.evaluate() + right.evaluate();
    }
    public String toString(){
        return "(" + left.toString() + " + " + right.toString() + ")";
    }
}

src/Division.java

public 
class Division extends BinaryExpression {
    public Division(Expression left, Expression right){
        super(left, right);
    }
    public double evaluate() {
        if(right.evaluate() == 0){
            throw new ArithmeticException("Loi chia cho 0");
        }
        return left.evaluate()/right.evaluate();
    }
    public String toString(){
        return "(" + left.toString() + "/" + right.toString() + ")";
    }
}

public 
class ExpressionTest {
    public void testExpression(){
        Numeral a = new Numeral(10);
        Numeral b = new Numeral(3);
        Numeral c = new Numeral(-3);
        Numeral d = new Numeral(4);
        Square sq = new Square(a);
        Addition add = new Addition(sq, c);
        Multiplication m = new Multiplication(d, b);
        Addition add1 = new Addition(add, m);

        assertEquals("((((10) ^ 2 + -3) + (4 * 3))) ^ 2",kq.toString());
    }

}
