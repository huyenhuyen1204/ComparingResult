import java.lang.ArithmeticException;
public 
class Division extends BinaryExpression {
    public Division(Expression left, Expression right){
        super(left, right);
    }

    public String toString(){
        return "(" + this.left.toString() + " / " + this.right.toString() + ")";
    }
    public double evaluate() throws ArithmeticException{
        if(this.right.evaluate()==0) throw new ArithmeticException("Lá»—i chia cho 0");
        else{
            return (this.left.evaluate()/this.right.evaluate());
        }
    }
}

Multiplication.java

public 
class Multiplication extends BinaryExpression {
    public Multiplication(Expression left, Expression right){
        super(left, right);
    }
    public String toString(){
        return ("(" + this.left.toString() + " * " + this.right.toString() + ")");
    }
    public double evaluate(){
        return this.right.evaluate()*this.left.evaluate();
    }
}

Numeral.java

public 
class Numeral extends Expression {
    private  double value;

    public Numeral(double value){
        this.value = value;
    }
    public Numeral(){
        this.value=0;
    }

    public String toString(){
        return String.format(String.valueOf(this.value)).replaceAll("\\.0+$", "");
    }
    public double evaluate(){
        return this.value;
    }
}

Subtraction.java

public 
class Subtraction extends BinaryExpression {
    public Subtraction(Expression left, Expression right){
        super(left, right);
    }

    public String toString(){
        return ("("+left.toString()+" - "+right.toString()+")");
    }
    public double evaluate(){
        return (this.left.evaluate()-this.right.evaluate());
    }
}
