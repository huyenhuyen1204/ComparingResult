/**

 */
public 
class Addition extends BinaryExpression{
    Addition(Expression left,Expression right){
        super(left,right);
    }

    public double evaluate() {
        return left.evaluate()+right.evaluate();
    }

    public String toString() {
        return " ("+left+" + " +right+") ";
    }
}

class Multiplication extends BinaryExpression{

public 
class Numeral extends  Expression{
    private  int value;
    Numeral(){}
    Numeral(int value){
        this.value=value;
    }

    public String toString() {
        return String.valueOf((int)value);
    }

    public double evaluate() {
        return value;
    }
}

src/Square.java

/**
 * Created by CCNE on 22/10/2019.
 */
public 
class Square extends Expression {
    private Expression expression;
    Square(Expression expression){
        this.expression=expression;
    }

    public double evaluate() {
        return (double)expression.evaluate()*expression.evaluate();
    }

    public String toString() {
        return "("+String.valueOf(expression)+")^2";
    }
}

class Subtraction extends BinaryExpression {
