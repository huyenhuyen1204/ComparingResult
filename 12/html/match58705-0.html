public class ExpressionTest {
    public static void main(String[] args) {
        Expression n1 = new Numeral(1);
        Expression n2 = new Numeral(2);
        Expression n3 = new Numeral(3);
        Expression n4 = new Numeral(4);
        Expression n5 = new Numeral(0);

        Expression a12 = new Addition(n1, n2);
        Expression s34 = new Subtraction(n3, n4);
        Expression sqr5 = new Square(n5);

        Expression m = new Multiplication(a12, s34);
        Expression d = new Division(m, sqr5);

        System.out.println(d);
        System.out.println(d.evaluate());
    }
}
