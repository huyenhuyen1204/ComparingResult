Addition.java

public class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }

    public double evaluate(){
        return left.evaluate() + right.evaluate();
    }

    public String toString(){
        return "(" + left + " + " + right + ")";
    }
}

ExpressionTest.java

public class ExpressionTest {
    public static void main(String[] args) {
        Expression first = new Addition(new Square(new Numeral(10)), new Numeral(-3));
        Expression second = new Multiplication(new Numeral(4), new Numeral(3));
        Expression expression = new Square(new Addition(first, second));

}
