src/Addition.java

public 
class Addition extends BinaryExpression {
	public Addition(Expression l, Expression r) {
		super(l, r);
	}

	public double evaluate() {
		return left.evaluate() + right.evaluate();
	}

	public String toString() {
		String left_str = left.toString();
		String right_str = right.toString();

		return "(" + left_str + " + " + right_str + ")";
	}
}

src/Square.java

public 
class Square extends Expression {
	private Expression expression;
	
	public Square(Expression expr) {
		expression = expr;
	}

	public double evaluate() {
		double val = expression.evaluate();
		return val * val;
	}

	public String toString() {
		String expr_str = expression.toString();
		return "(" + expr_str + ") ^ 2";
	}
}
