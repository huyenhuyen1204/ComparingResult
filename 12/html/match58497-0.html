public class ExpressionTest{


    public static void main(String[] args) {
        Numeral ten = new Numeral(10);
        Numeral two = new Numeral(2);
        Numeral three = new Numeral(3);
        Numeral four = new Numeral(4);

        Square s1 = new Square((Expression) ten);
        Multiplication multi = new Multiplication((Expression) four, (Expression) three);
        Subtraction minus = new Subtraction((Expression) s1, (Expression) three);
        Addition plus = new Addition((Expression) minus, (Expression) multi);
        Square s  = new Square((Expression) plus);
        System.out.println(s.evaluate());
        System.out.print(s.toString());

        Division div = new Division(new Numeral(2), new Numeral(0));
        System.out.print(div.evaluate());
    }
}

Multiplication.java

public class Multiplication extends BinaryExpression{
    public Multiplication(Expression left, Expression right) {
        super(left, right);
    }

    public double evaluate() {
        return this.left.evaluate() * this.right.evaluate();
    }

    public String toString() {
        return "(" + this.left.toString() + " * " + this.right.toString() + ")";
    }
}

Subtraction.java

public class Subtraction extends BinaryExpression{
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }

    public double evaluate() {
        return this.left.evaluate() - this.right.evaluate();
    }

    public String toString() {
        return "(" + this.left.toString() + " - " + this.right.toString() + ")";
    }
}
