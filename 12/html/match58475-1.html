public class Division extends BinaryExpression {

    public Division(Expression nLeft, Expression nRight){
        super(nLeft,nRight);
    }

    @Override
    public String toString() {
        return "(" + left.toString() + " / " + right.toString() + ")";
    }

    @Override
    public 
double evaluate() throws ArithmeticException{
        double l=left.evaluate();
        double r=right.evaluate();
        if (r == 0){
            throw new ArithmeticException("Lá»—i chia cho 0");
        }
        return l/r;
    }
}

public 
class Square extends Expression {
    private Expression expression;

    public Square(Expression nExpression){
        expression=nExpression;
    }

    @Override
    public String toString() {
        return "(" + expression + ")" + " ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}
