class ExpressionTest {
    public static void main(String[] args) {
        Expression a = new Numeral(10);
        Expression c = new Numeral(3);
        Expression d = new Numeral(4);
        Expression e = new Numeral(3);
        Expression result = new Square(a);
        Expression result1 = new Subtraction(result,c);
        Expression result2 = new Multiplication(d,e);
        Expression result3 = new Addition(result1,result2);
        Expression result4 = new Square(result3);
        System.out.println(result4.evaluate());
