public 
class Numeral extends Expression{
    private double value;

    public Numeral() {
    }

    public Numeral(double value) {
        this.value = value;
    }

    @Override
    public String toString() {
        int tmp = (int)value;
        if (value == tmp) {
            
return String.valueOf(tmp);
        }
        return String.valueOf(value);
    }

    @Override
    public double evaluate() {
        return value;
    }
}
