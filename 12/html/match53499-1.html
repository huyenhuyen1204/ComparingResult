Addition.java

public class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right){
        super (left, right);
    }

    @Override
    public double evaluate() {
        return left.evaluate()+right.evaluate();
    
}

    @Override
    public String toString() {
        String l=left.toString();
        String r=right.toString();
        return  "("+l+" + "+r+")";
    }
}

class Division extends BinaryExpression {
    public Division(Expression left, Expression right){
        super (left, right);
    }

    @Override
    public double evaluate() throws ArithmeticException {
        double ans= left.evaluate()/right.evaluate();
        if (right.evaluate()==0)
            throw new ArithmeticException("Lá»—i chia cho 0");
        return ans;
    }

    @Override
    public String toString() {
        String l=left.toString();
        String r=right.toString();
        return  "("+l+" / "+r+")";
    }
}

Numeral.java

public 
class Numeral extends Expression {
    private double value;
    public Numeral(double value){
        this.value=value;
    }
    public Numeral(){}

    @Override
    public double evaluate() {
        return value;
    }

    @Override
    public String toString() {
        if (value - (int)value==0) return Integer.toString((int)value);
        return Double.toString(value);
    }
}
