class Addition extends BinaryExpression{

    {
        super(left,right);
    }
     public double evaluate()
     {
         return left.evaluate()+right.evaluate();
     }
     public String toString(){
        return "("+left.toString()+" + "+right.toString()+")";
     }
}

import java.lang.ArithmeticException;
public 
class Division extends BinaryExpression {
    public Division (Expression left,Expression right)
    {
        super(left,right);
    }
    public double evaluate() throws ArithmeticException
    {
        if(right.evaluate()==0)
        {
            throw new ArithmeticException("Lá»—i chia cho 0");
        }
        return (double)left.evaluate()/right.evaluate();
    }
    public String toString()
    {
        return "("+left.toString()+"\\"+right.toString()+")";
    }
}

src/Numeral.java

public class Numeral extends Expression {
    private double value;
    public Numeral(double Value)
    {
        this.value=Value;
    }
    public Numeral(){
}
    public double evaluate()
    {
        return value;
    }
    public String toString()
    {
        return String.valueOf((int)value);
    }
}

class Square extends Expression {
    private Expression expression;
    public Square(Expression expression)
    {
        this.expression=expression;
    }
    public double evaluate()
    {
        return (double)expression.evaluate()*expression.evaluate();
    }
    public String toString()
    {
        return "("+String.valueOf(expression)+") ^ 2";
    }

}
