public class ExpressionTest {
    public static void main(String[] args) {
        Expression a= new Numeral(10);
        Expression c= new Square(a);
        Expression b= new Numeral(3);
        Expression d= new Subtraction(c,b );
        Expression a1= new Numeral(3);
        Expression a2= new Numeral(4);
        Expression a3= new Multiplication(a1,a2);
        Expression s= new Addition(d, a3);
        Expression result= new Square(s);
        System.out.println(result.toString());
        System.out.println(result.evaluate());
    }

}
