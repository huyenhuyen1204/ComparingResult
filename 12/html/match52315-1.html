class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        return "(" + this.leftString + " + " + this.leftString + ")";
    }
    @Override
    public double evaluate() {
        return this.right.evaluate() + this.left.evaluate();
    }
}

Multiplication.java

public 
class Multiplication extends BinaryExpression {
    public Multiplication(Expression left, Expression right) {
        super(left, right);
    }
    @Override
    public String toString() {
        return "("+ this.leftString + " * " + this.rightString + ")";
    }

    @Override
    public double evaluate() {
        return this.left.evaluate() * this.right.evaluate();
    }
}

public class Square extends Expression {
    private Expression expression;
    private String expressionString;

    public Square(Expression expression) {
        this.expression = new Numeral(expression.evaluate());
        this.expressionString = expression.toString();
    }

    public void setSquare(Expression expression) {
        this.expression = new Numeral(expression.evaluate());
    }

    public 
Expression getSquare() {
        return expression;
    }

    @Override
    public String toString() {
        return "(" + this.expressionString + ") ^ 2" ;
    }

    @Override
    public double evaluate() {
        return this.expression.evaluate()*this.expression.evaluate();
    }
}

class Subtraction extends BinaryExpression{
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }
    @Override
    public String toString() {
        return "(" + this.leftString + " - " + this.rightString + ")";
    }

    @Override
    public double evaluate() {
        return this.left.evaluate() - this.right.evaluate();
    }
}
