src/BinaryExpression.java

abstract public class BinaryExpression extends Expression {
    protected Expression left;
    protected Expression right;
    public BinaryExpression(Expression l, Expression r)
    {
        this.left = l;
        this.right = r;
    }
    public abstract String toString();
    public abstract double evaluate();
}

src/Division.java


public class Division extends BinaryExpression {
    public Division(Expression l, Expression r)
    {
        super(l,r);
    }
    public String toString()
    {
        return "("+super.left.toString()+" / "+super.right.toString()+")";
    }
    public double evaluate() throws java.lang.ArithmeticException
    {
        if(super.right.evaluate()==0)   {
            throw new java.lang.ArithmeticException("Lá»—i chia cho 0");
        }
        return super.left.evaluate()/super.right.evaluate();
    }
}

src/Multiplication.java

public class Multiplication extends BinaryExpression {
    public Multiplication(Expression l,Expression r)
    {
        super(l,r);
    }
    public String toString()
    {
        return "("+super.left.toString()+" * "+super.right.toString()+")";
    }
    public double evaluate()
    {
        return super.left.evaluate()*super.right.evaluate();
    }
}

src/Square.java

public class Square extends Expression {
    private Expression expression;
    public Square(Expression e)
    {
        this.expression=e;
    }
    public String toString()
    {
        return "("+this.expression.toString()+") ^ 2";
    }
    public double evaluate()
    {
        return this.expression.evaluate()*this.expression.evaluate();
    }
}

public class Subtraction extends BinaryExpression {
    public Subtraction(Expression l, Expression r)
    {
        super(l,r);
    }
    public String toString()
    {
        return "("+super.left.toString()+" + - "+super.right.toString()+")";
    }
    public double evaluate()
    {
        return super.left.evaluate()-super.right.evaluate();
    }
}
