class ExpressionTest {
    public static void main(String[] args) {
        Expression ten = new Numeral(10);
        Expression three = new Numeral(3);
        Expression four = new Numeral(4);
        Expression ten_square = new Square(ten);
        Expression ten_square_minus_three = new Subtraction(ten_square, three);
        Expression three_times_of_four = new Multiplication(four, three);
        Expression ten_square_minus_three_plus_three_times_of_four = new Addition(ten_square_minus_three, three_times_of_four);
        Expression ten_square_minus_three_plus_three_times_of_four_square = new Square(ten_square_minus_three_plus_three_times_of_four);
        System.out.println(ten_square_minus_three_plus_three_times_of_four_square);
        System.out.println("=" + ten_square_minus_three_plus_three_times_of_four_square.evaluate());
    }
}

public 
class Substraction extends BinaryExpression {
    public Substraction(Expression left, Expression right) {
        super(left, right);
    }
    
    public String toString() {
        return "(" + left + " - " + right + ")";
    }
    
    public double evaluate() {
        return left.evaluate() - right.evaluate();
    }
}
