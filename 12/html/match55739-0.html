public class ExpressionTest {
    public static void main(String[] args) {
        Expression num1 = new Numeral(10);
        Expression num2 = new Numeral(-3);
        Expression num3 = new Numeral(4);
        Expression num4 = new Numeral(3);
        Expression square = new Square(num1);
        Expression addition = new Addition(square,num2);
        Expression multi = new Multiplication(num3,num4);
        Expression sum = new Addition(addition,multi);
        Expression expression = new Square(sum);
        System.out.println(expression.evaluate());
    }
}

src/Square.java

public class Square extends Expression {
  private Expression expression;

    public Square(Expression expression) {
        this.expression = expression;
    }

    @Override
    public String toString() {
        return "(" + expression + ") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}
