Numeral.java


public 
class Numeral extends Expression{
	private double value;
	public Numeral(){};
	public Numeral(double x) {this.value = x;}
	
	@Override 
	public String toString() {
		int x = (int) this.value;
		if ((double)x == this.value) return (String.valueOf(x)); 
		else return (String.valueOf(this.value));
	}
	@Override
	public double evaluate() {return this.value;}
}
