public class ExpressionTest  {
    public static void main(String[] args) {
        Expression a=new Numeral(10);
        Expression b=new Numeral(-3);
        Expression c=new Numeral(4);
        Expression d=new Numeral(3);
        Expression e=new Numeral(2);
        Expression f=new Numeral(0);
        Square bao1=new Square(a);
        Addition bao2=new Addition(bao1,b);
        Multiplication bao3=new Multiplication(c,d);
        Addition bao4=new Addition(bao2,bao3);
        Square bao5=new Square(bao4);
        System.out.println(bao5.toString());
        System.out.println(bao5.evaluate());
        Division g=new Division(a,f);
        System.out.println(g.toString());
        System.out.println(g.evaluate());

    }
}
