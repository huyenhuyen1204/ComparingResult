Addition.java

public class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        String str1;
        if (left instanceof Numeral || left instanceof Square)
            str1 = left.toString();
        else
            str1 = "(" + left.toString() +")";
        String str2;
        if (right instanceof Numeral || right instanceof Square)
            str2 = right.toString();
        else
            str2 = "(" + right.toString() +")";
        return str1 + " + " + str2;
    }

    @Override
    public double evaluate() {
        return left.evaluate()+right.evaluate();
    }
}

public class Multiplication extends BinaryExpression{

    }

        if (left instanceof Numeral|| left instanceof Square)
            s1 = left.toString();
        else
            s1 = "(" + left.toString() +")";
        String s2;
        if (right instanceof Numeral || right instanceof Square)
            s2 = right.toString();
        else
            s2 = "(" + right.toString() +")";
        return s1 + " * " + s2;
    }

    @Override
    public double evaluate() {
        return right.evaluate()*left.evaluate();

Square.java

public class Square extends Expression {
    private Expression expression;

    public Square(Expression expression) {
        this.expression = expression;
    }

    @Override
    public String toString() {
        String str1;
        if (expression instanceof Numeral)
            str1 = expression.toString();
        else
            str1 =  "(" + expression.toString() + ")";
        return "(" + str1 + ") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}

public class Subtraction extends BinaryExpression {
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        String str1;
        if (left instanceof Numeral || left instanceof Square)
            str1 = left.toString();
        else
            str1 = "(" + left.toString() +")";
        String str2;
        if (right instanceof Numeral) return str1 + " + -"+right.toString();
        else if ( right instanceof Square)
            str2 = right.toString();
        else
            str2 = "(" + right.toString() +")";
        return str1 + " - " + str2;
    }

    @Override
    public double evaluate() {
        return left.evaluate() - right.evaluate();
    }
}
