public class Division extends BinaryExpression {
    public Division(Expression l, Expression r)
    {
        super(l,r);
    }
    public String toString()
    {
        return "("+super.left.toString()+" / "+super.right.toString()+")";
    }
    public double evaluate() throws java.lang.ArithmeticException
    {
        if(super.right.evaluate()==0)   {
            throw new java.lang.ArithmeticException("Lá»—i chia cho 0");
        }
        return super.left.evaluate()/super.right.evaluate();
    }
}

src/Numeral.java

public class Numeral extends Expression {
    private double value;
    public Numeral(){};
    public Numeral(double value)
    {
        this.value = value;
    }
    public String toString()
    {
        return ""+(int)this.value;
    }
    public double evaluate()
    {
        return this.value;
    }
}

src/Subtraction.java

public class Subtraction extends BinaryExpression {
    public Subtraction(Expression l, Expression r)
    {
        super(l,r);
    }
    public String toString()
    {
        return "("+super.left.toString()+" + - "+super.right.toString()+")";
    }
    public double evaluate()
    {
        return super.left.evaluate()-super.right.evaluate();
    }
}
