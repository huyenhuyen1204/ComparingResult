class Division extends BinaryExpression{
    public Division(Expression left, Expression right){
        super(left, right);

    }
    @Override
    public double evaluate() {
        if(this.right.evaluate() !=0)
            return this.left.evaluate()/this.right.evaluate();
        else{
            throw new ArithmeticException("Lá»—i chia cho 0");
        }
    }

    public String toString() {
        return ("("+this.left.toString()+" / "+this.right.toString()+")");
    }
}

class ExpressionTest {

        Expression ten = new Numeral(10);
        Expression ten_square = new Square(ten);

        Expression right = new Numeral(3);

        Expression add = new Addition(ten, div);

    }
