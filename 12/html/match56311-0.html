    public static void main(String[] args) {
        Expression e1 = new Numeral(10);
        Expression s1 = new Square(e1);
        Expression e2 = new Numeral(-3);
        Expression s2 = new Addition(s1, e2);
        Expression s3 = new Multiplication(new Numeral(4), new Numeral(3));

Numeral.java

public class Numeral extends Expression {
    private double value;

    public Numeral(){};
    public Numeral(double value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return "" + (int)value + "";
    }

    @Override
    public double evaluate() {
        return value;
    }
}
