/**

 */
public 
class Numeral extends Expression {

    private double value;

    public Numeral(double value){
        this.value = value;
    }

    public Numeral(){

    }

    @Override
    public String toString() {
        int val = (int) value;
        
if (val == value) {
            return Integer.toString(val);
        } else {
            return Double.toString(value);
        }
    }

    @Override
    public double evaluate() {
        return value;
    }

}
