Solution/Addition.java



public class Addition extends BinaryExpression{
	public Expression left;
	public Expression right;
	public Addition(Expression left , Expression right) {
		super(left , right);
		this.left = left;
		this.right = right;
	}
	
	public String toString() {
		String res = "";
		if (left instanceof Numeral || left instanceof Square) res += left.toString();
		else res += "(" + left.toString() + ")";
		res += " + ";
		
if (right instanceof Numeral || right instanceof Square) res += right.toString();
		else res += "(" + right.toString() + ")";
		return res;
	}
	
	public double evaluate()
	{
		return left.evaluate() + right.evaluate(); 
	}
}
