class Numeral extends Expression {
    private double value;

    public Numeral(double value)
    {
        this.value = value;
    }

    public Numeral()
    {

    }

    @Override
    public String toString() {
        return String.valueOf((int)value);
    }

    public double getValue() {
        return value;
    }

    @Override
    public double evaluate() {
        return this.value;
    }
}
