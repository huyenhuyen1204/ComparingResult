Addition.java

public 
class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }
    public double evaluate()
    {
        return this.left.evaluate()+ this.right.evaluate();
    }
    @Override
    public String toString()
    {
        return "("+ this.left + " + " + this.right + ")";
    }
}

Multiplication.java

public 
class Multiplication extends BinaryExpression{
    public Multiplication(Expression left, Expression right){
        super(left, right);
    }
    public double evaluate(){
        return left.evaluate() * right.evaluate();
    }

    @Override
    public String toString() {
        return "(" + left + " * " + right + ")";
    }
}

Numeral.java

public 
class Numeral extends Expression {
    private double value;
    public Numeral(double value)
    {
        this.value=value;
    }
    public Numeral(){};

    @Override
    public String toString() {
        int a=(int) this.value;
        return Integer.toString(a);
    }

    @Override
    public double evaluate() {
        return this.value;
    }
}

Square.java

public 
class Square extends Expression {
    private Expression expression;
    public Square(Expression expression)
    {
        this.expression=expression;
    }

    @Override
    public String toString() {
        return "(" + this.expression + ")" + " ^ " +"2";
    }

    @Override
    public double evaluate() {
        return this.expression.evaluate()*this.expression.evaluate();
    }
}

Subtraction.java

public 
class Subtraction extends BinaryExpression{
    public Subtraction(Expression left, Expression right){
        super(left, right);
    }
    public double evaluate(){
        return left.evaluate() - right.evaluate();
    }

    @Override
    public String toString() {
        return "(" + left + " + -" + right + ")";
    }
}
