    @Test

void testExpression()
    {
        Numeral ten = new Numeral(10);
        Numeral three = new Numeral(3);
        Numeral negaThree = new Numeral(-3);
        Numeral four = new Numeral(4);
        Square square1 = new Square(ten);
        Addition addition1 = new Addition(square1, negaThree);
        Multiplication multiplication = new Multiplication(four, three);
        Addition addition2 = new Addition(addition1, multiplication);
        Square total = new Square(addition2);
        assertEquals("((((10) ^ 2 + -3) + (4 * 3))) ^ 2",total.toString());

class Square extends Expression {

    private Expression expression;

    public Square(Expression expression) {
        this.expression = expression;
    }

    @Override
    public String toString() {
        return "("+ expression + ") ^ 2" ;
    }

    @Override
    public double evaluate() {
        return expression.evaluate() * expression.evaluate();
    }
}
