import java.lang.ArithmeticException;

public class Division extends BinaryExpression {
    public Division(Expression left, Expression right) {
        super(left, right);
        if (right instanceof Numeral) {
            Numeral numeral = (Numeral)right;
            if(numeral.evaluate() == 0)
             throw new ArithmeticException("Lá»—i chia cho 0");
        }
    }

    @Override
    public String toString() {
        String s;
        if (left instanceof Numeral || left instanceof Square)
            s = left.toString();
        else
            s = "(" + left.toString() +")";
        String s1;
        if (right instanceof Numeral || right instanceof Square)
            s1 = right.toString();
        else
            s1 = "(" + right.toString() +")";
        return s + " / " + s1;
    }

    @Override
    public double evaluate() {
        return left.evaluate()/right.evaluate();
    }
}

Square.java

public class Square extends Expression{
    private Expression expression;
    public Square(Expression expression)
    {
        this.expression=expression;
    }

    @Override
    public String toString() {
        String s;
        if(expression instanceof Numeral)
        {
            s=expression.toString();
        }
        else
        {
            s="("+expression.toString()+")";
        }
        return "("+ s+") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}

public class Subtraction extends BinaryExpression{
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public String toString() {
        String s;
        if (left instanceof Numeral || left instanceof Square)
            s = left.toString();
        else
            s = "(" + left.toString() +")";
        String s1;
        if (right instanceof Numeral) return s + " + -"+right.toString();
        else if ( right instanceof Square)
            s1 = right.toString();
        else
            s1 = "(" + right.toString() +")";
        return s + " - " + s1;
    }

    @Override
    public double evaluate() {
        return left.evaluate() - right.evaluate();
    }
}
