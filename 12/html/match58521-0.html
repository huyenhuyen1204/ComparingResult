import org.junit.Test;
import static junit.framework.Assert.assertEquals;
public class ExpressionTest {
    @Test
    public void testExpression()
    {
        Numeral ten = new Numeral(10);
        Numeral three = new Numeral(3);
        Numeral negaThree = new Numeral(-3);
        Numeral four = new Numeral(4);
        Square sq = new Square(ten);
        Addition add1 = new Addition(sq, negaThree);
        Multiplication mul = new Multiplication(four, three);
        Addition add2 = new Addition(add1, mul);
        Square all = new Square(add2);
        assertEquals("((((10) ^ 2 + -3) + (4 * 3))) ^ 2",all.toString());
    }

}
