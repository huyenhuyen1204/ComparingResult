BinaryExpression.java

public abstract class BinaryExpression extends Expression {
    protected Expression left;
    protected Expression right;
    public BinaryExpression (Expression left, Expression right){
        this.left = left;
        this.right = right;
    }

    @Override
    public abstract String toString();

    @Override
    public abstract double evaluate();
}

public class ExpressionTest {
    public static void main(String [] args){
        Expression n1= new Numeral(10); //10
        Expression s1= new Square(n1); //10^2
        Expression n2= new Numeral(3);//3
        Expression st1= new Subtraction(s1,n2); //10^2 -3
        Expression n3 = new Numeral(3);//3
        Expression n4 = new Numeral(4);//4
        Expression m1 = new Multiplication(n3,n4);//4*3
        Expression a1 = new Addition(st1, m1);
        Expression s2 = new Square(a1);

        System.out.println(n1);
        System.out.println(s1);
        System.out.println(n2);
        System.out.println(st1);
        System.out.println(n3);
        System.out.println(n4);
        System.out.println(m1);
        System.out.println(a1);
        System.out.println(s2.evaluate());
    }
}

Numeral.java

public class Numeral extends Expression {
    private double value;
    public Numeral(double value){
        this.value = value;
    }
    public Numeral(){
    }

    @Override
    public String toString() {
        return null;
    }

    @Override
    public double evaluate() {
        return value;
    }
}

Square.java

public class Square extends Expression {
    private Expression expression;

    public Square(Expression e) {
        expression = e;
    }

    @Override
    public String toString() {
        return null;
    }

    @Override
    public double evaluate() {
        return Math.pow(expression.evaluate(),2);
    }
}
