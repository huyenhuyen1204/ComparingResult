Numeral.java

public 
class Numeral extends Expression {
    private double value;

    public Numeral(double value) {
        this.value = value;
    }
    public Numeral(){

    }
    @Override
    public String toString(){
        int answer = (int) this.value;
        return Integer.toString(answer);
    }
    @Override
    public  double evaluate() {
        return this.value;
    }


}
