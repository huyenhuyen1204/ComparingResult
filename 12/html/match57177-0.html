import java.lang.ArithmeticException;

public class Division extends BinaryExpression {

    private Expression ex1,ex2;

    public Division(Expression l, Expression r){
    	super(l, r);
        this.ex2=r;
        this.ex1=l;
    }

    @Override
    public double evaluate(){
        if (ex2.evaluate() == 0)
        {
        	throw new ArithmeticException("Lá»—i chia cho 0");
        }
        
        return this.ex1.evaluate() / this.ex2.evaluate();
    }

    public String toString() {
        return "(" + this.ex1.toString()+" / "+this.ex2.toString() + ")";
    }
    
    
}

1/Multiplication.java

public class Multiplication extends BinaryExpression {

    private Expression l,r;

    public Multiplication (Expression l,Expression r){
    	super(l, r);
        this.l=l;
        this.r=r;
    }
    
    public double evaluate() {
        return this.l.evaluate()*this.r.evaluate();
    }

    public String toString() {
        return "(" + this.l.toString()+" * "+this.r.toString() + ")";
    }

    
}

1/Subtraction.java

public class Subtraction extends BinaryExpression {

    private Expression left,right;

    public Subtraction(Expression left, Expression right){
    	super(left, right);
        this.left=left;
        this.right=right;
    }

    public double evaluate() {
        return this.left.evaluate()-this.right.evaluate();
    }

    public String toString() {
        return "("+ this.left.toString() +" - "+this.right.toString()+")";
    }

    
}
