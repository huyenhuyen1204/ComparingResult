class ExpressionTest {
    public static void main(String[] args) {
        Expression number10 = new Numeral(10);
        Expression numberNeg3 = new Numeral(3);
        Expression number4 = new Numeral(4.12);
        Expression number3 = new Numeral(3);
        Expression number0 = new Numeral(0);
        Expression sqr1 = new Square(number10);
        Expression sub1 = new Subtraction(sqr1, numberNeg3);
        Expression mult1 = new Multiplication(number4, number3);
        Expression add1 = new Addition(sub1, mult1);
        Expression sqr2 = new Square(add1);
        Expression div1 = new Division(sqr2, number0);
        System.out.println(sqr2);
        System.out.println(sqr2.evaluate());
        System.out.println(div1);
        System.out.println(div1.evaluate());

    }
}
