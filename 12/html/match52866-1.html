public class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public 
String toString() {
        String s1;
        if (left instanceof Numeral || left instanceof Square)
            s1 = left.toString();
        else
            s1 = "(" + left.toString() +")";
        String s2;
        if (right instanceof Numeral || right instanceof Square)
            s2 = right.toString();
        else
            s2 = "(" + right.toString() +")";
        return s1 + " + " + s2;
    }

    @Override
    public double evaluate() {
        return left.evaluate()+right.evaluate();
    }
}

Square.java

public class Square extends Expression {
    private Expression expression;

    public Square(Expression expression) {
        this.expression = expression;
    }

    @Override
    public 
String toString() {
        String s1;
        if (expression instanceof Numeral)
            s1 = expression.toString();
        else
            s1 =  "(" + expression.toString() + ")";
        return "(" + s1 + ") ^ 2";
    }

    @Override
    public double evaluate() {
        return expression.evaluate()*expression.evaluate();
    }
}

public class Subtraction extends BinaryExpression {
    public Subtraction(Expression left, Expression right) {
        super(left, right);
    }

    @Override
    public 
String toString() {
        String s1;
        if (left instanceof Numeral || left instanceof Square)
            s1 = left.toString();
        else
            s1 = "(" + left.toString() +")";
        String s2;
        if (right instanceof Numeral) return s1 + " + -"+right.toString();
        else if ( right instanceof Square)
            s2 = right.toString();
        else
            s2 = "(" + right.toString() +")";
        return s1 + " - " + s2;
    }

    @Override
    public double evaluate() {
        return left.evaluate() - right.evaluate();
    }
}
