Numeral.java

public 
class Numeral extends Expression
{
    private double value;

    public Numeral(double value)
    {
        this.value = value;
    }

    public Numeral()
    {

    }

    @Override
    public String toString() {
        Double val = value;
        if (val == Math.floor(val))
        {
            Integer m = (int)Math.floor(val);
            return m.toString();
        }
        return val.toString();
    }

    @Override
    public double evaluate() {
        return value;
    }

}
