src/BinaryExpression.java

public abstract class BinaryExpression extends Expression {
    protected Expression left;
    protected Expression right;

    protected String leftString;
    protected String rightString;



    public BinaryExpression(Expression left, Expression right) {
        this.left = new Numeral(left.evaluate());
        this.right = new Numeral(right.evaluate());
        this.leftString = left.toString();
        this.rightString = right.toString();
    }



    @Override
    public String toString() {
        return this.leftString + " " + this.rightString;
    }

    @Override
    public double evaluate() {
        return this.right.evaluate()+this.left.evaluate();
    }
}

src/Division.java

public class Division extends BinaryExpression{
    public Division(Expression left, Expression right) {
        super(left, right);
    }
    @Override
    public String toString() {
        return "("+this.leftString+ " / " + this.rightString+")";
    }

    @Override
    public double evaluate() {
        double evaluate = 0;
        try{
            evaluate = this.left.evaluate() / this.right.evaluate();

        }catch (ArithmeticException e) {
            System.out.println(e.toString());
        }
        return evaluate;
    }
}

src/Numeral.java

public class Numeral extends Expression {
    private double value;

    public Numeral() {

    }
    public Numeral(double value) {
        this.value = value;
    }

    public void setValue(double value) {
        this.value = value;
    }

    public double getValue() {
        return value;
    }

    @Override
    public String toString() {
        return (int) this.getValue() +"";
    }
    public double evaluate() {
        return this.getValue();
    }
}

src/Square.java

public class Square extends Expression {
    private Expression expression;
    private String expressionString;


    public Square(Expression expression) {
        this.expression = new Numeral(expression.evaluate());
        this.expressionString = expression.toString();
    }

    public void setSquare(Expression expression) {
        this.expression = new Numeral(expression.evaluate());
    }

    public Expression getSquare() {
        return expression;
    }

    @Override
    public String toString() {
        return "(" + this.expressionString + ") ^ 2" ;
    }

    @Override
    public double evaluate() {
        return this.expression.evaluate()*this.expression.evaluate();
    }
}
