Addition.java

public class Addition extends BinaryExpression {
    public Addition(Expression left, Expression right) {
        super(left, right);
    }
    public String toString() {
        String s1;
        if (left instanceof Numeral || left instanceof Square)
            s1 = left.toString();
        else
            s1 = "(" + left.toString() +")";
        String s2;
        if (right instanceof Numeral || right instanceof Square)
            s2 = right.toString();
        else
            s2 = "(" + right.toString() +")";
        return s1 + " + " + s2;
    }
    public double evaluate() {
        return left.evaluate()+right.evaluate();
    }
}

public class Multiplication extends BinaryExpression{
    public Multiplication(Expression left, Expression right) {
        super(left, right);
    }
    public String toString() {
        String s1;
        if (left instanceof Numeral|| left instanceof Square)
            s1 = left.toString();
        else
            s1 = "(" + left.toString() +")";
        String s2;
        if (right instanceof Numeral || right instanceof Square)
            s2 = right.toString();
        else
            s2 = "(" + right.toString() +")";
        return s1 + " * " + s2;
    }
    public double evaluate() {
        return right.evaluate()*left.evaluate();
    }
}

public class Square extends Expression {
    private Expression expression ;

    public Square(Expression expression){
        this.expression = expression ;
    }
    public String toString() {
        String s1;
        if (expression instanceof Numeral)
            s1 = expression.toString();
        else
            s1 =  "(" + expression.toString() + ")";
        return "(" + s1 + ") ^ 2";
    }

    public double evaluate() {
        return this.expression.evaluate()*this.expression.evaluate();
    }
