import java.math.BigDecimal;

public class Circle {
    protected final double PI = 3.14;
    private double radius = 1.0;
    private String color = "red";

    public Circle(){

    };

//    Circle(int radius){
//        double k = radius;
//        this.radius = k;
//    }

    public Circle( double radius){
        this.radius = radius;
    }

    public Circle(double radius, String color){
        this.radius = radius;
        this.color = color;
    }

    public double getRadius() {
        return radius;
    }

    public void setRadius(double radius) {
        this.radius = radius;
    }

    public String getColor() {
        return color;
    }

    public void setColor(String color) {
        this.color = color;
    }

    public double getArea(){
        BigDecimal r = new BigDecimal(radius);
        BigDecimal p = new BigDecimal(PI);
        BigDecimal result = r.multiply(r.multiply(p));
        //double x = (double) result;
        return result.doubleValue();
    };

    public double getCircuit(){
        BigDecimal r = new BigDecimal(radius);
        BigDecimal i = new BigDecimal(2);
        BigDecimal p = new BigDecimal(PI);
        BigDecimal c = i.multiply(p.multiply(r));
        return  c.doubleValue();
    }

    public String toString(){
        //String a = getClass().getName();
        return "Circle[radius=" + radius +",color=" + color + "]";
    }
}

import java.math.BigDecimal;
import java.math.BigInteger;

public class Cylinder extends Circle{
    private double height = 1.0;

    public Cylinder(){

    };

    public Cylinder(double height){
        setHeight(height);
    }

    public Cylinder(double height, double radius){
        setRadius(radius);
        this.height = height;
    }
    public Cylinder(double height,double radius,  String color){
        setRadius(radius);
        this.height = height;
        setColor(color);
    }

    public double getHeight() {
        return height;
    }

    public void setHeight(double height) {
        this.height = height;
    }

    public double getVolume(){
        BigDecimal S = new BigDecimal(super.getArea());
        BigDecimal h = new BigDecimal(height);
        BigDecimal v = S.multiply(h);
        return v.doubleValue();
    }


    @Override
    public String toString(){
        return getClass().getName() + "[" + super.toString() + ",height=" + height + "]";
    }

    @Override
    public double getArea(){
        BigDecimal s = new BigDecimal(super.getArea());
        BigDecimal h = new BigDecimal(height);
        //BigDecimal c = new BigDecimal(getCircuit());
        BigDecimal as = (new BigDecimal(super.getCircuit()).multiply(h));

        BigDecimal all = as.add(s.multiply(new BigDecimal(2)));

        return all.doubleValue();
    }
