class Student extends Person {
    private String program;
    private int year;
    private double fee;
    public Student(String nName, String nAddress, String nProgram, int nYear, double nFee){
        super(nName,nAddress);
        this.setProgram(nProgram);
        this.setYear(nYear);
        this.setFee(nFee);
    }
    public String getProgram(){
        return this.program;
    }
    public void setProgram(String nProgram){
        this.program=nProgram;
    }
    public int getYear(){
        return this.year;
    }
    public void setYear(int nYear){
        this.year=nYear;
    }
    public double getFee(){
        return this.fee;
    }
    public void setFee(double nFee){
        this.fee=nFee;
    }
    public String toString(){
        return "Student[" + super.toString() + ",program=" + this.getProgram() + ",year=" + this.getYear() + ",fee=" + this.getFee() + "]";
    }
}
